class FlatMaker(object):
    """ Creates flat fields from input QE maps"""

    def __init__(self, data_dir: str,
                 grism_parameters: dict,
                 qe_files_list: List[str],
                 qe_range: Tuple[float, float],
                 logger: 'ElementsKernel.Logging'):
        """         
        Parameters
        ----------
            
        data_dir : str
            full paths for the input QE maps FITS files

        grism_parameters: dict
             nested dictionary having grism (blue/red specific entries 
             { 'RGS' : {'alpha'      : -0.9,
                        'hdu_extn'   : 'RGS000_ORDER_1'
                       },
               'BGS': {'alpha'      : -0.75,
                        'hdu_extn'   : 'BGS000_ORDER_1'
                       }
             }

        qe_files_list : List[str]
            A list of input QE FITS files

        qe_range : Tuple[float, float]
            A tuple of (max_qe, min_qe) cutoff values

        logger: ElementsKernel.Logging
            logs various messages and actions
        """

        self._logger = logger
        self._qe_files_list = qe_files_list
        self._grism_specific_items = grism_parameters

        self._weights = {}
        self._wavelengths = {}
        self._wavelength_ranges = {}

        qe_file_path = os.path.join(data_dir, self._qe_files_list[0])

        self._wave = []
        self._logger.info(f'Loading wavelengths from input QE file: {qe_file_path}')
 
        self._load_wavelengths(qe_file_path)

        self._qe_hi = qe_range[0] 
        self._qe_lo = qe_range[1]


    def _load_wavelengths(self, qe_file_path: str) -> None:
        """ loads full range of wavelengths from input QE maps

        Parameters
        ----------
  
        qe_file_path : str
            full path of QE input FITS file
 
        """

        with fits.open(qe_file_path) as hdu:
            self._wave = np.zeros(len(hdu), dtype=float)                 
            for i in range(1, len(self._wave)):
                self._wave[i] = hdu[i].header['WAVELEN']

    
    def _get_wavelength_range(self, wavelengths: np.ndarray, 
                              transmissions: np.ndarray) -> Tuple[int, int]:
        """Identifies useful range from the full range of wavelengths

        From input transmission curve, covering the full wavelength 
        range where throughput is > 10% of max. value

        Parameters
        ----------

        wavelengths : numpy array of floats
            contains complete set of wavelengths available in input
            QE map for a grism

        transmissions : numpy array of floats
            contains complete set of transmission values available in 
            input QE map for a grism

        Returns
        -------

        index_start : int
            starting index for wavelengths

        index_end : int
            ending index for wavelengths
        """

        transmissions_cutoff = 0.1 * transmissions.max()

        wavelength_low = wavelengths[min(np.where(transmissions > transmissions_cutoff)[0])]

        wavelength_high = wavelengths[max(np.where(transmissions > transmissions_cutoff)[0])]

        index_start = max(np.where(self._wave <= wavelength_low)[0])

        index_end = min(np.where(self._wave >= wavelength_high)[0])

        return index_start, index_end

    def load_grism_transmission_curves(self, grism_file_path: str) -> None:
        """ Interpolates transmission values from input grism curves

        Reads in NISP grism transmission data for various wavelengths
        from input FITS file. Subsequently, interpolates that data 
        for `self.wavelengths` points, Ends by passing the transmission 
        values to `_calculate_weights()`

        Parameters
        ----------

        grism_file_path : str
            full path for NISP Grism Curve FITS file

        """

        with fits.open(grism_file_path) as hdu:
            for grism in self._grism_specific_items: 

                data = hdu[self._grism_specific_items[grism]['hdu_extn']].data
                alpha = self._grism_specific_items[grism]['alpha']

                index_start, index_end = self._get_wavelength_range(data['wavelength'],
                                                                    data['transmission'])

                self._wavelength_ranges[grism] = (index_start, index_end)

                self._wavelengths[grism] = self._wave[index_start: index_end + 1]
                
                f = interp1d(data['wavelength'], data['transmission'], kind='cubic')

                trans = f(self._wavelengths[grism])

                self._weights[grism] = self._calculate_weights(
                    self._wavelengths[grism], trans, alpha)




    def _calculate_weights(self, wavelengths: np.ndarray, 
                           transmissions: np.ndarray, 
                           alpha: float) -> np.ndarray:
        """ Calculates weights for QE transmissions for each wavelength

        Parameters
        ----------

        wavelengths : numpy array of floats
            requisite wavelength range for a grism

        transmissions : numpy array of floats
            transmission values for aforementioned wavelengths

        alpha : float
            a constant (see Notes below for more)

        Returns
        -------

        weights : numpy array of floats
            weights calculated for corresponding input wavelengths

        Notes
        -----

        alpha used to estimate the following,
        Median zody spectrum intensity I_nu ~ lambda^alpha

        Delta nu also varies from 2.4 at 12500 to 1.12 at the red end

        """

        waves1 = 1.e-3 * wavelengths
        intensity_nu_zody = waves1 ** alpha
        delta_nu = self._check_delta_nu(wavelengths)

        const = consts.h * consts.c * 1.e6
        zody_photons = intensity_nu_zody * delta_nu * transmissions / (const / waves1)

        weights = zody_photons / np.sum(zody_photons)

        return weights


    def _check_delta_nu(self, wavelengths: np.ndarray) -> np.ndarray:
        """

        Parameters
        ----------

        wavelengths : numpy array of floats
            requisite range of wavelengths 

        Returns
        -------

        delta_nu : numpy array of floats
            change in nu values interpolated for `wavelengths`

        Notes
        -----

        `resolution` is nyquist sampled
        """

        lambda0 = wavelengths[0] 
        resolution = 500
        delta_lambda = lambda0 / (resolution * 2.)

        lambdas = np.arange(750, dtype=float)
        lo_lambdas = np.arange(750, dtype=float)
        hi_lambdas = np.arange(750, dtype=float)

        lambdas[0] = lambda0
        index = 0

        lo_lambdas[index] = lambdas[index] - 0.5 * delta_lambda
        hi_lambdas[index] = lambdas[index] + 0.5 * delta_lambda

        while lambdas[index] < wavelengths[-1]:
            index = index + 1
            lambdas[index] = lambdas[index - 1] + delta_lambda
            lo_lambdas[index] = lambdas[index] - 0.5 * delta_lambda
            hi_lambdas[index] = lambdas[index] + 0.5 * delta_lambda

        lambdas = lambdas[0:index]

        nanometers_per_meter = 1.0e9

        c_nms = consts.c * nanometers_per_meter

        delta_nu = c_nms * (1. / lo_lambdas[0: index] - 1. / hi_lambdas[0: index])

        f = interp1d(lambdas, delta_nu, kind='cubic', fill_value="extrapolate")

        delta_nu_interp = f(wavelengths)
        delta_nu_interp = 1.e-11 * delta_nu_interp

        return delta_nu_interp


    def _get_detid_scaid_qefilename_dict(self, data_dir: str) -> dict:
        """ Creates dictionary between Detector ID with SCA ID & QE input file

        Parameters
        ----------
        data_dir : str
            data directory where QE input FITS files are located

        Returns
        -------
        detid_scaid_qefilename_dict : dict
            dictionary for {Detector ID : List[SCA ID, QE filename]}

        """

        detid_scaid_qefilename_dict = {}
        for qe_filename in self._qe_files_list:
            qe_filepath = os.path.join(data_dir, qe_filename)

            with fits.open(qe_filepath) as hdu:
                det_id = hdu[1].header['SCE_POS']
                sca_id = hdu[1].header['SCA_ID']

                scaid_qefilename_list = [sca_id, qe_filename]
                detid_scaid_qefilename_dict[det_id] = scaid_qefilename_list

        return detid_scaid_qefilename_dict 


    def _get_mask(self, grism: str, qe_filepath: str) -> np.ndarray:
        """ Creates mask of good pixels for a detector for all wavelengths

        Parameters
        ----------

        grism : str
            grism color - BGS or RGS

        qe_filepath : full path for a detector's QE input file
        
        Returns
        -------

        mask : numpy array
            boolean mask of all pixels within acceptable QE range
            (self._qe_lo, self._qe_hi) for all wavelengths

        Notes
        -----

        A pixel is dubbed as "bad" at ALL wavelengths, as soon as its
        QE value is found to be outside of self._qe_lo > QE > self._qe_hi
        range for ANY single wavelength. That pixel is subsequently
        thrown out of the mask

        """

        mask = np.ma.make_mask(np.ones((2040, 2040), dtype=np.int32))

        range_start = self._wavelength_ranges[grism][0]
        range_end = self._wavelength_ranges[grism][1]

        self._logger.info(f'Creating mask for {qe_filepath}')

        for wave_index in range(range_start, range_end + 1):
            with fits.open(qe_filepath) as hdu:
                detflat = hdu[wave_index].data

                detflat[(detflat < self._qe_lo) | (detflat > self._qe_hi)] = np.nan
                detflat_noedge = detflat[4:2044, 4:2044]  
                current_mask = np.isfinite(detflat_noedge) 

            mask = mask & current_mask

        return mask

    def write_fits(self, data_dir: str, 
                   output_fits_filepath: str,
                   text_history: List[str]) -> None: 
        """ Writes flat-fielded QE maps into grism-specific FITS files

        Parameters
        ----------

        data_dir : str
            Directory for Input QE maps FITS files

        output_fits_filepath : str
            output FITS filepath 
  
        text_history: List[str]
            line-by-line text, as List, to describe the history

        """

        self._logger.info(f'Start writing output file: {output_fits_filepath}')

        grisms = list(self._grism_specific_items)
     
        detid_scaid_qefilename_dict = self._get_detid_scaid_qefilename_dict(data_dir)
        det_ids = sorted(detid_scaid_qefilename_dict)

        hdu_list = fits.HDUList()

        hdu = fits.PrimaryHDU()

        for line in text_history:
            hdu.header.add_history(line)

        hdu_list.append(hdu)

        for grism in grisms:
   
            weights = self._weights[grism]

            for det_id in det_ids: 

                sca_id = detid_scaid_qefilename_dict[det_id][0]

                qe_filename = detid_scaid_qefilename_dict[det_id][1]
                qe_filepath = os.path.join(data_dir, qe_filename)

                true_flat = np.zeros((2040, 2040), dtype=np.float32)

                range_start = self._wavelength_ranges[grism][0]
                range_end = self._wavelength_ranges[grism][1]

                mask = self._get_mask(grism, qe_filepath)

                self._logger.info(f'Loading input QE map from file {qe_filename}')

                for wave_index in range(range_start, range_end + 1): 
                    with fits.open(qe_filepath) as hdu:
                        detflat = hdu[wave_index].data 

                    detflat_noedge = detflat[4:2044, 4:2044]

                    true_flat[mask] = (true_flat[mask] +
                                        weights[wave_index - range_start] * detflat_noedge[mask])

                true_flat = true_flat / 100.

                hdu = fits.ImageHDU(true_flat)
                hdu.header['DETNAME'] = f'DET_{det_id}' 
                hdu.header['SCAID'] = sca_id
                hdu.header['CalibID'] = grism
                hdu.header['EXTNAME'] = f'DET_{det_id}.{grism}.FLAT'
                hdu_list.append(hdu)

                self._logger.info(f'{grism} ImageHDU for DET_{det_id}, SCA ID {sca_id} created')

        hdu_list.writeto(output_fits_filepath, overwrite=True)
        hdu_list.close()
        self._logger.info(f'Finished writing output file {output_fits_filepath}')
