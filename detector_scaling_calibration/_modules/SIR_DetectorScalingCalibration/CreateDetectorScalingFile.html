def mainMethod(args):
    """
    @brief The "main" method.
    
    @details This method is the entry point to the program. In this sense, it is
    similar to a main (and it is why it is called mainMethod()).
    """

    module_name = "CreateDetectorScalingFile"

    logger = log.getLogger(module_name)

    logger.info('#')
    logger.info(f'# Entering {module_name} mainMethod()')
    logger.info('#')

    work_dir = args.workdir
    data_dir = os.path.join(work_dir, 'data')

    mdb = DpdMdbDataBase(os.path.join(work_dir, args.mdb))

    grism_curve_filename = mdb.get_transmission_curve_file()
    grism_curve_filepath = os.path.join(work_dir, 'data', grism_curve_filename)

    qe_files_list = mdb.get_nisp_qe_file()

    grism_parameters = {'RGS': {'alpha': -0.9,
                                'hdu_extn': 'RGS000_ORDER_1'
                                },
                        'BGS': {'alpha': -0.75,
                                 'hdu_extn': 'BGS000_ORDER_1'
                                 }
                        }

    flat_maker = FlatMaker(data_dir=data_dir,
                           grism_parameters=grism_parameters,
                           qe_files_list=qe_files_list,
                           qe_range=[120.0, 5.0],
                           logger=logger)

    logger.info(f'Loading input grism transmission files.')

    flat_maker.load_grism_transmission_curves(grism_curve_filepath)
 
    txt_history = []
    txt_history.append(f'Date: {dt.date.today().isoformat()}')
    txt_history.append('Detector QE scaling file for SIR pipeline')
    txt_history.append('Total 32 extensions, 1 per detector, 16 per grism')
    txt_history.append('RGS alpha= -0.9 & BGS alpha= -0.75 used')
    txt_history.append('Median zody spectrum intensity I_nu ~ lambda^alpha used')
    txt_history.append('Wavelength range for input transmission curve throughput > 10% of max considered')
   
    instance_id = f'{args.outfile}' 

    # get unique FITS filename
    fits_output = DpdSirDetectorScaling.create_container_name(instance_id)
    fits_output_path = os.path.join(work_dir, 'data', fits_output)
    # write fits
    flat_maker.write_fits(data_dir, fits_output_path, txt_history)

    # write XML
    xml_obj = DpdSirDetectorScaling()
    xml_obj.append_data(fitsfile=fits_output_path)
    xml_obj.write(os.path.join(work_dir, args.outfile))

    logger.info('#')
    logger.info(f'# Exiting {module_name} mainMethod()')
    logger.info('#')
