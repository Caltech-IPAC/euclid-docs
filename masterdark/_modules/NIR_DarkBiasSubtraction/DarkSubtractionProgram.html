def defineSpecificProgramOptions():
    """
    @brief
        Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for DarkSubtractionProgram
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser("""Dark subtraction.  Usage syntax:
                ~> ERun NIR_DarkBiasSubtraction darkSubtraction      \\
                   --infile=$workdir/data/infile.fits                \\
                   --masterdark==$workdir/input/NIR-MasterDark.xml   \\
                   --outfile=outfile.fits                            \\
                   --workdir=./                                      \\
                   --logdir=./
              """)

    parser.add_argument('--infile', type=str, dest='infile',
                        help='Input exposure Fits file')
    parser.add_argument('--masterdark', type=str, dest='masterdark',
                        help='XML file for Master Dark file')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='Output Fits file')
    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method
    @details
        This method is the entry point to the program

        Example to use this script:
        ./run darkSubtPyProgram --infile=inexposure.fits
                                --masterdark=calfile.xml
                                --outfile=outfile.fits
                                --workdir=workdir --logdir=logdir
    """
    logger = log.getLogger('DarkSubtractionProgram')
    logger.info('Entering DarkSubtractionProgram mainMethod')

    imgfile = os.path.join(args.workdir, args.infile)
    if os.path.isfile(imgfile):
        logger.info('Input image file: ' + imgfile)
    else:
        msg = 'Image File {} not found'.format(imgfile)
        logger.info(msg)
        raise IOError(msg)

    xmlpath = os.path.join(args.workdir, args.masterdark)
    logger.info('The Master Dark xml file: ' + xmlpath)

    # Call XmlReader
    xmlR = XmlReader()
    fitsFile = xmlR.get_masterdark_fits(xmlpath)
    calfile = os.path.join(args.workdir, 'data', fitsFile)
    if os.path.isfile(calfile):
        logger.info('The Master Dark FITS file: ' + calfile)
    else:
        msg = 'The Master Dark File {} not found'.format(calfile)
        logger.info(msg)
        raise IOError(msg)

    outfile = os.path.join(args.workdir, args.outfile)
    logger.info('The output file: ' + outfile)

    obj = DarkSubtraction(imgfile, calfile, outfile)
    obj.apply()

    logger.info('Exiting DarkSubtractionProgram mainMethod')
