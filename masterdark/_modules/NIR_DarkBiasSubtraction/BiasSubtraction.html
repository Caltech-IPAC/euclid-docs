class BiasSubtraction(object):
    """
    @class BiasSubtraction

    @brief This class applies bias subtraction or reference pixel correction.
           Please refer to "Addendum to Reference Pixel Subtraction Method
           for EUCLID SCS Noise Specification", EUCL-IPN-TN-7-001a.

           This module will be required if fowler sampling or ramp is used for imaging.
           In this case reference pixels around the image provide the bias level
           and need to be analyzed to determine the bias level to remove. 

           If it is not in fowler or ramp mode, simply subset the data to
           2040x2040 and return.

    """

    logger = log.getLogger('BiasSubtraction')
    def __init__(self, imgfile, outfile):
        """
        @brief Constructor
        """

        self._imgfile = imgfile
        self._outfile = outfile

        self.NUM_CH = 32 #number of channels
        self.MPIX = 64   #number of pixels in j direction in Up/Down frame
        self.NPIX = 4    #number of pixesl in j direction in Left/Right frame
                         #it's also the number of pixels in i direction in Up/Down
        self.NN = 4      #the 2*nn+1 average window for Left/Right correction

    def apply(self):
        """
        @brief Method to apply the bias frame subtraction
        """
        #read in image data file
        self.logger.info("BiasSubtraction.apply(): read in input data")
        imgframe = NispSurveyExposure.readFits(self._imgfile)

        #init final output
        outframe = NispSurveyExposure(2040, 2040)
        outframe.setMetadata(imgframe.getMetadata())

        #check if it's in fowler or ramp mode
        #if not, simply subset the data and write out
        mode = str(imgframe.getMetadata().get("READMODE"))
        self.logger.info("BiasSubtraction.apply(): The input data READMODE is: " + mode)
        numDetectors = imgframe.getNumDetectors()

        if (mode != "UpTheRamp") and (mode != "FowlerSamples"):
           self.logger.info("BiasSubtraction.apply(): No correction is done. " \
                            "Simply crop the data from 2048x2048 to 2040x2040.")
           #subset the data and return
           for det in range(numDetectors):
               detPtr = imgframe.getDetector(det)

               newdet = detPtr.extractSubPortion(4,4,2040,2040)

               pl = detPtr.getMetadata()
               pl.set('CRPIX1', pl.get('CRPIX1') - 4.)  #update the meta
               pl.set('CRPIX2', pl.get('CRPIX2') - 4.)
               newdet.setMetadata(pl)

               #add to the final output
               outframe.addDetector(newdet)

           # saving all three layers in a new FITS file
           outframe.writeFits(self._outfile)
           self.logger.info("BiasSubtraction.apply(): Written to file: " + self._outfile)

           return

        #if it is in fowler or ramp mode, do the correction
        self.logger.info("BiasSubtraction.apply(): starting reference pixel correction... ")
        for det in range(numDetectors):
            detPtr = imgframe.getDetector(det)
            detId = detPtr.getMetadata().get("DET_ID")
            self.logger.info("BiasSubtraction.apply(): reference pixel correction at detector " + str(detId))

            # get the image data layer
            sciimg = detPtr.getScience()

            # get numpy arrays
            sciimgarray = sciimg.getArray()

            #Up-Down reference pixels correction
            self.updownCorr(sciimgarray)

            #Left-Right reference pixels correction
            self.leftrightCorr(sciimgarray)

            #subset the detector to the center 2040x2040
            newdet = detPtr.extractSubPortion(4,4,2040,2040)

            pl = detPtr.getMetadata()
            pl.set('CRPIX1', pl.get('CRPIX1') - 4.)  #update the meta
            pl.set('CRPIX2', pl.get('CRPIX2') - 4.)
            newdet.setMetadata(pl)

            #add to the final output
            outframe.addDetector(newdet)

        # saving to a new FITS file
        outframe.writeFits(self._outfile)
        self.logger.info("BiasSubtraction.apply(): Written to file: " + self._outfile)

        return

    def updownCorr(self, sciimgarray):
        """
         Up and Down reference pixel correction.
         
         Args:
             sciimgarray: 2D numpy array, containing the science data layer

         Returns:
             Implicit return.  The input sciimgarray got modified in this funtion.
        """

        self.logger.info("BiasSubtraction.apply(): Up-Down reference pixel correction")

        dims = sciimgarray.shape
        num_ch = self.NUM_CH
        npix = self.NPIX
        mpix = self.MPIX

        for ch in range(0, num_ch):
            punmasked = []

            #Up reference pixels
            pud = sciimgarray[0:npix, ch*mpix:(ch+1)*mpix]
            res, med, nmad = self.getstats(pud)
            punmasked += list(res.data[res.mask])

            #Down reference pixels
            pud = sciimgarray[dims[0]-npix:dims[0], ch*mpix:(ch+1)*mpix]
            res,med,nmad = self.getstats(pud)
            punmasked += list(res.data[res.mask])
            
            x = np.mean(np.array(punmasked))

            #sciimgarray[npix:dims[0]-npix, ch*mpix:(ch+1)*mpix] -= x  ##Old doc
            sciimgarray[0:dims[0], ch*mpix:(ch+1)*mpix] -= x

        return

    def leftrightCorr(self, sciimgarray):
        """
         Left-Right reference pixels correction

         Args:
             sciimgarray: 2D numpy array, containing the science data layer

         Returns:
             Implicit return.  The input sciimgarray got modified in this funtion.
        """
        self.logger.info("BiasSubtraction.apply(): Left-Right reference pixel correction")

        dims = sciimgarray.shape
        npix = self.NPIX
        nn = self.NN

        #Left reference pixels
        #plr = sciimgarray[npix:dims[0]-npix, 0:npix] ##Old doc
        plr = sciimgarray[0:dims[0], 0:npix]

        resl, med, nmad = self.getstats(plr)

        #Right reference pixels
        #plr = sciimgarray[npix:dims[0]-npix, dims[1]-npix:dims[1]] ##Old doc
        plr = sciimgarray[0:dims[0], dims[1]-npix:dims[1]]

        resr, med, nmad = self.getstats(plr)

        for i in range(npix, dims[0]-npix):
            #compute the average over a 2*n+1 window of the left and right
            #reference pixels
            punmasked = []
            pll = resl[i-nn:i+nn+1, :]          
            prr = resr[i-nn:i+nn+1, :]          
            punmasked += list(pll.data[pll.mask])
            punmasked += list(prr.data[prr.mask])
            y = np.mean(np.array(punmasked))

            sciimgarray[i, npix:dims[1]-npix] -= y

        return

    @staticmethod
    def getstats(pixelarray):
        med = np.median(pixelarray)
        nmad = 1.4826 * np.median(np.fabs(pixelarray- med))
        masku = (pixelarray > (med+5*nmad)) | (pixelarray < (med-5*nmad))
        res = ma.array(pixelarray, mask=~masku)
        #A True in the retrun mask means the pixel is NOT masked by the criteria!
        return res, med, nmad
