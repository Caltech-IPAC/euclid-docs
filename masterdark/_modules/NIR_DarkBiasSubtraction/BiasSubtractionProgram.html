def defineSpecificProgramOptions():
    """
    @brief
        Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for BiasSubtractionProgram
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser()
    parser.add_argument('--infile', type=str, dest='imgframefile',
                        help='Input image frame Fits file')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='Output Fits file')
    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method
    @details
        This method is the entry point to the program

        Example to use this script:
        ./run biasSubtPyProgram --infile imgframe.fits --outfile outfile.fits --workdir workdirpath --logdir logdirpath
    """
    logger = log.getLogger('BiasSubtractionProgram')
    logger.info('Entering BiasSubtractionProgram mainMethod')

    imgfile = os.path.join(args.workdir, args.imgframefile)
    if os.path.isfile(imgfile):
       logger.info('Input image frame file: ' + imgfile)
    else:
       msg = 'Image File {} not found'.format(imgfile)
       logger.info(msg)
       raise IOError(msg)

    outfile = os.path.join(args.workdir, args.outfile)
    logger.info('Outfile file: ' + outfile)

    obj = BiasSubtraction(imgfile, outfile)
    obj.apply()

    logger.info('Exiting BiasSubtractionProgram mainMethod')
