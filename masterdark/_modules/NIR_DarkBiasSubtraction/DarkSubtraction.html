class DarkSubtraction(object):
    """
    @class DarkSubtraction

    @brief This class subtracts the dark current from the detectors.  The
           subtracted dark current could potentially depend on the time the
           exposure was taken, temperature of the detector, and the exposure
           time and mode of detector operation.  This apply module will select
           the appropriate dark using the time-stamp, temperature, detector mode,
           and exposure time of the data frame.

    """

    def __init__(self, imgfile, calfile, outfile):
        """
        @brief Constructor
        """

        self.logger = log.getLogger('DarkSubtraction')
        self._imgfile = imgfile
        self._calfile = calfile
        self._outfile = outfile

    def apply(self):
        """
        @brief Method to apply the dark frame subtraction
        """
  
        # Read In the input exposure and dark calibration fits file
        # (file format: fits file with 3x16 extensions, wheres 3 means
        # DET##.SCI, DET##.VAR, DET##.DQ; 16 is for 16 detectors; the
        # primary HDU has no data but headers.)
        self.logger.info("DarkSubtraction.apply(): read in input data")
        calStr = str(self._calfile)
        cal = NispSurveyExposure.readFits(calStr)
        calmeta = cal.getMetadata()

        exp = NispSurveyExposure.readFits(self._imgfile)
        meta = exp.getMetadata()
        tempf = os.path.basename(calStr)
        tempf = tempf[0: tempf.rfind('T')]    #no long lines before astropy 4.0
        meta.set("MDARKCAL", tempf, "Master dark calibration file")
        meta.set("ORIGIN", "OU-NIR", "Euclid SGS origin")
        #meta.set("VERSION", calmeta.get("VERSION"), "Master Dark build version")

        # make sure both input data and calibraton data have equal
        # number of detectors
        numDect = exp.getNumDetectors()
        numDect0 = cal.getNumDetectors()
        if numDect != numDect0:
           self.logger.info("ERROR: DarkSubtraction.apply(): \
                unequal number of dete ctors in input data and calibration data.")
           raise Exception("ERROR: DarkSubtraction.apply(): \
                unequal number of dete ctors in input data and calibration data.")

        # loop through all detectors
        for i in range(numDect):
            self.logger.info("DarkSubtraction.apply(): \
                 dark subtraction on detector " + str(i+1))
            # get detector pointer from input data
            NDExp_ptr = exp.getDetector(i) 
            detid = (NDExp_ptr.getMetadata().get("DET_ID")).strip()
            cal_ptr = cal.getDetector(detid) 
            self.darkSubtract(NDExp_ptr, cal_ptr)
        self.logger.info("DarkSubtraction.apply(): \
             finish dark subtraction on all detectors")

        # saving all three layers in a new FITS file
        exp.writeFits(self._outfile)
        self.logger.info("DarkSubtraction.apply(): output writen to " + self._outfile)

    @staticmethod
    def darkSubtract(NDExp_ptr, cal_ptr):
        # get the 3 layers of a exposure
        sciimg = NDExp_ptr.getScience()      #type Image<float>
        scivar = NDExp_ptr.getVariance()     #type Image<float>
        scimsk = NDExp_ptr.getMask()         #type Mask<uint32_t>
        scimeta = NDExp_ptr.getMetadata()

        calimg = cal_ptr.getScience()
        calvar = cal_ptr.getVariance()
        calmsk = cal_ptr.getMask()
        calmeta = cal_ptr.getMetadata()

        #populate some metadata from the calibration product to the image frame (SCI layer only)
        scimeta.add("NDFILL", calmeta.get("NDFILL"), "Number of high noise pixels with filled dark values")
        scimeta.add("DARKFILL", calmeta.get("DARKFILL"), "Dark fill value in e-")

        # get numpy arrays
        sciimgarray = sciimg.getArray()
        scivararray = scivar.getArray()
        scimskarray = scimsk.getArray()
     
        calimgarray = calimg.getArray()
        calvararray = calvar.getArray()
        calmskarray = calmsk.getArray() 

        #from SC8 (as of January 15, 2021), the detector map are in 2048x2048
        #make sure the master dark has the same dimension with the image frame
        #the image frame should be 2040x2040 at this masking step in the NIR pipeline
        caldim = calimgarray.shape[0]
        if caldim == 2048:
           calimgarray = calimgarray[4:-4, 4:-4]
           calvararray = calvararray[4:-4, 4:-4]
           calmskarray = calmskarray[4:-4, 4:-4]
        
        #sanity check, in case of NaNs, do not apply the calibration
        nan_ind = np.isnan(calimgarray)
        calimgarray[nan_ind] = 0.
        nan_ind = np.isnan(calvararray)
        calvararray[nan_ind] = 0.

        # subtract calibration data from the image layer
        sciimgarray -= calimgarray

        # add dark frame var layer to the image data in quadrature
        #scivararray[:] = np.sqrt(np.square(scivararray) + np.square(calvar))
        res = scivararray + calvararray
        scivar.setArray(res)

        #TODO Dec, 14m 2020, commented out for SC8 because of the large number pixels
        #got flagged in the master dark, and downstream PFs can't handle it!
        # propagate the master dark flag to the DQ layer in the input frame
        darkbit = EUC_NIR_MASK_DICT["DARKNODET"]
        darkpixvalue = 1 << darkbit
        darkpix = calmskarray & darkpixvalue
        darkpixind = np.where(darkpix)
        scimskarray[darkpixind] |= darkpixvalue
