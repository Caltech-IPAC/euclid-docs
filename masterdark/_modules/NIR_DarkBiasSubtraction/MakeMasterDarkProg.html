def defineSpecificProgramOptions():
    """
    @brief
        Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for MakeMasterDarkProg
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser()
    parser.add_argument('--darklist', type=str, dest='darklist',
                        help='JSON file for a list of input dark files')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='Output XML file for the Master Dark file')
    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')
    parser.add_argument('--config', type=str, dest='config',
                        help='The NIR configuration set')
    parser.add_argument('--mdbfile', type=str, dest='mdbfile',
                        help='MDB XML file name')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method

    @details
        This method is the entry point to the program
    """
    logger = log.getLogger('MakeMasterDarkProg')
    logger.info('Entering MakeMasterDarkProg mainMethod')

    #the filename of the list of darks 
    darkfn = os.path.join(args.workdir, args.darklist)
    if os.path.isfile(darkfn):
       logger.info('Input JSON file for a list of darks in FITS: ' + darkfn)
    else:
       msg = 'Dark file list {} not found'.format(darkfn)
       logger.info(msg)
       raise IOError(msg)

    #output file
    outfn = os.path.join(args.workdir, args.outfile)
    logger.info('Output file: ' + outfn)

    #the configuration file containing the parameters
    if args.config:
       if os.path.isabs(args.config):
          config_set = args.config
       else:
          config_set = os.path.join(args.workdir, args.config)
       logger.info('NIR configuration set file: ' + config_set)
    else:
       config_set = None
       logger.info('NIR configuration set file is not specified.')
       
    #the MDB XML file
    if args.mdbfile:
       mdbfn = os.path.join(args.workdir, args.mdbfile)
       if os.path.isfile(mdbfn):
          logger.info('The MDB XML file: ' + mdbfn)
       else:
          msg = 'The MDB XML file {} not found'.format(mdbfn)
          logger.info(msg)
          raise IOError(msg)
    else:
       raise IOError("Missing MDB file!")

    obj = MakeMasterDark(args.workdir, darkfn, outfn, config_set, mdbfn)
    obj.apply()


    logger.info('Exiting MakeMasterDarkProg mainMethod')
