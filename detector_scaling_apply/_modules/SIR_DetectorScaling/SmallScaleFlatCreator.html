class SmallScaleFlatCreator(object):
    """ Class to handle application of small scale flat """
    sir_invalid_bitmask = np.uint32(2**EUC_SIR_MASK_DICT['INVALID'])


    def __init__(self, input_exposure_fits: str, zodi_model: str):
        """ 
        Parameters
        ----------
        input_exposure_fits: str
            The input FITS filename, w.r.t workdir. User may choose to append
            a directory in front of the FITS filename.

        zodi_model: str
            Input zodiacal light model FITS file, with full path

        output_fits: str
            The output FITS filename, w.r.t workdir. User may choose append
            a directory in front of the FITS filename.
        """

        self._exposure = NispSurveyExposure.readFits(input_exposure_fits)
        self._grism = self._exposure.getGrism()[:3]
        self._zodi_model = zodi_model

    def apply_correction(self) -> None:  
        """ Applies the flat field correction """

        with fits.open(self._zodi_model) as zodi_hdu:

            for detector_index in range(self._exposure.getNumDetectors()):
                detector = self._exposure.getDetector(detector_index)
                detector_id = detector.getDetectorID()

                flat_array = zodi_hdu[f'DET_{detector_id}.{self._grism}.FLAT'].data

                invalid_pixel_mask = flat_array == 0.0
                flat_array[invalid_pixel_mask] = np.nan

                science_array = np.copy(detector.getScience().getArray())
                science_array_flatfielded = science_array / flat_array

                pixels_with_inf = np.isinf(science_array_flatfielded)
                science_array_flatfielded[pixels_with_inf] = np.nan
                detector.getScience().setArray(science_array_flatfielded)            

                variance_array = np.copy(detector.getVariance().getArray())
                variance_array_flatfielded = variance_array / np.square(flat_array)

                pixels_with_inf = np.isinf(variance_array_flatfielded)
                variance_array_flatfielded[pixels_with_inf] = np.nan
                detector.getVariance().setArray(variance_array_flatfielded)

                mask_array = np.copy(detector.getMask().getArray())
                mask_array[invalid_pixel_mask] |= self.sir_invalid_bitmask
                detector.getMask().setArray(mask_array)
