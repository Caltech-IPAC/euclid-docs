def defineSpecificProgramOptions():
    """
    @brief Allows to define the (command line and configuration file) options
    specific to this program

    @details
        See the Elements documentation for more details.
    @return
        An  ArgumentParser.
    """

    parser = argparse.ArgumentParser()

    # Generic parameters

    parser.add_argument('--ver', action='version', version='0.1')

    parser.add_argument('--workdir', type=str, required=True,
                        help='The root working directory where the input files and data/ directroy are located.')

    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='The logging directory')

    parser.add_argument('--config', type=str, dest='config',
                        help='Configuration File')

    # Input data filenames:

    parser.add_argument('--mdb', type=str, required=True,
                        help='The Mission Database XML file, containing all key instrument and mission parameters.')

    parser.add_argument('--sci_frame_fits', type=str, required=True, metavar='FRAME',
                        help='Input science frame FITS file for all detectors. May include directory w.r.t workdir')

    parser.add_argument('--zodiacal_light', type=str, required=True, metavar='ZODI',
                        help='The zodiacal light model.')

    # Options / Behavior-modifying parameters:

    parser.add_argument('--pipeline', dest='pipeline',  nargs='?', default=False, const=True, type=str_to_bool,
                        help='Specify if running in manual mode or in pipeline')

    # Output data filename(s):

    parser.add_argument('--output_frame_fits', type=str, required=True, metavar='OUTPUT',
                        help='Flat-fielded output FITS file for all detectors. May include directory w.r.t workdir')

    return parser

def mainMethod(args):
    """
    @brief The "main" method.
    @details
        This method is the entry point to the program. In this sense, it is
        similar to a main (and it is why it is called mainMethod()).
    """

    module_name = 'SIR_ApplyDetectorScaling'

    logger = log.getLogger(module_name)

    logger.info('#')
    logger.info(f'# Entering {module_name} mainMethod()')
    logger.info('#')

    args = addProgramOptions('SIR_DetectorScaling.{module_name}', args, module_name, logger)

    if args.pipeline:
        logger.info("Running within a pipeline.")
    else:
        logger.info("Running in manual / stand-alone mode.")

    nisp_exposure_fits_file = os.path.join(args.workdir, args.sci_frame_fits)

    zodi_dataproduct = DpdSirDetectorScaling(os.path.join(args.workdir, args.zodiacal_light))
    zodiacal_light_file = os.path.join(args.workdir, 'data', zodi_dataproduct.get_zodi_light_fits_file())

    flatfielded_output_fits_file = os.path.join(args.workdir, args.output_frame_fits)

    logger.info("Loading science exposure file.")

    flat_maker = SmallScaleFlatCreator(nisp_exposure_fits_file, zodiacal_light_file)

    logger.info("Loading zodiacal light file and pplying flat-fielding correction....")

    flat_maker.apply_correction()

    logger.info("Saving flat-fielded output FITS file")

    flat_maker.save_flatfielded_fits(flatfielded_output_fits_file)

    logger.info('#')
    logger.info(f'# Exiting {module_name} mainMethod()')
    logger.info('#')
