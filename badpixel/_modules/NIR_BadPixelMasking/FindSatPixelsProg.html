def defineSpecificProgramOptions():
    """
    @brief
      Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for FindSatPixelsProg
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser("""Find saturated pixels.  Usage syntax:
                ~> ERun NIR_BadPixelMasking findSatPixels            \\
                   --mdbfile=$workdir/theMDBfile.xml                 \\
                   --config=$workdir/theNIRconfigset.xml             \\
                   --exposurelist=$workdir/ExposureList.json         \\
                   [--scaid=18221,18548,...]                         \\
                   --outfile=./mask_xml_files.json                   \\
                   --workdir=./                                      \\
                   --logdir=./
              """)

    parser.add_argument('--mdbfile', type=str, dest='mdbfile',
                        help='MDB XML file name')
    parser.add_argument('--config', type=str, dest='config',
                        help='NIR configuration set file name, optional in MDB mode')
    parser.add_argument('--exposurelist', type=str, dest='explist',
                        help='Exposure list file name, optional in MDB mode')
    parser.add_argument('--scaid', type=str, dest='scaid',
                        default=None,
                        help='SCA ID, optional argument if config file provided')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='List of output mask xml file name for all SCA_ID specified')

    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method

    @details
        This method is the entry point to the program
    """
    logger = log.getLogger('FindSatPixelsProg')
    logger.info('Entering FindSatPixelsProg mainMethod')

    # the MDB XML file
    if args.mdbfile:
        mdbfn = os.path.join(args.workdir, args.mdbfile)
        if os.path.isfile(mdbfn):
            logger.info('The MDB XML file: ' + mdbfn)
        else:
            msg = 'The MDB XML file {} not found'.format(mdbfn)
            logger.info(msg)
            raise IOError(msg)
    else:
        raise IOError("Missing MDB file!")

    # the configuration file containing the parameters
    if args.config:
        config_set = os.path.join(args.workdir, args.config)
        logger.info('NIR configuration set file: ' + config_set)

        module_name = NIR_module_name.NIR_BadPixelMasking
        conf_file = XmlReader().get_config_file(xml_conf_file=config_set,
                                                module_name=module_name, sub_module='satpixel_parameters')
    else:
        conf_file = None
        logger.info('NIR configuration set file is not specified.')

    if conf_file:
        configfn = os.path.join(args.workdir, 'data', conf_file)
        logger.info('User input parameter file: ' + configfn)
    else:
        configfn = None
        logger.info('No user parameters specified.  Using default.')

    # for MDB mode, a SCA_ID must be given if a config file is not spcified
    if (not configfn) and (not scaid):
        raise IOError("A SCA_ID must be specified without a config file!")

    # exposure list file
    if args.explist:
        explistfn = os.path.join(args.workdir, args.explist)
        if os.path.isfile(explistfn):
            logger.info('Input exposure list file: ' + explistfn)
        else:
            msg = 'Exposure list file {} not found'.format(explistfn)
            logger.info(msg)
            raise IOError(msg)
    else:
        explistfn = None
        logger.info('No exposure list file specified.  Using default.')

    # scaids
    # there are a few ways specifying the SCAIDs:
    # on the command line using ERun, one can specify a list of SCAIDs
    # or using a config file to specify a list of SCAIDs
    # or if config file is present by no SCAID params is empty, then
    # the code will loop through all 16 SCAIDs.
    scaids = None
    if configfn is not None:
        with open(configfn, 'r') as fh:
            params = json.load(fh)
            if not params['SCAID']:
                scaids = list(readDetectorLayout(mdbfn).values())
            else:
                scaids = params['SCAID']
                #scaids = params['SCAID'].split(",").strip()

    if args.scaid:
        scaids = args.scaid.split(",")
        scaids = [s.strip() for s in scaids]

    # output json file, a list of xml files for all specified scaids
    output_json = os.path.join(args.workdir, args.outfile)
    parent_task_folder = os.path.dirname(args.outfile)

    output_list = []
    for scaid in scaids:
        filename = os.path.join(parent_task_folder, f"outfile_{scaid}.xml")
        out_file = os.path.join(args.workdir, filename)
        logger.info(f'Output file for {scaid}: {out_file}')
        obj = FindSatPixels(args.workdir, mdbfn, configfn,
                            explistfn, scaid, out_file)
        obj.findthem()
        output_list.append(filename)

    with open(output_json, 'w') as json_xml_list:
        json.dump(output_list, json_xml_list)

    logger.info('Exiting FindSatPixelsProg mainMethod')
