class FindDeadPixels(object):
    """Class FindDeadPixels
    This class identifies pixels which are measured to have zero signal in a
    set of Level 1 SCIENCE exposures.

    The algorithm can process either MDB, RAW (full ramp) or SCI (fitted ramps) exposures:

    * In RAW data, the first and last groups are averaged and differenced. A
    pixel is disconnected if the difference is exactly 0, and flagged dead in
    the final mask if it is dead to a threshold repeatability across N input
    exposures.  If N is less than the threshold, the repeatability must be 100%.

    * In SCI data, a pixel is disconnected if the standard deviation of the
    estimated flux is exactly zero across all N input exposures.  

    * In MDB mode, the disconnected pixel maps from the MDB is used directly.

    The output for a single detector is a 2D dead pixel mask stored as FITS,
    named dpix_SCAID_[timestamp].fits, including a table extension listing the
    input data.  In the case of RAW input data, a FITS cube [N,2040,2040]
    containing N image masks, and corresponding ascii lists of the flagged pixels
    are also created.

    The output of this class is to match what's in the DCL data that are in the
    Euclid MDB.  This is the not the final dead pixel mask.  The otuput of this
    class is part of the inputs to CreateBadPixelsMask class.

    This class works on a detector basis, since: (a) this allows each
    to be monitored on their own timescales; (b) it is easier to define a smaller
    homogenous set of ground or on orbit observations to be processed; (c) runs
    faster with simpler I/O.

    Bad Pixel flavors:
    1 = Disconnected
    2 = Zero QE (formerly Zero Full Well)
    3 = High CDS noise (stand-in for RTN)
    4 = Snowball
    5 = Hot (formerly 'Halo')
    6 = High Low Baseline
    """

    def get_scaid(self, a_expfile):
        """
        Retrieve the SCA_ID from one input exposure FITS file a_expfile
        The SCA_ID should be in the input exposure FITS header!
        """
        ff = fits.open(a_expfile)
        try:
            self._scaid = ff[0].header['SCA_ID']

            raise IOError("ERROR: the data is missing header key SCA_ID!")
        finally:
            ff.close()

        return self._scaid

    def get_detid(self, a_expfile=None):
        """
        Get the detector id from either the input file or the detector slot table.
        The DET_ID will be in the input exposure FITS header.
        If the exposure is not specified or if the DET_ID is not in the input exposure data file, 
        retrieve the detector id from the detector slots table given the SCA_ID.

        :return: string, detector id
        """
        if a_expfile:
            ff = fits.open(a_expfile)
            hdr = ff[0].header
            if 'DET_ID' in hdr:
                self._detid = ff[0].header['DET_ID']
                ff.close()
                return self._detid

        # read in the detector slots table
        detslots = readDetectorLayout(self._mdbfile)

        # find the detector id corresponding to the scaid
        detid = None
        for key, val in detslots.items():
            if str(val) == str(self._scaid):
                detid = key
        if detid is not None:
            self._detid = detid
        else:
            raise Exception(
                "Can not find the corresponding detector ID for SCA_ID {}".format(self._scaid))

        return detid

    def get_thresh_pcnt(self, num_exposures, in_percent):
        """
        Get the threshold percentage for merging the bad pixel masks from all exposures
        based on number of available exposures.

        The criteria is as following:
         X=threshhold_percent  for N >= MIN_NUM_EXPOSURES
         X=100%=1.             for N <  MIN_NUM_EXPOSURES
         N is the number of total number of exposures

        :param in_percent: (float) percentage of the threshold
        :param num_exposures: (int) number of exposures

        :return: float, the threshold for dead pixel
        """
        if num_exposures >= self._min_num_exposures:
            threshold = in_percent
        else:
            threshold = 1.

        self._com_threshold_used = threshold

        return threshold

    def merge_masks(self, num_exposures, sum_deadpix):
        """
        Check for threshold and determine if the pixel is indeed dead. Make the final mask for this SCA_ID.

        The criteria is as following:
        For each pixel, if the total number of dead pixels identified from all the exposures (sum_deadpix)
        is larger than threshold X, then it is a dead pixel.
        Where:
         X=80% of N  for N >= MIN_NUM_EXPOSURES
         X=100% of N for N <  MIN_NUM_EXPOSURES
         N is the number of total number of exposures

        :param num_exposures: (int) number of exposures used
        :param sum_deadpix:  (2D numpy array) sum of bad pixels at each pixel

        :return: (2D int number array) the dead pixel mask for this SCA_ID       

        """

        threshold_pcnt = self.get_thresh_pcnt(
            num_exposures, self._com_threshold)

        # check for threshold
        ind = np.where(sum_deadpix >= (threshold_pcnt * num_exposures))
        self.set_numbdpix(len(ind[0]))

        self.logger.info("Final merged mask has {} dead pixels in {} exposures at >= {}% repeatability".
                         format(self._numbdpix, num_exposures, threshold_pcnt*100))

        mask = np.zeros(sum_deadpix.shape, dtype=np.int32)
        mask[ind] = 1

        return mask

    def find_per_exposure(self, ramps, fnum):
        """
        Find the bad pixels for each exposure file which contains n_ramps. 

        :param ramps: 2D numpy array, ramp data for this exposure
        :param fnum: int, the fnum-th exposure in the input exposure list
        """
        nz, ny, nx = ramps.shape  # number of up-the-ramp reads, rows, columns
        self.logger.info("Input ramps dimension: {},{},{}".format(nz, ny, nx))

        if ny > 2040:
            ramps = ramps[:, 4:2044, :]  # Remove the border if present
        if nx > 2040:
            ramps = ramps[:, :, 4:2044]  # Remove the border if present

        # chenck for NaN in the data.  These NaN are from faulty readouts or data
        # analysis or other reasons that do not necessairly mean the pixel is
        # physically bad
        ramps_nan_masked = np.ma.MaskedArray(ramps, mask=np.isnan(ramps))
        # average first 16 frames
        pedestal = np.ma.average(ramps_nan_masked[:16, :, :], axis=0)
        totalsignal = np.ma.average(
            ramps_nan_masked[nz-16:, :, :], axis=0)  # average last 16 frames
        netsignal = totalsignal - pedestal  # difference to find dead pixels
        ind_nan = np.where(netsignal.mask)  # pixels where NaN occurs
        num_nans = len(ind_nan[0])
        self.logger.info(
            "There are {} NaN pixels in this exposure".format(num_nans))

        # find zero signal pixels.  Some are negative, these have
        # problems e.g. zero full well, but not defined as disconnected.
        goodpixs = np.greater(np.abs(netsignal.data), self._bad_threshold,
                              where=~(netsignal.mask))
        # pixels where NaN occurs are not dead pixels, thus set to True
        goodpixs[ind_nan] = True
        deadpixs = ~goodpixs
        ind_deadpixs = np.where(deadpixs)  # where the deak pixels are
        total_num_deadpixels_det = len(ind_deadpixs[0])
        self.logger.info("There are {} dead pixels in this exposure".format(
            total_num_deadpixels_det))

        # boolean --> int
        deadpixs_int = deadpixs * 1

        # Write out a list of dead pixels
        if self._verbose:
            now = datetime.datetime.now().isoformat().replace(
                ':', '').split(".")[0]
            outlisttxtfile = './deadpixels_list_SCA' + str(self._scaid) + '_'
            outlisttxtfile += str(fnum) + '_' + str(now) + '.txt'
            outlisttxtfile = os.path.join(
                self._workdir, 'data', outlisttxtfile)
            xpixs, ypixs = ind_deadpixs
            with open(outlisttxtfile, 'w') as of:
                print("SCA_ID   X    Y   Bad-reason", file=of)
                for x, y in zip(xpixs, ypixs):
                    print(self._scaid, ',', x, ',', y,
                          ',', self._bad_reason, file=of)
            self.logger.info(
                "Written dead pixels list to {}".format(outlisttxtfile))

        return deadpixs_int, total_num_deadpixels_det, num_nans

    def findthem_raw(self):
        """Function to find dead pixels using full ramps (RAW mode)
        """
        # read the list of exposures
        expfiles = self.getExposureList()

        # get detector id
        #detid = self.get_detid(expfiles[0])
        self.get_detid()

        # init the sum array for checking if the pixel are indeed dead given the threshold
        deadpixsum = np.zeros((2040, 2040), dtype=np.int32)

        # init the FITS data structure
        # this FITS file contains len(expfiles) extensions, each for one exposure
        hdulist = fits.HDUList([fits.PrimaryHDU()])

        # get the signal ramps for each pixel
        fnum = 1
        for f in expfiles:
            self.logger.info("Process exposure file: {}".format(f))
            exphdu = fits.open(f)
            ramps = exphdu[0].data
            exphdu.close()

            # find the bad pixels in the ramps of this exposure file
            deadpixs_int, total_num_deadpixels_det, num_nans = self.find_per_exposure(
                ramps, fnum)
            self.logger.info("File {}, SCA {}, there are total {} dead pixels, {} NaNs".
                             format(os.path.basename(f), self._scaid, total_num_deadpixels_det, num_nans))

            # since the value is either 1 or 0 for bad pixels, we can add them up to get
            # the total number of found bad ones at each pixel grid from all of the exposures
            deadpixsum += deadpixs_int

            # append dead pixel array to FITS HDU
            newhdu = fits.ImageHDU(deadpixs_int.astype(np.uint8))
            newhdu.name = 'DQ_' + str(fnum+1)
            hdr = newhdu.header
            hdr.set('SCA_ID', self._scaid, 'SCA Identifier')
            hdr.set('INDATA', os.path.basename(f), 'Data source')
            hdr.set('NBADPIX', total_num_deadpixels_det,
                    'Number of bad pixels of type BADTYPE')
            hdr.set('NUMNAN', num_nans,
                    'Number of NaNs, not counted with NBADPIX')
            hdulist.append(newhdu)

            fnum += 1

        # merge all the masks from each exposure and make the final mask for this DET_ID/SCA_ID
        deadmask = self.merge_masks(len(expfiles), deadpixsum)

        # make a table HDU containing the list of input exposure file names
        t_hdu = make_files_hdu(expfiles)

        # append the list of input files as a table to FITS HDU
        hdulist.append(t_hdu)

        # update primary header of both hdulist
        prihdr = hdulist[0].header
        prihdr.set('TELESCOP', 'Euclid', 'Telescope name')
        prihdr.set('INSTRUME', 'NISP', 'Instrument name')
        prihdr.set('ORIGIN', 'OU-NIR', 'SGS-OU origin')
        prihdr.set('DET_ID', self._detid, 'Detector ID')
        prihdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        prihdr.set('BADTYPE', self._bad_reason, 'See COMMENT')
        prihdr.set('COMMENT', 'Badtypes: 1=Disconnected, 2=ZeroQE, 3=HighNoise/RTN, ' +
                   '4=Snowball, 5=Hot, 6=BaselineOOL, 7=LowQE, 8=SuperQE')
        prihdr.set('DATEUTC', datetime.datetime.utcnow(
        ).isoformat().split(".")[0], 'File creation date')
        #prihdr.set('BADTHRES', self._bad_threshold, 'Thrshold for the dead pixels')
        prihdr.set('BADVAL', 1, 'Bad pixel value')

        # write the FITS file for each exposure
        if self._verbose:
            utcnow = datetime.datetime.utcnow().isoformat().replace(':',
                                                                    '').split(".")[0]
            aoutfile = 'dpix_SCA' + \
                str(self._scaid) + '_maskcube_' + str(utcnow) + '.fits'
            aoutfile = os.path.join(self._workdir, 'data', aoutfile)
            hdulist.writeto(aoutfile)
            self.logger.info("Mask cube written to file {}".format(aoutfile))

        return deadmask, t_hdu

    def findthem_macc(self):
        """Function to find dead pixels using fitted ramps (SCI mode)

        :return: int, 2D numpy array, the dead pixel mask
        """
        # read the list of exposures
        expfiles = self.getExposureList()
        num_exposures = len(expfiles)

        # get detector id
        self.get_detid()
        self.logger.info(f'Detector ID: {self._detid}')

        # get the threshold, though not used but it sets self._com_threshold_used which is need in the header of final product
        threshold = self.get_thresh_pcnt(num_exposures, 1.)

        """PM 7/23/24.
       The new condition for a dead pixel is lack of photometric response over a range of
       exposures, and selected over a low-level flux range determined from
       the signal properties of previously known disconnected pixels.  The list of disconnected
       pixels is extracted from MDB reference (ground-based) mask EUC_NISP_DISCONNECTED-TV1_01.00.fits.
       The signal properties of the known disconnected pixels are measured in the median of a
       stack of input LE1 frames listed in 'expfiles', then a flux range (in ADU) is set
       based on the median of the disconnected pixels at x-sigma below the most probable value of
       the signal in the bulk of the detector, excluding saturated pixels.  Pixels falling in the
       low flux range, and not previously flagged as DISCONNECTED, are declared dead, and effectively 
       in the same category as DISCONNECTED.

       Similar to the "MDB" option below, the ground-based reference
       EUC_NISP_DISCONNECTED-TV1_01.00.fits must be read in.  The final mask will be a
       union of the pixels flagged in that mask ("True" boolean) and the unresponsive pixels
       selected here.

       Input data in 'expfiles' must be LE1 slope images, i.e., without signal processing
       except for datatype conversion, subtraction of the bias offset, and trimming the
       reference border pixels.  The ideal observatations are LED flatfield exposures from
       PV-010 and/or F-007, and ideally at maximum fluence 730 e/sec with LED configurations
       given in calibration.pages.euclid-sgs.uk/CalTQ-Framework/calobs/calobs-block-f-visNL-nispLED.html,
       Table 56.
       
       For the LED flatfield data, use self._dataorig == "SIM".

       HERE IS HOW TO DO IT:
       """
        #read in data
        allimgs = self.load_data(expfiles)

        # subtract the bias offset
        allimgs -= 1024
        self.logger.info(f'Subtracted the bias offset 1024')

        # calculate median and std deviation, check for negative pixels
        self.logger.info(f'Checking for negative pixels...')
        imgs_med = np.median(allimgs, axis=0)
        imgs_std = np.std(allimgs, axis=0)
        ind = np.where(imgs_med < 0)
        num_negative = ind[0].size

        # get the reference map of known (pre-launch) disconnected pixels from MDB (16 extensions, dead pixel=1)
        self.logger.info(f'Retrieving the pre-launch disconnected pixels...')
        mdb = Mdb(self._mdbfile)
        disctfile = mdb.get_value(
            "SpaceSegment.Instrument.NISP.NISPDisconectedPixelsFile")[0]
        self.logger.info(
            "The pre-launch disconnected pixle map from MDB: {}".format(disctfile))

        # make a table HDU containing the list of input exposure file names
        t_hdu = make_files_hdu(expfiles + [disctfile])

        # read in the disconnected pixle map data for the current scaid
        disctfile = os.path.join(self._workdir, 'data', disctfile)
        hdus = fits.open(disctfile)

        # find the hdu for the corresponding scaid
        hdu = [h for h in hdus[1:17:1] if str(
            h.header['SCA_ID']) == str(self._scaid)]
        if len(hdu) != 1:  # there should be only 1 extension (SCI) containing the SCA_ID
            raise Exception(
                f'No extension found for SCA_ID={str(self._scaid)} in the pre-launch disconnected pixel file: {disctfile}')

        discopix = hdu[0].data
        hdus.close()

        nx, ny = discopix.shape
        if ny > 2040:
            discopix = discopix[4:2044, :]
        if nx > 2040:
            discopix = discopix[:, 4:2044]
        self.logger.info(f'Read in the pre-launch disconnected pixels')
        self.logger.info(
            f'There are {np.count_nonzero(discopix)} pre-launch disconnected pixels')

        # determine the median signal of all pixels except disconnected
        sigmed = np.median(imgs_med[discopix != 1])
        sigstd = np.std(imgs_med[discopix != 1])

        # make a copy of the median array to manipulate according to the flux range found
        # where the dicosnnected pixels show lowest signal (non-zero due to charge in the readout
        # integrated circuit)
        imgs_med1 = imgs_med.copy()
        rng = np.where((imgs_med > (np.median(imgs_med) - 5*np.std(imgs_med))))

        # nullify all pixels with signal outside of rng
        imgs_med1[rng] = np.nan

        # nullify all remaining pixels that are not disconnected
        imgs_med1[discopix != 1] = np.nan

        # now find the median (MPV) and std deviation of the pixels that have not been nullified.
        # these are the disconnected pixels at lowest signal on the array
        discoMPV = np.median(imgs_med1[~np.isnan(imgs_med1)])
        discoStd = np.std(imgs_med1[~np.isnan(imgs_med1)])

        # set a factor for how many sigma above the discoMPV to find dead pixels
        discoScale = 3.

        # make the dead pixel mask
        imgs_deadmask = np.zeros(imgs_med.shape, dtype=np.int32)
        imgs_deadmask[imgs_med <= (discoMPV + discoScale*discoStd)] = 1
        # imgs_deadmask[imgs_med > (discoMPV + discoScale*sigStd)] = 0   #alread zeroed-out

        # include the previously known disconnected pixels to keep track of.  The final mask will
        # be a union of dead and disconnected. """
        imgs_deadmask[discopix == 1] = -1

        goodpixs = np.where(imgs_deadmask == 0)
        # All = previously known disconnected + new dead pixels
        deadpixs = np.where(imgs_deadmask != 0)
        self.set_numbdpix(len(deadpixs[0]))

        self.logger.info("Final mask has {} dead pixels in {} exposures".
                         format(self._numbdpix, num_exposures))
        self.logger.info('Median frame {} pixels < 0 ADU'.format(num_negative))
        self.logger.info('Lower MPV, STD DEV of known DISCONNECTED pixels: {}, {}'.format(
                         np.round(discoMPV, 3), np.round(discoStd, 3)))
        self.logger.info(
            'Num known (ground-based) DISCONNECTED pixels: {}'.format(discopix[discopix == 1].size))
        self.logger.info('DISCONNECTED pixel MEDIAN, MIN, MAX in the median image: {}, {}, {}'.format(
                         np.median(imgs_med[discopix == 1]), np.min(
                             imgs_med[discopix == 1]),
                         np.max(imgs_med[discopix == 1])))
        self.logger.info('DETECTOR MEDIAN, STD DEV, MIN, MAX excl. DISCONNECTED pixels: {}, {}, {}, {}'.format(
                         np.median(imgs_med[discopix != 1]), np.std(
                             imgs_med[discopix != 1]),
                         np.min(imgs_med[discopix != 1]), np.max(imgs_med[discopix != 1])))
        self.logger.info('Num DISCONNECTED pixels in median image in the selected range: {}'.format(
                         discopix[np.where((discopix == 1) & (imgs_med <= (discoMPV + 3*discoStd)))].size))
        self.logger.info('Num DEAD pixels in median image in the selected range: {}'.format(
                         imgs_deadmask[imgs_deadmask == 1].size))

        if self._verbose:
            hdu1 = fits.PrimaryHDU()
            hdu1.header['TELESCOP'] = ('Euclid  ', ' ')
            hdu1.header['INSTRUME'] = ('NISP    ', ' ')
            hdu1.header['DET_ID'] = (self._detid, 'Detector ID')
            hdu1.header['DATE'] = (datetime.datetime.utcnow(
            ).isoformat(), 'UT date when this file was created')
            hdu1.header['INDATA'] = (
                self._explistfn, 'Input FITS exposure list')
            hdu1.header['NUMEXP'] = (
                num_exposures, 'Number of input exposures')

            hdu2 = fits.ImageHDU(imgs_med, name='DET' +
                                 str(self._detid)+'.MEDIAN')
            hdu2.header['DET_ID'] = (hdu1.header['DET_ID'], 'Detector ID')
            hdu2.header['IMAGETYP'] = (
                'median', 'Image data type, e.g., median, mean, ...')

    def findthem_mdb(self):
        """Function to find dead pixels using the disconnected pixel map in the MDB

        :return: int, 2D numpy array, the dead pixel mask
        """
        # get the corresponding detector id
        self.get_detid()

        # get the DISCONNECTED pixel map file from MDB  (16 extensions, dead pixel=1)
        mdb = Mdb(self._mdbfile)
        disctfile = mdb.get_value(
            "SpaceSegment.Instrument.NISP.NISPDisconectedPixelsFile")[0]
        self.logger.info(
            "The disconnected pixle map from MDB: {}".format(disctfile))

        self._num_exposures = 1
        self._com_threshold_used = 0.  # not used in this mode but for the output header

        # make a table HDU containing the list of input exposure file names
        t_hdu = make_files_hdu([disctfile])

        # read in the disconnected pixle map data for the current scaid
        disctfile = os.path.join(self._workdir, 'data', disctfile)
        hdus = fits.open(disctfile)

        # find the hdu for the corresponding scaid
        hdu = [h for h in hdus[1:17:1] if str(
            h.header['SCA_ID']) == str(self._scaid)]
        if len(hdu) != 1:  # there should be only 1 extension (SCI) containing the SCA_ID
            raise Exception("No extension found for SCA_ID=" +
                            str(self._scaid) + " in expousre file: " + disctfile)

        dat = hdu[0].data
        hdus.close()

        nx, ny = dat.shape

        if ny > 2040:
            dat = dat[4:2044, :]  # Remove the border if present
        if nx > 2040:
            dat = dat[:, 4:2044]  # Remove the border if present

        ind = np.where(dat)
        self.set_numbdpix(len(ind[0]))

        return dat * 1, t_hdu

    def load_data(self, expfiles):
        """Load the data based on the data origin for SCI mode.
        The data origin is one of the lab test, simulated and operation data: "TV1", "SIM", "OPS".

        :param expfiles: list, the list of exposure file names
        :return: int, 3D numpy array
        """

        #allimgs = np.zeros((len(expfiles), 2040, 2040), dtype=np.int32)
        allimgs = np.zeros((len(expfiles), 2040, 2040), dtype=np.float32)

        n = 0
        for f in expfiles:
            f = os.path.join(self._workdir, 'data', f)
            # TV1 test data (one file per scaid, 5 hdus)
            if self._dataorig == "TV1":
                self.logger.info(
                    "Read in exposure file {} : {}".format(n+1, f))
                hdu = fits.open(f)
                dat = hdu["SCIENCE"].data
                hdu.close()
            # simulated or in-flight data (32 hdus, each detector has SCI & DQ layers)
            elif self._dataorig == "SIM" or self._dataorig == "OPS":
                # parse the exposure list containing XML files
                root = ET.parse(f).getroot()
                for el in root.iter():
                    # if el.tag == "NirFrameFitsFile":
                    if el.tag == "NispFrameFitsFile":  # 7/29/2024
                        fn = el.findall('.//FileName')
                        fn = fn[0].text

                self.logger.info(
                    "Read in exposure file {} : {}".format(n+1, fn))
                # open exposure file
                hdus = fits.open(os.path.join(self._workdir, 'data', fn))

                # find the hdu for the corresponding scaid
                hdu = [h for h in hdus[1:32:2] if str(
                    h.header['SCA_ID']) == str(self._scaid)]
                if len(hdu) != 1:  # there should be only 1 extension (SCI) containing the SCA_ID
                    raise Exception("No extension found for SCA_ID=" +
                                    str(self._scaid) + " in expousre file: " + fn)

                hdu = hdu[0]

                # read in the hdu data
                dat = hdu.data
                hdus.close()

            nx, ny = dat.shape

            if ny > 2040:
                dat = dat[4:2044, :]  # Remove the border if present
            if nx > 2040:
                dat = dat[:, 4:2044]  # Remove the border if present

            allimgs[n, :, :] = dat
            n += 1

        return allimgs

    def findthem(self):
        if self._expmode == "MDB":
            deadmask, t_hdu = self.findthem_mdb()
        elif self._expmode == "RAW":
            deadmask, t_hdu = self.findthem_raw()
        elif self._expmode == "SCI":
            deadmask, t_hdu = self.findthem_macc()
        else:
            raise Exception(
                "Unsupported exposure mode: {}!".format(self._expmode))

        #!!Note: in the DCL data, 1=True=good pixel, 0=False=bad pixel,
        # we need to flip the mask since we are producing a mask that emulates what is in DCL data!!
        # deadmask = 1 - deadmask    #June 29, 2021, no longer needed

        # write out the final output
        badpixpath = self.write(deadmask.astype(np.uint8), t_hdu)

        # write out the XML file
        self.logger.info("Creating XML file " + self._outfile)
        result = writeXml(badpixpath, self._outfile)

        # check the result
        if result:
            self.logger.info("Wrote XML file " + self._outfile)
        else:
            msg = 'FindDeadPixels Xml Creation error.'
            self.logger.error(msg)
            raise Exception("ERROR: FindDeadPixels.writeXml(): " + msg)

    def write(self, deadpixmask, files_hdu):
        """Write out the final output

        :param deadpixmask: 2D numpy array, the dead pixel mask, according to DCL convention (1=good, 0=bad)
        """

        # init the FITS data structure
        # this FITS file contains one extension, for the combined maks for this DET_ID/SCA_ID
        hdu = fits.HDUList([fits.PrimaryHDU()])

        # update primary header of hdu
        hdr = hdu[0].header
        hdr.set('TELESCOP', 'Euclid', 'Telescope name')
        hdr.set('INSTRUME', 'NISP', 'Instrument name')
        hdr.set('ORIGIN', 'OU-NIR', 'SGS-OU origin')
        hdr.set('IMG_CAT', 'CALIB')
        hdr.set('DET_ID', self._detid, 'Detector ID')
        hdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        hdr.set('BADTYPE', self._bad_reason, 'See COMMENT')
        hdr.set('COMMENT', 'Badtypes: 1=Disconnected, 2=ZeroQE, 3=HighNoise/RTN, ' +
                '4=Snowball, 5=Hot, 6=BaselineOOL, 7=LowQE, 8=SuperQE')
        hdr.set('BADVAL', 1, 'Bad pixel value')
        hdr.set('NUMEXP', self._num_exposures, 'Number of input exposures')
        hdr.set('EXPTHRES', self._com_threshold_used,
                'Percent exposure threshold for flagging')
        hdr.set('NUMBAD', self._numbdpix,
                'Number of flagged pixels of BADTYPE')
        hdr.set('DATEUTC', datetime.datetime.utcnow(
        ).isoformat().split(".")[0], 'File creation date')
        hdr.set('FITS_DEF', 'nir.logicBadPixelFlavorMask',
                'FITS name definition')
        hdr.set('FITS_VER', '0.1', 'FITS version')
        hdr.set('VAL_STA', '2020-12-01T00:00:00.0',
                'Valid start time for this product(UTC)')
        hdr.set('VAL_END', '2028-12-31T00:00:00.0',
                'Valid end time for this product(UTC)')

        imghdu = fits.ImageHDU(deadpixmask)
        #imghdu.name = 'DQ_' + str(self._detid)
        imghdu.name = 'Disconnected Pixels'
        ihdr = imghdu.header
        ihdr.set('DET_ID', self._detid, 'Detector ID')
        ihdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        ihdr.set('NUMBAD', self._numbdpix,
                 'Number of flagged pixels of BADTYPE')
        #ihdr.set('BADTHRES', self._bad_threshold, 'Threshold for the dead pixels')
        hdu.append(imghdu)
        hdu.append(files_hdu)

        # write the FITS file for the mask for this SCA_ID
        now = datetime.datetime.now().isoformat().replace(
            ':', '').split(".")[0]
        outfile = FileNameProvider().get_allowed_filename(type_name="C-BadPixelFlavor",
                                                          instance_id='MASK',
                                                          processing_function='NIR',
                                                          release=None)

        outfile = os.path.join(self._workdir, 'data', outfile)
        hdu.writeto(outfile, overwrite=True)
        self.logger.info("Written to file {}".format(outfile))

        return outfile
