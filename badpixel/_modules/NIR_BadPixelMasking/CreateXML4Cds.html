def defineSpecificProgramOptions():
    """
    @brief
        Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for CreateBadPixelsMaskProg
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser("""Create the XML file for CDS noise dataset.  Usage syntax:
             ~> ERun NIR_BadPixelMasking createXmls4Cds   \\
                --infile=The_input_data_fits_file         \\
                --outfile=BadPixelMask.xml                \\
                --workdir=./                              \\
                --logdir=./                               \\
             """)
    parser.add_argument('--infile', type=str, dest='infile',
                        help='Input FITS file name')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='Output XML file for the CDS noise Bad Pixels file')
    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method

    @details
        This method is the entry point to the program
    """
    logger = log.getLogger('CreateXML4Cds')
    logger.info('Entering CreateXML4Cds mainMethod')

    # proper official product name
    niroutfn = FileNameProvider().get_allowed_filename(type_name="C-BadPixelFlavor",
                                                       instance_id='MASK',
                                                       processing_function='NIR',
                                                       release=None)
    niroutfn = os.path.join(args.workdir, 'data', niroutfn)
    print(niroutfn)

    # write it out
    copyfile(os.path.join(args.workdir, 'data', args.infile), niroutfn)

    # write out the XML file
    # create an instance of XmlWriter
    xmlW = XmlWriter()

    # Call the method to create the MasterDark class with FITS file content
    xmlW.create_bad_pixel_flavor(niroutfn)

    # write the xml
    result = xmlW.write_xml(os.path.join(args.workdir, args.outfile))
    print(os.path.join(args.workdir, args.outfile))

    logger.info('Exiting CreateXML4Cds mainMethod')
