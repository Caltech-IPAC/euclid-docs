class FindHotPixels(object):
    """
    This task identifies pixels which are measured to have signal in excess of the
    median +/- a threshold constant factor (bad_threshold) of the standard deviation.
    The input is a set of Level 1 SCIENCE exposures, taken with the shutter closed (dark).

    In Ver. 1.0, the algorithm can process SCI exposure (fitted ramps) and RAW mode.  

    The output for a single detector is a 2D hot pixel mask stored as FITS,
    named hotpix_SCAID_[timestamp].fits, including a table extension listing the
    input data.  

    This task works on a detector basis, since: (a) this allows each
    to be monitored on their own timescales; (b) it is easier to define a smaller
    homogenous set of ground or on orbit observations to be processed; (c) runs
    faster with simpler I/O.

    Bad Pixel flavors:
    1 = Disconnected
    2 = Zero QE (formerly Zero Full Well)
    3 = High CDS noise (stand-in for RTN)
    4 = Snowball
    5 = Hot (formerly 'Halo')
    6 = High Low Baseline
    """

        self._outfile = outfile

    def get_detid(self, a_expfile=None):
        """
        Get the detector id from either the input file or the detector slot table.
        The DET_ID will be in the input exposure FITS header.
        If the exposure is not specified or if the DET_ID is not in the input exposure data file, 
        retrieve the detector id from the detector slots table given the SCA_ID.

        :return: string, detector id
        """
        if a_expfile:
            ff = fits.open(a_expfile)
            hdr = ff[0].header
            if 'DET_ID' in hdr:
                self._detid = ff[0].header['DET_ID']
                ff.close()
                return self._detid

        # read in the detector slots table
        detslots = readDetectorLayout(self._mdbfile)

        # find the detector id corresponding to the scaid
        detid = None
        for key, val in detslots.items():
            if str(val) == str(self._scaid):
                detid = key
        if detid is not None:
            self._detid = detid
        else:
            raise Exception(
                "Can not find the corresponding detector ID for SCA_ID {}".format(self._scaid))

        return detid

    def findthem(self):

            #hotmask, t_hdu = self.findthem_raw()

                "Unsupported exposure mode: {}!".format(self._expmode))

        # write out the final output
        badpixpath = self.write(hotmask.astype(np.uint8), t_hdu)

        # write out the XML file
        self.logger.info("Creating XML file " + self._outfile)

        # check the result
        if result:
            self.logger.info("Wrote XML file " + self._outfile)
        else:
            msg = 'FindDeadPixels Xml Creation error.'
            self.logger.error(msg)
            raise Exception("ERROR: FindDeadPixels.writeXml(): " + msg)

    def findthem_macc(self):
        """Function to find hot pixels usning fitted ramps (SCI mode)

           PM update 8/12/24:  The bad_threshold definition is being changed from a pixels
           with positive signal at a sigma (stdev) from the median and then requiring 100%
           repeatability in all exposures to a charge rate.  E.g., bad_threshold = 1.0
           = 1.0 e-/sec, which is a more conventional definition in the literature on H2RG
           performances and hot pixels.

           Also, two constants are needed:
            - Detector gain to convert ADU --> e-.  It should be read from the MDB gain file.
            - Saturation limit according to findSaturatedPixels.  It is not equal to the 64511 ADU
              A/D limit, but a fraction lower.  For now this will be the same as the one in
              default in findSaturatedPixels.

        :return: int, 2D numpy array, the hot pixel mask
        """
        # read the list of exposures
        expfiles = self.getExposureList()
        num_exposures = len(expfiles)

        # sort by creation date to avoid precision error during np.sum/np.mean for different order of input data
        expfiles = self.sortByCreationDate(expfiles)

        # find the elapsed time
        self._tElapsed = self.get_telapsed(expfiles)

        # make a table HDU containing the list of input exposure file names
        t_hdu = make_files_hdu(expfiles)

        # integration time
        nr1, nd1, ng1, ft1 = self.get_header0(expfiles[0])
        # integration time in seconds
        tint1 = (nr1+nd1)*(ng1-1)*ft1

        # get detector id
        self.get_detid()
        det_mapid = 'H2RG_' + str(self._detid)[0] + '_' + str(self._detid[1])

        # read in the gain map
        gainmapfile = XmlReader().get_from_mdb_nisp_instrument(
            mdb_xml=self._mdbfile, field="NISPGainFile")[0]
        self.logger.info("Gain map file: {}".format(gainmapfile))
        gain_hdus = fits.open(os.path.join(self._workdir, 'data', gainmapfile))
        gains = gain_hdus[det_mapid].data  # in AUD/e-
        if gains.shape[1] == 2048:
            gains = gains[4:-4, 4:-4]

        # sigma clip threshold
        satLimit = self._satThresh * self._adLimit
        satLimitEPS = satLimit/tint1/gains   # convert saturation limit ADU --> e-/sec

        # read in data in small tiles
        numTiles = int(2040/self._tilesize)
        signalMed = np.zeros((2040, 2040), dtype=np.float32)
        signalStd = np.zeros((2040, 2040), dtype=np.float32)
        for i in range(numTiles):
            # row start/end index
            r1 = i * self._tilesize  # start row
            r2 = (i+1) * self._tilesize  # end row, non-inclusive
            for j in range(numTiles):
                # column start/end index
                c1 = j * self._tilesize
                c2 = (j+1) * self._tilesize
                self.logger.info(
                    f'Row and column start:end index for current tile [{r1}:{r2}, {c1}:{c2}]')

                # init data array
                tilecube = np.zeros(
                    (num_exposures, self._tilesize, self._tilesize), dtype=np.float32)

                # subset gains to match input data
                gain_tile = gains[r1:r2, c1:c2]

                # read in all files in tiles
                self.logger.info('Reading exposure files in tiles')
                for ifile, f in enumerate(expfiles):
                    self.logger.info("Process exposure file: {}".format(f))
                    # r1/r2,c1/c2 refers to image in 2040x2040 shape
                    signal_ADU = self.load_data(f, [r1, r2, c1, c2])
                    signal_ADU -= 1024  # total ADU

                    signal_APS = signal_ADU/tint1                      # ADU/sec
                    signal_EPS = np.divide(signal_APS, gain_tile)      # e-/sec

                    tilecube[ifile, :, :] = signal_EPS

                # calculate median and std
                tileMed = np.nanmedian(tilecube, axis=0)
                tileStd = np.nanstd(tilecube, axis=0)

                # combine all tiles
                signalMed[r1:r2, c1:c2] = tileMed
                signalStd[r1:r2, c1:c2] = tileStd

        # check for hot pixels
        #signalMed = np.nanmedian(cube, axis=0)
        #signalStd = np.nanstd(cube, axis=0)
        indHot = np.where((signalMed < satLimitEPS) &
                          (signalMed >= self._bad_threshold))
        self._numbdpix = len(indHot[0])
        self.logger.info(f'Number hot pixels: {self._numbdpix}')

        # the hot pixels mask
        hotmask = np.zeros(signalMed.shape, dtype=np.int8)
        hotmask[indHot] = 1

        # save intermediate results
        if self._verbose:
            hdu1 = fits.PrimaryHDU()
            hdu1.header['TELESCOP'] = ('Euclid  ', ' ')
            hdu1.header['INSTRUME'] = ('NISP    ', ' ')
            hdu1.header['DET_ID'] = (self._detid, 'Detector ID')
            hdu1.header['DATE'] = (datetime.datetime.utcnow(
            ).isoformat(), 'UT date when this file was created')
            hdu1.header['INDATA'] = (
                self._explistfn, 'Input FITS exposure list')
            hdu1.header['NUMEXP'] = (
                num_exposures, 'Number of input exposures')
            hdu1.header['TELAPS'] = (
                np.round(self._tElapsed, 2), 'Elasped time of input exposures (h)')

            hdu2 = fits.ImageHDU(signalMed, name='DET' +
                                 str(self._detid)+'.MEDIAN')
            hdu2.header['IMAGETYP'] = (
                'median', 'Image data type, e.g., median, mean, ...')
            hdu2.header['BUNIT'] = ('ADU', 'Pixel data unit')

            hdu3 = fits.ImageHDU(signalStd, name='DET' +
                                 str(self._detid)+'.STDEV')
            hdu3.header['IMAGETYP'] = (
                'stdev', 'Image data type, e.g., median, mean, ...')
            hdu3.header['BUNIT'] = ('ADU', 'Pixel data unit')

            hdu4 = fits.ImageHDU(hotmask, name='DET'+str(self._detid)+'.HOT')
            hdu4.header['HOTTHRSH'] = (
                self._bad_threshold, 'Threshold for hot pixels [e-/s]')
            hdu4.header['NUMHOT'] = (
                self._numbdpix, 'Number hot pixels in NUMEXP exposures')

            new_hdu = fits.HDUList([hdu1, hdu2, hdu3, hdu4])
            new_hdu.writeto('HOTPIX_Det'+str(self._detid) +
                            '.fits', overwrite=True)

        return hotmask, t_hdu

    def sortByCreationDate(self, expfiles):
        """
         Sort the XML list by the creation date of the data.
        """
        cdates = []
        for axml in expfiles:
            root = ET.parse(axml).getroot()
            for el in root.iter():
                if el.tag == "Header":
                    cdate = el.findall('./CreationDate')
                    cdates.append(cdate[0].text)

        # sort by date
        np_lst = np.array(cdates)
        sorted_indices = np_lst.argsort()

        np_simlst = np.array(expfiles)
        sorted_expfiles = np_simlst[sorted_indices]

        return sorted_expfiles.tolist()

    def get_telapsed(self, expfiles):
        """Get elapsed time.
        """
        # red in the 'DATE-OBS' and 'DATE-END' in the input files
        datestart = []
        dateend = []
        for axml in expfiles:
            root = ET.parse(axml).getroot()
            for el in root.iter():
                if el.tag == "NispFrameFitsFile":
                    fn = el.findall('.//FileName')
                    fn = fn[0].text
            datestart.append(fits.getheader(os.path.join(
                self._workdir, 'data', fn))['DATE-OBS'])
            dateend.append(fits.getheader(os.path.join(
                self._workdir, 'data', fn))['DATE-END'])

        # sort by date
        sorted_dateS = np.array(datestart)
        sorted_dateS.sort()
        sorted_dateE = np.array(dateend)
        sorted_dateE.sort()

        tstart = sorted_dateS[0]
        tend = sorted_dateE[len(expfiles)-1]
        td = datetime.datetime.strptime(tend, '%Y-%m-%dT%H:%M:%S.%fZ') - \
            datetime.datetime.strptime(tstart, '%Y-%m-%dT%H:%M:%S.%fZ')

        tElapsed = td.total_seconds()/3600.  # hours
        self.logger.info(f'Input data elapsed time: {tElapsed} hours')

        return tElapsed

    def get_header0(self, expfile0):
        """Get some headers from the first file in the exposures list.

        :param exfile0: string, the exposure file name in XML

        # parse the exposure list containing XML files

                fn = el.findall('.//FileName')
                fn = fn[0].text

        self.logger.info(

        nd1 = fits.getheader(fn, ext=0)["ND"]
        ng1 = fits.getheader(fn, ext=0)["NG"]
        ft1 = fits.getheader(fn, ext=0)["FRTIME"]

        return nr1, nd1, ng1, ft1

    def load_data(self, expfile, tile):
        """Load the data based on the data origin for SCI mode.
        The data origin is one of the test, simulated and operation data: "TV1", "SIM", "OPS".

        Read a FITS file by tiles from the specified HDU.
        For large FITS file and many of them, there is a need to read in
        small tiles at a time.

        :param expfile: string, the exposures list file name
        :param tile: list, [r1,r2, c1,c2] the index range [r1:r2, c1:c2] in the image array with dimension of 2040x2040!
        :return: int, 2D numpy array
        """
        expfile = os.path.join(self._workdir, 'data', expfile)
        # TV1 test data (one file per scaid, 5 hdus)
        if self._dataorig == "TV1":
            self.logger.info("Process exposure file: {}".format(expfile))
            hdu = fits.open(expfile)
            dat = hdu["SCIENCE"].data
            hdu.close()
        # simulated or in-flight data (32 hdus, each detector has SCI & DQ layers)
        elif self._dataorig == "SIM" or self._dataorig == "OPS":
            # parse the exposure list containing XML files
            root = ET.parse(expfile).getroot()
            for el in root.iter():
                if el.tag == "NispFrameFitsFile":
                    fn = el.findall('.//FileName')
                    fn = fn[0].text

            self.logger.info("Process exposure file: {}".format(fn))
            # open exposure file
            hdus = fits.open(os.path.join(self._workdir, 'data', fn))

            # find the hdu for the corresponding scaid
            hdu = [h for h in hdus[1:32:2] if str(
                h.header['SCA_ID']) == str(self._scaid)]
            if len(hdu) != 1:  # there should be only 1 extension (SCI) containing the SCA_ID
                raise Exception("No extension found for SCA_ID=" +
                                str(self._scaid) + " in expousre file: " + fn)

            hdu0 = hdu[0]

            # what is the dimension of the input data?
            naxis1 = hdu0.header["NAXIS1"]

            # read in the hdu data and trim its borders
            if naxis1 > 2040:
                dat = hdu0.data[4+tile[0]:4+tile[1], 4 +
                                tile[2]:4+tile[3]].astype("float32")
            else:
                dat = hdu0.data[tile[0]:tile[1],
                                tile[2]:tile[3]].astype("float32")
            hdus.close()

        return dat

    def write(self, hotpixmask, files_hdu):
        """Write out the final output

        :param hotpixmask: 2D numpy array, the hot pixel mask.  1=bad, 0=good
        """

        # init the FITS data structure
        # this FITS file contains one extension, for the combined maks for this DET_ID/SCAID
        hdu = fits.HDUList([fits.PrimaryHDU()])

        # update primary header of hdu
        hdr = hdu[0].header
        hdr.set('TELESCOP', 'Euclid', 'Telescope name')
        hdr.set('INSTRUME', 'NISP', 'Instrument name')
        hdr.set('ORIGIN', 'OU-NIR', 'SGS-OU origin')
        hdr.set('IMG_CAT', 'CALIB')
        hdr.set('DET_ID', self._detid, 'Detector ID')
        hdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        hdr.set('BADTYPE', self._bad_reason, 'See COMMENT')
        hdr.set('COMMENT', 'Badtypes: 1=Disconnected, 2=ZeroQE, 3=HighNoise/RTN, ' +
                '4=Snowball, 5=Hot, 6=BaselineOOL, 7=LowQE, 8=SuperQE, 9=Saturated')
        hdr.set('BADVAL', 0, 'Bad pixel value')
        hdr.set('NUMEXP', self._num_exposures, 'Number of input exposures')
        hdr.set('HOTTHRES', self._bad_threshold,
                'Signal threshold for flagging [elec/sec]')
        hdr.set('TELAPS', np.round(self._tElapsed, 2),
                'Elasped time of input exposures (h)')
        #hdr.set('EXPTHRES', self._merge_threshold, 'Percent exposure threshold for flagging')
        hdr.set('DATEUTC', datetime.datetime.utcnow(
        ).isoformat().split(".")[0], 'File creation date')
        hdr.set('FITS_DEF', 'nir.logicBadPixelFlavorMask',
                'FITS name definition')
        hdr.set('FITS_VER', '0.1', 'FITS version')
        hdr.set('VAL_STA', '2020-12-01T00:00:00.0',
                'Valid start time for this product(UTC)')
        hdr.set('VAL_END', '2028-12-31T00:00:00.0',
                'Valid end time for this product(UTC)')

        imghdu = fits.ImageHDU(hotpixmask)
        #imghdu.name = 'DQ_' + str(self._detid)
        imghdu.name = 'Hot Pixels'
        ihdr = imghdu.header
        ihdr.set('DET_ID', self._detid, 'Detector ID')
        ihdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        ihdr.set('NUMBAD', self._numbdpix,
                 'Number of flagged pixels of BADTYPE')
        hdu.append(imghdu)
        hdu.append(files_hdu)

        # write the FITS file for the mask for this SCAID
        now = datetime.datetime.now().isoformat().replace(
            ':', '').split(".")[0]
        # outfile = 'hotpix_SCA_' + scaID + '_' + str(now)+'.fits'     #TODO this is the output file name
        outfile = FileNameProvider().get_allowed_filename(type_name="C-BadPixelFlavor",
                                                          instance_id='MASK',
                                                          processing_function='NIR',
                                                          release=None)
        outfile = os.path.join(self._workdir, 'data', outfile)
        hdu.writeto(outfile, overwrite=True)
        self.logger.info("Written to file {}".format(outfile))

        return outfile

    def get_thresh_pcnt(self, num_exposures, in_percent):
        """
        Get the threshold percentage for merging the bad pixel masks from all exposures
        based on number of available exposures.

        The criteria is as following:
         X=in_percent      for N >= MIN_NUM_EXPOSURES
         X=100%=1.         for N <  MIN_NUM_EXPOSURES
         N is the number of total number of exposures

        :param num_exposures: (int) number of exposures
        :param in_percent: (float) percentage of the threshold

        :return: float, the threshold for the bad pixels
        """
        if num_exposures >= self._min_num_exposures:
            threshold = in_percent
        else:
            threshold = 1.

        self._merge_threshold = threshold

        return threshold

    def merge_masks(self, num_exposures, sum_badpix):
        """
        Check for threshold and determine if the pixel is indeed bad. Make the final mask for this SCAID.

        The criteria is as following:
        For each pixel, if the total number of bad pixels identified from all the exposures (sum_badpix)
        is larger than threshold X, then it is a bad pixel.
        Where:
         X=80% of N  for N >= MIN_NUM_EXPOSURES
         X=100% of N for N <  MIN_NUM_EXPOSURES
         N is the number of total number of exposures

        :param num_exposures: (int) number of exposures used
        :param sum_badpix:  (2D numpy array) sum of bad pixels at each pixel

        :return: (2D int number array) the bad pixel mask for this SCAID

        """

        threshold_pcnt = self.get_thresh_pcnt(
            num_exposures, self._com_threshold)

        # check for threshold
        ind = np.where(sum_badpix >= (threshold_pcnt * num_exposures))
        self._numbdpix = len(ind[0])

        self.logger.info("Final merged mask has {} hot pixels in {} exposures at >= {}% repeatability".
                         format(self._numbdpix, num_exposures, threshold_pcnt*100))

        mask = np.zeros(sum_badpix.shape, dtype=np.int32)
        mask[ind] = 1

        return mask
