class FindZeroQEPixels(object):
    """Class FindZeroQEPixels
    This task identifies pixels which have zero or negative Quantum Efficiencies (QE)
    in the median over the nominally responsive range of 1.0 - 2.05 microns.  Pixels
    are flagged bad where median(QE) <= 0 or NaN.

    The input is the QE map produced at the Goddard DCL in 2016 for the selecteed detector.

    The output is a 2D Zero QE  mask stored as FITS, named zeroQE_SCAID_[timestamp].fits.  

    This class works on a detector basis, since: (a) this allows each
    to be monitored on their own timescales; (b) it is easier to define a smaller
    homogenous set of ground or on orbit observations to be processed; (c) runs
    faster with simpler I/O.

    Bad Pixel flavors:
    1 = Disconnected
    2 = Zero QE (formerly Zero Full Well)
    3 = High CDS noise (stand-in for RTN)
    4 = Snowball
    5 = Hot (formerly 'Halo')
    6 = High Low Baseline
    """

    def __init__(self, workdir, mdbfile, configfn, scaid, outfile):
        """
        @brief Constructor
        """
        self.logger = log.getLogger('FindZeroQEPixels')
        self._workdir = workdir
        self._mdbfile = mdbfile
        self._configfn = configfn
        self._scaid = scaid
        self._outfile = outfile
        self._qefiles = None

        self.setParams()

    def setParams(self):
        if not self._configfn:
            if not self._scaid:  # if scaid not given as well
                self._scaid = 18221
            self._bad_reason = 2
            self._0qe_threshold = 6
        else:
            with open(self._configfn, 'r') as fh:
                params = json.load(fh)

            if not self._scaid:  # if scaid is not given
                self._scaid = str(params["SCAID"])
            self._bad_reason = int(params["BAD_REASON"])
            self._0qe_threshold = params["0QE_THRESHOLD"]

        self.logger.info("Parameters to be used:")
        self.logger.info("Bad reason: " + str(self._bad_reason))
        self.logger.info("SCA ID: " + str(self._scaid))

    def get_detid(self):
        """
        Get the detector id from the detector slot table.
        The DET_ID will be in the input exposure FITS header.

        :return: string, detector id
        """
        # read in the detector slots table
        detslots = readDetectorLayout(self._mdbfile)

        # find the detector id corresponding to the scaid
        detid = None
        for key, val in detslots.items():
            if str(val) == str(self._scaid):
                detid = key
        if detid is not None:
            self._detid = detid
        else:
            raise Exception(
                "Can not find the corresponding detector ID for SCA_ID {}".format(self._scaid))

        return detid

    def getQEfile(self):
        """Get the corresponding QE file from the MDB XML file for the SCA_ID
        """
        if self._qefiles is None:
            mdb = Mdb(self._mdbfile)
            self._qefiles = mdb.get_value(
                "SpaceSegment.Instrument.NISP.NISPQuantumEfficiencyFile")

        qefile = None
        for qf in self._qefiles:
            if str(self._scaid) in qf:
                qefile = os.path.join(self._workdir, "data", qf)

        return qefile

    def findthem(self):
        """Function to find zero QE pixels using the DCL QE data

        :return: int, 2D numpy array, the zero QE pixel mask
        """
        # find the corresponding QE file from the MDB
        qefile = self.getQEfile()
        self._num_exposures = 1

        # get detector id
        self.get_detid()

        # make a table HDU containing the list of input exposure file names
        t_hdu = make_files_hdu([qefile])

        # wave lengths to be used
        qe_wav_lens = [wlen for wlen in range(
            1000, 2110, 50)]  # [1000nm - 2100nm]
        # hud names based on the wavelength
        extnames = ['WAVELEN' + str(wlen) + 'nm' for wlen in qe_wav_lens]
        num_qestack = len(qe_wav_lens)

        # read in the data
        qestack = np.zeros((num_qestack, 2040, 2040))
        hdus = fits.open(os.path.join(self._workdir, 'data', qefile))
        nxy = hdus[1].header["NAXIS1"]
        n = 0
        for ext in extnames:
            if nxy == 2048:
                qestack[n, :, :] = hdus[ext].data[4:-4:, 4:-4]
            else:
                qestack[n, :, :] = hdus[ext].data

            n += 1
        hdus.close()

        # mask the NaNs before calculating the median
        qestack_nan = np.ma.MaskedArray(qestack, mask=np.isnan(qestack))

        # calculate the median in wavelength direction
        qemed = np.ma.median(qestack_nan, axis=0)

        # make the mask by checking for QE <= 0 and NaN
        zeroQEmask = np.zeros(qemed.shape, dtype=np.int32)
        ind = np.where(qemed <= self._0qe_threshold)
        zeroQEmask[ind] = 1
        ind = np.where(qemed.mask)
        zeroQEmask[ind] = 1

        ind = np.where(zeroQEmask)
        self._numbdpix = len(ind[0])
        self.logger.info(
            "Number of pixels with Median(QE) <= 0 or NaN: {}".format(self._numbdpix))

        # write out the final output
        badpixpath = self.write(zeroQEmask.astype(np.uint8), t_hdu)

        # write out the XML file
        self.logger.info("Creating XML file " + self._outfile)
        result = writeXml(badpixpath, self._outfile)

        # check the result
        if result:
            self.logger.info("Wrote XML file " + self._outfile)
        else:
            msg = 'FindDeadPixels Xml Creation error.'
            self.logger.error(msg)
            raise Exception("ERROR: FindDeadPixels.writeXml(): " + msg)

    def write(self, zeroqepixmask, files_hdu):
        """Write out the final output

        :param zeroqepixmask: 2D numpy array, the zero QE pixel mask, 0=good, 1=bad
        """

        # init the FITS data structure
        # this FITS file contains one extension, for the combined maks for this DET_ID/SCAID
        hdu = fits.HDUList([fits.PrimaryHDU()])

        # update primary header of hdu
        hdr = hdu[0].header
        hdr.set('TELESCOP', 'Euclid', 'Telescope name')
        hdr.set('INSTRUME', 'NISP', 'Instrument name')
        hdr.set('ORIGIN', 'OU-NIR', 'SGS-OU origin')
        hdr.set('IMG_CAT', 'CALIB')
        hdr.set('DET_ID', self._detid, 'Detector ID')
        hdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        hdr.set('BADTYPE', self._bad_reason, 'See COMMENT')
        hdr.set('COMMENT', 'Badtypes: 1=Disconnected, 2=ZeroQE, 3=HighNoise/RTN, ' +
                '4=Snowball, 5=Hot, 6=BaselineOOL, 7=LowQE, 8=SuperQE')
        hdr.set('BADVAL', 1, 'Bad pixel value')
        hdr.set('NUMEXP', self._num_exposures, 'Number of input exposures')
        hdr.set('EXPTHRES', 0., 'Not used for this BADTYPE')
        hdr.set('BADTHRES', self._0qe_threshold,
                'Threshold for zero QE pixels')
        #hdr.set('NZEROQE', self._numbdpix, 'Number of flagged pixels of BADTYPE')
        hdr.set('DATEUTC', datetime.datetime.utcnow(
        ).isoformat().split(".")[0], 'File creation date')
        hdr.set('FITS_DEF', 'nir.logicBadPixelFlavorMask',
                'FITS name definition')
        hdr.set('FITS_VER', '0.1', 'FITS version')
        hdr.set('VAL_STA', '2020-12-01T00:00:00.0',
                'Valid start time for this product(UTC)')
        hdr.set('VAL_END', '2028-12-31T00:00:00.0',
                'Valid end time for this product(UTC)')

        imghdu = fits.ImageHDU(zeroqepixmask)
        imghdu.name = 'Zero QE Pixels'
        ihdr = imghdu.header
        ihdr.set('DET_ID', self._detid, 'Detector ID')
        ihdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        #ihdr.set('BADTYPE', self._bad_reason, 'See COMMENT')
        #ihdr.set('BADVAL', 1, 'Bad pixel value')
        # ihdr.set('COMMENT','Badtypes: 1=Disconnected, 2=ZeroQE, 3=HighNoise/RTN, '+
        #         '4=Snowball, 5=Hot, 6=BaselineOOL, 7=LowQE, 8=SuperQE')
        #ihdr.set('NZEROQE', self._numbdpix, 'Number of flagged pixels of BADTYPE')
        ihdr.set('NUMBAD', self._numbdpix,
                 'Number of flagged pixels of BADTYPE')
        hdu.append(imghdu)
        hdu.append(files_hdu)

        # write the FITS file for the mask for this SCAID
        now = datetime.datetime.now().isoformat().replace(
            ':', '').split(".")[0]
        # outfile = 'zeroQE_SCA_' + scaID + '_' + str(now)+'.fits'     #TODO this is the output file name
        outfile = FileNameProvider().get_allowed_filename(type_name="C-BadPixelFlavor",
                                                          instance_id='MASK',
                                                          processing_function='NIR',
                                                          release=None)
        outfile = os.path.join(self._workdir, 'data', outfile)

        hdu.writeto(outfile, overwrite=True)
        self.logger.info("Written to file {}".format(self._outfile))

        return outfile
