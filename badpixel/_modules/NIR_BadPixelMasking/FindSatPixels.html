class FindSatPixels(object):
    """Class FindSatPixels
    Consistently saturated pixels are identified across a set of dark current frames.
    The method involves a running pair-wise comparision of frames and extracting the
    minimum population that always appear at the saturation limit, which is defined 
    as a threshold close to the A/D limit.  The threshold takes readout and numerical
    noise into account. Each pair-wise comparison is basically the 2D intersection of
    signal at the limit in pixel coordinates.  In any frame there may be cosmic ray
    related saturated pixel transients; these are not counted.

    This class works on a detector basis, since: (a) this allows each
    to be monitored on their own timescales; (b) it is easier to define a smaller
    homogenous set of ground or on orbit observations to be processed; (c) runs
    faster with simpler I/O.

    Bad Pixel flavors:
    1 = Disconnected
    2 = Zero QE (formerly Zero Full Well)
    3 = High CDS noise (stand-in for RTN)
    4 = Snowball
    5 = Hot (formerly 'Halo')
    6 = High Low Baseline
    7 = Low QE
    8 = Super QE
    9 = Saturated
    """

        self._mdbfile = mdbfile
        self._configfn = configfn
        self._explistfn = explistfn
        self._scaid = scaid
        self._outfile = outfile

        self.setParams()

    def setParams(self):
        if not self._configfn:
            if not self._scaid:  # if scaid not given as well
                self._scaid = 18221
            self._bad_reason = 9
            self._sat_threshold = 0.985
            self._ad_limit = 64511
        else:
            with open(self._configfn, 'r') as fh:
                params = json.load(fh)

            if not self._scaid:  # if scaid is not given
                self._scaid = str(params["SCAID"])
            self._bad_reason = int(params["BAD_REASON"])
            self._sat_threshold = params["SAT_THRESHOLD"]
            self._ad_limit = params["AD_LIMIT"]

        self._sat_limit = self._ad_limit * self._sat_threshold

        self.logger.info("Parameters to be used:")
        self.logger.info("Bad reason: " + str(self._bad_reason))
        self.logger.info("SCA ID: " + str(self._scaid))
        self.logger.info("Saturation threshold: " + str(self._sat_threshold))
        self.logger.info("Saturation limit: " + str(self._sat_limit))

    def get_detid(self):
        """
        Get the detector id from the detector slot table.
        The DET_ID will be in the input exposure FITS header.

        :return: string, detector id
        """
        # read in the detector slots table
        detslots = readDetectorLayout(self._mdbfile)

        # find the detector id corresponding to the scaid
        detid = None
        for key, val in detslots.items():
            if str(val) == str(self._scaid):
                detid = key
        if detid is not None:
            self._detid = detid
        else:
            raise Exception(
                "Can not find the corresponding detector ID for SCA_ID {}".format(self._scaid))

        self.logger.info(f'Current detector ID: {detid}')
        return detid

    def load_data(self, expfile):
        """Read in the data from a XML file

        :param expfiles: string, the XML file of the exposure
        :return: int, 2D numpy array
        """

        # parse the exposure XML files
        root = ET.parse(expfile).getroot()
        for el in root.iter():
            # if el.tag == "NirFrameFitsFile":
            if el.tag == "NispFrameFitsFile":  # 7/29/2024
                fn = el.findall('.//FileName')
                fn = fn[0].text

        # open exposure file
        self.logger.info(f'Read in exposure file: {fn}')
        hdus = fits.open(os.path.join(self._workdir, 'data', fn))

        # find the hdu for the corresponding scaid
        hdu = [h for h in hdus[1:32:2] if str(
            h.header['SCA_ID']) == str(self._scaid)]
        if len(hdu) != 1:  # there should be only 1 extension (SCI) containing the SCA_ID
            raise Exception("No extension found for SCA_ID=" +
                            str(self._scaid) + " in expousre file: " + fn)

        hdu = hdu[0]

        # read in the hdu data
        dat = hdu.data.astype("float32")
        hdus.close()

        # remove borders
        nx, ny = dat.shape
        if ny > 2040:
            dat = dat[4:2044, :]  # Remove the border if present
        if nx > 2040:
            dat = dat[:, 4:2044]  # Remove the border if present

        return dat

    def findthem(self):
        """Function to find saturated pixels.

        :return: int, 2D numpy array, the saturated pixel mask
        """
        # read in the exposures list
        expfiles = self.getExposureList()

        # sort by creation date to avoid precision error during np.sum/np.mean for different order of input data
        expfiles = self.sortByCreationDate(expfiles)

        # find the elapsed time
        self._tElapsed = self.get_telapsed(expfiles)

        # get detector id
        self.get_detid()

        # make a table HDU containing the list of input exposure file names
        infiles_thdu = make_files_hdu(expfiles)

        # loop through the exposures, pair-wise comparision, find the pixels where
        # both exposure are above the limit
        nsatarray = np.zeros((self._num_exposures-1,), dtype=np.int32)
        for i in range(self._num_exposures - 1):
            # read in 2 exposures at a time
            if i == 0:
                signal1 = self.load_data(expfiles[i])
            else:
                signal1 = signal2_save  # use the previously read in file
            signal2 = self.load_data(expfiles[i+1])
            signal2_save = signal2.copy()  # save for the next iteration

            signal1 -= 1024
            signal2 -= 1024

            # intersect where both signal >= satLimit
            ind = np.where((signal1 >= self._sat_limit) &
                           (signal2 >= self._sat_limit))
            # number of pixels where both signals > satLimit for this pair
            nsatarray[i] = len(ind[0])

        # where does minimum number of pixels occur
        ind_min = np.where(nsatarray == np.min(nsatarray))[0][0]

        self.logger.info("Saturation array min, max, mean, # occurences of min, 1st occurrence of min: ".
                         format(np.min(nsatarray), np.max(nsatarray), np.mean(nsatarray),
                                nsatarray[nsatarray == np.min(nsatarray)].size,
                                np.where(nsatarray == np.min(nsatarray))[0][0]))

        self.logger.info("FIDUCIAL INTERSECTION: ")
        self.logger.info(expfiles[ind_min])
        self.logger.info(expfiles[ind_min+1])

        # compare the two exposures where the minimum number of pixels occures from above
        signal1 = self.load_data(expfiles[ind_min])
        signal2 = self.load_data(expfiles[ind_min+1])
        signal1 -= 1024
        signal2 -= 1024

        # intersect where both signal >= satLimit
        ind = np.where((signal1 >= self._sat_limit) &
                       (signal2 >= self._sat_limit))
        self.logger.info(f'Number saturated pixels in common: {len(ind[0])}')

        # make the mask
        satmask = np.zeros((2040, 2040), dtype=np.uint8)
        satmask[ind] = 1
        self._numbdpix = len(ind[0])

        # save intermediate result to a table HDU
        col1 = fits.Column(name='Saturated pixel count',
                           array=nsatarray, format='I')
        t_hdu = fits.BinTableHDU.from_columns([col1])
        t_hdu.name = 'SATURATED COUNT'
        t_hdu.header['SNUMMIN'] = (np.min(nsatarray), 'Saturation array MIN')
        t_hdu.header['SNUMMAX'] = (np.max(nsatarray), 'Saturation array MAX')
        t_hdu.header['SNUMMED'] = (
            np.median(nsatarray), 'Saturation array MEDIAN')
        t_hdu.header['SNUMMEAN'] = (
            np.mean(nsatarray), 'Saturation array MEAN')
        t_hdu.header['SNUMSTD'] = (np.std(nsatarray), 'Saturation array STDEV')

        # write out the final output
        badpixpath = self.write(satmask.astype(np.uint8), infiles_thdu, t_hdu)

        # write out the XML file
        self.logger.info("Creating XML file " + self._outfile)
        result = writeXml(badpixpath, self._outfile)

        # check the result
        if result:
            self.logger.info("Wrote XML file " + self._outfile)
        else:
            msg = 'FindSatPixels Xml Creation error.'
            self.logger.error(msg)
            raise Exception("ERROR: FindSatPixels.writeXml(): " + msg)

    def sortByCreationDate(self, expfiles):
        """
         Sort the XML list by the creation date of the data.
        """
        cdates = []
        for axml in expfiles:
            root = ET.parse(axml).getroot()
            for el in root.iter():
                if el.tag == "Header":
                    cdate = el.findall('./CreationDate')
                    cdates.append(cdate[0].text)

        # sort by date
        np_lst = np.array(cdates)
        sorted_indices = np_lst.argsort()

        np_simlst = np.array(expfiles)
        sorted_expfiles = np_simlst[sorted_indices]

        return sorted_expfiles.tolist()

    def get_telapsed(self, expfiles):
        """Get elapsed time.
        """
        # red in the 'DATE-OBS' and 'DATE-END' in the input files
        datestart = []
        dateend = []
        for axml in expfiles:
            root = ET.parse(axml).getroot()
            for el in root.iter():
                if el.tag == "NispFrameFitsFile":
                    fn = el.findall('.//FileName')
                    fn = fn[0].text
            datestart.append(fits.getheader(os.path.join(
                self._workdir, 'data', fn))['DATE-OBS'])
            dateend.append(fits.getheader(os.path.join(
                self._workdir, 'data', fn))['DATE-END'])

        # sort by date
        sorted_dateS = np.array(datestart)
        sorted_dateS.sort()
        sorted_dateE = np.array(dateend)
        sorted_dateE.sort()

        tstart = sorted_dateS[0]
        tend = sorted_dateE[len(expfiles)-1]
        td = datetime.datetime.strptime(tend, '%Y-%m-%dT%H:%M:%S.%fZ') - \
            datetime.datetime.strptime(tstart, '%Y-%m-%dT%H:%M:%S.%fZ')

        tElapsed = td.total_seconds()/3600.  # hours
        self.logger.info(f'Input data elapsed time: {tElapsed} hours')

        return tElapsed

    def write(self, satpixmask, files_hdu, t_hdu):
        """Write out the final output

        :param satpixmask: 2D numpy array, the saturated pixel mask, 0=good, 1=bad
        """

        # init the FITS data structure
        # this FITS file contains one extension, for the combined maks for this DET_ID/SCAID
        hdu = fits.HDUList([fits.PrimaryHDU()])

        # update primary header of hdu
        hdr = hdu[0].header
        hdr.set('TELESCOP', 'Euclid', 'Telescope name')
        hdr.set('INSTRUME', 'NISP', 'Instrument name')
        hdr.set('ORIGIN', 'OU-NIR', 'SGS-OU origin')
        hdr.set('IMG_CAT', 'CALIB')
        hdr.set('DET_ID', self._detid, 'Detector ID')
        hdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        hdr.set('BADTYPE', self._bad_reason, 'See COMMENT')
        hdr.set('COMMENT', 'Badtypes: 1=Disconnected, 2=ZeroQE, 3=HighNoise/RTN, ' +
                '4=Snowball, 5=Hot, 6=Baseline, 7=LowQE, 8=SuperQE, 9=Saturated')
        hdr.set('BADVAL', 1, 'Bad pixel value')
        hdr.set('NUMEXP', self._num_exposures, 'Number of input exposures')
        hdr.set('EXPTHRES', 0., 'Not used for this BADTYPE')
        hdr.set('TELAPS', np.round(self._tElapsed, 2),
                'Elasped time of input exposures (h)')
        hdr.set('DATEUTC', datetime.datetime.utcnow(
        ).isoformat().split(".")[0], 'File creation date')
        hdr.set('FITS_DEF', 'nir.logicBadPixelFlavorMask',
                'FITS name definition')
        hdr.set('FITS_VER', '0.1', 'FITS version')
        hdr.set('VAL_STA', '2020-12-01T00:00:00.0',
                'Valid start time for this product(UTC)')
        hdr.set('VAL_END', '2028-12-31T00:00:00.0',
                'Valid end time for this product(UTC)')

        imghdu = fits.ImageHDU(satpixmask)
        imghdu.name = 'Saturated Pixels'
        ihdr = imghdu.header
        ihdr.set('DET_ID', self._detid, 'Detector ID')
        ihdr.set('SCA_ID', self._scaid, 'SCA Identifier')
        ihdr.set('SATLIMIT', self._sat_limit,
                 "Threshold for saturated pixels [ADU]")
        ihdr.set('NUMBAD', self._numbdpix,
                 'Number of flagged pixels of BADTYPE')
        hdu.append(imghdu)
        hdu.append(files_hdu)
        hdu.append(t_hdu)

        # write the FITS file for the mask for this SCAID
        now = datetime.datetime.now().isoformat().replace(
            ':', '').split(".")[0]
        # outfile = 'zeroQE_SCA_' + scaID + '_' + str(now)+'.fits'     #TODO this is the output file name
        outfile = FileNameProvider().get_allowed_filename(type_name="C-BadPixelFlavor",
                                                          instance_id='MASK',
                                                          processing_function='NIR',
                                                          release=None)
        outfile = os.path.join(self._workdir, 'data', outfile)

        hdu.writeto(outfile, overwrite=True)
        self.logger.info("Written to file {}".format(self._outfile))

        return outfile
