def defineSpecificProgramOptions():
    """
    @brief
        Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for CreateBadPixelsMaskProg
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser("""Create the NIR Bad Pixel Mask.  Usage syntax:
             ~> ERun NIR_BadPixelMasking makeNirBadPixels                         \\
                --mdbfile=The_MDB_file                                            \\
                --deadpixeldatalist=The_deadpixel_data_list                       \\
                --hotpixeldatalist=The_hotpixel_data_list                         \\
                --badbaselinepixeldatalist=The_badbaselinepixel_data_list         \\
                --lowqepixeldatalist=The_lowqepixel_data_list                     \\
                --superqepixeldatalist=The_superqepixel_data_list                 \\
                --zeroqepixeldatalist=The_zeroqepixel_data_list                   \\
                --satpixeldatalist=The_saturatedpixel_data_list                   \\
                --outfile=BadPixelMask.xml                                        \\
                --workdir=./                                                      \\
                --logdir=./                                                       \\
             """)
    parser.add_argument('--mdbfile', type=str, dest='mdbfile',
                        help='MDB XML file')
    parser.add_argument('--deadpixeldatalist', type=str, dest='deadpixeldatalist',
                        help='List of XML files which reference to the dead pixel input data files')
    parser.add_argument('--hotpixeldatalist', type=str, dest='hotpixeldatalist',
                        help='List of XML files which reference to the hot pixel input data files')
    parser.add_argument('--badbaselinepixeldatalist', type=str, dest='badbaselinepixeldatalist',
                        help='List of XML files which reference to the bad baseline pixel input data files')
    parser.add_argument('--lowqepixeldatalist', type=str, dest='lowqepixeldatalist',
                        help='List of XML files which reference to the low QE pixel input data files')
    parser.add_argument('--superqepixeldatalist', type=str, dest='superqepixeldatalist',
                        help='List of XML files which reference to the super QE pixel input data files')
    parser.add_argument('--zeroqepixeldatalist', type=str, dest='zeroqepixeldatalist',
                        help='List of XML files which reference to the zero QE pixel input data files')
    parser.add_argument('--satpixeldatalist', type=str, dest='satpixeldatalist',
                        help='List of XML files which reference to the saturated pixel input data files')
    parser.add_argument('--flowerpixeldatalist', type=str, dest='flowerpixeldatalist',
                        help='List of XML files which reference to the flower pixel input data files')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='Output XML file for the NIR Bad Pixels file')
    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method

    @details
        This method is the entry point to the program
    """
    logger = log.getLogger('CreateBadPixelsMaskProg')
    logger.info('Entering CreateBadPixelsMaskProg mainMethod')

    # the MDB XML file
    mdbfn = os.path.join(args.workdir, args.mdbfile)
    if os.path.isfile(mdbfn):
        logger.info('The MDB XML file: ' + mdbfn)
    else:
        msg = 'The MDB XML file {} not found'.format(mdbfn)
        logger.info(msg)
        raise IOError(msg)

    # the filename of the list of dead pixel XML files
    deadpixfn = os.path.join(args.workdir, args.deadpixeldatalist)
    if os.path.isfile(deadpixfn):
        logger.info('The list file for the XML files: ' + deadpixfn)
    else:
        msg = 'The list file {} not found'.format(deadpixfn)
        logger.info(msg)
        raise IOError(msg)

    # the filename of the list of hot pixel XML files
    hotpixfn = os.path.join(args.workdir, args.hotpixeldatalist)
    if os.path.isfile(hotpixfn):
        logger.info('The list file for the XML files: ' + hotpixfn)
    else:
        msg = 'The list file {} not found'.format(hotpixfn)
        logger.info(msg)
        raise IOError(msg)

    # the filename of the list of bad baseline pixel XML files
    badbaselinepixfn = os.path.join(
        args.workdir, args.badbaselinepixeldatalist)
    if os.path.isfile(badbaselinepixfn):
        logger.info('The list file for the XML files: ' + badbaselinepixfn)
    else:
        msg = 'The list file {} not found'.format(badbaselinepixfn)
        logger.info(msg)
        raise IOError(msg)

    # the filename of the list of low QE pixel XML files
    lowqepixfn = os.path.join(args.workdir, args.lowqepixeldatalist)
    if os.path.isfile(lowqepixfn):
        logger.info('The list file for the XML files: ' + lowqepixfn)
    else:
        msg = 'The list file {} not found'.format(lowqepixfn)
        logger.info(msg)
        raise IOError(msg)

    # the filename of the list of super QE pixel XML files
    superqepixfn = os.path.join(args.workdir, args.superqepixeldatalist)
    if os.path.isfile(superqepixfn):
        logger.info('The list file for the XML files: ' + superqepixfn)
    else:
        msg = 'The list file {} not found'.format(superqepixfn)
        logger.info(msg)
        raise IOError(msg)

    # the filename of the list of zero QE pixel XML files
    zeroqepixfn = os.path.join(args.workdir, args.zeroqepixeldatalist)
    if os.path.isfile(zeroqepixfn):
        logger.info('The list file for the XML files: ' + zeroqepixfn)
    else:
        msg = 'The list file {} not found'.format(zeroqepixfn)
        logger.info(msg)
        raise IOError(msg)

    # the filename of the list of saturated pixel XML files
    satpixfn = os.path.join(args.workdir, args.satpixeldatalist)
    if os.path.isfile(satpixfn):
        logger.info('The list file for the XML files: ' + satpixfn)
    else:
        msg = 'The list file {} not found'.format(satpixfn)
        logger.info(msg)
        raise IOError(msg)
    
    # the filename of the list of flower pixel XML files
    flowerpixfn = os.path.join(args.workdir, args.flowerpixeldatalist)
    if os.path.isfile(flowerpixfn):
        logger.info('The list file for the XML files: ' + flowerpixfn)
    else:
        msg = 'The list file {} not found'.format(flowerpixfn)
        logger.info(msg)
        raise IOError(msg)
    

    # output file
    outfn = os.path.join(args.workdir, args.outfile)
    logger.info('Output file: ' + outfn)

    inlistfiles = [deadpixfn, hotpixfn, badbaselinepixfn,
                   lowqepixfn, superqepixfn, zeroqepixfn, satpixfn, flowerpixfn]
    obj = CreateBadPixelsMask(args.workdir, mdbfn, inlistfiles, outfn)
    obj.create_master_mask()

    logger.info('Exiting CreateBadPixelsMaskProg mainMethod')
