class ApplyBadPixels(object):
    """
    @class ApplyBadPixels
    @brief This class applies a mask of bad and unreliable pixels in the
           near-infrared detectors.  It applies the mask to each data frame.
           This module only masks pixels that are truly dead or can not be
           calibrated to any reasonable level, where reasonable is TBD by the
           calibration group. This module does not mask pixels with low transmittance,
           high dark current, saturated pixels, non-linear pixels or pixels that
           are affected by persistence from pervious exposures.  These will be
           flagged by other modules if they violet their calibration margins. 
           The image combination module will determine if a pixel is usable
           for a given science product.
    """

    def __init__(self, workdir, infile, xmlfile, outfile):
        """
        @brief Constructor
        """
        self.logger = log.getLogger('ApplyBadPixels')

        self._fitspath = os.path.join(workdir, 'data')

        self._infile = infile
        self._xmlfile = xmlfile
        self._outfile = outfile

        self.logger.info(self._infile)
        self.logger.info(self._xmlfile)

    def apply(self):
        """
        @brief Method to apply bad pixel mask to the master mask based
               on calibration dat.  This method applies on 16 detectors
               data.
        """
        self.logger.info("ApplyBadPixels.apply(): read in input data")

        # Get the Badpixel fits filenames
        calfile = XmlReader().get_bad_pixel_mask_fits(self._xmlfile)
        self.logger.info("Bad pixel mask file from MDB: " + calfile)

        calfile = os.path.join(self._fitspath, calfile)
        self.logger.info("Bad pixel mask file : " + calfile)
        cal = fits.open(calfile)

        # Read in data file (file format: fits file with 3x16 extensions,
        # wheres 3 means DET##.SCI, DET##.VAR, DET##.DQ; 16 is for 16
        # detectors; the primary HDU has no data but headers.)
        nse = NispSurveyExposure.readFits(self._infile)
        meta = nse.getMetadata()
        tempf = os.path.basename(calfile)
        tempf = tempf[0: tempf.rfind('T')]  # no long lines before astropy 4.0
        meta.set("BPIXCAL", tempf, "Bad pixel calibraiton file")

        # make sure both input data and calibraton data have equal
        # number of detectors
        num_dect = nse.getNumDetectors()
        if num_dect != (len(cal)-1):
            raise Exception(
                "ERROR: ApplyBadPixels.apply(): unequal number of detectors in input data and calibration data")

        # loop through all detectors
        for i in range(num_dect):
            self.logger.info(
                "ApplyBadPixels.apply(): apply bad pixel mask on detector " + str(i+1))
            # get detector pointer from input data
            ndexp_ptr = nse.getDetector(i)
            expmeta = ndexp_ptr.getMetadata()
            det_id = (expmeta.get("DET_ID")).strip()

            bpmeta = cal["DQ_" + det_id].header
            bparray = cal["DQ_" + det_id].data  # data is an numpy record array
            self.logger.info(
                "Number of bad pixels for this detector: {}".format(len(bparray)))

            # populate some metadata from the calibration product to the image frame (SCI layer only)
            expmeta.add(
                "NBADPIXT", bpmeta["NBADPIXT"], "Total number of bad pixels for this detector")

            # apply calibration mask to the master for each detector
            self.apply_badpixels(ndexp_ptr, bparray)
        self.logger.info(
            "ApplyBadPixels.apply(): finish applying bad pixel mask on all detectors")

        # write out to Fits
        nse.writeFits(self._outfile)
        self.logger.info(
            "ApplyBadPixels.apply(): output writen to " + self._outfile)

    @staticmethod
    def apply_badpixels(imgptr, calarray):
        mastermask = imgptr.getMask()
        expmeta = imgptr.getMetadata()

        # gettting the numpy arrays from the mask, Pixel memory
        # buffers are shared between each image and the corresponding array
        maskarray = mastermask.getArray()

        # input bad pixel mask is a numpy record array for SC456
        x = calarray['X']
        y = calarray['Y']
        flag = calarray['FLAG']

        # the bad pixel mask indexing is in 2048x2048
        # we need to check the image frame dimensions and apply BPM accordingly
        dim = maskarray.shape[0]
        # setting the bad pixel mask in the master Mask
        if dim == 2048:
            xx = x
            yy = y
        elif dim == 2040:
            xx, yy = ApplyBadPixels.from2048to2040(x, y)
        maskarray[xx, yy] |= flag.astype(
            np.uint32)  # maskarray is in np.uint32!

    @staticmethod
    def from2048to2040(xpix, ypix):
        """ Subset image from 2048x2048 to 2040x2040.
        """
        ind = np.where((xpix > 3) & (xpix < 2044))
        xpix = xpix[ind]
        ypix = ypix[ind]

        ind = np.where((ypix > 3) & (ypix < 2044))
        xpix = xpix[ind]
        ypix = ypix[ind]

        return xpix-4, ypix-4
