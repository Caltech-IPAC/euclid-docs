def load_list_from_file(fname):
    """ Read in the list file which is either a list or in JSON format

    Args:
         fname - file name for the list file

    Returns:
         listobject -- list of data files
    """
    with open(fname, 'r') as f:
        if f.read(1) in '{[':  # JSON format
            f.seek(0)
            listobject = flattenList(json.load(f))
            #listobject = json.load(f)
        else:
            f.seek(0)
            lines = f.readlines()
            # get rid of the carriage return at end of the lines if any
            listobject = [line.rstrip() for line in lines]

        return listobject

def prepend_dir(alist, ddir):
    """ Prepend ddir to a list of files

    Args:
         alist -- Python list, a list of file names

    Returns:
         olist -- Python list, a list of file names prepended with data_dir
    """

    olist = []
    for f in alist:
        fp = os.path.join(ddir, f)
        olist.append(fp)

    return olist

def utcnow():
    # figure out the current UTC time in ISO format
    utc_offset = datetime.datetime.utcnow() - datetime.datetime.now()
    local_datetime = datetime.datetime.now()
    result_utc_datetime = local_datetime + utc_offset
    return result_utc_datetime.isoformat()

def make_files_hdu(expfiles):
    """Make a table FITS HDU containing the list of input exposure file names

    :param expfiles: list, the list of exposure file names
    """
    # table FITS HDU
    infileslist = [os.path.basename(x) for x in expfiles]
    col1 = fits.Column(name='input data', array=infileslist,
                       format=str(64)+'A')
    t_hdu = fits.TableHDU.from_columns([col1])
    t_hdu.name = "RAW_FILES"
    return t_hdu

def writeXml(badpixpath, outfilepath):
    """ Write the XML file for the bad pixel FITS file

    Args:
       badpixpath: the bad pixel FITS file path
       outfilepath: the output XML file path
    """

    # create an instance of XmlWriter
    xmlW = XmlWriter()

    # Call the method to create the MasterDark class with FITS file content
    xmlW.create_bad_pixel_flavor(badpixpath)

    # write the xml
    result = xmlW.write_xml(outfilepath)

    return result
