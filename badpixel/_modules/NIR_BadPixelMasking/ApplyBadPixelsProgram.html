def defineSpecificProgramOptions():
    """
    @brief
        Allows to define the (command line and configuration file)
        options specific to this program
    @details
        Input options definition for ApplyBadPixelsProgram
    @return
        An  ArgumentParser.
    """
    parser = argparse.ArgumentParser()
    parser.add_argument('--infile', type=str, dest='infile',
                        help='Input image frame Fits file')
    parser.add_argument('--xmlfile', type=str, dest='xmlfile',
                        help='XML file for Bad pixel calibration Fits files')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='Output Fits file')

    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method.
    @details
        This method is the entry point to the program.

        Example to use this script:
        ./run badpixPyProgram --infile=EUC-TEST-GRED180D11-2015-07-16T054123.848Z.fits
                              --xmlfile=xmlfile.xml --outfile=out.fits
                              --wordir=wordir --logdir=logdir
    """
    logger = log.getLogger('ApplyBadPixelsProgram')
    logger.info('Entering ApplyBadPixelsProgram mainMethod()')

    #
    #  Log some of the arguments
    #
    logger.info('Input file: ' + args.infile)
    logger.info('XML file: ' + args.xmlfile)
    logger.info('Output file: ' + args.outfile)

    imgframefn = os.path.join(args.workdir, args.infile)
    if os.path.isfile(imgframefn):
        logger.info('Input image frame file: ' + imgframefn)
    else:
        msg = 'Image File {} not found'.format(imgframefn)
        logger.info(msg)
        raise IOError(msg)

    xmlfn = os.path.join(args.workdir, args.xmlfile)
    if os.path.isfile(xmlfn):
        logger.info('Input XML file: ' + xmlfn)
    else:
        msg = 'The XML {} not found'.format(xmlfn)
        logger.info(msg)
        raise IOError(msg)

    outfn = os.path.join(args.workdir, args.outfile)
    logger.info('Output file: ' + outfn)

    #
    #  Calling the constructor and a method member
    #
    obj = ApplyBadPixels(args.workdir, imgframefn, xmlfn, outfn)
    obj.apply()

    logger.info('Exiting ApplyBadPixelsProgram mainMethod()')
