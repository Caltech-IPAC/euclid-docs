    def makeTableHDU1(self):
        """Create a Table HDU for the combined mask for this detector."""

        # first combine all the masks
        mask = self.combineByBits()

        # make a Table HDU
        x, y = np.where(mask)
        flag = mask[x, y]

        c1 = fits.Column(name='X',    format='I', array=x)
        c2 = fits.Column(name='Y',    format='I', array=y)
        c3 = fits.Column(name='FLAG', format='J', array=flag)

        thdu = fits.BinTableHDU.from_columns([c1, c2, c3])
        hdr = thdu.header
        hdr.set('DET_ID', self._detid, 'Detector ID')
        hdr.set('SCA_ID', self._scaid, 'SCA identifier')
        hdr.set('EXTNAME', "DQ_" + str(self._detid))
        hdr.set('NBADPIX1', self._NBADPIX1, 'Number of dis-connected pixels')
        hdr.set('NBADPIX2', self._NBADPIX2, 'Number of zero QE pixels')
        hdr.set('NBADPIX3', self._NBADPIX3, 'Number of RTN noise pixels')
        hdr.set('NBADPIX4', self._NBADPIX4, 'Number of snowball pixels')
        hdr.set('NBADPIX5', self._NBADPIX5, 'Number of hot pixels')
        hdr.set('NBADPIX6', self._NBADPIX6, 'Number of bad baseline pixels')
        hdr.set('NBADPIX7', self._NBADPIX7, 'Number of low QE pixels')
        hdr.set('NBADPIX8', self._NBADPIX8, 'Number of super QE pixels')
        hdr.set('NBADPIX9', self._NBADPIX9, 'Number of saturated pixels')
        hdr.set('NBADPIX10', self._NBADPIX10, ' Number of flower pixels')
        hdr.set('NBADPIXT', self._NBADPIXT,
                'Total number of bad pixels for this detector')

        return thdu

    def combineByBits(self):
        """Combine bad pixels types/flavors.
        These 8 bad pixels types are combined:
        Dis-connected, zero-qe (previously full well), RTN (previously high cds noise), snowball pixels, hot,
        low-qe, super-qe.
        Bits assignments:
        1. Dis-connected = Dead = 2
        2. Zero-QE = 3
        3. Bad Baseline = 4
        4. Low-QE = 5
        5. Super-QE = 6
        6. Hot = 7
        7. RTN = 8
        8. Snowball = 9
        9. Saturated = 10
        """
        #bits = {"HOT":0, "DEAD":8, "RTN":22, "SNOWBALL":24}
        #bits = {"DEAD": 2, "ZeroQE": 3, "BadBL": 4, "LowQE": 5, "SuperQE": 6, "HOT": 7, "RTN": 8, "SnowBall": 9}

        # init the combined mask
        self._mask = np.zeros((2048, 2048), dtype=np.int32)

        # dead pixels(disconnected pixles)
        if self._deadpixels is not None:
            arr = self._deadpixels
            ind = np.where(arr)
            self._NBADPIX1 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["DISCONNECTED"]
            self._mask[ind] |= badpixvalue
            # invalid flag on if any of (DISCONNECTED/ZEROQE/SuperQE/BADBASE)
            badpixvalue = 1 << EUC_NIR_MASK_DICT["INVALID"]
            self._mask[ind] |= badpixvalue
        else:
            self._NBADPIX1 = 0

        # zero QE (previously zero full well) pixels
        if self._zeroqe is not None:
            arr = self._zeroqe
            ind = np.where(arr)
            self._NBADPIX2 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["ZEROQE"]
            self._mask[ind] |= badpixvalue
            # invalid flag on if any of (DISCONNECTED/ZEROQE/SuperQE/BADBASE/SATURATED)
            badpixvalue = 1 << EUC_NIR_MASK_DICT["INVALID"]
            self._mask[ind] |= badpixvalue
        else:
            self._NBADPIX2 = 0

        # bad baseline pixels
        if self._badbl is not None:
            arr = self._badbl
            ind = np.where(arr)
            self._NBADPIX6 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["BADBASE"]
            self._mask[ind] |= badpixvalue
            # invalid flag on if any of (DISCONNECTED/ZEROQE/SuperQE/BADBASE/SATURATED)
            badpixvalue = 1 << EUC_NIR_MASK_DICT["INVALID"]
            self._mask[ind] |= badpixvalue
        else:
            self._NBADPIX6 = 0

        # low QE pixels
        if self._lowqe is not None:
            arr = self._lowqe
            ind = np.where(arr)
            self._NBADPIX7 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["LOWQE"]
            self._mask[ind] |= badpixvalue
        else:
            self._NBADPIX7 = 0

        # super QE pixels
        if self._superqe is not None:
            arr = self._superqe
            ind = np.where(arr)
            self._NBADPIX8 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["SUPERQE"]
            self._mask[ind] |= badpixvalue
            # invalid flag on if any of (DISCONNECTED/ZEROQE/SuperQE/BADBASE/SATURATED)
            badpixvalue = 1 << EUC_NIR_MASK_DICT["INVALID"]
            self._mask[ind] |= badpixvalue
        else:
            self._NBADPIX8 = 0

        # hot pixels
        if self._hot is not None:
            arr = self._hot
            ind = np.where(arr)
            self._NBADPIX5 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["HOT"]
            self._mask[ind] |= badpixvalue
        else:
            self._NBADPIX5 = 0

        # saturated pixels
        if self._sat is not None:
            arr = self._sat
            ind = np.where(arr)
            self._NBADPIX9 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["SATUR"]
            self._mask[ind] |= badpixvalue
            # invalid flag on if any of (DISCONNECTED/ZEROQE/SuperQE/BADBASE/SATURATED)
            badpixvalue = 1 << EUC_NIR_MASK_DICT["INVALID"]
            self._mask[ind] |= badpixvalue
        else:
            self._NBADPIX9 = 0
            
        # flower pixels
        if self._flower is not None:
            arr = self._flower
            ind = np.where(arr)
            self._NBADPIX10 = len(ind[0])
            badpixvalue = 1 << EUC_NIR_MASK_DICT["FLOWER"]
            self._mask[ind] |= badpixvalue
            # setting also the invalidity for flower pixels
            badpixvalue = 1 << EUC_NIR_MASK_DICT["INVALID"]
            self._mask[ind] |= badpixvalue     
        else:
            self._NBADPIX10 = 0      

        # RTN (previously high cds noise) pixels
        """
       if self._cdsnoise is not None:
          arr = self._cdsnoise
          ind = np.where(arr)
          self._NBADPIX3 = len(ind[0])
          badpixvalue = 1 << EUC_NIR_MASK_DICT["RTN"]
          self._mask[ind] |= badpixvalue
       """
        self._NBADPIX3 = 0

        # snowball pixels
        self._NBADPIX4 = 0

        ind = np.where(self._mask)
        self._NBADPIXT = len(ind[0])

        self.logger.info("Detector {}, number of disconnected bad pixels: {}".format(
            self._detid, self._NBADPIX1))
        self.logger.info("Detector {}, number of zero QE bad pixels: {}".format(
            self._detid, self._NBADPIX2))
        self.logger.info("Detector {}, number of RTN noise bad pixels: {}".format(
            self._detid, self._NBADPIX3))
        self.logger.info("Detector {}, number of snowball pixels: {}".format(
            self._detid, self._NBADPIX4))
        self.logger.info("Detector {}, number of hot pixels: {}".format(
            self._detid, self._NBADPIX5))
        self.logger.info("Detector {}, number of bad baseline pixels: {}".format(
            self._detid, self._NBADPIX6))
        self.logger.info("Detector {}, number of low QE bad pixels: {}".format(
            self._detid, self._NBADPIX7))
        self.logger.info("Detector {}, number of super QE bad pixels: {}".format(
            self._detid, self._NBADPIX8))
        self.logger.info("Detector {}, number of saturated bad pixels: {}".format(
            self._detid, self._NBADPIX9))
        self.logger.info("Detector {}, number of flower bad pixels: {}".format(
            self._detid, self._NBADPIX10))
        self.logger.info("Detector {}, total number of bad pixels: {}".format(
            self._detid, self._NBADPIXT))

        return self._mask
