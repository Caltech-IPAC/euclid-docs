    obj.split()

    logger.info('Exiting SplitData mainMethod')

class SplitData(object):
    """
    @class SplitData

    @brief This class splits the input data into group of 2-detector pairs
     to be fed as input to FitCoef.py.  This split will enable the parallel
     processing in IAL.
     The input is a list of XML files in JSON.
     The output is a nested list of XML files in JSON.
    """
    #DCUs are group of pair of detectors
    DCUs = [[11, 31], [12, 32], [13, 33], [14, 34], [21, 41], [22, 42], [23, 43], [24, 44]]

    def __init__(self, workdir, xmllist, outfile):
        """
        @brief Constructor
        """

        self.logger = log.getLogger('SplitData')
        self._workdir = workdir
        self._xmllist = xmllist
        self._outfile = outfile

    def split(self):
        """
        @brief Method to split the input list into nested lists.
        """
  
        self.logger.info("split(): read in input data")
        #load the list
        inlist = load_list_from_file(self._xmllist)
        listxmls = prepend_dir(inlist, self._workdir)
        
        #loop through the DCUs and split the pairs into nested list
        fin_list = []
        for pair in self.DCUs:
            print(pair[0], pair[1])
            pairxml = []
            #loop though the XML list and find those belog to the pair
            for xmlfn in listxmls:
                root = ET.parse(xmlfn).getroot()
                for el in root.iter():
                    if el.tag == "DetectorList":
                       detid = el.findall('.//DetectorId')
                       detid = detid[0].text
                       if (detid == str(pair[0])) or (detid == str(pair[1])):
                          pairxml.append(xmlfn)
            if len(pairxml) !=0 :
               fin_list.append(pairxml)
                
        #dump as a JSON file
        with open(self._outfile, 'w') as listfile:
             finlist_json = json.dumps(fin_list)
             listfile.write(finlist_json)

def load_list_from_file(fname):
    """ Read in the list file which is in either JSON format or plain list of files

    Args:
         fname - file name for the list of data file

    Returns:
         listobject -- list of data files
    """
    with open(fname,'r') as f:
         if f.read(1) in '{[':    #JSON format
            f.seek(0)
            listobject = flattenList(json.load(f))
            #listobject = json.load(f)
         else:
            f.seek(0)
            lines = f.readlines()
            #get rid of the carriage return at end of the lines if any
            listobject = [line.rstrip() for line in lines]

         return listobject

def prepend_dir(alist, ddir):
    """ Prepend ddir to a list of files

    Args:
         alist -- Python list, a list of file names

    Returns:
         olist -- Python list, a list of file names prepended with data_dir
    """

    olist = []
    for f in alist:
        fp = os.path.join(ddir, f)
        olist.append(fp)

    return olist
