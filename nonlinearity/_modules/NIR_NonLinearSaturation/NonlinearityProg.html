def defineSpecificProgramOptions():
    """
    @brief
        Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for NonlinearityProg 
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser("""Nonlinearity correction.  Usage syntax:
             ~> ERun NIR_NonLinearSaturation correctNonlinearity   \\
                --xmlfile=The_MDB_file                             \\
                --calxmlfile=ListOfCalProductsXmlFiles             \\
                --config=$workdir/theNIRconfigset.xml              \\
                --infile=input_data.fits                           \\
                --outfile=output_data.fits                         \\
                --workdir=./                                       \\
                --logdir=./                                        \\
             """)
    parser.add_argument('--infile', type=str, dest='infile',
                        help='Input image frame Fits file')
    parser.add_argument('--xmlfile', type=str, dest='xmlfile',
                        help='The MDB XML file')
    parser.add_argument('--calxmlfile', type=str, dest='calxmlfile',
                        help='JSON file listing the calibration products XML files')
    parser.add_argument('--outfile', type=str, dest='outfile',
                        help='Output Fits file')
    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')
    parser.add_argument('--config', type=str, dest='config',
                        help='The NIR configuration set, optional if using the nonlinearity model')

    return parser

def mainMethod(args):
    """
    @brief
        The "main" method
    @details
        This method is the entry point to the program
    """
    logger = log.getLogger('NonlinearityProg')
    logger.info('Entering NonlinearityProg mainMethod')

    #image frame file
    imgframefn = os.path.join(args.workdir,  args.infile)
    if os.path.isfile(imgframefn):
       logger.info('Input image frame file: ' + imgframefn)
    else:
       msg = 'Image File {} not found'.format(imgframefn)
       logger.info(msg)
       raise IOError(msg)

    #MDB XML file
    xmlfn = os.path.join(args.workdir, args.xmlfile)
    if os.path.isfile(xmlfn):
       logger.info('Input MDB XML file: ' + xmlfn)
    else:
       msg = 'The MDB XML {} not found'.format(xmlfn)
       logger.info(msg)
       raise IOError(msg)

    #Calibration products XML file
    calxmlfn = os.path.join(args.workdir, args.calxmlfile)
    if os.path.isfile(calxmlfn):
       logger.info('Input list file for the calibration products XML files : ' + calxmlfn)
    else:
       msg = 'The input list file {} not found'.format(calxmlfn)
       logger.info(msg)
       raise IOError(msg)

    #output file
    outfn = os.path.join(args.workdir, args.outfile)
    logger.info('Output file: ' + outfn)

    ##the configuration file containing the parameters
    if args.config:
       config_set = os.path.join(args.workdir, args.config)
       logger.info('NIR configuration set file: ' + config_set)

       try:
          module_name = NIR_module_name.NIR_NonLinearSaturation
          conf_file = XmlReader().get_config_file(xml_conf_file=config_set,
                      module_name=module_name, sub_module='nonlinear_validity_limits')
          configfn = os.path.join(args.workdir, 'data', conf_file)
          logger.info('User input parameter file: ' + configfn)
       except:
          configfn = None
          logger.info('No user parameters specified.  Using the nonlinearity  model.')
    else:
       conf_file = None
       configfn = None
       logger.info('NIR configuration set file is not specified.')

    #
    obj = NonLinearityCorrection(args.workdir, imgframefn, xmlfn, calxmlfn, configfn, outfn)
    obj.apply()

    logger.info('Exiting NonlinearityProg mainMethod')
