def polyfit2(x, y, order=2):
    """
     Do 2nd (default) order polynomial fit: y(x) = a0 + a1*x + a2*x^2
     
     Args:
        y -- 1D numpy array
        x -- 1D numpy array

     Returns:
        a -- (a2, a1, a0),  n-th order polynomial coefficients
    """
    a = np.polyfit(x, y, order)
    return a     #a2, a1, a0

def load_list_from_file(fname):
    """ Read in the list file which is in JSON format

    Args:
         fname - JSON file, file name for the list of XML file

    Returns:
         listobject -- list of XML files
    """
    with open(fname,'r') as f:
         listobject = flattenList(json.load(f))
         return listobject

def parse_xml(xml_list):
    """Parse a list of XML files for the content of FileName tag
     Args:
       xml_list - list of XML files

     Return:
       file_list - list of file names found inside the XML
    """
    file_list = []
    for axml in xml_list:
        root = ET.parse(axml).getroot()
        for fn in root.iter("FileName"):
            file_list.append(fn.text)

    return file_list

def readFitsByChunk(filename, num_hdus, firsthdu, segment):
    """ Read a FITS file by chunks and return data from num_hdus HDUs per detector
        For large FITS file and many of them, there is a need to read in small chunks at a time.
   
     Args:
        filename -- the FITS file name
        num_hdus -- number of HDUs per detector in the fits file, this is to accomadate different datasets.
                    e.g., a NISP image frame typically has 3 layers per detector
        firsthdu -- the index (within 16xnum_hdus) of the 1st hdu of the num_hdus HDUs to read
        segment  -- [m1,m2,n1,n2] the index range [m1:m2,n1:n2] in the image array
  
     Returns:
        hdudata: the data for this chunk and hdu, dimension [num_hdus, segment]
    """

    m1 = segment[0]
    m2 = segment[1]
    n1 = segment[2]
    n2 = segment[3]

    if num_hdus == 1:  #assuming 1 layer per detector, and it's a image layer
       dd = np.zeros((num_hdus, m2-m1, n2-n1), dtype=np.float32)
    elif num_hdus == 3: #assuming 3 layers per detector
       dd = np.zeros((num_hdus-1, m2-m1, n2-n1), dtype=np.float32)
       dq = np.zeros((m2-m1, n2-n1), dtype=np.int32)
       
    with fits.open(filename, mode='denywrite', memmap=True, do_not_scale_image_data=True) as hdus:
        for i in range(num_hdus):
           if i < 2:
              dd[i, :, :] = hdus[firsthdu+i].data[m1:m2, n1:n2]
           if i == 2:  #the DQ layer for a NISP image frame
              dq = hdus[firsthdu+i].data[m1:m2, n1:n2]
              hdr = hdus[firsthdu+i].header
              if 'BZERO' in hdr.keys():     #need to shift the unsigned to signed integer
                 dq = dq * hdr['BSCALE'] + hdr['BZERO']
              else:
                 dq = dq + 2 ** (hdr['BITPIX'] - 1)

           del hdus[firsthdu+i].data  #force mmap to close
 
    if num_hdus == 1:
       return dd
    else:
       return dd, dq
