class Combine2Dets(object):
    """
    @class Combine2Dets

    @brief This class combines a list of 2-detector-pairs input data
     (coef.fits, cov.fits. failmask.fits) into 3 FITS files each of which
     contains 16 detectors. 

     These input data is the outputs from FitCoef.py.
     Using the default split output format of the IAL, the input list
     is of a JSON file that has 8 nexted elements.
     It looks like:
     [["iteratio1/nl_coef.fits", "iteratio1/nl_cov.fits", "iteratio1/nl_failedmask.fits"],
      ["iteratio2/nl_coef.fits", "iteratio2/nl_cov.fits", "iteratio2/nl_failedmask.fits"],
      ...
     ]

    """
    #detector ids
    detids = [11,21,31,41, 12,22,32,42, 13,23,33,43, 14,24,34,44]

    def __init__(self, workdir, filelist, outfile):
        """
        @brief Constructor
        """

        self.logger = log.getLogger('Combine2Dets')
        self._workdir = workdir
        self._filelist = filelist
        self._outfile = outfile

    def combine(self):
        """
        @brief Method to combine the 2-detector-pairs FITS into one 16-detector FITS.
        """
  
        self.logger.info("combine(): read in input data")
        #load the list
        inlist = load_list_from_file(self._filelist)
        listfiles = prepend_dir(inlist, self._workdir)
        
        #group Coef, Cov and FailedMask files
        coeflist = listfiles[::3]
        covlist = listfiles[1::3]
        masklist = listfiles[2::3]

        #what is the channel?
        obstype = fits.getheader(coeflist[0], 0)["OBSTYPE"]
        if obstype == "IMAGE":
           self._mode = "PHOTO"
        else: 
           self._mode = "SPECTRO"
        self.logger.info("combine(): it is {} mode".format(self._mode))

        #make final prodducts
        coefhdus = fits.HDUList(fits.PrimaryHDU())
        covhdus  = fits.HDUList(fits.PrimaryHDU())
        maskhdus = fits.HDUList(fits.PrimaryHDU())

        #copy primary headers from input to output
        phdr = fits.getheader(coeflist[0], 0)
        coefhdus[0].header = phdr.copy()
        phdr = fits.getheader(covlist[0], 0)
        covhdus[0].header = phdr.copy()
        phdr = fits.getheader(masklist[0], 0)
        maskhdus[0].header = phdr.copy()

        #loop through the input file and add the data to the final HDUs
        for f in coeflist:
            ff = fits.open(f)
            coefhdus.append(ff[1].copy())
            coefhdus.append(ff[2].copy())
            ff.close(ff)

        for f in covlist:
            ff = fits.open(f)
            covhdus.append(ff[1].copy())
            covhdus.append(ff[2].copy())
            ff.close(ff)

        for f in masklist:
            ff = fits.open(f)
            maskhdus.append(ff[1].copy())
            maskhdus.append(ff[2].copy())
            ff.close(ff)

        #write out the FITS files
        coefspath = FileNameProvider().get_allowed_filename(type_name="NL"+self._mode,
                                                           instance_id="COEF-IPAC",
                                                           processing_function='NIR',
                                                           release=None)

        coefspath = os.path.join(self._workdir, 'data', coefspath)
        coefhdus.writeto(coefspath, overwrite=True)
        self.logger.info("combine(): written to {}".format(coefspath))

        covpath = FileNameProvider().get_allowed_filename(type_name="NL"+self._mode,
                                                           instance_id="COVMAT-IPAC",
                                                           processing_function='NIR',
                                                           release=None)

        covpath = os.path.join(self._workdir, 'data', covpath)
        covhdus.writeto(covpath, overwrite=True)
        self.logger.info("combine(): written to {}".format(covpath))

        maskpath = FileNameProvider().get_allowed_filename(type_name="NL"+self._mode,
                                                           instance_id="FAILED-IPAC",
                                                           processing_function='NIR',
                                                           release=None)

        maskpath = os.path.join(self._workdir, 'data', maskpath)
        maskhdus.writeto(maskpath, overwrite=True)
        self.logger.info("combine(): written to {}".format(maskpath))

        #write out the XML file
        self.logger.info("combine(): creating XML file " + self._outfile)
        xmlW = XmlWriter()
        xmlW.create_nonlinearity_mask(coeff_fits=coefspath, cov_fits=covpath, mask_fits=maskpath)
        result = xmlW.write_xml(self._outfile)

        # check the result 
        if result:
           self.logger.info("combine(): wrote XML file " + self._outfile)
        else:
           msg = 'Non-linearity coefficient Xml Creation error.'
           self.logger.error(msg)
           raise Exception("combine(): ERROR:" + msg)

        return

def load_list_from_file(fname):
    """ Read in the list file which is in either JSON format or plain list of files

    Args:
         fname - file name for the list of data file

    Returns:
         listobject -- list of data files
    """
    with open(fname,'r') as f:
         if f.read(1) in '{[':    #JSON format
            f.seek(0)
            listobject = flattenList(json.load(f))
            #listobject = json.load(f)
         else:
            f.seek(0)
            lines = f.readlines()
            #get rid of the carriage return at end of the lines if any
            listobject = [line.rstrip() for line in lines]

         return listobject

def prepend_dir(alist, ddir):
    """ Prepend ddir to a list of files

    Args:
         alist -- Python list, a list of file names

    Returns:
         olist -- Python list, a list of file names prepended with data_dir
    """

    olist = []
    for f in alist:
        fp = os.path.join(ddir, f)
        olist.append(fp)

    return olist
