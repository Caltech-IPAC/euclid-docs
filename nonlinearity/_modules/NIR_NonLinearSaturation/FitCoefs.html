   def make_nl_cal(self):
       #read in the input list
       self.logger.info("Reading in: " + self._xmllist)
       inlist = load_list_from_file(self._xmllist)
       listsims = prepend_dir(inlist, self._workdir)

       #sort by creation date to avoid precision error during np.sum/np.mean for different order of input data
       listsims = self.sortByCreationDate(listsims)

       #retrieve some info from the xml file
       self.getDetectorIds(listsims[0])
       self.logger.info("This run processes for detectors: {} and {}".format(self._detids[0], self._detids[1]))
       self.getReadoutMode(listsims[0])
       if self._ng == 15:
          self.logger.info("This run processes readout mode S")
          self._mode = "SPECTRO"
       if self._ng == 4:
          self.logger.info("This run processes readout mode P")
          self._mode = "PHOTO"
       self.logger.info("NG={}, NF={}, ND={}".format(self._ng, self._nf, self._nd))

       #read in the detector slots
       self._detslots = readDetectorLayout(self._mdbxml)

       #assign flux ids to each flux level
       fid_lut = self.assignFluxIds(listsims)

       #fitting the ramp
       self.logger.info("Fitting the ramp for all pixels in the two detectors...")
       a = self.avg_Beta_Slope(listsims)
       
       #mask bad pixels
       self.logger.info("Masking bad NL fits...")
       mask1 = mask_beta(a['beta1'])
       mask2 = mask_beta(a['beta2'])
    
       #filling the bad pixels
       self.logger.info("Filling in the bad pixels with slope values to skip polynomial fitting issues in the next step...")
       for i in range(np.shape(a['beta1'])[0]):
           a['beta1'][i,mask1==1]=a['slope1'][i,mask1==1]
       for i in range(np.shape(a['beta2'])[0]):
           a['beta2'][i,mask2==1]=a['slope2'][i,mask2==1]
    
       #insert column of zeros to the 1st dimension, hack to fit to the 4th order
       a['slope1'] = np.insert(a['slope1'][:], 0, 0, axis=0)
       a['beta1']  = np.insert(a['beta1'][:],  0, 0, axis=0)
       a['sigma1'] = np.insert(a['sigma1'][:], 0, 0, axis=0)
       a['slope2'] = np.insert(a['slope2'][:], 0, 0, axis=0)
       a['beta2']  = np.insert(a['beta2'][:],  0, 0, axis=0)
       a['sigma2'] = np.insert(a['sigma2'][:], 0, 0, axis=0)

       numcores = int(os.environ.get('PIPELINE_CPU_CORES', 1))
       self.logger.info(f'{numcores} cores to be used on each node')
       #fitting the 1st detector
       self.logger.info("Fitting pixels in first Detector...")
       #coef1,covar1 = zip(*Parallel(n_jobs=-1)
       #               (delayed(np.polyfit)(a['slope1'][:,pix],a['beta1'][:,pix],self._polyorder,w=a['sigma1'][:,pix]/max(a['sigma1'][:,pix]),cov=True)
       #                for pix in tqdm(range(np.shape(a['beta1'])[1]), position=0, leave=True)))

        
      # Retrieve results, filtering out None values (where an exception occurred)
       with ThreadPoolExecutor(max_workers=numcores) as executor:
           futures = [executor.submit(compute_polyfit_and_covar, pix, a, self._polyorder,self._effectiveframes,det=1) for pix in range(np.shape(a['beta1'])[1])]
           results = [future.result() for future in tqdm(futures, position=0, leave=True)]
       coef1, covar1 = zip(*results)

       #mask the pixel that can't be fitted
       badfit_pix = [i for i in range(len(coef1)) if coef1[i][0] != coef1[i][0]] 
       if len(badfit_pix) > 0:
          self.logger.info("Detector 1 has {} pixels can't be fitted".format(len(badfit_pix))) 
          mask1[badfit_pix] = 1

       #reshape coef1 [2048x2048,5] to [2048,2048,5]
       coef_map1 = np.reshape(coef1,(2048,2048,-1))                   #np.polyfit returns highest order first                   
       coef_matrix1 = np.zeros([self._polyorder+3, np.shape(coef_map1)[0], np.shape(coef_map1)[1]])    
       coef_matrix1[0,:,:] = 0 #Fmin
       coef_matrix1[1,:,:] = 64086.2069#(59000 - 3000) * ng*(nf+nd) / ((ng-1)*nd + ng*nf) #Fmax
       for i in range(self._polyorder+1):
           coef_matrix1[2+i,:,:] = coef_map1[:,:,-1-i]                #reverse to 'from smallest order to highest'
           #coef_matrix1[2+i,:,:] = coef_map1[:,:,i]                  #no reverse for curve_fit

       #reverse covar1 to 'from smallest order to highest' by rotating 5x5 by 180 degrees
       covar1 = np.asarray(covar1)    #tuple to numpy array
       reversed_covar1 = Parallel(n_jobs=-1)(delayed(np.rot90)(covar1[pix, :,:], k=2) for pix in tqdm(range(np.shape(covar1)[0])))
       #reshape reversed_covar1 [2048x2048,5,5] to [5,5,2048,2048]
       cov_map1 = np.reshape(reversed_covar1, (2048,2048, np.shape(reversed_covar1)[1], np.shape(reversed_covar1)[2])) 
       cov_map1 = np.moveaxis(cov_map1, (2,3), (0,1))

       #reshape mask1 [2048x2048] to [2048,2048]
       mask_map1 = np.reshape(mask1, (2048,2048))

       #fitting the 2nd detector
       self.logger.info("Fitting pixels in second Detector...")
       with ThreadPoolExecutor(max_workers=numcores) as executor:
           futures = [executor.submit(compute_polyfit_and_covar, pix, a, self._polyorder,self._effectiveframes,det=2) for pix in range(np.shape(a['beta2'])[1])]
           results = [future.result() for future in tqdm(futures, position=0, leave=True)]
       coef2, covar2 = zip(*results)

       #mask the pixel that can't be fitted
       badfit_pix = [i for i in range(len(coef2)) if coef2[i][0] != coef2[i][0]] 
       if len(badfit_pix) > 0:
          self.logger.info("Detector 2 has {} pixels can't be fitted".format(len(badfit_pix))) 
          mask2[badfit_pix] = 1

       #reshape coef2 [2048x2048,5] to [2048,2048,5]
       coef_map2 = np.reshape(coef2,(2048,2048,-1))                   #np.polyfit returns highest order first
       coef_matrix2 = np.zeros([self._polyorder+3, np.shape(coef_map2)[0], np.shape(coef_map2)[1]])    
       coef_matrix2[0,:,:] = 0 #Fmin
       coef_matrix2[1,:,:] = 64086.2069#(59000 - 3000) * ng*(nf+nd) / ((ng-1)*nd + ng*nf) #Fmax
       for i in range(self._polyorder+1):
            coef_matrix2[2+i,:,:] = coef_map2[:,:,-1-i]                #reverse to 'from smallest order to highest'
            #coef_matrix2[2+i,:,:] = coef_map2[:,:,i]                  # no reverse for curve_fit

       #reverse covar2 to 'from smallest order to highest' by rotating 5x5 by 180 degrees
       covar2 = np.asarray(covar2)    #tuple to numpy array
       reversed_covar2 = Parallel(n_jobs=-1)(delayed(np.rot90)(covar2[pix, :,:], k=2) for pix in tqdm(range(np.shape(covar2)[0]))) 
       #reshape reversed_covar2 [2048x2048,5,5] to [5,5,2048,2048]
       cov_map2 = np.reshape(reversed_covar2, (2048,2048, np.shape(reversed_covar2)[1], np.shape(reversed_covar2)[2])) 
       cov_map2 = np.moveaxis(cov_map2, (2,3), (0,1))

       #reshape mask2 [2048x2048] to [2048,2048]
       mask_map2 = np.reshape(mask2, (2048,2048))

       #write out
       self.logger.info("Writing out the products for both detectors... ")
       coeffile = self.writeCoefs(coef_matrix1, coef_matrix2)
       covfile = self.writeCov(cov_map1, cov_map2)
       failfile = self.writeFailedMask(mask_map1, mask_map2)

       return

   def sortByCreationDate(self, listsims):
       """
        Sort the XML list by the creation date of the data.
       """
       cdates = []
       for axml in listsims:
           root = ET.parse(axml).getroot()
           for el in root.iter():
               if el.tag == "Header":
                  cdate = el.findall('./CreationDate')
                  cdates.append(cdate[0].text)

       #sort by date
       np_lst = np.array(cdates)
       sorted_indices = np_lst.argsort()

       np_simlst = np.array(listsims)
       sorted_listsims = np_simlst[sorted_indices]

       return sorted_listsims.tolist()

   def getReadoutMode(self, axmlfile):

   def getFluxIdByCurrentPwm(self, axmlfile):
       """
        Method to calculate a flux_id value from MDB files and
        Intensity and PwmDutyCycle in the XML file.
        Note the corresponding LED_INT and LED_PWM are in the FITS file header after NIR_Init step.
       """
       #get the Intensity and PwmDutyCycle from the xmlfile if any
       root = ET.parse(axmlfile).getroot()
       for el in root.iter():
           if el.tag == "CalibUnit":
              st = el.findall('Status')
              if str(st[0].text) == 'true':
                 led_id = el.findall('.//LedId')         #led identifier (A, B, C, D, E)
                 led_id = str(led_id[0].text)
                 led_int = el.findall('.//Intensity')    #led commanded intensity
                 led_int = float(led_int[0].text)
                 led_pwm = el.findall('.//PwmDutyCycle') #led commanded PWM
                 led_pwm = float(led_pwm[0].text)
              else:
                 self.logger.info(f'Calibration Unit is not active from input file {axmlfile}!!')
                 return None
       #self.logger.info(f'Flat from LED {led_id} with Intensity {led_int} and PWM {led_pwm}')
        
       #
       #self.logger.info('Calculate the Flux ID ')
       # get from the MDB the list of CU LED file and select the one corresponding to LED ID
       file_list = XmlReader().get_from_mdb_nisp_instrument(mdb_xml=self._mdbxml, field='NISPCULEDFluxLUT')

       led_string = 'nicu_1-LED' + led_id
       led_cu_file = None
       for file in file_list:
           if led_string in file:
               led_cu_file = file

       if not led_cu_file:
           err_msg = f'File for {led_string} not found!'
           self.logger.error(err_msg)
           raise FileNotFoundError(err_msg)

       # read the cvs file with pandas
       file_led_path = os.path.join(self._workdir, "data", led_cu_file)

       csv_data = pd.read_csv(file_led_path, skiprows=1)

       if not len(csv_data.columns) == 4:
           err_msg = f'CSV file format not valid. Got {len(csv_data.columns)} columns instead of 4'
           self.logger.error(err_msg)
           raise FileNotFoundError(err_msg)
       else:
           #self.logger.info(f'csv data ok ')
           pass

       # got the columns names
       column_I = csv_data.columns[0]  # intensity name
       column_PWM = csv_data.columns[1]  # PWM name
       column_flux_e = csv_data.columns[3]  # flux_e name

       # group data for intensity
       grouped_I = csv_data.groupby(column_I)

       # get the list of groups Intensity keys and save them in a list
       int_list = grouped_I.groups.keys()
       groups_key_list = list(int_list)

       # search for the closer intensity available
       intensity_best_value = get_closer_value(input_list=groups_key_list, target_value=led_int)
       #self.logger.info(f'For LED_INT = {led_int} found intensity closer value as {intensity_best_value}')

       # get the group with closer intensity
       best_value_int_group = grouped_I.get_group(intensity_best_value)

       # get the column PWM
       pwm_list = best_value_int_group[column_PWM]

       pwm_best_value = get_closer_value(input_list=pwm_list, target_value=led_pwm)
       #self.logger.info(f'For LED_PWM = {led_pwm} found intensity closer value as {pwm_best_value}')

       best_row = best_value_int_group.loc[best_value_int_group[column_PWM] == pwm_best_value]

       electron_flux = best_row[column_flux_e]  # get the value of electron flux in the selected row
       #self.logger.info(f'Found a flux of {int(electron_flux)}')

       return int(electron_flux)

   def assignFluxIds(self, listsims):
       """
        Assign a flux id to an unique flux_e so that the IDs are in range of [1, n],
        where n is the number of unique fluxes in the list of input data.
       """
       #get flux_e
       fluxes = []
       for axml in listsims:
           flux = self.getFluxIdByCurrentPwm(axml)
           if flux is not None:
              fluxes.append(flux)
 
       #find unique fluxes
       fluxes_u = list(set(fluxes))
       fluxes_u.sort()

       #assign ids
       nid = len(fluxes_u)
       lut = {}
       for i, flux in enumerate(fluxes_u):
           lut[flux] = i + 1
           self.logger.info(f'Flux-e {flux} at flux-id {i+1}')

       self._fluxid_lut = lut

       return lut

   def readSlope(self, in_xml):
       '''read from science extension and convert to ADU/frame '''
       scifile_name = self.getSciFileName(in_xml)
       scifile_name = os.path.join(self._workdir, "data", scifile_name)

       header = fits.getheader(scifile_name)
       s_offset = header.get('S_OFFSET', 'Key not found')
       o1 = fits.getdata(scifile_name,1) # one detector 
       o2 = fits.getdata(scifile_name,3) # one detector 
       if s_offset == "Key not found":
          raise Exception("Keyword S_OFFSET not found in the header of file {}".format(os.path.basename(scifile_name)))
       else:
          self.logger.info("Subtracting the offset: {}".format(s_offset))
          o1 -= s_offset
          o2 -= s_offset
       return o1,o2

   def framesArray(self):
       ng = self._ng
       nf = self._nf
       nd = self._nd
       nreads = int((ng*nf + (ng-1.)*nd))
       t2 = np.float64()
       for i in range(0,nreads+1):
           if i % int(nf+nd) in (range(nf)):
              t = i+1
              t2 = np.append(t2, t) 
    
       tgrp = np.reshape(t2[1:],[ng,nf])
       tgrpavg =[]
       for i in range(ng):
           tga = int(np.mean(tgrp[i,:]))
           tgrpavg = np.append(tgrpavg, tga)
       return tgrpavg,t2[1:]

   def avg_Beta_Slope(self, listsims):
       nfluxlev = 5  #total 5 flux levels
       lst = [[] for _ in range(nfluxlev)]
       list_betas_det1,list_slopes_det1,list_sigmas_det1 = copy.deepcopy(lst), copy.deepcopy(lst), copy.deepcopy(lst) 
       list_betas_det2,list_slopes_det2,list_sigmas_det2 = copy.deepcopy(lst), copy.deepcopy(lst), copy.deepcopy(lst)
    
       for sim in tqdm(listsims):
           #get flux ID for this input data
           #self.getFluxId(sim)
           fluxe = self.getFluxIdByCurrentPwm(sim)
           if fluxe is None:
              continue
           self._fluxid = self._fluxid_lut[fluxe]
           #self.logger.info(f'This run processes flux-id {self._fluxid} and flux {fluxe}e-/s...')

           frameAvgs,frames = self.framesArray()
        
           #Measuring Beta1 from the average ramps for two detectors over flattened pixels    
           o = Ortho_Polyfit(frameAvgs,3)  
           ramp_det1,ramp_det2 = self.readRamp(sim)
           flatramp_det1 = np.reshape(ramp_det1,(self._ng,-1))
           betas_det1 = o.fit(flatramp_det1)
           flatramp_det2 = np.reshape(ramp_det2,(self._ng,-1))
           betas_det2 = o.fit(flatramp_det2)
           b1_det1,b1_det2 = betas_det1[:,-2],betas_det2[:,-2]

           # slopes of the two detectors for flattened pixels
           slope_det1,slope_det2 = self.readSlope(sim)
           sig_det1,sig_det2 = self.readSigma(sim)
           s_det1, s_det2 = np.reshape(slope_det1,-1), np.reshape(slope_det2,-1)
           x_det1, x_det2 = np.reshape(sig_det1,-1), np.reshape(sig_det2,-1)
    
           # saving in arrays depending on flux level
           list_betas_det1[self._fluxid-1].append(b1_det1)
           list_betas_det2[self._fluxid-1].append(b1_det2)
           list_slopes_det1[self._fluxid-1].append(s_det1)
           list_slopes_det2[self._fluxid-1].append(s_det2)
           list_sigmas_det1[self._fluxid-1].append(x_det1)
           list_sigmas_det2[self._fluxid-1].append(x_det2)
        
       #make sure there is enough data to do the fitting 
       if len(list_betas_det1[self._polyorder]) == 0:
          #num_fluxlel = [i for i in range(self._polyorder,-1,-1) if len(list_betas_det1[i]) != 0][0] + 1
          #err_msg = f'STOP: not enough ramp data to fit to the {self._polyorder}-th order. Only {num_fluxlel} flux levels were found.'
          num_fluxlel = [i for i in range(self._polyorder,-1,-1) if len(list_betas_det1[i]) != 0]
          if len(num_fluxlel) != 0:
             err_msg = f'STOP: not enough ramp data to fit to the {self._polyorder}-th order. Only {num_fluxlel[0]+1} flux levels were found.'
          else:
             err_msg = f'STOP: not enough ramp data to fit to the {self._polyorder}-th order. Zero flux level was found.'
          self.logger.info(err_msg)
          raise Exception(err_msg)

       # Averaging the repeated simulations with the same LED flux level, 5 (nfluxlev) =number of flux levels
       beta1_det1,beta1_det2 = np.zeros((nfluxlev,2048*2048)), np.zeros((nfluxlev,2048*2048))
       slope_det1,slope_det2 = np.zeros((nfluxlev,2048*2048)), np.zeros((nfluxlev,2048*2048))
       sigma_det1,sigma_det2 = np.zeros((nfluxlev,2048*2048)), np.zeros((nfluxlev,2048*2048))
    
       for i in range(nfluxlev):
           beta1_det1[i,:] = average_arrays(list_betas_det1[i])
           beta1_det2[i,:] = average_arrays(list_betas_det2[i])

def a_prime_j(j, sigma):
    res = list()
    sig = np.copy(sigma)
    for jj in range(1, j + 1):
        sig[:] = sigma[:]
        det_sig = np.linalg.det(sig[:, 1:])
        sig[:, jj] = sig[:, 0] * -1.0
        det_sig_jj = np.linalg.det(sig[:, 1:])
        res.append(det_sig_jj / det_sig)
    return np.asarray(res)

def a_matrix(x, m):
    D = len(x)
    xsp = sum_power(x, 2 * m)
    a = np.zeros((m + 1, m + 1))
    a[:] = np.nan
    a[0, 0] = 1 / np.sqrt(D)
    a[1, 1] = np.sqrt(D / (D * xsp[2] - xsp[1]**2))
    a[1, 0] = -xsp[1] * a[1, 1] / D
    if m < 2:
        return a

    sigma = np.zeros((m, m + 1))
    sigma[:] = np.nan
    for jj in range(m + 1):
        sigma[0, jj] = sigma_n_m(0, jj, a, xsp)
        sigma[1, jj] = sigma_n_m(1, jj, a, xsp)

    for j in range(2, m + 1):

        # compute a prime
        a_prime = np.zeros(j + 1)
        a_prime[0] = 1
        a_prime[1:] = a_prime_j(j, sigma[:j, :j + 1])

        # normalization ortho poly => <Pj, Pj> = 1
        powers = range(j + 1)
        s = 0
        for xx in x:
            s += np.sum(a_prime * np.power(xx, powers))**2
        a[j, 0] = np.sqrt(1 / s)
        a[j, 1:j + 1] = a_prime[1:] * a[j, 0]

        if j < m:
            for jj in range(m + 1):
                sigma[j, jj] = sigma_n_m(j, jj, a, xsp)

    return a

def beta_k_i(x, N):
    x = np.asarray(x).astype(np.float64)
    D = len(x)
    a = a_matrix(x, N)
    powers = range(N + 1)

    beta_k_i = np.zeros((N + 1, D))

    def _z_m_i(m, a_mk, out):
        for i, xx in enumerate(x):
            xx_pow = np.power(xx, powers[:m + 1])
            out[i] += np.sum(a[m, 0:m + 1] * xx_pow) * a_mk

    for k in range(N + 1):
        for m in range(k, N + 1):
            _z_m_i(
                m,
                a[m, k],
                beta_k_i[-1 -
                         k, :]  # compatible numpy polyfit: coeff ak,...,a0
            )
    return beta_k_i

def betas(x, N):
    # calcul les betas pour chaque vector x[0:i] avec i de 0...len(x)
    D = len(x)
    beta = np.zeros((D + 1, N + 1, D))
    beta[0:2] = np.nan
    for i in range(2, D + 1):
        beta[i, :, :i] = beta_k_i(x[:i], N)
    return beta

def sigma_n_m(n, m, a, xsp):
    # \gamma_{nm} = \sum\limits_{j=0}^n(a_{nj}\sum\limits_{i=1}^D x_{i}^{m+j})
    res = 0
    for j in range(n + 1):
        res += a[n, j] * xsp[m + j]
    return res

def sum_power(x_vect, to_pow):
    # calcul la somme des x de puissance 0  to_pow inclus
    # \sum\limits_{i=1}^D x_{i}^pow avec D=len(x) et pow \in {0...to_pow}
    powers = range(to_pow + 1)
    res = np.zeros(len(powers), dtype=np.float64)
    for x in x_vect:
        res += np.power(x, powers)
    return res

    fout = np.zeros_like(full_frame)
    fout1 = np.zeros_like(full_frame)

    frame_science_channel_sls = ([(slice(4, -4), slice(4, 64))] + [(slice(4, -4), slice(i * 64, (i + 1) * 64)) for i in range(1, 31)] + [(slice(4, -4), slice(-64, -4))])
    saturated_mask = np.where(full_frame >= sat_value)

    h, w = np.int32(full_frame.shape)
    n_r, c_s, n_c, l_w_s = nb_ref_pix, channel_size, np.int32(w / channel_size), lr_window_size

    # up/down ref pixels mean
    ud_mean = np.ndarray(n_c, dtype=np.uint16)
    for ch in range(n_c):
        ud_mean[ch] = 0.5 * (
            np.mean(full_frame[0:n_r, ch * c_s:(ch + 1) * c_s]) +  # up
            np.mean(full_frame[-n_r:, ch * c_s:(ch + 1) * c_s])  # down
        )
        sl = frame_science_channel_sls[ch]
        # substract ref pixels
        fout1[sl] = full_frame[sl].astype(np.uint16) - ud_mean[ch]
        
    lr_s = (h - 2 * n_r)
    lr_mean = np.ndarray(lr_s, dtype=np.uint16)
    # left/right ref pixels mean
    for i in range(lr_s):
        j =   i + n_r - l_w_s // 2
        lr_mean[i] = sum(
            np.median(fout1[j:j + l_w_s, 0:n_r] -
                    ud_mean[0], axis=0) +  # left
            np.median(fout1[j:j + l_w_s, -n_r:] -
                    ud_mean[-1], axis=0)  # right
        ) / (2 * n_r)
    fout[n_r:-n_r, n_r:-n_r] = fout1[n_r:-n_r, n_r:-n_r].astype(np.uint16) - lr_mean[:, np.newaxis]
    fout[saturated_mask] = sat_value 
    return fout.astype(full_frame.dtype)
