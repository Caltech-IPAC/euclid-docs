def defineSpecificProgramOptions():
    """
    @brief
        Define the command line and configuration file options
        specific to this program
    @details
        Input options definition for FitCoefs
    @return
        An ArgumentParser
    """
    parser = argparse.ArgumentParser("""Create the Non-linearity coefficients.  Usage syntax:
             ~> ERun NIR_NonLinearSaturation makeNLCoefs         \\
             --mdbxmlfile=The_MDB_file                           \\
             --xmllistfile=xml_list_for_ramps                    \\
             --polyorder=polyorder                               \\
             --nl_coef=nonlinearity_coef.fits                    \\
             --nl_cov=nonlinearity_cov.fits                      \\
             --nl_failedmask=nonlinearity_failedmask.fits        \\
             --workdir=./                                        \\
             --logdir=./                                         \\
             """)
    parser.add_argument('--mdbxmlfile', type=str, dest='mdbxmlfile',
                        help='The MDB XML file')
    parser.add_argument('--xmllistfile', type=str, dest='xmllist',
                        help='File name for the list of XML files describing the calibration/simulation products (txt,JSON)')
    parser.add_argument('--polyorder', type=int, dest='polyorder', default=4,
                        help='The order of the polynomial fitting')
    parser.add_argument('--nl_coef', type=str, dest='nl_coef',
                        help='Output non-linearity coefficient FITS file')
    parser.add_argument('--nl_cov', type=str, dest='nl_cov',
                        help='Output non-linearity covariance FITS file')
    parser.add_argument('--nl_failedmask', type=str, dest='nl_failed',
                        help='Output non-linearity failed mask FITS file')

    parser.add_argument('--workdir', type=str, dest='workdir',
                        help='Working directory path')
    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='Logging directory path')
    return parser

def mainMethod(args):
    """The main method"""
    logger = log.getLogger('NLModelCalibration')
    logger.info('Entering Nonlinearity Calibration FitCoefs mainMethod')

    #MDB XML file
    mdbxmlfn = os.path.join(args.workdir, args.mdbxmlfile)
    if os.path.isfile(mdbxmlfn):
       logger.info('Input MDB XML file: ' + mdbxmlfn)
    else:
       msg = 'The MDB XML {} not found'.format(mdbxmlfn)
       logger.info(msg)
       raise IOError(msg)

    #the input XML list file
    xmllistfn = os.path.join(args.workdir, args.xmllist)
    if os.path.isfile(xmllistfn):
       logger.info("Input ramps list file: " + xmllistfn)
    else:
       msg = 'The input ramps list file {} not found'.format(xmllistfn)
       logger.info(msg)
       raise IOError(msg)

    #poly order
    polyorder = args.polyorder
    logger.info("The polynomial order is: " + str(polyorder))

    #output files
    coef_outfn = os.path.join(args.workdir, args.nl_coef)
    logger.info('Nonlinearity coefficients output file: ' + coef_outfn)
    cov_outfn = os.path.join(args.workdir, args.nl_cov)
    logger.info('Nonlinearity covariance output file: ' + cov_outfn)
    failed_outfn = os.path.join(args.workdir, args.nl_failed)
    logger.info('Nonlinearity failed mask output file: ' + failed_outfn)

    obj = FitCoefs(args.workdir, mdbxmlfn, xmllistfn, polyorder, coef_outfn, cov_outfn, failed_outfn)
    obj.make_nl_cal()

    logger.info('Exiting FitCoefs mainMethod')
