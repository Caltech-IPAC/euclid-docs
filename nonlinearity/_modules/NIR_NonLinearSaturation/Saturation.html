class Saturation(object):
    """
    @class Saturation

    @brief This class masks pixels where the true flux cannot be
           recovered at better than a TBD level.  These pixels will
           be set to a set number of counts and be flagged as saturated.
           Pixels which are  not saturated but do not meet the non-linearity
           requirements will be flagged by the non-linearity module. 

    """

    def __init__(self, imgfile, calfile, outfile):
        """
        @brief Constructor
        """

        self.logger = log.getLogger('Saturation')
        self._imgfile = imgfile
        self._calfile = calfile
        self._outfile = outfile

    def apply(self):
        """
        @brief Method to apply the saturation correction
        """
  
        # Read In the input exposure and saturation calibration fits files.
        # (data file format: fits file with 3x16 extensions, wheres 3 means
        # DET##.SCI, DET##.VAR, DET##.DQ; 16 is for 16 detectors; the
        # primary HDU has no data but headers.)
        
        # (cal file format: fits file with 16 extensions (empty primary HDU),
        # each extension is a 2D array
        
        self.logger.info("Saturation.apply(): read in input data")
        exp = NispSurveyExposure.readFits(self._imgfile)
        expmeta = exp.getMetadata()

        calfits = fits.open(self._calfile)

        tempf = os.path.basename(self._calfile)
        tempf = tempf[0: tempf.rfind('T')]    #no long lines before astropy 4.0
        expmeta.set("SATCAL", tempf, "Saturation limit calibration file")

        # make sure both input data and calibraton data have equal
        # number of detectors
        numDect = exp.getNumDetectors()
        numDect0 = len(calfits) - 1   #not counting the primary HDU
        if numDect != numDect0:
           self.logger.info("ERROR: Saturation.apply(): \
                unequal number of dete ctors in input data and calibration data.")
           raise Exception("ERROR: Saturation.apply():  \
                unequal number of dete ctors in input data and calibration data.")

        # loop through all detectors
        for i in range(numDect):
            self.logger.info("Saturation.apply():  \
                 flag saturation on detector " + str(i+1))
            # get detector pointer from input data
            NDExp_ptr = exp.getDetector(i) 
            det_id = NDExp_ptr.getMetadata().get("DET_ID")
            soffset = NDExp_ptr.getMetadata().get("S_OFFSET")

            # get calibration data for the current detector
            calarr = calfits["DET" + str(det_id)].data
            calarr -= soffset   #issue 14536

            #from SC8 (as of January 15, 2021), the detector map are in 2048x2048
            #make sure the saturation cal data has the same dimension with the image frame
            #the image frame should be 2040x2040 at this step in the NIR pipeline
            caldim = calarr.shape[0]
            if caldim == 2048:
                calarr = calarr[4:-4, 4:-4]

            # flag saturation
            self.flagSaturation(NDExp_ptr, calarr)
            self.logger.info("Saturation.apply():  \
                 finish flag saturation on all detectors")

        calfits.close()

        # saving all three layers in a new FITS file
        exp.writeFits(self._outfile)
        self.logger.info("Saturation.apply(): output writen to " + self._outfile)

    def flagSaturation(self, NDExp_ptr, calarr):
        # get the 3 layers of a exposure
        sciimg = NDExp_ptr.getScience()      #type Image<float>
        scimsk = NDExp_ptr.getMask()         #type Mask<uint32_t>
        scimeta = NDExp_ptr.getMetadata()
        
        # get numpy arrays
        sciimgarray = sciimg.getArray()
        scimskarray = scimsk.getArray()
        
        # find saturated pixels
        satpix = sciimgarray >= calarr
        ind = np.where(satpix)

        if np.sum(satpix) > 0:
            # find the nearest neighboring pixles   6/9/2022
            nns = self.find_nns(satpix) 

            #set saturated pixel flag at the saturated pixels
            scimskarray[satpix] |= (1 << EUC_NIR_MASK_DICT["SATUR"])   #SATUR=10, 4/1/2022
            scimskarray[satpix] |= (1 << EUC_NIR_MASK_DICT["INVALID"]) #4/14/2022

            #set saturated pixel flag at the neighboring pixels
            nns_tuple = tuple((nns[:,0], nns[:,1]))    #tuple of (i,j)
            scimskarray[nns_tuple] |= (1 << EUC_NIR_MASK_DICT["SATUR"])
            scimskarray[nns_tuple] |= (1 << EUC_NIR_MASK_DICT["INVALID"])

            num_sat = len(ind[0]) + len(nns_tuple[0])
        else:
            num_sat = 0

        scimeta.add("NSATPIX", num_sat, "Number of pixels flagged for saturation")
        self.logger.info("Number of pixels flagged for saturation: {}".format(num_sat))

    def find_nns(self, satmsk):
       """ Given a pixel mask of saturation, find the pixels that border the pixels.

       :param satmsk: 2D bool numpy array, True=saturated, False=not saturated

       :retrun:  2D numpy array in dimension of [num_nns, 2], (i,j) index of the found bordering pixels
       """
       #saturated pixel index
       satmsk = 1*satmsk
       dims = satmsk.shape
       satpix = np.argwhere(satmsk == 1).tolist()
       self.logger.info("There are {} saturated pixels".format(len(satpix)))

       #query for the nearest neighbors for each saturated pixel
       xx, yy = np.mgrid[0:dims[0], 0:dims[1]]
       points = np.c_[xx.ravel(), yy.ravel()]
       tree = spatial.cKDTree(points)
       neighbors = tree.query_ball_point(satpix, 1)

       #rid of identity point (the center point of each neighborhood, which is the saturated pixel)
       all_nns = [[points[idx] for idx in nn_ind if idx != (satpix[i][0]*dims[0]+satpix[i][1])]  \
                  for i, nn_ind in enumerate(neighbors)]

       #flatten the list of list to list of tuples
       nns = [(xy[0], xy[1]) for xys in all_nns for xy in xys]
 
       ##check against the saturated pixels, and delete those in the found neighboring pixels
       ##this can happen when saturated pixels are next to each other
       satpix = [(l[0], l[1]) for l in satpix]   #from list of list to list of tuple
       res = list(set(nns).difference(satpix))
       final_nns = np.array(res)
       self.logger.info("Found {} neighboring pixels".format(final_nns.shape[0]))

       """
       #plot all the bad pixels and its nearest neighbors

       plt.plot(final_nns[:,0], final_nns[:,1], 'o')
       plt.show()
       """

       return final_nns
