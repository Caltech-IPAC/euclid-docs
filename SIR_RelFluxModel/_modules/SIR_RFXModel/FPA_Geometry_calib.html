    def pix2fp(self, det, x, y):
        """@brief Convert the location of pixel(s) in a detector to
        the corresponding position(s) in the focal plane.

        @param det: a scalar detector name (among the allowed ones, see detnames());
        @param x: list or numpy array of pixel coordinates along horizontal direction;
        @param y: list or numpy array of pixel coordinates along vertical direction;

        @returns 2-element tuple of N-element lists, each containing
        the locations on the focal plane (in mm) along the horizontal
        and vertical directions respectively of the N pixel
        coordinates provided as input.

        Note: the pixel coordinates are supposed to be in the range
        0:det_npix_horiz-1 or 0:det_npix_vert-1.  However, it is
        possible to provide numbers outside this limit, to obtain
        positions in the focal plane corresponding to the gaps between
        detectors.

        Example:
        from NIR_misc_utils.FocalPlaneModel import *
        fpmodel = FPModel('NISPDetectorSlots.csv', 2048, 2048)
        fp_x, fp_y = fpmodel.pix2fp('33', [3, 2043], [3, 2043])
        # ([1.601895000000006, 38.160735], [2.704857000000004, 397.683697])
        """
        cc = self._fpwcs[det].pixel_to_world(x, y)
        outx = [v.value for v in cc[0]]
        outy = [v.value for v in cc[1]]
        return (outx, outy)

    def fp2pix(self, x_fp, y_fp):
        """
        Convert focal-plane coordinates to detector pixel coordinates.

        Parameters
        ----------
        x_fp : float or ndarray
            Normalised focal-plane X coordinate.
        y_fp : float or ndarray
            Normalised focal-plane Y coordinate.

        Returns
        -------
        det_id : int or ndarray
            Detector index for each input position.
        x_pix : float or ndarray
            Pixel coordinate along the detector X axis.
        y_pix : float or ndarray
            Pixel coordinate along the detector Y axis.
        """
        # existing implementation...

        if use_det is not None:
            assert use_det in self.detnames()
        outd = np.repeat('  '  , x.size)
        outx = np.repeat(np.NaN, x.size)
        outy = np.repeat(np.NaN, x.size)
        dist = np.repeat(np.Inf, x.size)
        for det in self.detnames():
            if use_det is not None:
                if det != use_det:
                    continue
            (ox, oy) = self._fpwcs[det].world_to_pixel(x * units.mm, y * units.mm)
            dd = np.sqrt((ox - self.det_npix_horiz() / 2)**2 +
                         (oy - self.det_npix_vert()  / 2)**2)
            i = np.where(dd < dist)[0]
            if len(i) > 0:
                outd[i] = det
                outx[i] = ox[i]
                outy[i] = oy[i]
                dist[i] = dd[i]
        return (outd, outx, outy)

    def __init__(self, det, x, y):
        if np.isscalar(det):
            det = np.array((det,))
        if np.isscalar(x):
            x = np.array((x,))
        if np.isscalar(y):
            y = np.array((y,))
        if (det.size == 1)  and  (x.size > 1):
            det = np.repeat(det, x.size)
        assert type(det) == type(x) == type(y) == np.ndarray
        assert det.size == x.size == y.size
        self.det = copy.copy(det)
        self.x = copy.copy(x)
        self.y = copy.copy(y)

    def toFPCoords(self):
        outx = np.repeat(np.NaN, self.x.size)
        outy = np.repeat(np.NaN, self.x.size)
        for det in NIRFP().detnames():
            i = np.where(self.det == det)[0]
            if len(i) > 0:
                (ox, oy) = NIRFP().pix2fp(det, self.x[i], self.y[i])
                outx[i] = ox
                outy[i] = oy
        assert not np.any(np.isnan(outx))  # No detector found for coordinates
        return FPCoords(outx, outy)

def boundaries(x):
    """Return boundaries in an array of coordinates.

    Given an input 1D array of sorted coordinates, returns the array
    of boundaries (or edges).  Note that output arrays has 1 extra
    element with respect to input one.
    """
    out = (x[0:-1] + x[1:]) / 2
    d = out[1]-out[0]
    out = np.append(out, out[-1]+d)
    out = np.insert(out, 0, out[0]-d)
    return out

def plt_scatter(object, fp, val=None, **kwargs):
    assert type(fp) == FPCoords
    if val is None:
        return object.scatter(fp.x, fp.y, **kwargs)
    else:
        # I need to transpose here to get consistent results
        return object.scatter(fp.x, fp.y, val.transpose(), **kwargs)
