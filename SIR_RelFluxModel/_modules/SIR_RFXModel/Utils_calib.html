    def pix2fp(self, det, x, y):
        """@brief Convert the location of pixel(s) in a detector to
        the corresponding position(s) in the focal plane.

        @param det: a scalar detector name (among the allowed ones, see detnames());
        @param x: list or numpy array of pixel coordinates along horizontal direction;
        @param y: list or numpy array of pixel coordinates along vertical direction;

        @returns 2-element tuple of N-element lists, each containing
        the locations on the focal plane (in mm) along the horizontal
        and vertical directions respectively of the N pixel
        coordinates provided as input.

        Note: the pixel coordinates are supposed to be in the range
        0:det_npix_horiz-1 or 0:det_npix_vert-1.  However, it is
        possible to provide numbers outside this limit, to obtain
        positions in the focal plane corresponding to the gaps between
        detectors.

        Example:
        from NIR_misc_utils.FocalPlaneModel import *
        fpmodel = FPModel('NISPDetectorSlots.csv', 2048, 2048)
        fp_x, fp_y = fpmodel.pix2fp('33', [3, 2043], [3, 2043])
        # ([1.601895000000006, 38.160735], [2.704857000000004, 397.683697])
        """
        cc = self._fpwcs[det].pixel_to_world(x, y)
        outx = [v.value for v in cc[0]]
        outy = [v.value for v in cc[1]]
        return (outx, outy)

    def fp2pix(self, x_fp, y_fp):
        """
        Convert focal-plane coordinates to detector pixel coordinates.

        Parameters
        ----------
        x_fp : float or ndarray
            Normalised focal-plane X coordinate.
        y_fp : float or ndarray
            Normalised focal-plane Y coordinate.

        Returns
        -------
        det_id : int or ndarray
            Detector index for each input position.
        x_pix : float or ndarray
            Pixel coordinate along the detector X axis.
        y_pix : float or ndarray
            Pixel coordinate along the detector Y axis.
        """
        # existing implementation...

        if use_det is not None:
            assert use_det in self.detnames()
        outd = np.repeat('  '  , x.size)
        outx = np.repeat(np.NaN, x.size)
        outy = np.repeat(np.NaN, x.size)
        dist = np.repeat(np.Inf, x.size)
        for det in self.detnames():
            if use_det is not None:
                if det != use_det:
                    continue
            (ox, oy) = self._fpwcs[det].world_to_pixel(x * units.mm, y * units.mm)
            dd = np.sqrt((ox - self.det_npix_horiz() / 2)**2 +
                         (oy - self.det_npix_vert()  / 2)**2)
            i = np.where(dd < dist)[0]
            if len(i) > 0:
                outd[i] = det
                outx[i] = ox[i]
                outy[i] = oy[i]
                dist[i] = dd[i]
        return (outd, outx, outy)

def boundaries(x):
    """Return boundaries in an array of coordinates.

    Given an input 1D array of sorted coordinates, returns the array
    of boundaries (or edges).  Note that output arrays has 1 extra
    element with respect to input one.
    """
    out = (x[0:-1] + x[1:]) / 2
    d = out[1]-out[0]
    out = np.append(out, out[-1]+d)
    out = np.insert(out, 0, out[0]-d)
    return out

def replace_nan_with_interpolation(image):
    """
    Replace NaN values in an image with interpolated values from non-NaN neighbors.
    """
    x, y = np.indices(image.shape)
    non_nan_elements = ~np.isnan(image)
    
    interpolated_values = interpolate.griddata(
        (x[non_nan_elements], y[non_nan_elements]),  # Points where the data is not NaN
        image[non_nan_elements],  # Non-NaN values
        (x[~non_nan_elements], y[~non_nan_elements]),  # Points where the data is NaN
        method='nearest'
    )
    
    result = image.copy()
    result[~non_nan_elements] = interpolated_values
    return result

def nan_helper(arr, sigma=3):
    """
    Apply 2D Gaussian smoothing to a 3D array (3x100x100) where each 100x100 matrix is an image.
    NaN values are handled by interpolation.
    
    :param arr: Input 3D array of shape (3, 100, 100).
    :param sigma: Standard deviation for Gaussian kernel.
    :return: Smoothed 3D array.
    """
    smoothed_array = np.zeros_like(arr)
    
    for i in range(arr.shape[0]):  # Iterate through the first dimension (wavelengths)
        image = arr[i, :, :]
        image_with_no_nan = replace_nan_with_interpolation(image)
        smoothed_image = gaussian_filter(image_with_no_nan, sigma=sigma)
        smoothed_array[i, :, :] = smoothed_image
    
    return smoothed_array

def sigma_clipped_mean(data, sigma=1, axis=0):
    # Calculate the Z-scores
    z_scores = zscore(data, axis=axis)
    
    # Mask the data entries that are beyond the sigma threshold
    data[np.abs(z_scores) > sigma] = np.median(data)
    
    # Compute the mean along the specified axis while ignoring masked values
    return data.mean(axis=axis)

def findMatchingObs(id):
    (out1, out2) = sortmerge(id, id)
    # Discard identities and reversed pairs
    i = np.where(out2 > out1)[0]
    assert i.size > 0
    out1 = out1[i]
    out2 = out2[i]
    assert max(np.abs(id[out1] - id[out2])) == 0
    return (out1, out2)

def measureChunkMagnitude1D(wave,flux, var, quality, center_w=15000,delta_w=300):
    '''This returns the magnitude of a wavelength chunk of a 1D spectra'''   
    u1 = np.where(wave>(center_w-delta_w))[0][0]
    u2 = np.where(wave<(center_w+delta_w))[0][-1]  
    f = flux[u1:u2]   
    v = var[u1:u2]
    q = quality[u1:u2]   

    good = np.where((f!=0)&(q>=0.8)&(v!=0)&(v<50000) & (np.isfinite(f)))
    if (np.sum(good)>0.5*len(f)):
        if (np.median(f[good])>0.0):
            #mag = -2.5*np.log10(np.sum(f[good])/len(good[0]))+25.0
            mag = -2.5*np.log10(np.median(f[good]))+25.0
            dmag = 1.08*(np.sqrt(np.nansum(v[good])))/np.nansum(f[good])
        else:
            mag=99.0
            dmag=99.0
    else:
        mag = 99.0
        dmag = 99.0
        
    return mag,dmag

def find_bounds(arr, value):
    min_x, max_x = float('inf'), -float('inf')
    min_y, max_y = float('inf'), -float('inf')

    for y in range(len(arr)):
        for x in range(len(arr[y])):
            if arr[y][x] == value:
                min_x = min(min_x, x)
                max_x = max(max_x, x)
                min_y = min(min_y, y)
                max_y = max(max_y, y)

    return (min_x-1, max_x+2, min_y-1, max_y+2)

def illuminate(imsize= 100,wsize = 3):
    '''Illuminating the detector with a asymmetric gaussian for dither simulation'''
    x, y = np.meshgrid(np.linspace(0,imsize-1,imsize), np.linspace(0,imsize-1,imsize))
    d = np.sqrt((x-imsize/2.1)**2+(y-imsize/1.9)**2)
    sigma, mu = 200, 0
    cube = np.zeros((wsize,imsize,imsize))
    for w in range(wsize):
        cube[w,:,:]=-2.5*np.log10(np.exp(-((1.2*d-mu)**2 / ( 2.0 * (sigma+10.0*w)**2 ))))
    return cube

def get_euclid_wcs(ra_boresight, dec_boresight, roll = None, oversample = 1,
                   pixel_size = 0.13104):
    """
    A routine to generate the World Coordinate System corresponding Euclid FOV

    Parameters
    ----------
    ra_boresight : float
        The boresight right ascencion in degrees.
        If ecliptic=True, this is actually the ecliptic longitude.
    dec_boresight : float
        The boresight declination in degrees.
        if ecliptic=True, this is actually ecliptic latitude
    roll : float, optional
        A roll angle around the telescope line of sight in degrees. The default is None.

    pixel_size : float, optional
        pixel size in arcseconds. The default is 0.13104, chosen such that a pixel value of
        10,000 is the edge of the FOV.

    Returns
    -------
    w : WCS
        An astropy WCS (world-coordinate system) object.

    """
  
    # set up the wcs
    w = wcs.WCS(naxis=2)
    w.wcs.crval = [ra_boresight, dec_boresight] # boresight pointing
    w.wcs.crpix = [0,0]        
    w.wcs.cdelt = [pixel_size / oversample / 3600.0 , pixel_size / oversample / 3600.0 ]
    w.wcs.ctype = ["RA---SIN","DEC--SIN"]
            
    # if roll is present, insert that
    if roll is not None:
        w.wcs.crota = [ 0.0, roll ]
    else:
        w.wcs.crota = [ 0.0, 0.0 ]
            
    return w

def normaldet(xy,maxx=2040,maxy=2040):
    """gets 
    - the detector coordinate in pixel
    returns:
    - the detector coordinates in normalized [-1,1] range """
    xslope = 2.0/(maxx)
    x = xslope*(xy[0]-(maxx/2.0))
        
    yslope = 2.0/(maxy)
    y = yslope*(xy[1]-(maxy/2.0))
    return np.array([x,y])

def normalfov(box,fov):
    """gets 
    - the detector model which has the corners of the FOV in mm 
    - the fov in mm
    returns:
    - the focal plane coordinates in normalized [-1,1] range """
    xslope = 2.0/(box[1][0] - box[0][0])
    xint = (1+(box[1][0] / box[0][0]))/(1.0-(box[1][0] / box[0][0]))
    x = xslope*fov[0]+xint
        
    yslope = 2.0/(box[1][1] - box[0][1])
    yint = (1+(box[1][1] / box[0][1]))/(1.0-(box[1][1] / box[0][1]))
    y = yslope*fov[1]+yint
    return np.array([x,y])

def normalfov_r(box,fov):
    """gets 
    - the detector model which has the corners of the FOV in normalized [-1,1]
    - the fov in mm
    returns:
    - the focal plane coordinates in mm """
    xslope = 2.0/(box[1][0] - box[0][0])
    xint = (1+(box[1][0] / box[0][0]))/(1.0-(box[1][0] / box[0][0]))
    x = (fov[0]-xint)/xslope
        
    yslope = 2.0/(box[1][1] - box[0][1])
    yint = (1+(box[1][1] / box[0][1]))/(1.0-(box[1][1] / box[0][1]))
    y = (fov[1]-yint)/yslope
    return np.array([x,y])

    i2a = 0
    count = 0
    for i1 in range(0, id1.size):
        for i2 in range(i2a, id2.size):
            j1 = isort1[i1]
            j2 = isort2[i2]
            if id1[j1] == id2[j2]:
                if count == out1.size: # resize arrays when needed
                    out1.resize(out1.size + size_incr)
                    out2.resize(out2.size + size_incr)
                out1[count] = j1  # register indices for matching pair
                out2[count] = j2
                count += 1
            else:
                if id1[j1] > id2[j2]:
                    i2a += 1
                else:
                    break
    out1 = out1[0:count]
    out2 = out2[0:count]
    return (out1, out2)

def det_neighbour(det1, det2):
    if det1 == det2:
        return False  # same detector
    dd = np.abs(int(det1[0]) - int(det2[0])) + \
        np.abs( int(det1[1]) - int(det2[1]))
    if dd != 1:
        return False  # non-neighbours
    return True

def FPCoordsGaps(det1, det2, n=1):
    assert det_neighbour(det1, det2)
    c1 = DetCoords(det1, 0., 0.).toFPCoords()
    c2 = DetCoords(det2, 0., 0.).toFPCoords()
    edges =  DetCoords(det1, np.array((-0.9, 0.9)), np.array((-0.9, 0.9))).toFPCoords()
    if np.abs(c1.x[0] - c2.x[0]) > np.abs(c1.y[0] - c2.y[0]):
        cc = FPCoords(np.repeat((c1.x[0] + c2.x[0]) / 2, n),
                      np.linspace(edges.y[0], edges.y[1], n))
    else:
        cc = FPCoords(np.linspace(edges.x[0], edges.x[1], n),
                      np.repeat((c1.y[0] + c2.y[0]) / 2, n))
    return cc

def cartesianProduct(x, y):
    (x, y) = np.meshgrid(x, y, indexing='ij')
    x = x.reshape(x.size)
    y = y.reshape(y.size)
    return (x, y)

def gaussian(x, mu, sig):
    return np.exp(-np.power(x - mu, 2.) / (2 * np.power(sig, 2.))) / np.sqrt(2 * np.pi) / sig
