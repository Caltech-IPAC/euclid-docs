def coordGapsX(n=1):
    """ get coordinates of gap along X """
    out = np.array(())
    c = 0.
    for i in range(0, 4):
        out = np.append(out, c + sizeDetX() * np.linspace(1, n, n) / (n+1))
        c += sizeDetX()
        if i <= 2:
            out = np.append(out, c + sizeGapsX()[i] / 2)
            c += sizeGapsX()[i]
    out = np.asarray(out) / sizeFPX() * 2 - 1
    return out

def coordGaps(n=1):
    """ get Gaps coordinates """
    x = coordGapsX(n)
    y = coordGapsY(n)
    outX = []
    outY = []
    for ix in range(0, len(x)):
        for iy in range(0, len(y)):
            # Discard points fallig on a detector
            (idet, _x, _y) = focalplane2det(x[ix], y[iy])
            if idet != 0:
                continue

            # Discard points fallig on the corner between 4 detectors
            (idet1, _x, _y) = focalplane2det(x[ix], -1)
            (idet2, _x, _y) = focalplane2det(-1, y[iy])
            if (idet1 == 0)  and  (idet2 == 0):
                continue

            outX.append(x[ix])
            outY.append(y[iy])
    return np.asarray(outX), np.asarray(outY)

def allcombinations(x, y):
    """ do alllpossible combination """
    # NOTE: this is correct: first y, then x.
    (y, x) = np.meshgrid(x, y)
    x = x.reshape(x.size)
    y = y.reshape(y.size)
    return x, y

def cartesianProduct(x, y):
    """ cartesian product """
    (x, y) = np.meshgrid(x, y)
    x = x.reshape(x.size)
    y = y.reshape(y.size)
    return x, y

def closestDetectors(x, y):
    """ get closest detectors """
    c = np.array((-0.75, -0.25, 0.25, 0.75))
    (cx, cy) = allcombinations(c.copy(), c.copy())
    idet = np.argsort(np.sqrt( (cx-x)**2  +  (cy-y)**2 ))[0:2] + 1
    (idetx1, idety1) = idetseq2xy(idet[0])
    (idetx2, idety2) = idetseq2xy(idet[1])
    assert (np.abs(idetx1-idetx2) + np.abs(idety1-idety2)) == 1
    if (idetx1 > idetx2)  or  (idety1 > idety2):
        return idet[[1,0]]
    return idet

def gaussian(x, mu, sig):
    """ gaussian shape """
    return np.exp(-np.power(x - mu, 2.) / (2 * np.power(sig, 2.))) / np.sqrt(2 * np.pi) / sig
