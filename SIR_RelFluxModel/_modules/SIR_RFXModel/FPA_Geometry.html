#  ___________________________________________________________________
#  NOTATION
#
#  Acronyms:
#    FP  = Focal Plane
#    FPA = Focal Plane Assembly
#
#  References:
#    https://wiki.cosmos.esa.int/euclid/images/9/92/NISP-Detector-orient.png
#
#  The geometry of the FPA is as follows:
#
#                                               R_{NI-DS} Y direction -->
#                -1                                        0                                        1
#               -1|---------------|-----|---------------|-----|---------------|-----|---------------|-1
#                 |   DET11 ( 1)  | gap |   DET12 ( 5)  | gap |   DET13 ( 9)  | gap |   DET14 (13)  |
#                 |      gap      | gap |      gap      | gap |      gap      | gap |      gap      |
#   R_{NI-DS}     |   DET21 ( 2)  | gap |   DET22 ( 6)  | gap |   DET23 (10)  | gap |   DET24 (14)  |
#  Z direction   0|      gap      | gap |      gap      | gap |      gap      | gap |      gap      |0
#       |         |   DET31 ( 3)  | gap |   DET32 ( 7)  | gap |   DET33 (11)  | gap |   DET34 (15)  |
#       |         |      gap      | gap |      gap      | gap |      gap      | gap |      gap      |
#      \|/        |   DET41 ( 4)  | gap |   DET42 ( 8)  | gap |   DET43 (12)  | gap |   DET44 (16)  |
#                1|---------------|-----|---------------|-----|---------------|-----|---------------|1
#                -1                                        0                                        1
#
#  The numbers around the frame are given in FP normalized
#  coordinates, spanning the range from -1 (on the left or upper edge)
#  to 1 (right or lower edge).  The FP normalized coordinate (0,0)
#  fall exactly in the center of the FPA.
#
#  Note that the vertical axis of the FPA is named "Z", while the horizontal one
#  is named "Y" in the official documentation.  For simplicity here we will assume the
#  following notation:
#   Python source code  |                 Meaning                |  Direction in FPA
#            x          |  horizontal, first index in a 2D array |       Y
#            y          |   vertical, second index in a 2D array |       Z
#
#  Also note that in the vertical direction the FP normalized
#  coordinates increase downwards.
#
#  Detectors have both a row-column ID (from 11 to 44), and a
#  sequential one (from 1 to 16).  Note that the XY identifier is the
#  reverse of the row-column one.
#
#  Pixel coordinates range from 0 (upper or left edge of first pixel)
#  to 2048 (lower or right edge of last pixel).  Pixel coordinates can
#  be non-integers: the middle point of each pixels are 0.5, 1.5, ...,
#  2047.5.
#
#  Pixel scheme for each detector:
#  |-------------|-------------|-------------|-------------|-------------|
#  | (   1,   1) | (   1,   2) | (   1, ...) | (   1,2047) | (   1,2048) |
#  | (   2,   1) | (   2,   2) | (   2, ...) | (   2,2047) | (   2,2048) |
#  |     ...     |     ...     |     ...     |     ...     |     ...   ) |
#  | (2047,   1) | (2047,   2) | (2047, ...) | (2047,2047) | (2047,2048) |
#  | (2048,   1) | (2048,   2) | (2048, ...) | (2048,2047) | (2048,2048) |
#  |-------------|-------------|-------------|-------------|-------------|
#
#  Note that the pixel with x=1, y=2048 is located in the upper right
#  corner (rather than lower left). Hence a transpose operation must
#  be used when converting from pixel to detector and viceversa.
#
#  Also note that detectors in the bottom half of the FPA (i.e. those
#  with FP normalized coordinates > 0 in the vertical directions) are
#  rotated by 180 degrees.
#

class GlobalSettings:
    """Class containing global settings.

    The properties are:

    - pairCoordXY (boolean, default False): whether the coordinate pairs
      of pixel should be interpreted as first and second index in the 2D
      array (as read with astropy.io.fits);

    - enableDetRotation (boolean, default False): set to True to enable
      rotation of detectors in the bottom part of the focal plane;

    - binning (int, default 3): pixel binning factor to display images.
      Must be a positivie integer number smaller than 9.  Increase this
      number to improve performances;

    - markpixels (boolean, default False): set to True to mark reference
      pixels in the image.  Note: this will erase all pixel information;

    - markplot (boolean, default False): set to True to mark reference
      pixels in the plot.
        """

        self.enableDetRotation = False
        self.binning = 3
        self.markpixels = False
        self.markplot = False

def sizeFPX():
    """Return the total size of the focal plane along the horizontal direction."""
    return 4 * sizeDetX() + np.sum(sizeGapsX())





[docs]
def sizeFPY():
    """Return the total size of the focal plane along the vertical direction."""
    return 4 * sizeDetY() + np.sum(sizeGapsY())

def focalplaneStructX():
    """Return an array with FP normalized coordinates of the gap edges along the horizontal direction."""
    c = np.array((0.,                          \
                  sizeDetX(), sizeGapsX()[0],  \
                  sizeDetX(), sizeGapsX()[1],  \
                  sizeDetX(), sizeGapsX()[2],  \
                  sizeDetX())) / sizeFPX()
    c = np.cumsum(c) * 2 - 1
    c[0] = -1.;  c[-1] = 1.
    return c

def idetseq2xy(idet):
    """Convert a sequential ID into an X, Y pair."""
    assert np.all(idet >= 1);  assert np.all(idet <= 16)
    idetx =       (idet-1) // 4  + 1
    idety = np.mod(idet-1,    4) + 1
    return (idetx, idety)





[docs]
def idetseq2string(idet):
    """Convert a sequential ID into a string (suitable to be used as 'DET_ID' keyword)."""
    (idetx, idety) = idetseq2xy(idet)
    return str(idety) + str(idetx)  # this is correct: first Y, then X.

def idetxy2seq(idetx, idety):
    """Convert an X, Y pair into a sequential ID."""
    assert np.all(idetx >= 1);  assert np.all(idetx <= 4)
    assert np.all(idety >= 1);  assert np.all(idety <= 4)
    return (idetx-1) * 4 + (idety)

def detectorRotated(idet):
    """Return True if the chip of a detector is rotated by 180 degrees in the FPA."""
    if settings.enableDetRotation:
        (idetx, idety) = idetseq2xy(idet)
        return (idety >= 3)
    return False

def det2pixel(idet, x, y):
    """Convert from detector normalized coordinates into pixels."""
    #assert np.all(np.abs(x) <= 1)
    #assert np.all(np.abs(y) <= 1)
    x = (x + 1) / 2 * 2048
    y = (y + 1) / 2 * 2048
    if np.isscalar(idet):
        if detectorRotated(idet):
            x = 2048 - x
            y = 2048 - y
    else:
        i = np.where(detectorRotated(idet))[0]
        if len(i) > 0:
            x[i] = 2048 - x[i]
            y[i] = 2048 - y[i]
    if settings.pairCoordXY:
        return (y, x)
    else:
        return (x, y)

def pixel2det(idet, _x, _y):
    """Convert from pixels into detector normalized coordinates."""
    x = copy.copy(_x)
    y = copy.copy(_y)
    #assert np.all(0. <= x);  assert np.all(x <= 2048)
    #assert np.all(0. <= y);  assert np.all(y <= 2048)
    if np.isscalar(idet):
        if detectorRotated(idet):
            x = 2048 - x
            y = 2048 - y
    else:
        i = np.where(detectorRotated(idet))[0]
        if len(i) > 0:
            x[i] = 2048 - x[i]
            y[i] = 2048 - y[i]
    x = 2 * x / 2048 - 1
    y = 2 * y / 2048 - 1
    if settings.pairCoordXY:
        return (y, x)
    else:
        return (x, y)

def det2focalplane(idet, x, y):
    """Convert from detector to focal plane normalized coordinates."""
    outx = np.full_like(x, np.NaN, dtype=float)
    outy = np.full_like(y, np.NaN, dtype=float)

    idetx, idety = idetseq2xy(idet)
    idetx -= 1
    idety -= 1
    x0 = CX[idetx*2]
    x1 = CX[idetx*2 + 1]
    y0 = CY[idety*2]
    y1 = CY[idety*2 + 1]

    outx = (x + 1) / 2 * (x1 - x0) + x0
    outy = (y + 1) / 2 * (y1 - y0) + y0
    return outx, outy

def focalplane2det(x, y):
    """Convert from focal plane to detector normalized coordinates."""
    scalar_x = np.isscalar(x)
    scalar_y = np.isscalar(y)
    if scalar_x:
        x = np.array((x,))
    if scalar_y:
        y = np.array((y,))
    assert np.all(np.abs(x) <= 1)
    assert np.all(np.abs(y) <= 1)
    assert x.size == y.size

    idet = np.zeros(len(x), dtype=int)
    outx = np.repeat(np.NaN, len(x))
    outy = np.repeat(np.NaN, len(y))

    for idetx in range(0, 4):
        x0 = CX[idetx*2  ]
        x1 = CX[idetx*2+1]
        for idety in range(0, 4):
            y0 = CY[idety*2  ]
            y1 = CY[idety*2+1]
            i = np.where((x >= x0)  &  (x <= x1)  &  \
                         (y >= y0)  &  (y <= y1))[0]
            if len(i) > 0:
                outx[i] = (x[i] - x0) * 2 / (x1 - x0) - 1
                outy[i] = (y[i] - y0) * 2 / (y1 - y0) - 1
                idet[i] = idetxy2seq(idetx+1, idety+1)

    outx = np.round(outx * 1e5) / 1e5
    outy = np.round(outy * 1e5) / 1e5
    if scalar_x:
        outx = outx[0]
    if scalar_y:
        outy = outy[0]
    if scalar_x  and  scalar_y:
        idet = idet[0]
    return (idet, outx, outy)

def focalplane2pixel(x, y):
    """Convert from focal plane normalized coordinates into pixels."""
    (idet, x, y) = focalplane2det(x, y)
    (x, y) = det2pixel(idet, x, y)
    return (idet, x, y)

def pixel2focalplane(idet, x, y):
    """Convert from pixel into focal plane normalized coordinates."""
    (x, y) = pixel2det(idet, x, y)
    (x, y) = det2focalplane(idet, x, y)
    return (x, y)

def boundaries(x):
    """Return boundaries in an array of coordinates.

    Given an input 1D array of sorted coordinates, returns the array
    of boundaries (or edges).  Note that output arrays has 1 extra
    element with respect to input one.
    """
    out = (x[0:-1] + x[1:]) / 2
    d = out[1]-out[0]
    out = np.append(out, out[-1]+d)
    out = np.insert(out, 0, out[0]-d)
    return out
