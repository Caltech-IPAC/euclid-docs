[docs]class AstroObjectInfo(object):
    """
    Stores astrophysical data about a specific galaxy or star.

    This class expands upon the data available in the AstronomicalObject class
    by adding the Y, J and H band fluxes (rather than the magnitudes), a model for the
    1-dimensional spectrum of the object, and a lookup data structure for quickly
    finding which dithers the object appears in, which detectors within each detector,
    which spectral orders appear on each dither. The lookup structure also points to
    the SpectrumInfo objects associated with this particular star or galaxy.
    """

    STAR = 0
    """Indicates that the object is a star"""

    GALAXY = 1
    """Indicates that the object is a galaxy"""

    ZEROTH_ORDER_THRESHOLD = 20
    """Zeroth orders fainter than this threshold, in all bands, will be ignored."""

    SECOND_ORDER_THRESHOLD = 16
    """Second orders fainter than this threshold, in all bands, will be ignored."""

    WAVELENGTHS = None
    """The wavelengths used by all 1D model spectra"""

    EUC_TO_2MASS = {'J': 'J2M', 'H': 'H2M'}
    """The mapping from Euclid band name to 2MASS band name."""

    __slots__ = ['_invalid_mag',
                 '_usable',
                 '_id',
                 '_mag',
                 '_flux',
                 '_dflux',
                 '_cutout',
                 '_spectra_info',
                 '_1d_continuum',
                 '_lines',
                 '_interp_mask',
                 '_type',
                 '_measured',
                 '_has_2mass_mags',
                 '__weakref__']

    def __init__(self, location, cutouts, phot_info, limit_band='J', limit_mag=np.inf):
        """
        Constructs the AstroObjectInfo instance.
        Parameters
        ----------

        location: LocationObject
            The object describing the astronomical object, the position of its spectrum, and the dispersion solution.

        cutouts: CutoutCollection or None
            Contains information for all cutouts.

        phot_info: PhotometerInfo
            Contains information about the photometric bands

        limit_band: str
            The string specifying the name of the filter / band that is being used to limit the objects, based on
            brightness.

        limit_mag: float
            The threshold magnitude, above which the object will be considered too faint to be usable.
        """

        metadata = location.getAstronomicalObject()

        self._invalid_mag = False
        self._usable = False
        self._id = None
        self._mag = {'Y': None, 'J': None, 'H': None}
        self._flux = {'Y': None, 'J': None, 'H': None}
        self._dflux = {'Y': None, 'J': None, 'H': None}
        self._cutout = None

        # {dither: {detector : {order: SpectrumInfo}}}
        self._spectra_info = NestedDict(SpectrumInfo, 'dither', 'detector', 'order')

        self._1d_continuum = None
        self._lines = None
        self._measured = False
        self._interp_mask = True
        self._type = None

        has_magnitudes = metadata.hasMagnitude('Y') and metadata.hasMagnitude('H') and metadata.hasMagnitude('J')
        has_2mass_mags = metadata.hasMagnitude('H2M') and metadata.hasMagnitude('J2M')

        self._has_2mass_mags = has_2mass_mags

        if not has_magnitudes and not has_2mass_mags:
            # the magnitude data is unavailable or incomplete: we need to know the Y, J, and H mags or 2MASS mags.
            self._invalid_mag = True
            # it is usable in the sense that its location will be masked out.
            self._usable = True
            self._id = int(metadata.getId())
            return
        else:
            if has_2mass_mags:
                mag_in_limit_band = metadata.getMagnitude(self.EUC_TO_2MASS[limit_band]).getValue()
            else:
                if metadata.hasMagnitude(limit_band):
                    mag_in_limit_band = metadata.getMagnitude(limit_band).getValue()
                else:
                    mag_in_limit_band = np.nan

            # handle the possibility that mag_in_limit_band is None:

            bright_enough = mag_in_limit_band < limit_mag if mag_in_limit_band is not None else False

            if not bright_enough:
                # the object is not usable because the magnitude is missing or it is too faint.
                if mag_in_limit_band is None or np.isnan(mag_in_limit_band):
                    self._invalid_mag = True
                return
            else:
                # Add metadata specific to the astronomical object:
                self._load_metadata(metadata, phot_info)

                # the object is usable.
                self._usable = True

                # fetch the specific cutout for this object, if it exists in the list of cutouts
                if self._cutout is None and cutouts is not None:
                    self._cutout = cutouts[self._id] if self._id in cutouts else None
                    if self._cutout is not None:
                        self._cutout.pixels[~np.isfinite(self._cutout.pixels)] = 0.0


[docs]    def load_spectra(self, location_object, grism_model, padding):
        """
        Loads all spectra associated with the object

        Parameters
        ----------

        location_object: LocationObject
            Contains all known information about all spectra in a particular detector

        grism_model: DetectorGrism
            The DetectorGrism object associated with the detector and dither that corresponds to the LocationTable
            object.

        padding: int
            The number of pixels that should be inserted between the bounding box of each spectral cutout
            (specified in a `SpectrumInfo` object) and the rectangle defined by the `LocationSpectrum` object
            in the LocationTable.

        Side Effects
        ------------

        Modifies `self._spectra_info`
        """

        # location_object.getAllSpectra() does not return an iterable container, so we resort to finding its length
        # and indexing into the object that is returned:

        n_spectra = location_object.getNumberOfSpectra()

        if n_spectra == 0:
            return

        if n_spectra > 1:
            specs = location_object.getAllSpectra()
            for s in range(n_spectra):
                spec = specs[s]
                self._add_spec_info(spec, grism_model, padding)
        elif n_spectra == 1:
            spec = location_object.getAllSpectra()[0]
            self._add_spec_info(spec, grism_model, padding)

[docs]    def get_1d_spectrum(self) -> Union[np.ndarray, None]:
        """
        Computes and returns the full 1D spectral model of the object (continuum + lines).

        Returns
        -------

        one_dimensional_spectrum: np.ndarray or None
            The 1D model spectrum of the object, if the spectrum has already been measured. Otherwise,
            None is returned.
        """
        if self.continuum is None:
            return None

        if self.lines is None:
            return self.continuum
        else:
            line_models = self.lines.line_models
            lines = np.zeros_like(self.continuum)
            if line_models is not None:
                for line_model in line_models:
                    lines += line_model.eval(self.WAVELENGTHS)

            return self.continuum + lines

    def _add_spec_info(self, spectrum_location, grism_model, padding):
        order = spectrum_location.getOrder()
        detector = spectrum_location.getDetectorNumber()

        if not self._invalid_mag:
            min_mag = min(self.mag('H'), self.mag('J'), self.mag('Y'))
            if order == 0 and min_mag > self.__class__.ZEROTH_ORDER_THRESHOLD:
                # This zeroth order is too faint to be included
                return

            if order == 2 and min_mag > self.__class__.SECOND_ORDER_THRESHOLD:
                # This second order is too faint to be included
                return

        spec_info = SpectrumInfo(self, spectrum_location, grism_model, padding)

        if order == 0:
            zeroth_mag = 0.0 if self._invalid_mag else self.mag('J')
            spec_info.zeroth_order_size = grism_model.get_zeroth_order_boxsize(zeroth_mag)

        left, bottom, width, height = spec_info.bbox

        is_on_detector = left < grism_model.xres and bottom < grism_model.yres

        dimensions_are_positive = width > 0 and height > 0

        if is_on_detector and dimensions_are_positive:
            self._spectra_info[grism_model.dither, detector, order] = spec_info

    def _load_metadata(self, metadata, phot_info):

        self._id = int(metadata.getId())

        if not self.has_2mass_mags:
            self._mag['H'] = hmag = metadata.getMagnitude('H').getValue()
            self._mag['J'] = jmag = metadata.getMagnitude('J').getValue()
            self._mag['Y'] = ymag = metadata.getMagnitude('Y').getValue()
            hmag_error = metadata.getMagnitude('H').getError()
            jmag_error = metadata.getMagnitude('J').getError()
            ymag_error = metadata.getMagnitude('Y').getError()
        else:
            self._mag['H'] = hmag = metadata.getMagnitude(self.EUC_TO_2MASS['H']).getValue()
            self._mag['J'] = jmag = metadata.getMagnitude(self.EUC_TO_2MASS['J']).getValue()
            # Y mag will not be used during model-making because self.has_2mass_mags is True; just set to 1000.
            self._mag['Y'] = ymag = 1000.0
            ymag_error = 1.0
            hmag_error = metadata.getMagnitude(self.EUC_TO_2MASS['H']).getError()
            jmag_error = metadata.getMagnitude(self.EUC_TO_2MASS['J']).getError()

        if None in (hmag, jmag, ymag) or np.isnan(hmag) or np.isnan(jmag) or np.isnan(ymag):
            self._invalid_mag = True
            return

        if math.fabs(hmag_error) > 1.0:
            hmag_error = 1.0

        if math.fabs(jmag_error) > 1.0:
            jmag_error = 1.0

        if math.fabs(ymag_error) > 1.0:
            ymag_error = 1.0

        # compute fluxes in microJy

        h_microjansky = Utils.abmag_to_ujy(hmag)
        j_microjansky = Utils.abmag_to_ujy(jmag)
        y_microjansky = Utils.abmag_to_ujy(ymag)

        h_plus_microjansky = Utils.abmag_to_ujy(hmag + hmag_error)
        j_plus_microjansky = Utils.abmag_to_ujy(jmag + jmag_error)
        y_plus_microjansky = Utils.abmag_to_ujy(ymag + ymag_error)

        # convert from microJy to erg/s/cm^2/Angstrom

        self._flux['H'] = Utils.ujy_to_angstrom(h_microjansky, phot_info.effective_wavelength('H'))
        self._flux['J'] = Utils.ujy_to_angstrom(j_microjansky, phot_info.effective_wavelength('J'))
        self._flux['Y'] = Utils.ujy_to_angstrom(y_microjansky, phot_info.effective_wavelength('Y'))

        self._dflux['H'] = math.fabs(self._flux['H'] - Utils.ujy_to_angstrom(h_plus_microjansky,
                                                                             phot_info.effective_wavelength('H')))
        self._dflux['J'] = math.fabs(self._flux['J'] - Utils.ujy_to_angstrom(j_plus_microjansky,
                                                                             phot_info.effective_wavelength('J')))
        self._dflux['Y'] = math.fabs(self._flux['Y'] - Utils.ujy_to_angstrom(y_plus_microjansky,
                                                                             phot_info.effective_wavelength('Y')))

        self._type = self.STAR if metadata.Star == metadata.getType() else self.GALAXY
