from typing import Union

from SIR_SpectraLocation.H5LocationTable import H5LocationTable
import SIR_SpectraDecontamination.Utils as Utils



[docs]class SpectrumContainer:
    __slots__ = ('_id',
                 '_dither',
                 '_detector',
                 '_h5object',
                 '_x_offset',
                 '_y_offset',
                 '__weakref__')

    def __init__(self, h5object: Union['h5py.Dataset', 'h5py.Group']=None):
        self._id = None
        self._dither = None
        self._detector = None
        self._h5object = h5object  # contains an h5py.Group or h5py.Dataset
        self._x_offset = None
        self._y_offset = None

    def __del__(self):
        self._h5object = None

    @property
    def id(self) -> int:
        """
        The ID of the object, as a string.
        """
        if self._id is None and self._h5object is not None:
            self._id = int(self._h5object.attrs['object_id'])
        return self._id

    @id.setter
    def id(self, object_id):
        if Utils.represents_int(object_id):
            self._id = int(object_id)
        else:
            raise TypeError('Expected an integer or a string that is convertible to an integer.')

    @property
    def dither(self) -> int:
        """
        The dither sequence number as an integer.
        """
        if self._dither is None and self._h5object is not None:
            self._dither = int(self._h5object.file.attrs[H5LocationTable.DITHER_LABEL])
        return self._dither

    @dither.setter
    def dither(self, dither: int):
        if Utils.represents_int(dither):
            self._dither = int(dither)
        else:
            raise TypeError('Expected an integer or a string that is convertible to an integer.')

    @property
    def detector(self) -> int:
        """
        The detector number, as an integer (1 - 16)
        """
        if self._detector is None and self._h5object is not None:
            self._detector = int(self._h5object.file.attrs['Detector'])
        return self._detector

    @detector.setter
    def detector(self, detector: int):
        if Utils.represents_int(detector):
            self._detector = int(detector)
        else:
            raise TypeError('Expected an integer or a string that is convertible to an integer.')

    @property
    def x_offset(self) -> int:
        """
        The x-coordinate of the lower-left pixel of the spectral cutout [int].
        """
        if self._x_offset is None and self._h5object is not None:
            self._x_offset = int(self._h5object.attrs['x_offset'])
        return self._x_offset

    @x_offset.setter
    def x_offset(self, xoff):
        if Utils.represents_int(xoff):
            self._x_offset = int(xoff)
        else:
            raise TypeError('Expected an integer or a string that is convertible to an integer.')

    @property
    def y_offset(self) -> int:
        """
        The y-coordinate of the lower-left pixel of the spectral cutout [int].
        """
        if self._y_offset is None and self._h5object is not None:
            self._y_offset = int(self._h5object.attrs['y_offset'])
        return self._y_offset

    @y_offset.setter
    def y_offset(self, yoff):
        if Utils.represents_int(yoff):
            self._y_offset = int(yoff)
        else:
            raise TypeError('Expected an integer or a string that is convertible to an integer.')

    @property
    def offset(self) -> tuple:
        """
        The offset as a tuple: (x_offset, y_offset)
        """
        return self.x_offset, self.y_offset

    @offset.setter
    def offset(self, offset: tuple):
        if not isinstance(offset, tuple) or len(offset) != 2:
            raise TypeError(f'The offset is expected to be a 2-tuple. Encountered: {offset}')

        xoff, yoff = offset

        if Utils.represents_int(xoff) and Utils.represents_int(yoff):
            self.x_offset, self.y_offset = offset
        else:
            raise TypeError(f'The offset must be a tuple of integers. Encountered: {offset}')

    def __str__(self):
        if self._id is None and self._h5object is None:
            string = "Uninitialized Spectrum"
        else:
            string = (f"Object {self.id}"
                      f"\ndither: {self.dither}"
                      f"\ndetector: {self.detector}"
                      f"\noffset: {self.offset}")
        return string
