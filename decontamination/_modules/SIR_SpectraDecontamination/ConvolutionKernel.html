    """Creates the 2-dimensional kernel patches which are used when creating a 2D spectrum"""

    SCALE_RATIO = 3
    """
    The number of cutout pixels per spectroscopic pixel: spectroscopic_pixel_scale / cutout_pixel_scale
    """

    OBJECT_SIGMA_IN_PIXELS = 0.46
    """
    The typical object PSF size (sigma) in units of cutout pixels. We use this hard-coded value to specify the 
    approximate size of a Gaussian kernel PSF corresponding to a point-source.
    """

    MAX_PIXEL_VALUE = Utils.max_peak(SCALE_RATIO * OBJECT_SIGMA_IN_PIXELS)
    """The maximum pixel value of a 2-D Gaussian kernel with sigma = OBJECT_SIGMA_IN_PIXELS"""

    def __init__(self, spectrum_info, exposure_direction):
        """
        Creates a 2-dimensional kernel from a Cutout, if a suitable one exists. Otherwise, a Gaussian kernel is created.

        Parameters
        ----------

        spectrum_info: SpectrumInfo
            The SpectrumInfo object describing the spectrum that is being modelled.

        exposure_direction: float
            The direction angle, in degrees, of the +RA axis in the exposure containing the 2-D spectra.
            The angle is expected to be measured counterclockwise from the +x axis of the image. If the image is
            a NISP Exposure, one can use the `compute_nisp_exposure_direction()` method to compute this angle.

        Raises
        ------

        ValueError: if the kernel_size or exposure_direction are out of range.
        """
        if exposure_direction < 0 or exposure_direction > 360:
            raise ValueError(f'exposure_direction mush be within [0, 360] degrees; encountered {exposure_direction}')

        self._spectrum_info = spectrum_info
        self._exposure_direction = exposure_direction
        self._wavelength = (spectrum_info.grism_model.MIN_WAV_INNER + spectrum_info.grism_model.MAX_WAV_INNER) / 2
        self._prev_wavelength = self._wavelength
        self._psf_model = spectrum_info.grism_model.PSF_MODEL
        if self._spectrum_info.astro_object_info.stamp is not None:
            self._use_gaussian_fit = self._spectrum_info.astro_object_info.stamp.use_gaussian_fit
        else:
            self._use_gaussian_fit = False
        self._sigma_major = None
        self._sigma_minor = None
        self._gaussian_theta = None
        self._normed_stamp = self._get_normalized_stamp(spectrum_info)
        self._using_cutout = self._normed_stamp is not None
        self._kernel_size = None
        self._const_kernel = self._make_kernel()
        self._psf_convolved_kernel = None
        self._kernel = None
        self._prev_kernel = None
        self._region = None
        self._prev_region = None
        self._center = Utils.gaussian_centroid(self._const_kernel)
        self._position = None
        self._prev_position = None
        self._in_spectral_box = False
        self._history = []

    def __del__(self):
        self._spectrum_info = None
        self._psf_model = None
        self._normed_stamp = None
        self._const_kernel = None
        self._psf_convolved_kernel = None
        self._kernel = None

        self._region = None

    def _update_position(self):
        """Makes the kernel and region consistent with the current position and wavelength"""
        if self._position is None:
            raise UserWarning("Set the position before requesting a kernel or region.")

        half_kernel_size = 0.5 * self._kernel_size / self.SCALE_RATIO

        x_offset, y_offset, width, height = self._spectrum_info.bbox

        # the new position of the kernel's centroid, in the coordinate system of the detector:

        detector_x_lambda, detector_y_lambda = self._position

        # the new position in the coordinate system of the spectrum:

        spectrum_x_lambda = detector_x_lambda - x_offset
        spectrum_y_lambda = detector_y_lambda - y_offset

        # the coordinates of the lower-left corner of the kernel, in the coordinate system of the spectrum:

        x_0 = round(spectrum_x_lambda - half_kernel_size)
        y_0 = round(spectrum_y_lambda - half_kernel_size)

        # the coordinates of the new centroid, in the coordinate system of the kernel (at detector scale):

        kernel_x_lambda = spectrum_x_lambda - x_0
        kernel_y_lambda = spectrum_y_lambda - y_0

        # the vector by which the kernel needs to be shifted in order to move the centroid to the specified position.
        # This vector is in pixel units of the cutout

        shift_x = self.SCALE_RATIO * kernel_x_lambda - self._center[0]
        shift_y = self.SCALE_RATIO * kernel_y_lambda - self._center[1]

        kernel = self._get_wavelength_dependent_kernel()

        shifted_kernel = Utils.shift(kernel, (shift_y, shift_x))

        # re-scale the pixels of the shifted_kernel to be consistent with the pixels of the spectrum:

        shifted_kernel = Utils.normalize_image_fast(Utils.rebin3(shifted_kernel))

        # crop out the region of the cutout that falls within the spectral box

        left_s = max(0, x_0)
        bottom_s = max(0, y_0)
        right_s = min(width, x_0 + self.kernel_size)
        top_s = min(height, y_0 + self.kernel_size)

        left_k = left_s - x_0
        right_k = right_s - x_0
        bottom_k = bottom_s - y_0
        top_k = top_s - y_0
        new_kernel = np.copy(shifted_kernel[bottom_k:top_k, left_k:right_k])

        if new_kernel.size != 0 and new_kernel.shape == (top_s - bottom_s, right_s - left_s):
            self._in_spectral_box = True
            self._prev_kernel = self._kernel
            self._kernel = new_kernel
            self._prev_region = self._region
            self._region = (slice(bottom_s, top_s), slice(left_s, right_s))
            self._prev_position = self._position
        else:
            self._in_spectral_box = False

    def _make_kernel(self):
        """Make the kernel patch."""
        if self._normed_stamp is None:
            # Use a generic Gaussian kernel
            kernel = self._make_gaussian_kernel()
        else:
            # use the Cutout to make the kernel
            kernel = self._make_kernel_from_cutout()

        normed_kernel = Utils.normalize_image(kernel)

        if normed_kernel is None:
            # the kernel couldn't be normalized; use the default
            kernel = self._make_gaussian_kernel()
            normed_kernel = Utils.normalize_image(kernel)

        return normed_kernel

    def _make_gaussian_kernel(self):
        """Make a Gaussian kernel in a square stamp."""
        kernel_size = self.SCALE_RATIO * int(self._spectrum_info.location_spectrum.getApertureSize() + 2)

        self._kernel_size = kernel_size

        center = 0.5 * (kernel_size - 1)

        kernel = np.zeros((kernel_size, kernel_size), dtype=np.float32)

        for amp, sigma in self._psf_params(self._wavelength):
            tmp_kernel = np.zeros((kernel_size, kernel_size), dtype=np.float32)

            native_sigma = self.SCALE_RATIO * sigma

            tmp_kernel = Utils.gauss_kernel(tmp_kernel,
                                            (center, center),
                                            (native_sigma, native_sigma))

            kernel += amp * tmp_kernel

        return self._shift_kernel_to_center(kernel)

    def _make_kernel_from_cutout(self):
        """
        Make the kernel patch using a Cutout image.

        Makes the kernel by rotating the cutout so that it is aligned with the exposure.

        Returns
        -------

        2-D NumPy array containing the rotated stamp
        """
        cutout = self._spectrum_info.astro_object_info.stamp

        cutout_direction = Utils.atan(cutout.wcs.CD2_1, cutout.wcs.CD2_2)

        rotation_angle = self._exposure_direction - cutout_direction

        rotated_kernel = ndimage.rotate(self._normed_stamp, angle=rotation_angle, order=1)

        self._kernel_size = max(rotated_kernel.shape)

        if self._use_gaussian_fit:
            smoothed_kernel = self._make_gaussian_from_image(rotated_kernel)

            if smoothed_kernel is None:
                smoothed_kernel = self._make_gaussian_kernel()

            output = self._shift_kernel_to_center(smoothed_kernel)
        else:
            output = self._shift_kernel_to_center(rotated_kernel)

        return output

    def _get_wavelength_dependent_kernel(self):
        """Returns the wavelength-dependent PSF-convolved kernel for the current wavelength."""
        if self._prev_wavelength != self._wavelength or self._psf_convolved_kernel is None:
            self._prev_wavelength = self._wavelength
            if self._using_cutout:
                self._psf_convolved_kernel = self._correct_stamp(self._const_kernel, self._wavelength)
            else:
                self._psf_convolved_kernel = self._make_gaussian_kernel()

        return self._psf_convolved_kernel

    def _shift_kernel_to_center(self, kernel_image):
        """
        Repositions the centroid of the kernel to the actual center of the image.

        Parameters
        ----------

        kernel_image: NumPy.ndarray

        Returns
        -------

        shifted_kernel: 2-D NumPy array

        A square array of pixels with sides of length self._kernel_size. The array contains a kernel centered at
        the nominal center of the image.
        """
        kernel = np.zeros((self._kernel_size, self._kernel_size), dtype=np.float32)

        actual_center = Utils.gaussian_centroid(kernel_image)

        nominal_x, nominal_y = (0.5 * (self._kernel_size - 1), 0.5 * (self._kernel_size - 1))
        actual_x, actual_y = actual_center
        shift = (nominal_y - actual_y, nominal_x - actual_x)

        height, width = kernel_image.shape

        if height >= self._kernel_size and width >= self._kernel_size:
            # shift to the lower-left corner of kernel_image and copy region into kernel
            shifted_image = Utils.shift(kernel_image, shift)
            kernel[:, :] = shifted_image[:self._kernel_size, :self._kernel_size]
        else:
            # At least one dimension of the kernel_image is smaller than the kernel.
            # Copy the rotated image into kernel then shift
            rotated_height, rotated_width = kernel_image.shape
            kernel[:rotated_height, :rotated_width] = kernel_image[:, :]
            kernel = Utils.shift(kernel, shift)

        return kernel

    def _psf_params(self, wavelength: float) -> Tuple[Tuple[float, float], Tuple[float, float]]:
        """
        Determines the parameters of the model PSF, given as a mixture of 2D Gaussians

        Parameters
        ----------

        wavelength: float
            The wavelength, in units of Angstroms.

        Returns
        -------

        psf_parameters: Tuple[Tuple[float, float], Tuple[float, float]]

        A tuple of ((psf1_amplitude, sigma1), (psf2_amplitude, sigma2)).
        """
        return self._psf_model.get_parameters(self._spectrum_info.grism_model.grism_name,
                                              self._spectrum_info.detector_id,
                                              wavelength)

    def _correct_stamp(self, stamp: np.ndarray, wavelength: float) -> np.ndarray:
        """
        Modifies the direct image stamp by convolving it with a 2D Gaussian PSF kernel.

        Parameters
        ----------

        stamp: np.ndarray
            The raw stamp image.

        wavelength: float
            The wavelength to which you wish to adjust stamp.

        Returns
        -------

        corrected_stamp: np.ndarray
            The original image after being convolved with a correction Gaussian, which
            brings the stamp into agreement with the PSF at the specified wavelength.
        """
        correction_sigma = self.SCALE_RATIO * float(self._spectrum_info.grism_model.psf_correction_size(wavelength))
        canvas_size = 5 * self.SCALE_RATIO
        center = canvas_size // 2
        canvas = np.zeros((canvas_size, canvas_size), dtype=np.float32)
        psf_correction = Utils.gauss_kernel(canvas, (center, center), (correction_sigma, correction_sigma))

        return Utils.convolve2d(stamp, psf_correction)

    def _make_gaussian_from_image(self, image: np.ndarray) -> Union[np.ndarray, None]:
        """
        Makes a 2D Gaussian kernel from an input image.

        Parameters
        ----------

        image: np.ndarray
            The 2D Gaussian approximation to the content of the center of the input image.
        """
        height, width = image.shape
        left = width // 4
        right = left + width // 2
        bottom = height // 4
        top = bottom + height // 2
        center_region = image[bottom: top, left: right]

        if center_region.sum() < 1e-2:
            # the central region of the cutout is empty
            return None

        best_fit = self._fit_gaussian2d(center_region, 1, 1)

        if best_fit is None:
            return None

        amp = best_fit.amplitude.value
        x_mean = best_fit.x_mean.value + left
        y_mean = best_fit.y_mean.value + bottom
        x_std = best_fit.x_stddev.value
        y_std = best_fit.y_stddev.value
        theta = best_fit.theta.value
        y, x = np.mgrid[:height, :width]
        self._sigma_major = max(x_std, y_std)
        self._sigma_minor = min(x_std, y_std)
        theta_deg = np.rad2deg(theta)
        if x_std > y_std:
            self._gaussian_theta = 90 + theta_deg
        else:
            if theta_deg < 0:
                self._gaussian_theta = 180 + theta_deg
            else:
                self._gaussian_theta = theta_deg

        return best_fit.evaluate(x, y, amp, x_mean, y_mean, x_std, y_std, theta)

    @classmethod
    def _get_normalized_stamp(cls, spectrum_info: 'SpectrumInfo') -> Union[np.ndarray, None]:
        """
        Processes the stamp by normalizing it and checking to see if it is usable.

        Parameters
        ----------

        spectrum_info: SpectrumInfo
            The SpectrumInfo object for the spectrum whose associated normalized direct image stamp is being requested.

        Returns
        -------

        normalized_stamp: NumPy.ndarray or None

        A clipped, normalized version of the stamp stored in the SpectrumInfo object or None if the stamp is None or if
        the pixels of the stamp do not appear to contain a usable image. The image is expected to contain a distribution
        of pixels that are more spread-out than a simple point-source Gaussian with sigma=OBJECT_SIGMA_IN_PIXELS (if the
        distribution is more peaked than a point-source, something is wrong with the cutout image).
        """
        stamp = spectrum_info.astro_object_info.stamp
        y_mag = spectrum_info.astro_object_info.mag('Y')
        j_mag = spectrum_info.astro_object_info.mag('J')
        h_mag = spectrum_info.astro_object_info.mag('H')

        if stamp is None or min(y_mag, j_mag, h_mag) < 16.0:
            return None

        finite_stamp = np.copy(stamp.pixels)

        finite_stamp[~np.isfinite(finite_stamp)] = 0.0

        normed_stamp = Utils.normalize_image(finite_stamp)

        if normed_stamp is None or cls._is_too_compact(normed_stamp) or cls._is_on_edge(normed_stamp):
            # the normalized kernel stamp either does not exist or it has issues
            return None
        else:
            return normed_stamp

    def compute_nisp_exposure_direction(nisp_exposure) -> float:
        """
        Computes the direction of the +RA axis in a NISP exposure, in degrees, counterclockwise from +x.

        Parameters
        ----------

        nisp_exposure: NispSurveyExposure
            The NISP exposure with which the spectrum is associated.

        Returns
        -------

        angle: float

        The direction angle of the +DEC axis with respect to the +y axis. The angle is always
        positive.
        """
        metadata = nisp_exposure.getDetector(1).getMetadata()

        dy = metadata.get('CD2_1')

        dx = metadata.get('CD2_2')

        return Utils.atan(dy, dx)


    @classmethod
    def _is_on_edge(cls, normed_stamp: np.ndarray) -> bool:
        """
        Checks whether the kernel appears to be on the edge of the square stamp.

        Parameters
        ----------

        normed_stamp: np.ndarray
            A normalized 2D array of floats.

        Returns
        -------

        True if the kernel appears to be on the edge of the stamp. False otherwise.
        """
        peak_position = np.unravel_index(normed_stamp.argmax(), normed_stamp.shape)

        _, sigma_x = Utils.gaussian_moments(normed_stamp.sum(axis=0))

        _, sigma_y = Utils.gaussian_moments(normed_stamp.sum(axis=1))

        x_threshold = max(1.5, 1.5 * sigma_x)

        y_threshold = max(1.5, 1.5 * sigma_y)

        in_x = x_threshold < peak_position[1] < normed_stamp.shape[1] - x_threshold

        in_y = y_threshold < peak_position[0] < normed_stamp.shape[0] - y_threshold

        peak_near_edge = not in_x or not in_y

        if peak_near_edge:
            return True

        left_sum = normed_stamp[:, 0].sum()
        edge_max = normed_stamp[:, 0].max()

        right_sum = normed_stamp[:, -1].sum()
        edge_max = max(edge_max, normed_stamp[:, -1].max())

        top_sum = normed_stamp[0, 1:-1].sum()
        edge_max = max(edge_max, normed_stamp[0, 1:-1].max())

        bottom_sum = normed_stamp[-1, 1:-1].sum()
        edge_max = max(edge_max, normed_stamp[-1, 1:-1].max())

        edge_max_too_large = edge_max > 0.25 * normed_stamp.max()

        edge_sum = left_sum + right_sum + top_sum + bottom_sum

        edge_sum_too_large = edge_sum > 0.07

        return edge_sum_too_large or edge_max_too_large

    @classmethod
    def _is_too_compact(cls, normed_stamp: np.ndarray) -> bool:
        """
        Checks whether the kernel is too peaked (too compact) relative to a Gaussian comparable with the PSF size.

        Parameters
        ----------

        normed_stamp: np.ndarray
            A normalized 2D array of floats.

        Returns
        -------

        True if the kernel appears to be more compact than the most compact point source. False otherwise.
        """
        # in the presence of noise, the kernel could potentially be more peaked than the PSF. This factor allows for
        # this possibility

        noise_allowance = 1.05

        return normed_stamp.max() > noise_allowance * cls.MAX_PIXEL_VALUE

    @staticmethod
    def _fit_gaussian2d(image: np.ndarray, sigma_major: float, sigma_minor: float) -> Union[models.Gaussian2D, None]:
        """
        Fits a 2D Gaussian profile to the pixels in the center of the provided image

        Parameters
        ----------

        image: np.ndarray
            A 2D NumPy array of floats, representing pixels in an image of an astronomical object.

        sigma_major: float
            An initial guess for the size of the major axis of the elliptical 2D Gaussian.

        sigma_minor: float
            An initial guess for the size of the minor axis of the elliptical 2D Gaussian.

        Returns
        -------

        parameters: models.Gaussian2D or None
            The Gaussian2D model that best fits the image. If the fit was not successful, returns None.
        """
        fit_g = fitting.LevMarLSQFitter()
        y_dim, x_dim = image.shape
        x_center, y_center = Utils.gaussian_centroid(image)
        g_init = models.Gaussian2D(amplitude=image.max(),
                                   x_mean=x_center,
                                   y_mean=y_center,
                                   x_stddev=sigma_minor,
                                   y_stddev=sigma_major)
        g_init.x_mean.fixed = True
        g_init.y_mean.fixed = True
        y, x = np.mgrid[:y_dim, :x_dim]
        with warnings.catch_warnings():
            warnings.filterwarnings('ignore', category=AstropyUserWarning)
            try:
                parameters = fit_g(g_init, x, y, image, filter_non_finite=True)
            except fitting.NonFiniteValueError:
                # Even when the input data is finite, the fitting routine can apparently encounter a div/0 error
                # internally, which apparently cannot be prevented by the user. Essentially, the fit fails. So we treat
                # this the same way as other failed fits.
                parameters = None

        if parameters is None:
            return None

        max_size = max(x_dim, y_dim)
        failed_fit = fit_g.fit_info['nfev'] > 80 or fit_g.fit_info['ierr'] not in (1, 2, 3, 4)
        too_big = parameters.x_stddev.value > max_size or parameters.y_stddev.value > max_size
        too_small = parameters.x_stddev.value < 1 or parameters.y_stddev.value < 1
        bad_amplitude = parameters.amplitude.value < 1e-20 or parameters.amplitude.value > 1.0
        failed = failed_fit or too_big or bad_amplitude or too_small
        return None if failed else parameters
