[docs]def defineSpecificProgramOptions():
    """
    Defines the (command line and configuration file) options specific to this task

    Returns:
       An  ArgumentParser.
    """
    parser = argparse.ArgumentParser(prog='SIR_SpectraDecontamination')

    # Generic parameter

    parser.add_argument('--ver', action='version', version='0.2')

    parser.add_argument('--workdir', type=str, required=True,
                        help='The root working directory where the data is located.')

    parser.add_argument('--logdir', type=str, dest='logdir',
                        help='The logging directory')

    parser.add_argument('--config', type=str, dest='config',
                        help='Configuration File')

    # Input data filenames:

    parser.add_argument('--sci_frames', type=str, required=True, metavar='FRAMES',
                        help='A JSON file containing the list of input science frames.')

    parser.add_argument('--loc_tables', type=str, required=True, metavar='LOCTAB',
                        help='A JSON file containing a list of XML files associated with each location table file.')

    parser.add_argument('--mdb', type=str, required=True,
                        help='The Mission Database XML file, containing all key instrument and mission parameters.')

    parser.add_argument('--relflux_flat', type=str, required=False, default='None', metavar='FLUXCOR',
                        help='The relative flux correction master flat cube (XML filename or HDF5 filename).')

    parser.add_argument('--sens', type=str, required=False, default='None', metavar='FLUXSENS',
                        help='The absolute flux sensitivity function (XML filename for SirAbsoluteFluxScaling data '
                             'product).')

    group = parser.add_mutually_exclusive_group(required=True)

    group.add_argument('--cutouts', type=str, metavar='CUTOUTS',
                       help='a JSON file, containing a list of files containing cutouts from a stacked mosaic of '
                            'direct images of the galaxies. REQUIRED unless --passthrough is specified.')

    # Options / Behavior-modifying parameters:

    group.add_argument('--passthrough', dest='passthrough', action='store_true',
                       help='Skip the actual decontamination step; simply create cutouts, leaving the content '
                            'unmodified.')

    parser.add_argument('--padding', type=int, dest='padding', default=0,
                        help='Padding around the spectrum, in pixels, used when creating spectral cutouts.')

    parser.add_argument('--limit_mag', type=float, dest='limit_mag', default=1000, metavar="MAG",
                        help='Upper magnitude limit of objects in the band specified by --limit_band. '
                             'Objects fainter than this limit will not be treated as contaminants.')

    parser.add_argument('--limit_band', type=str, dest='limit_band', default='J', metavar='BAND',
                        help='The filter/band name that will be used to limit the number of objects that are '
                             'treated as contaminants.')

    parser.add_argument('--zeroth_order_limit', type=float, dest='zeroth_order_limit', default=20.0,
                        help='The magnitude threshold below which zeroth order contaminants will be flagged.')

    parser.add_argument('--second_order_limit', type=float, dest='second_order_limit', default=16.0,
                        help='The magnitude threshold below which second order contaminants will be flagged.')

    parser.add_argument('--detectors', dest='detectors', required=False, default='*',
                        help='Instructs the program to run using only on the specified detector(s)'
                             '(defaults to all detectors \'*\', if none are specified).')

    parser.add_argument('--pipeline', dest='pipeline',  nargs='?', default=False, const=True, type=str_to_bool,
                        help='Specify if PE is in single-PE mode or in pipeline')

    parser.add_argument('--overwrite', dest='overwrite', action='store_true',
                        help='Overwrite the output decontaminated collection file.')

    parser.add_argument('--gaussian_profile', dest='gaussian_profile', action='store_true',
                        help='Fit 2D Gaussian to cutouts, rather than using the cutouts as they are provided.')

    parser.add_argument('--order', type=int, dest='order', default=1,
                        help='Order name to decontaminate')

    parser.add_argument('--no_measure', dest='no_measure', action='store_true',
                        help='Skip the spectral measurement; only use broadband fluxes to create spectral models.')

    parser.add_argument('--max_spectra', type=int, dest='max_spectra', required=False, default=0,
                        help='The maximum number of decontaminated spectra per output file. The value 0 means no limit')

    # Output data filename(s):

    parser.add_argument('--dec_spectra', type=str, required=True, metavar='OUTPUT',
                        help='The name of the JSON file which will contain a list of DecontaminatedSpectraCollection '
                             'files---one for each input detector.')

    return parser




[docs]def mainMethod(args):
    """
    The task main method.

    This method is the entry point to the program.
    See the ElementsProgram documentation for more details.
    """
    logger = log.getLogger('SpectraDecontaminationMain')
    logger.info('Entering Spectra Decontamination mainMethod()')

    module_name = 'SpectraDecontaminationMain'
    args = addProgramOptions('SIR_SpectraDecontamination.SpectraDecontaminationMain', args,
                             module_name, logger)

    decontaminator = Executor(args, logger)

    decontaminator.exec()
