"""
Contains the classes, SpectraCollectionIndex and SpectraCollectionIndexEntry, which handle the creation, loading, and
filtering of the index table of a DecontaminatedSpectraCollection file or a set of such files.

[docs]class SpectraCollectionIndexEntry:
    """
    Contains the information corresponding to a single item in the SpectraCollectionIndex.
    """

    def __init__(self, table_row: np.ndarray):
        self._row = table_row
        self._bbox = None  # this will store the decoded self._row['bbox']

    @property
    def dither(self) -> int:
        return self._row['dither']

    @property
    def detector(self) -> int:
        return self._row['detector']

    @property
    def id(self) -> int:
        return self._row['id']

    @property
    def j_mag(self) -> float:
        return float(self._row['j_mag'])

    @property
    def h_mag(self) -> float:
        return float(self._row['h_mag'])

    @property
    def bbox(self) -> tuple:
        """
        A tuple of ints: (left, bottom, width, height).
        """
        if self._bbox is None:
            self._bbox = tuple(np.frombuffer(self._row['bbox'], dtype=np.int16).astype(np.uint32))
        return self._bbox

    @property
    def x_offset(self) -> int:
        """
        The horizontal pixel offset from the left edge of the detector.
        """
        return int(self.bbox[0])

    @property
    def y_offset(self) -> int:
        """
        The vertical pixel offset from the bottom of the detector.
        """
        return int(self.bbox[1])

    @property
    def width(self) -> int:
        return int(self.bbox[2])

    @property
    def height(self) -> int:
        return int(self.bbox[3])

    @property
    def offset(self) -> tuple:
        """
        A tuple of ints: (x_offset, y_offset)
        """
        return self.bbox[:2]

    def __str__(self):
        return (f"ID:       {self.id}\n"
                f"dither:   {self.dither}\n"
                f"detector: {self.detector}\n"
                f"J mag:    {self.j_mag}\n"
                f"H mag:    {self.h_mag}\n"
                f"bbox:     {self.bbox}\n")




[docs]class SpectraCollectionIndex:
    """
    A class for efficiently indexing the spectra stored in a single
    `DecontaminatedSpectraCollection` file so that the file can be
    efficiently searched. The class can create an index from an HDF5
    object and write the contents of the index to the file. It can also
    load the contents of a previously-written index.
    """
    TABLE_FORMAT = [('dither', np.int32),
                    ('detector', np.int8),
                    ('id',    np.int64),
                    ('j_mag', np.float16),
                    ('h_mag', np.float16),
                    ('bbox', 'V8')
                    ]

    INDEX_LABEL = 'Index'

    DECONTAMINATED_SPECTRA_LABEL = 'Decontaminated Spectra'

    def __init__(self, hdf5_file: 'h5py.File'=None, table=None):
        if hdf5_file is not None and table is not None:
            raise UserWarning("Both an hdf5_file and a table were provided. The table will be ignored.")

        # this will contain the table, structured as specified by TABLE_FORMAT
        self._table = None

        # this will be of the form {object_id: [SpectraCollectionIndexEntry]}
        self._table_by_id = None

        if hdf5_file is not None:
            self.load_index_from_hdf5(hdf5_file)
        elif table is not None:
            self._table = table

    @property
    def table(self):
        if self._table is None:
            raise RuntimeError(f"The {self.__class__.__name__} object has not been populated with data.")
        return self._table

    @property
    def dithers(self) -> tuple:
        if self._table is not None:
            return tuple(set(self._table['dither'].astype(int)))
        else:
            return tuple()

    @property
    def ids(self) -> tuple:
        return tuple(np.unique(self.table['id']))


[docs]    def get_detectors(self, dither: int) -> tuple:
        matches = self.table['dither'] == np.int32(dither)
        return tuple(np.unique(self.table['detector'][matches]).astype(int))



[docs]    def filter_by_dither(self, dither: Union[int, None]) -> 'SpectraCollectionIndex':
        if dither is None:
            return self
        matches = self.table['dither'] == np.int32(dither)
        table = np.copy(self.table[matches])
        return SpectraCollectionIndex(table=table)

        if detector is None:

        table = np.copy(self.table[matches])

[docs]    def filter_by_magnitude(self, h_max=np.inf, h_min=-np.inf, j_max=np.inf, j_min=-np.inf):
        j_mag = self.table['j_mag']
        h_mag = self.table['h_mag']
        matches = (j_mag >= j_min) & (j_mag <= j_max) & (h_mag >= h_min) & (h_mag <= h_max)
        table = np.copy(self.table[matches])
        return SpectraCollectionIndex(table=table)



[docs]    def filter_by_size(self, min_width=1, max_width=np.inf, min_height=1, max_height=np.inf):
        """
        Filters by the dimensions of the spectrum.

        By default, the dimensions are measured with respect to the image (detector) coordinates, in which case the
        width is the horizontal extent of the bounding box of the spectrum and the height is the vertical extent.

        Parameters
        ----------
        min_width: int
            The minimum extent of the spectra in the x-direction.

        max_width: int
            The maximum extent of the spectra in the x-direction.

        min_height: int
            The minimum extent of the spectra in the y-direction.


        max_height: int
            The maximum extent of the spectra in the y-direction.

        Returns
        -------

        filtered_index: SpectraCollectionIndex
            The filtered index (containing only the entries that passed through the filter).
        """
        size_table = np.zeros((len(self.table), 2), dtype=np.int16)

        for i, row in enumerate(self.table):
            size_table[i] = np.frombuffer(row['bbox'], dtype=np.int16)[-2:]

        widths = size_table[:, 0]
        heights = size_table[:, 1]

        matches = (widths >= min_width) & (widths <= max_width) & (heights >= min_height) & (heights <= max_height)

        table = np.copy(self.table[matches])
        return SpectraCollectionIndex(table=table)



[docs]    def load_index_from_hdf5(self, hdf5_file: 'h5py.File'):
        """
        Loads the table data from the h5py File, if the table already exists in the file.
        If the table has not been created yet, then use self.create_index() instead.
        """
        if self._table is None:
            self._table = self._load_index(hdf5_file)
        else:
            self._table = np.concatenate([self._table, self._load_index(hdf5_file)])


    def __getitem__(self, object_id):
        """
        Fetches the SpectraCollectionIndexEntry (or entries) corresponding to the specified object_id.
        Note that a generator is returned, so this has to be used as follows:

        for entry in collection_index:
            do_something(entry)

        where collection_index is an instance of this class.

        Parameters
        ----------

        object_id: int
            The id of the object whose spectrum is being requested.

        Returns
        -------

        spectrum_entry: a generator of SpectraCollectionIndexEntry objects
            A generator which yields SpectraCollectionIndexEntry objects describing each spectrum
            of the object, object_id.
        """
        if object_id in self:
            if self._table_by_id is None:
                self._make_table_by_id()
            return (entry for entry in self._table_by_id[object_id])
        else:
            raise KeyError(f"{object_id} is not present in this {self.__class__.__name__}.")

    def __contains__(self, object_id: int):
        if self._table is None:
            return False
        if self._table_by_id is None:
            return np.int64(object_id) in self.table['id']
        else:
            return object_id in self._table_by_id

    def __iter__(self):
        return (SpectraCollectionIndexEntry(row) for row in self.table)

    def _make_table_by_id(self):
        self._table_by_id = {}

        for row in self.table:
            object_id = row['id']
            if object_id in self._table_by_id:
                self._table_by_id[object_id].append(SpectraCollectionIndexEntry(row))
            else:
                self._table_by_id[object_id] = [SpectraCollectionIndexEntry(row)]

    def __len__(self):
        if self._table is None:
            return 0

        return len(self.table)


[docs]    @classmethod
    def add_index_to_hdf5(cls, hdf5_file: 'h5py.File'):
        """
        Stores the index table to an h5py File.
        """
        table = cls._create_index(hdf5_file)

        cls._write_index_to_file(hdf5_file, table)


    @classmethod
    def _write_index_to_file(cls, hdf5_file: 'h5py.File', table: np.ndarray):
        """
        Writes the specified Index table to the specified HDF5 File.

        Parameters
        ----------

        hdf5_file: h5py.File
            The file to which the table will be written.

        table: np.ndarray
            The index table, summarizing the contents of the file.
        """
        hdf5_file.create_dataset(cls.INDEX_LABEL, data=table)

    @classmethod
    def _create_index(cls, hdf5_file: 'h5py.File') -> np.ndarray:
        """
        Parses an HDF5 File and generates the index table, summarizing the data stored in the file.

        Parameters
        ----------
        hdf5_file: h5py.File
            The HDF5 file in question. This needs to be a file generated by DecontaminatedSpectraCollection.

        Returns
        -------

        index_table
            An np.ndarray, formatted as defined in cls.TABLE_FORMAT, containing entries for every decontaminated
            spectrum in the file.
        """
        spectra = hdf5_file[cls.DECONTAMINATED_SPECTRA_LABEL]

        n_spectra = len(spectra)

        table = np.zeros((n_spectra,), dtype=cls.TABLE_FORMAT)

        table['dither'][:] = np.int32(hdf5_file.attrs['Dither'])
        table['detector'][:] = np.int8(hdf5_file.attrs['Detector'])

        for i, spec_id in enumerate(spectra):
            spec = spectra[spec_id]
            spec_attrs = spec.attrs
            row = table[i]
            row['id'] = np.int64(spec_id)
            row['j_mag'] = np.float16(spec_attrs['j_mag'])
            row['h_mag'] = np.float16(spec_attrs['h_mag'])
            x_off = spec_attrs['x_offset']
            y_off = spec_attrs['y_offset']
            height, width = spec['science'].shape
            row['bbox'] = np.bytes_(np.array((x_off, y_off, width, height), dtype=np.int16))

        return table

    @classmethod
    def _load_index(cls, hdf5_file: 'h5py.File') -> np.ndarray:
        """
        If the HDF5 file contains an Index table, then load it. Otherwise, generate the index from the contents of
        the file.

        Parameters
        ----------

        hdf5_file: h5py.File

        Returns
        -------

        index_table
            An np.ndarray, formatted as defined in cls.TABLE_FORMAT, containing entries for every decontaminated
            spectrum in the file.
        """
        if cls.INDEX_LABEL in hdf5_file:
            return np.array(hdf5_file[cls.INDEX_LABEL])
        else:
            return cls._create_index(hdf5_file)
