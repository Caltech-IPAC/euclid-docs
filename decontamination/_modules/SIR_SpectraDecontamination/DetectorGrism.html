[docs]class DetectorGrism(object):
    """
    Handles the manipulation and creation of 1-D and 2-D spectra and stores configuration data.

    Provides a means to convert spectra from physical units to pixel values. This involves using both the
    features of the DetectorModel and the GrismModel, as well as other data from the Euclid Mission Database.
    """

    MIN_WAV_INNER = None
    MAX_WAV_INNER = None
    PSF_MODEL = None

    def __init__(self, mdb, config, nisp_exposure, sens=None):
        """
        Parameters
        ----------

        mdb: DpdMdbDataBase
            A SIR Utilities Mdb class instance containing data from the official Mission Database and the
            additional information, specific to SIR, that is not part of the official Euclid MDB.

        config: ConfigObj
            Contains the parsed contents of the spectra decontamination .ini file.

        nisp_exposure: NispSurveyExposure
            The NispSurveyExposure corresponding to the dither that this DetectorGrism is in charge
            of modelling.

        sens: astropy.io.fits.hdu.hdulist.HDUList or None
            The FITS file containing the sensitivity curve. If this is not specified, the sensitivity function from
            the MDB will be used.
        """
        self._mdb = mdb

        self._config = config

        self._nisp_exposure = nisp_exposure

        # The name of the Grism associated with this dither (exposure)
        self._grism_name = nisp_exposure.getMetadata().get('GWA_POS')

        # the nominal tilt angle of the grism
        self._grism_tilt = nisp_exposure.getMetadata().get('GWA_TILT')

        # The exposure time in seconds
        self._exposure_time = nisp_exposure.getMetadata().get('EXPTIME')

        # The number of this dither in the dither pattern
        self._dither = nisp_exposure.getMetadata().get('DITHOBS')

        # The DetectorModel object, used for determining information about the detector.
        self._detector_model = DetectorModel(self._grism_name, mdb)

        # The size of a pixel in millimeters
        self._pixel_size_in_mm = self._detector_model.getPixelScale()

        # The number of pixels along the x-direction of a detector
        self._xres = self._detector_model.getDetXPixels()

        # The number of pixels along the y-direction of a detector
        self._yres = self._detector_model.getDetYPixels()

        self._get_psf_model(mdb, config, nisp_exposure)

        # Create the interpolator object that will provide the aperture flux corrections
        self._flux_corrector = self._make_flux_correction_interpolator()

        # Create the interpolator object that will provide the wavelength-dependent size correction to the PSF.
        self._psf_size_corrector = self._make_psf_size_correction_interpolator()

        if sens is None:
            sensitivity = SensitivityModel(mdb=mdb, grism=self._grism_name)
            grism_wavs = sensitivity.getWaves(GrismModel.Order_1)
            grism_sensitivity = sensitivity.getSensitivity(GrismModel.Order_1)
            grism_mask = None
        else:
            grism_wavs, grism_sensitivity, grism_mask = self._get_sensitivity_from_dataproduct(sens)

        # The sensitivity curve, used to transform from erg/s/cm^2/Angstrom to detector counts / s / Angstrom.
        self._grism_sensitivity = grism_sensitivity  # needs to eventually account for 1st and 2nd order

        # The wavelengths of the sensitivity curve, in units of Angstroms.
        self._grism_wavs = grism_wavs  # needs to eventually account for 1st and 2nd order

        # The inverse sensitivity of the grism + detector. This needs to eventually account for 1st and 2nd order.
        self._grism_inverse_sensitivity = Utils.div0(1.0, grism_sensitivity)

        self._full_wavelength_range = self._determine_full_wavelength_range(grism_wavs, grism_sensitivity)

        self._determine_inner_wavelenth_range(grism_wavs, grism_sensitivity, grism_mask)

        self._angstroms_per_pixel = self._determine_angstroms_per_pixel()

        start, stop = self._full_wavelength_range

        self._first_order_length = int((stop - start) / self._angstroms_per_pixel)

        self._zeroth_order_bounds = np.array(config['ZEROTH_ORDER']['bounds'], dtype=np.float32)
        padding = float(config['ZEROTH_ORDER']['padding'])
        self._zeroth_order_length = np.array(config['ZEROTH_ORDER']['spectral_size'], dtype=np.float32) + 2 * padding
        self._zeroth_order_thickness = np.array(config['ZEROTH_ORDER']['spatial_size'], dtype=np.float32) + 2 * padding

[docs]    def flambda_to_detector_counts(self, wavelengths: np.ndarray, flux_densities: np.ndarray):
        """
        Converts units of erg/s/cm^2/Angstrom to detector_counts/Angstrom.

        Parameters
        ----------

        wavelengths: 1-D NumPy array of floats
            The wavelengths of the flux measurements, in units of Angstroms.

        flux_densities: 1-D NumPy array of floats
            The flux density measurements, in units of erg/s/cm^2/Angstrom.

        Returns
        -------

        flux_density: 1-D NumPy array of floats
            The flux density in units of detector counts per Angstrom.
        """
        transmitted_spectrum = Utils.interp_multiply(wavelengths,
                                                     flux_densities,
                                                     self._grism_wavs,
                                                     self._grism_sensitivity)

        # transmitted_spectrum is now in units of detector_counts / second / Angstrom.
        # We have to multiply by exposure time to get detector_counts / Angstrom:

        return transmitted_spectrum * self._exposure_time

[docs]    def disperse(self, wavelengths, spectrum_info):
        """
        Disperses (convolves) a 1-D spectrum onto a 2-D canvas to create a 2-D model spectrum.

        Parameters
        ----------

        wavelengths: 1-D NumPy array of floats
            The wavelengths of the flux measurements, in units of Angstroms.

        spectrum_info: SpectrumInfo
            The `SpectrumInfo` object associated with the model spectrum that will be created by this function.

        Returns
        -------

        model_spectrum: 2-D NumPy array of floats
            The 2-D model spectrum corresponding to the spectrum described by `spectrum_info`.

        interpolation_mask: 2-D NumPy array of bools
            A mask indicating whether the flux of each pixel of the model was computed using interpolation.
        """
        counts_per_segment, segment_centers = self._compute_counts_per_segment(wavelengths, spectrum_info)

        size = spectrum_info.convolution_kernel.kernel_size

        correction = self._flux_corrector(size, segment_centers).flatten()

        counts_per_segment = counts_per_segment * correction

        if len(spectrum_info.convolution_kernel.history) == 0:
            model = self._convolve2d(spectrum_info, counts_per_segment, segment_centers)
        else:
            model = self._convolve2d_from_history(spectrum_info, counts_per_segment)

        if spectrum_info.astro_object_info.interp_mask:
            mask = (model > 0.0)
        else:
            mask = np.zeros(model.shape, bool)

        return model, mask



[docs]    def get_zeroth_order_boxsize(self, mag: float) -> Tuple[int, int]:
        """
        Determines the size of the zeroth-order in units of pixels.

        Parameters
        ----------

        mag: int
            the magnitude of the object

        Returns
        -------

        size: tuple

        A tuple of (spectral_size, spatial_size)
        """
        idx = np.argmin(np.fabs(self._zeroth_order_bounds - mag))

        return self._zeroth_order_length[idx], self._zeroth_order_thickness[idx]


    def _compute_counts_per_segment(self, wavelengths, spectrum_info):
        """
        Determines the number of counts (electrons) that will be deposited in each spectral bin (one-pixel-wide bins).
        """
        if spectrum_info.relflux_correction is None:
            flux_densities = spectrum_info.one_dim_spectrum
        else:
            flux_densities = spectrum_info.one_dim_spectrum / spectrum_info.relflux_correction

        detector_counts_per_angstrom = self.flambda_to_detector_counts(wavelengths, flux_densities)

        cumulative_counts = cumtrapz(detector_counts_per_angstrom, wavelengths, initial=0)

        integrated_counts = interp1d(wavelengths, cumulative_counts, copy=False, kind='linear', assume_sorted=True)

        segment_counts = integrated_counts(wavelengths[1:]) - integrated_counts(wavelengths[:-1])

        segment_centers = 0.5 * (wavelengths[1:] + wavelengths[:-1])

        return segment_counts, segment_centers

    def _make_flux_correction_interpolator(self):
        grism = 'RGS' if self.is_red() else 'BGS'
        table = self._config['FLUX_CORRECTIONS'][grism]
        wavelengths = np.array(table['wavelengths'], dtype=float)
        size_labels = table['apertures']
        sizes = np.array(size_labels, dtype=float)

        correction_table = [np.array(table[sl], dtype=float) for sl in size_labels]

        correction = np.stack(correction_table, axis=1)

        return interp2d(sizes, wavelengths, correction, fill_value=1.0)

    def _make_psf_size_correction_interpolator(self):
        grism = 'RGS' if self.is_red() else 'BGS'
        table = self._config['PSF_CORRECTIONS'][grism]
        wavelengths = np.array(table['wavelengths'], dtype=float)
        correction_sigmas = np.array(table['sigmas'], dtype=float)

        return interp1d(wavelengths, correction_sigmas, fill_value='extrapolate')

    def _convolve2d(self, spectrum_info, counts_per_segment, segment_centers):
        conv_kernel = spectrum_info.convolution_kernel

        left, bottom, width, height = spectrum_info.bbox

        model = np.zeros((height, width), dtype=np.float32)

        for segment, count in enumerate(counts_per_segment):
            wavelength = float(segment_centers[segment])
            x, y = spectrum_info.get_trace_position(wavelength)
            conv_kernel.move_to(x, y)
            conv_kernel.set_wavelength(wavelength)
            if conv_kernel.in_spectral_box:
                model[conv_kernel.region] += count * conv_kernel.kernel
                conv_kernel.add_to_history(segment, wavelength)

        return model

    def _determine_angstroms_per_pixel(self):
        angstroms_per_pixel = None
        if self.is_blue():
            # the average dlambda for the blue grism
            angstroms_per_pixel = self._mdb.get_nisp_blue_average_dlambda()
        elif self.is_red():
            # the average dlambda for the red grism
            angstroms_per_pixel = self._mdb.get_nisp_red_average_dlambda()
        else:
            raise ValueError(f'Unexpected grism name: {self._grism_name}')

        return angstroms_per_pixel

    def _get_sensitivity_from_dataproduct(self, sens) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
        """
        Fetches the sensitivity curve from the FITS file produced by the absolute calibration module.

        Parameters
        ----------

        sens: astropy.io.fits.hdu.hdulist.HDUList
            The FITS file containing the sensitivity curve.

        Returns
        -------

        wavelengths: np.ndarray

        sensitivity: np.ndarray

        mask: np.ndarray
        """
        if self._grism_tilt == 0.0:
            sep = ''
            suffix = ''
        else:
            sep = 'm' if self._grism_tilt < 0 else 'p'
            suffix = int(abs(self._grism_tilt))

        hdu_name = f"{self._grism_name}{sep}{suffix}"
        sens_hdu = sens[hdu_name]
        grism_sensitivity = sens_hdu.data['Sensitivity']
        grism_wavs = sens_hdu.data['Wavelength']
        grism_mask = sens_hdu.data['Mask'].astype(np.int64)

        # force endpoints to zero if they are not already zero

        grism_sensitivity[0] = 0.0
        grism_sensitivity[-1] = 0.0

        return grism_wavs, grism_sensitivity, grism_mask


    @staticmethod
    def _convolve2d_from_history(spectrum_info, counts_per_segment):
        conv_kernel = spectrum_info.convolution_kernel

        left, bottom, width, height = spectrum_info.bbox

        model = np.zeros((height, width), dtype=np.float32)

        for entry in conv_kernel.history:
            i, _, kernel, region, _ = entry
            count = counts_per_segment[i]
            model[region] += count * kernel

        return model

    @staticmethod
    def _determine_full_wavelength_range(grism_wavs: np.ndarray, grism_sensitivity: np.ndarray) -> Tuple[float, float]:
        """
        Determines the full wavelength range of the grism sensitivity

        Parameters
        ----------

        grism_wavs: np.ndarray
            The wavelengths of the grism sensitivity.

        grism_sensitivity: np.ndarray
            The sensitivity values of the grism.

        Returns
        -------

        limits: Tuple
            A tuple of the lower and upper wavelength limits of the grism transmission (lower, upper), as floats.
        """
        indices = np.where(grism_sensitivity > 0.009 * grism_sensitivity.max())[0]
        left = grism_wavs[indices[0]]
        right = grism_wavs[indices[-1]]
        pad = 0.01 * (right - left)

        return left - pad, right + pad

    @classmethod
    def _get_psf_model(cls, mdb: 'DpdMdbDataBase', config: 'ConfigObj', nisp_exposure: 'NispSurveyExposure'):
        if cls.PSF_MODEL is None:
            if not hasattr(config, 'data_dir'):
                ValueError("The config object is expected to have a 'data_dir' attribute.")
            full_filenames = [os.path.join(config.data_dir, f) for f in mdb.get_nisps_psf_filenames()]
            # determine arcsec_per_pixel from the exposures
            det = nisp_exposure.getDetector(1)
            dx = det.getMetadata().get('CD1_1')
            dy = det.getMetadata().get('CD1_2')
            arcsec_per_pixel = 3600 * np.hypot(dx, dy)
            cls.PSF_MODEL = PsfModel(full_filenames, arcsec_per_pixel)

    @classmethod
    def _determine_inner_wavelenth_range(cls,
                                         grism_wavs: np.ndarray,
                                         grism_sensitivity: np.ndarray,
                                         grism_mask: Union[np.ndarray, None]) -> None:
        """
        Determines the inner wavelength range of the grism (the region safely far from the ends to avoid edge effects.)

        Parameters
        ----------

        grism_wavs: np.ndarray
            The wavelengths of the grism sensitivity.

        grism_sensitivity: np.ndarray
            The sensitivity values of the grism.

        Side Effects
        ------------

        Sets the MIN_WAV_INNER and MAX_WAV_INNER class variables, if they are not yet set.
        """
        if cls.MIN_WAV_INNER is None:
            # determine the inner wavelength range within a margin of the cut-on and cut-off wavelengths.
            indices = np.where(grism_sensitivity > 0.009 * grism_sensitivity.max())[0]
            mask = grism_sensitivity > np.quantile(grism_sensitivity[indices], 0.09)
            margin_quantile = 0.05
            left = float(np.quantile(grism_wavs[mask], margin_quantile))
            right = float(np.quantile(grism_wavs[mask], 1 - margin_quantile))
            shift = 0.01 * (right - left)
            left = left + shift
            right = right - shift

            if grism_mask is not None:
                end_flag = SPECTRUM_MASK['SUSP_AFX']
                mid_wav = np.median(grism_wavs)
                low_end_idx = np.where((grism_wavs < mid_wav) & ((grism_mask & end_flag) == end_flag))
                high_end_idx = np.where((grism_wavs > mid_wav) & ((grism_mask & end_flag) == end_flag))
                left_wav = np.max(grism_wavs[low_end_idx])
                right_wav = np.min(grism_wavs[high_end_idx])
                if left < left_wav:
                    left = left_wav
                if right > right_wav:
                    right = right_wav

            cls.MIN_WAV_INNER = left
            cls.MAX_WAV_INNER = right
