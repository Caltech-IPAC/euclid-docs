    """
    Loads metadata about a set of h5LocationTable files.
    """

    def __init__(self, workdir: str, filename: str, sir_config: str, logger=None):
        """
        Loads the h5LocationTable files.

        Parameters
        ----------

        workdir: str
            The path to the working directory (i.e., the directory that contains the data/ directory).

        filename: str
            The filename of the JSON file containing a list of objects that will be loaded.
            The list within the JSON file should contain the names of XML files associated with h5LocationTable
            files. Each h5LocationTable file is an HDF5 file.

        sir_config str:
            The filename of the XML file containing configuration parameters for the SIR Pipeline.

        logger: logging.Logger
            [optional] If provided, the class will print progress messages to the log.

        Raises
        ------

        FileNotFoundError: if the input file does not exist.
        """
        self._logger = logger

        self._sir_config = sir_config

        self._tables = []

        self._json_filename = filename

        self._work_dir = os.path.abspath(workdir)

        self._data_dir = os.path.join(self._work_dir, 'data')

        self._log("Loading location table info from ", filename)

        with open(filename) as f:
            xml_filenames = json.load(f)

        self._selection = set()

        self._hdf5_filenames = []

        self._xml_filenames = []

        self._data_products = []

        self._data = NestedDict(list, 'dither', 'detector')

        self._filename_map = dict()

        # unique identifier for loaded LT
        self._LTs_unique_id = []

        with open(filename,'r') as j_file:
            for idx, table_name_xml in enumerate(json.load(j_file)):
                self._selection.add(idx)
                full_xml_filename = os.path.join(self._work_dir, table_name_xml)
                self._xml_filenames.append(full_xml_filename)
                table_obj = DpdSirLocationTable(full_xml_filename)
                self._data_products.append(table_obj)

                # create list to identify each loaded LT by "GRISM_TILT_POINTING_OBSERVATION_DITHER" string 
                dith_id = table_obj.get_dithobs()
                obs_id = table_obj.get_obs_id()
                ptg_id = table_obj.get_ptg_id()
                gwa_pos = table_obj.get_gwa_pos()
                gwa_tilt = table_obj.get_gwa_tilt()
                self._LTs_unique_id.append(f'{gwa_pos}_{gwa_tilt}_{ptg_id}_{obs_id}_{dith_id}')

                table_name_h5 = os.path.join(workdir, 'data', table_obj.get_location_h5_file())
                self._hdf5_filenames.append(table_name_h5)
                table = H5LocationTable.load(table_name_h5, only_metadata=True)
                dither = table.getDither()
                detector = table.getDetectorNumber()
                self._filename_map[(dither, detector)] = table_name_h5

                if (dither, detector) in self._data:
                    self._data[dither, detector].append(table)
                else:
                    self._data[dither, detector] = [table]

        self._location_factory_params = self._get_factory_param_filename(self._sir_config)

[docs]    def get_detectors(self, dither: int) -> List[int]:
        """
        Gets the detectors present for the specified dither.

        Parameters
        ----------

        dither: int
            The dither sequence number of interest.

        Returns
        -------
        detectors: List[int]
            For the specified dither, this returns a list of the detectors present in the set of LocationTables.

        Raises
        ------

        IndexError:
            If the value provided for `dither` is not present in the set of LocationTables.
        """
        if dither not in self._data.keys():
            raise IndexError(f"Invalid dither {dither} (The specified dither is not present.)")
        return list(self._data.get_subdict(dither).keys())



[docs]    def get_hdf5_filename(self, dither: int, detector: int) -> str:
        """
        Gets the name of the HDF5 file associated with the specified dither and detector.

        Parameters
        ----------

        dither: int
            The dither sequence number of interest.

        detector: int
            The detector number of interest
        """
        if (dither, detector) in self._filename_map:
            filename = self._filename_map[(dither, detector)]
        else:
            warnings.warn(f'This set does not contain dither {dither}, detector {detector}.')
            filename = ""

        return filename


    @property
    def detector_map(self) -> Dict[int, List[int]]:
        det_map = dict()
        for dither in self.get_dithers():
            det_map[dither] = self.get_detectors(dither)
        return det_map

    def _get_factory_param_filename(self, sir_config: str):
        """
        Gets the fully-qualified filename of the .ini file containing LocationObjectFactory parameters.

        Parameters
        ----------

        sir_config: str
            The filename of the XML file containing configuration parameters for the SIR Pipeline.

        Returns
        -------

        factory_parameter_filename: str
            Fully-qualified name of the .ini parameter file for the LocationObjectFactory class.
        """
        config_set = DpdSirConfigurationSet(filename=os.path.join(self._work_dir, sir_config))
        parameter_file = config_set.get_parameter_file('LocationObjectFactory')
        ini_file = os.path.join(self._work_dir, 'data', parameter_file)

        if not os.path.isfile(ini_file):
            raise FileNotFoundError(f"No parameter file {ini_file}")

        return ini_file

    def _log(self, *message):

        if self._logger is not None:
            self._logger.info(" ".join(message))
