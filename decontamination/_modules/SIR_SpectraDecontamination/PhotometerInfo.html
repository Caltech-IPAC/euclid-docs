import SIR_SpectraDecontamination.Utils as Utils



[docs]class PhotometerInfo(object):
    """
    Extracts pertinent information about the photometric filters from the Mission Database.
    """

    def __init__(self, mdb):
        """
        Constructs the PhotometerInfo object.

        Parameters
        ----------

        mdb: Mdb
            An Mdb object, loaded with data. This is the object returned by `SIR_Utilities.Mdb.Mdb.load()`
        """

        # The transmission curve for the Y band filter (wavelengths in nanometers and transmission fraction)

        y_wavs, y_transmission = mdb.get_nisp_Y_transmission_curves()

        # The transmission curve for the J band filter (wavelengths in nanometers and transmission fraction)
        j_wavs, j_transmission = mdb.get_nisp_J_transmission_curves()

        # The transmission curve for the H band filter (wavelengths in nanometers and transmission fraction)
        h_wavs, h_transmission = mdb.get_nisp_H_transmission_curves()

        # Convert from nanometers to Angstroms

        h_wavs *= 10
        j_wavs *= 10
        y_wavs *= 10

        self._transmission = {'Y': (y_wavs, y_transmission),
                              'J': (j_wavs, j_transmission),
                              'H': (h_wavs, h_transmission)}

        self._effective_wav = {'Y': Utils.effective_wavelength(y_wavs, y_transmission),
                               'J': Utils.effective_wavelength(j_wavs, j_transmission),
                               'H': Utils.effective_wavelength(h_wavs, h_transmission)}


[docs]    def effective_wavelength(self, band):
        return self._effective_wav[band.upper()]
