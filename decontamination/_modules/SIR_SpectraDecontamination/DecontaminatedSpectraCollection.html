[docs]    class InitMethod(Enum):
        """Specifies the Initialization method."""

        NONE = 0
        """NONE implies that the initialization has not happened yet."""

        FROM_DECONTAMINATOR = 1
        """
        FROM_DECONTAMINATOR indicates that the collection was initialized using data from the SpectraDecontaminator.
        """

        FROM_FILE = 2
        """FROM_FILE indicates that the collection was initialized by loading one or more files."""


    def __init__(self, filename: str=None, data_dir: str=None, decontaminated_spectra: NestedDict=None):
        """
        Construct the object.

        Parameters
        ----------

        filename: str, optional
            The name of a file to be loaded. This is either a JSON file containing a list of XML files or HDF5 files
            or the name of a single HDF5 file.

        data_dir: str, optional
            The directory where the HDF5 files are stored.

        decontaminated_spectra: NestedDict, optional
            A `NestedDict` of the form: {dither: {detector: [SpectrumCutout]}}, where `dither` and `detector` are
            integers specifying the exposure number (dither) within the dither sequence and `detector` number within
            the detector array. Each detector is mapped to a list of `SpectrumCutout` objects.
        """
        self._logger = log.getLogger('DecontaminatedSpectraCollection')

        if decontaminated_spectra is None and filename is None:
            self._logger.info(f"Creating an empty {self.__class__.__name__}. The load() method will need to "
                              "be called in order to fill the object with data.")
        else:
            self._logger.info(f"Creating a {self.__class__.__name__} object and populating it with spectra.")

        self._field_id = None

        # a dictionary of the form {dither: exposure time in seconds}
        self._exposure_time = {}

        # a dictionary of the form {dither: observation ID}
        self._observation_id = {}

        # a dictionary of the form {dither: pointing ID}
        self._pointing_id = {}

        # a dictionary of the form {dither: grism position}
        self._exposure_grism_position = {}

        # If the object is being used to load one or more files, or if the object has been used to save one or more
        # output files, then this contains a SpectraCollectionIndex object with a complete listing of all of the
        # decontaminated spectra in the collection. It is not populated until HDF5 files have been written or loaded.
        self._complete_index = None

        # If the object is being used to load one or more files, or if the object has been used to save one or more
        # output files, then this contains a SpectraCollectionIndex object with a subset of the decontaminated spectra
        # in the collection. It is not populated until HDF5 files have been written or loaded. If no filters have been
        # applied to the collection, then this is identical to self._complete_index
        self._filtered_index = None

        # the wavelengths of the 1D model spectra (it's the same vector for all spectra)
        self._one_dim_wavelengths = None

        # the decontaminated spectra are stored here (dither: int, detector: int, ob_id: int)
        self._hdf5_spectra = NestedDict(DecontaminatedSpectrum, 'dither', 'detector', 'ob_id', plural_key='dithers')

        # The 2D spectral models are stored here:
        self._hdf5_models = NestedDict(ModelSpectrum, 'dither', 'detector', 'ob_id', 'order', plural_key='dithers')

        # The 1D spectral models are stored here. These are tuples of (wavelengths, fluxes)
        self._hdf5_1d_models = NestedDict(np.ndarray, 'dither', 'detector', 'ob_id', plural_key='dithers')

        # A mapping from object ID to 1D model flux: {object_id: np.ndarray(flux)}
        self._one_dim_spectra_by_object_id = {}

        # The HDF5 files associated with the collection:
        self._hdf5_files = NestedDict(list, 'dither', 'detector', plural_key='dithers')

        # A dict of the form {object_id: {set of DecontaminatedSpectrum objects}}
        self._spectra_by_object_id = {}

        self._location_tables = None

        self._detector_models = dict()

        self._mdb = None

        # The method my which the object was initialized with data
        self._initialization_method = self.InitMethod.NONE

        if decontaminated_spectra is not None:
            self._initialization_method = self.InitMethod.FROM_DECONTAMINATOR
            self._set_metadata_from_cutouts(decontaminated_spectra)
            self._populate_hdf5_models(decontaminated_spectra)
            self._populate_hdf5_spectra(decontaminated_spectra)
        elif filename is not None:
            self.load(filename, data_dir)

    def __del__(self):
        self._hdf5_spectra = None
        self._hdf5_models = None
        self._hdf5_1d_models = None
        self._one_dim_spectra_by_object_id = None
        self._hdf5_files = None
        self._spectra_by_object_id = None
        self._location_tables = None

[docs]    def get_object_ids(self, dither: int=None, detector: int=None) -> Union[Tuple[int], Tuple]:
        """
        Gets the IDs of the objects whose spectra appear in the specified detector of the specified dither.

        Parameters
        ----------

        dither: int (optional)
            An integer in (1, 2, 3, 4) associated with the exposure withing the dither sequence.

        detector: int (optional)
            The number of the detector of interest within the specified dither (1, ..., 16).

        Returns
        -------

            A tuple of IDs of objects whose decontaminated spectra are located on the specified detector.
            If there are no object IDs,then an empty tuple is returned.
        """
        if not isinstance(dither, (int, type(None))) and not isinstance(detector, (int, type(None))):
            raise TypeError(f"dither and detector must be integers or None, not {type(dither)} and {type(detector)}.")

        ids = None

        if dither is None and detector is None:
            if self._initialization_method == self.InitMethod.FROM_FILE:
                ids = self._filtered_index.ids
            elif self._initialization_method == self.InitMethod.FROM_DECONTAMINATOR:
                ids = tuple(set((spec.id for spec in self)))
        else:
            if self._initialization_method == self.InitMethod.FROM_FILE:
                table = self._filtered_index.filter_by_dither(dither).filter_by_detector(detector)
                unique_ids = set((spec.id for spec in table))
                ids = tuple(unique_ids)
            elif self._initialization_method == self.InitMethod.FROM_DECONTAMINATOR:
                if detector is None:
                    ids = tuple(set((spec.id for spec in self if spec.dither == dither)))
                elif dither is None:
                    ids = tuple(set((spec.id for spec in self if spec.detector == detector)))
                else:
                    ids = tuple(self._hdf5_spectra.keys(dither, detector))

        if ids is None:
            ids = ()

        return ids

[docs]    def get_spectrum(self, dither: int, detector: int, object_id: int) -> DecontaminatedSpectrum:
        """
        Get the DecontaminatedSpectrum object corresponding to the object with the specified ID on the specified dither
        and the specified detector.

        Parameters
        ----------

        dither: int
            The dither index, an integer (1, 2, 3, 4), specifying the exposure number within the dither sequence.

        detector: int
           The detector on which the spectrum of interest is located (1, ..., 16).

        object_id: int
            The ID of the object whose spectrum will be returned.

        Returns
        -------

        DecontaminatedSpectrum:
            The DecontaminatedSpectrum object associated with the specified object in the specified detector of the
            specified dither (exposure) within the dither pattern. If the specified object is not present, returns None.

        """
        spec = None
        if (dither, detector, object_id) in self._hdf5_spectra:
            spec = self._hdf5_spectra[dither, detector, object_id]
        elif self._initialization_method == self.InitMethod.FROM_FILE:
            spec = self._load_spectrum_from_file(dither, detector, object_id)
            self._cache_spectrum(spec)
        return spec

        """
        Get all of the DecontaminatedSpectrum objects corresponding with the specified dither and detector.

        Parameters
        ----------

        dither: int
            The dither index, an integer (1, 2, 3, 4), specifying the exposure number within the dither sequence.

        detector: int
           The detector on which the spectrum of interest is located (1, ..., 16).

        Returns
        -------

        generator
            A generator that can be iterated to obtain a DecontaminatedSpectrum object for each spectrum falling
            in the specified detector of the specified exposure.
        """
        for object_id in self.get_object_ids(dither, detector):
            yield self.get_spectrum(dither, detector, object_id)

        """
        Get the model spectrum of the object in the specified detector of the specified dither with ID = object_id.

        Parameters
        ----------

        dither: int
            The dither index, an integer (1, 2, 3, 4), specifying the exposure number within the dither sequence.

        detector: int
           The detector on which the spectrum of interest is located (1, ..., 16).

        object_id: int
            The ID of the object whose spectrum will be returned.

        order: int
            The spectral order of the model in question.

        Returns
        -------

        ModelSpectrum
            A ModelSpectrum object containing the requested model. If the model does not exist, returns None.

        Raises
        ------

        ValueError
            If the specified order is not 1 or 2. Models are not created for the zeroth order, for instance.

        if order not in (1, 2):
            raise ValueError(f"Models are only created for spectral orders, 1, and 2---not {order}.")

        model = None

        if (dither, detector, object_id, order) in self._hdf5_models:
            model = self._hdf5_models[dither, detector, object_id, order]
        elif self._initialization_method == self.InitMethod.FROM_FILE:
            model = self._load_model_from_file(dither, detector, object_id, order)

        return model

[docs]    def write(self, workdir:str, 
                    json_name:str, 
                    location_tables: Union['LocationTableSet', None]=None,
                    mode:str='create',
                    parallel: bool=False):
        """
        Saves a JSON file containing a list of XML files. The JSON file will only contain the names of the XML files,
        without the path information.

        Parameters
        ----------

        workdir: str
            The path to the work directory.

        json_name: str
            The name of the JSON file that will contain the list of XML files.

        location_tables: LocationTableSet
            The set of location tables that were used when producing this DecontaminatedSpectraCollection.

        mode: str (optional)
            Specifies the saving mode. Allowed values are {'create', 'overwrite', 'append'}. Default: 'create'.

        parallel: bool
            Specifies whether the files will be written in parallel. This should only set to True when
            decontaminating entire detectors, rather than partial detectors. This is incompatible with the 'append'
            mode.
        """
        overwrite = bool(mode == 'overwrite')

        if mode == 'append':
            if not os.path.isfile(json_name):
                raise FileNotFoundError(f"Cannot append; the file {json_name} does not exist.")

            with open(json_name, 'r') as f:
                list_of_xml_names = json.load(f)

            if not isinstance(list_of_xml_names, list):
                raise TypeError(f"The data stored in {json_name} is expected to be a list.")
        else:
            list_of_xml_names = []

        if self._initialization_method == self.InitMethod.FROM_FILE:
            self._populate_data_structures()

        def determine_part(dither, detector_id, filenames):
            part = 1
            for name in filenames:
                if f"-{dither}-DET_{detector_id}-" in name:
                    part += 1
            return part

        # serially write products
        for dither in self._hdf5_spectra.dithers():
            for detector in self._hdf5_spectra.keys(dither):
                
                # compute DQC
                dqc_stats = QualityStatistics(self)
                dqc_stats.add_statistics(dither, detector)

                # get ID for current file
                detector_id = DetectorModel.getNumbers2IDs()[detector]

                if hasattr(self, 'is_final_part'):
                    part = determine_part(dither, detector_id, list_of_xml_names)
                    file_id = f'{self._observation_id[dither]}-{dither}-DET_{detector_id}-part-{part}'
                else:
                    file_id = f'{self._observation_id[dither]}-{dither}-DET_{detector_id}'
 
                # create H5
                h5_filename = DpdSirDecontaminatedSpectra.create_container_name(file_id)
                loctab_filename = location_tables.info.get_hdf5_filename(dither, detector)

                self._save_hdf5_v2(os.path.join(workdir, 'data', h5_filename),
                                   loctab_filename,
                                   dither,
                                   detector,
                                   overwrite)

                xml_filename = f'{os.path.splitext(json_name)[0]}-{file_id}.xml'
                xml_obj = DpdSirDecontaminatedSpectra()
                xml_obj.append_data(h5file=os.path.join(workdir,'data',h5_filename))
                xml_obj.append_dqc(dqc_stats.get_dict_dqc_parameters())
                xml_obj.write(xml_filename)

                list_of_xml_names.append(xml_filename)

        # create final JSON file with list of written products
        with open(os.path.join(workdir, json_name), 'w') as out_file:
            json.dump(list_of_xml_names, out_file)



[docs]    def load(self, filename, data_dir=None):
        """
        Loads a DecontaminatedSpectraCollection HDF5 file or a list of files.

        Parameters
        ----------

        filename: str
            The name of the file containing the DecontaminatedSpectraCollection that will be loaded (if the extension is
            one of ('h5', 'hde', 'hdf', 'hdf5', or 'he5'). If the file extension is `json`, then all files listed in
            the JSON file will be loaded.

        data_dir: str
            The path to the directory containing the HDF5 files (optional; only needed if loading a JSON file).

        Raises
        ------

        ValueError
            If the filename extension is not recognized or if the filename does not include the fully-qualified
            (absolute) path of an existing file.

        UserWarning
            If the user tries to load data into an instance that has been initialized from a NestedDict.

        Side Effects
        ------------

        Populates all of the internal data structures.
        """
        if self._initialization_method == self.InitMethod.FROM_DECONTAMINATOR:
            raise UserWarning("Loading data into this instance is not supported; it was initialized from a NestedDict.")

        filename_extension = filename.split('.')[-1]

        self._check_filename_extension(filename_extension)

        if filename_extension == 'json':
            self._load_json(data_dir, filename)
        elif filename_extension == 'xml':
            xml_obj = DpdSirDecontaminatedSpectra(filename)
            self._load_hdf5(os.path.join(data_dir, xml_obj.get_hdf5_filename()))
        else:
            self._load_hdf5(filename)

        self._initialization_method = self.InitMethod.FROM_FILE

    def concatenate(cls,
                    dither: int,
                    detector: int,
                    json_file: str,
                    data_dir: str,
                    location_tables: 'LocationTableSet',
                    mdb: 'DpdMdbDataBase',
                    delete: bool=False):
        """
        Concatenates file parts into full detector HDF5 files.

        Takes a list of XML files for parts of detectors and outputs one file per detector without loading all
        data into memory at once: only one detector worth of data is loaded at any given moment.

        Parameters
        ----------

        dither: int
            The dither number

        detector: int
            The detector number

        json_file: str
            The filename of a JSON file containing a list of XML files.

        data_dir: str
            The path to the directory in which the HDF5 files are stored.

        location_tables: LocationTableSet
            The locationTableSet containing location table information relevant for these decontaminated
            spectra collections.

        mdb: DpdMdbDataBase
            A Mission Database object.

        delete: bool
            If true, all input files listed in the original JSON file will be removed.
        """
        with open(json_file) as f:
            input_xmls = json.load(f)

        detector_id = DetectorModel.getNumbers2IDs()[detector]
        pattern = f'-{dither}-DET_{detector_id}'
        xmls = [x for x in input_xmls if pattern in x]

        temp_json_in = f'{json_file.removesuffix(".json")}-temporary-{dither}-{detector}.json'
        with open(temp_json_in, 'w') as f:
            json.dump(xmls, f)
        temp_json_out = f'{json_file.removesuffix(".json")}-{dither}-{detector}.json'
        collection = cls(temp_json_in, data_dir)
        collection.mdb = mdb
        work_dir = str(pathlib.Path(data_dir).parent)
        collection.write(work_dir, temp_json_out, location_tables, 'overwrite')
        del collection

        with open(temp_json_out) as f:
            xml_filename, = json.load(f)

        for x in xmls:
            input_xmls.remove(x)

        input_xmls.append(xml_filename)

        if delete:
            cls._delete_files(temp_json_in, data_dir)
            os.unlink(temp_json_in)
            os.unlink(temp_json_out)

        with open(json_file, 'w') as f:
            json.dump(input_xmls, f)


    def __getitem__(self, object_id: int) -> List[DecontaminatedSpectrum]:
        oid = object_id
        if oid not in self:
            raise KeyError(f"This {self.__class__.__name__} object does not contain {oid}.")

        if self._initialization_method == self.InitMethod.FROM_FILE and oid not in self._spectra_by_object_id:
            self._load_all_spectra_of_object(oid)

        return list(self._spectra_by_object_id[oid])

    def __contains__(self, object_id):
        if self._filtered_index is not None:
            return object_id in self._filtered_index
        else:
            return object_id in self._spectra_by_object_id

    def __iter__(self):
        if self._initialization_method == self.InitMethod.FROM_FILE:
            return (self._spectrum_from_index_entry(item) for item in self._filtered_index)
        else:
            return self._hdf5_spectra.values()

    def __len__(self):
        if self._initialization_method == self.InitMethod.FROM_FILE:
            return len(self._filtered_index)
        elif self._initialization_method == self.InitMethod.FROM_DECONTAMINATOR:
            return len(list(self._hdf5_spectra.values()))

    def __str__(self):
        if self._initialization_method == self.InitMethod.FROM_FILE:
            about = f"A {self.__class__.__name__} containing {len(self)} spectra, loaded from the following files:\n\n"
            for hdf_file in self._hdf5_files.values():
                for part in hdf_file:
                    about += f"{part.filename}\n"
        elif self._initialization_method == self.InitMethod.FROM_DECONTAMINATOR:
            about = f"A {self.__class__.__name__} containing {len(self)} spectra, constructed from a NestedDict."
        elif self._initialization_method == self.InitMethod.NONE:
            about = f"An empty {self.__class__.__name__}."

        return about

    def _cache_spectrum(self, spectrum: DecontaminatedSpectrum):
        """
        Stores a reference to the spectrum in the dictionaries: _spectra_by_object_id and _hdf5_spectra.

        Parameters
        ----------

        spectrum: DecontaminatedSpectrum
            The decontaminated spectrum that will be cached.
        """
        if spectrum.id not in self._spectra_by_object_id:
            self._spectra_by_object_id[spectrum.id] = set()

        self._spectra_by_object_id[spectrum.id].add(spectrum)

        self._hdf5_spectra[spectrum.dither, spectrum.detector, spectrum.id] = spectrum

    def _load_spectrum_from_file(self, dither, detector, object_id):
        h5files = self._hdf5_files[dither, detector]
        oid = str(object_id)
        decon_spec = None
        for part in h5files:
            spectra = part[SpectraCollectionIndex.DECONTAMINATED_SPECTRA_LABEL]
            if oid in spectra:
                decon_spec = DecontaminatedSpectrum(self, h5group=spectra[oid])
                break
        return decon_spec

    def _load_all_spectra_of_object(self, object_id):
        for entry in self._filtered_index[object_id]:
            self._cache_spectrum(self._spectrum_from_index_entry(entry))

    def _spectrum_from_index_entry(self, index_entry: SpectraCollectionIndexEntry) -> DecontaminatedSpectrum:
        return self.get_spectrum(index_entry.dither, index_entry.detector, index_entry.id)

    def _load_model_from_file(self, dither, detector, object_id, order) -> Union[ModelSpectrum, None]:
        h5files = self._hdf5_files[dither, detector]
        model = None
        oid = str(object_id)
        for part in h5files:
            models = part[DecontaminatedSpectraCollection.MODEL_SPECTRA_LABEL]
            if oid in models:
                model_dataset = models[oid][f"order_{order}"]
                model = ModelSpectrum(model_dataset)
                break
        return model

    def _set_metadata_from_cutouts(self, spectral_cutouts):
        """
        Sets the exposure / dither metadata.

        Side Effects
        ------------

        Modifies self._field_id, self._exposure_time, self._observation_id, self._pointing_id and
        self._exposure_grism_position
        """
        for dither in spectral_cutouts.dithers():
            if dither not in self._exposure_time:
                for detector in spectral_cutouts.keys(dither):
                    if len(spectral_cutouts[dither, detector]) > 0:
                        spec_cutout = spectral_cutouts[dither, detector][0]
                        spec_info = spec_cutout.spec_info
                        self._field_id = spec_info.field_id
                        self._exposure_time[dither] = spec_info.grism_model.exposure_time
                        self._observation_id[dither] = spec_info.observation_id
                        self._pointing_id[dither] = spec_info.pointing_id
                        self._exposure_grism_position[dither] = spec_info.grism_position
                        break

    def _populate_hdf5_spectra(self, spectral_cutouts):
        """
        Loads all objects in the spectral_cutouts nested dictionary.

        Parameters
        ----------

        spectral_cutouts: NestedDict
            A nested dict containing lists of SpectrumCutout objects, in the format:
            {dither: {detector: [SpectrumCutout]}}.

        Side Effects
        ------------

        Sets self._hdf5_spectra.
        """
        for dither, detector, cutouts in spectral_cutouts.items:
            for cutout in cutouts:
                # zero-out the surrounding pixels that are not part of the spectrum
                model = self.get_model(dither, detector, cutout.spec_info.object_id, cutout.spec_info.order)

                if model is not None:
                    mask = np.array(model.pixels > 1e-3, dtype=np.float32)
                    cutout.science = cutout.science * mask
                    cutout.flags = cutout.flags * mask.astype(np.uint32)
                    infinite_variance = ~np.isfinite(cutout.variance)
                    cutout.variance[infinite_variance] = 0.0
                    cutout.variance = cutout.variance * mask
                    cutout.variance[(mask > 0) & infinite_variance] = np.inf

                spectrum = DecontaminatedSpectrum(self, cutout)

                self._cache_spectrum(spectrum)

    def _populate_hdf5_models(self, spectral_cutouts):
        """
        Loads all model spectra referenced in the spectral_cutouts nested dictionary.

        Parameters
        ----------

        spectral_cutouts: NestedDict
            A nested dict containing lists of SpectrumCutout objects, in the format:
            {dither: {detector: [SpectrumCutout]}}

        Side Effects
        ------------

        Adds data to self._hdf5_models
        """
        for dither, detector, cutouts in spectral_cutouts.items:
            for cutout in cutouts:
                for contaminant in cutout.contaminants:
                    if contaminant.order != 0 and contaminant.model_spec is not None:
                        path = (dither, detector, contaminant.object_id, contaminant.order)
                        self._hdf5_models[path] = contaminant.model_spec
                        flux_1d = contaminant.one_dim_spectrum
                        self._hdf5_1d_models[path[:3]] = flux_1d
                        if self._one_dim_wavelengths is None:
                            self._one_dim_wavelengths = contaminant.one_dim_wavelengths

    def _populate_data_structures(self):
        """
        Loads data from HDF5 files into the internal data structures.

        This is only needed when saving files from data loaded from other files, such as when concatenating files.
        """
        self._one_dim_wavelengths = self._fetch_1d_wavelengths()
        for spec in self:
            dither = spec.dither
            detector = spec.detector
            oid = spec.id
            if spec.model is not None:
                model = spec.model
                order = model.order
                self._hdf5_models[dither, detector, oid, order] = model
                model_1d = self.get_1d_model(oid)
                self._hdf5_1d_models[dither, detector, oid] = model_1d
                self._one_dim_spectra_by_object_id[oid] = model_1d
            for contam_id, order in spec.contaminants:
                # handle the possible case in which a model does not have a corresponding decontaminated spectrum:
                if order > 0:
                    model = self.get_model(dither, detector, contam_id, order)
                    if model is not None:
                        self._hdf5_models[dither, detector, contam_id, order] = model
                        model_1d = self.get_1d_model(contam_id)
                        self._hdf5_1d_models[dither, detector, contam_id] = model_1d
                        self._one_dim_spectra_by_object_id[contam_id] = model_1d
            self._hdf5_spectra[dither, detector, oid] = spec

    def _load_json(self, data_dir: str, filename: str):
        """
        Loads all of the HDF5 files listed in the input JSON file.

        Parameters
        ----------

        filename: str
            The name of a JSON file containing the names of HDF5 files produced by a DecontaminatedSpectraCollection.

        data_dir: str
            The path to the directory containing the HDF5 files.
        """
        self._initialization_method = self.InitMethod.FROM_FILE

        with open(filename,'r') as f:
            for data_file_name in json.load(f):
                filename_extension = data_file_name.split('.')[-1]
                if filename_extension == 'xml':
                    xml_obj = DpdSirDecontaminatedSpectra(data_file_name)
                    self._load_hdf5(os.path.join(data_dir, xml_obj.get_hdf5_filename()))
                elif filename_extension in {'h5', 'hde', 'hdf', 'hdf5', 'he5'}:
                    self._load_hdf5(data_file_name)

    def _load_hdf5(self, filename):
        """
        Loads the contents of a single HDF5 file into this DecontaminatedSpectraCollection object.

        Parameters
        ----------

        filename: str
            The fully-qualified name (absolute path) of an HDF5 file containing a DecontaminatedSpectraCollection for
            one NISP detector.

        Raises
        ------

        ValueError
           If the filename is not a fully-qualified filename (absolute path) or the format of the file is incorrect, or
           if values within the file are not in the expected ranges.
        """
        h5_file = h5py.File(filename, 'r')

        self._load_metadata_from_hdf5(h5_file)

        if self._complete_index is None:
            self._complete_index = SpectraCollectionIndex(h5_file)
            self._filtered_index = self._complete_index
        else:
            self._complete_index.load_index_from_hdf5(h5_file)

    def _load_metadata_from_hdf5(self, h5_file):
        if h5_file.attrs['file_format'] != DecontaminatedSpectraCollection.FILE_FORMAT_NAME:
            raise ValueError(f"The hf_file is not a valid DecontaminatedSpectrumCollection file.")

        dither = h5_file.attrs[H5LocationTable.DITHER_LABEL]

        detector = int(h5_file.attrs['Detector'])

        opt = h5_file[H5OpticalModel.MODEL_LABEL]

        self._detector_models[(dither, detector)] = H5OpticalModel.new_from_HDF5_group(opt).det_model

        if detector < 1 or detector > 16:
            raise ValueError(f'The detector value, {detector}, is out of the expected range.')

        exposure_time = h5_file.attrs[H5LocationTable.EXPTIME_LABEL]

        if exposure_time <= 0.0:
            raise ValueError(f'The exposure time must be > 0.0 seconds.')

        self._exposure_time[dither] = exposure_time
        self._exposure_grism_position[dither] = GrismModel.getPositionNumber(h5_file.attrs[H5LocationTable.GWA_POSITION_LABEL])
        self._observation_id[dither] = h5_file.attrs[H5LocationTable.OBSERVATION_ID_LABEL]
        self._pointing_id[dither] = h5_file.attrs[H5LocationTable.POINTING_ID_LABEL]

        field_id = h5_file.attrs[H5LocationTable.FIELD_ID_LABEL]

        if self._field_id is not None:
            if self._field_id != field_id:
                raise ValueError("Encountered a new field ID; The DecontaminatedSpectraCollection can only "
                                 "contain spectra from a single field.")
        else:
            self._field_id = field_id

        if (dither, detector) not in self._hdf5_files:
            self._hdf5_files[dither, detector] = []

        self._hdf5_files[dither, detector].append(h5_file)

    def _save_hdf5_v2(self, filename, loctab_filename, dither, detector, overwrite=False):
        """
        Saves an HDF5 file containing all of the decontaminated spectra for the specified dither and detector.

        Parameters
        ----------

        filename: str
            The name of the HDF5 file that will be written.

        dither: int
            The dither index, an integer (1, 2, 3, 4), specifying the exposure number within the dither sequence.

        detector: int
           The detector on which the spectrum of interest is located (1, ..., 16).

        overwrite: bool (optional)
            Specifies whether or not to overwrite an existing file with the name specified by `filename`.
            Default: False
        """
        if os.path.exists(filename):
            if overwrite:
                self._logger.info(f"{filename} already exists; overwriting.")
                os.remove(filename)
            else:
                err_message = f"{filename} already exists and 'overwrite' is set to False."
                self._logger.error(err_message)
                raise FileExistsError(err_message)

        h5_file = h5py.File(filename, 'w')

        h5_file.attrs['file_format'] = DecontaminatedSpectraCollection.FILE_FORMAT_NAME
        h5_file.attrs[H5LocationTable.FIELD_ID_LABEL] = self.field_id
        h5_file.attrs[H5LocationTable.OBSERVATION_ID_LABEL] = self.get_observation_id(dither)
        h5_file.attrs[H5LocationTable.POINTING_ID_LABEL] = self.get_pointing_id(dither)
        h5_file.attrs[H5LocationTable.GWA_POSITION_LABEL] = GrismModel.getPositionName(self.get_grism_position(dither))
        h5_file.attrs[H5LocationTable.DITHER_LABEL] = dither
        h5_file.attrs['Detector'] = detector
        h5_file.attrs['DetectorId'] =  DetectorModel.getNumbers2IDs()[detector]
        h5_file.attrs[H5LocationTable.EXPTIME_LABEL] = self.get_exposure_time(dither)

        if loctab_filename is not None:
            loctab_file = h5py.File(loctab_filename, 'r')
            for att  in [H5LocationTable.GWA_TILT_LABEL,
                         H5LocationTable.PATCH_ID_LABEL,
                         H5LocationTable.EXPOSURE_LABEL,
                         H5LocationTable.TOTAL_EXPOSURE_LABEL,
                         H5LocationTable.UTC_LABEL,
                         H5LocationTable.OBT_LABEL,
                         H5LocationTable.MJD_LABEL] :
                h5_file.attrs[att] = loctab_file.attrs[att]
    
            loctab_file.copy(H5GlobalCRVModel.MODEL_LABEL, h5_file)
            loctab_file.copy(H5GlobalIDSModel.MODEL_LABEL, h5_file)
            loctab_file.copy(H5OpticalModel.MODEL_LABEL, h5_file)
        else:
            self._logger.info(f"Location tables were not provided; location information will not be saved.")

        self._add_spectra_to_hdf5_file(h5_file, dither, detector)

        self._add_models_to_hdf5_file(h5_file, dither, detector)

        self._add_1d_model_spectra_to_hdf5_file(h5_file, dither, detector)

        SpectraCollectionIndex.add_index_to_hdf5(h5_file)

        h5_file.close()

        self._logger.info(f"Saved {filename}.")

    def _add_spectra_to_hdf5_file(self, hdf5_file, dither, detector):
        """
        Adds all decontaminated spectra for the specified dither and detector to the specified HDF5 file.

        Parameters
        ----------

        hdf5_file: H5PY File object
            The HDF5 file object (referencing the root of the file).

            dither: int
            The dither index, an integer (1, 2, 3, 4), specifying the exposure number within the dither sequence.

        detector: int
           The detector on which the spectrum of interest is located (1, ..., 16).
        """
        spectra_group = hdf5_file.create_group(SpectraCollectionIndex.DECONTAMINATED_SPECTRA_LABEL)

        spectra_group.attrs['info'] = ("Contains a set of decontaminated spectral cutouts as well as a table "
                                       "listing the spectra which were subtracted from each cutout.")

        for spec in self._hdf5_spectra.values(dither, detector):
            self._add_spectrum_to_hdf5_group(spectra_group, spec)

    def _add_models_to_hdf5_file(self, hdf5_file, dither, detector):
        """
        Adds all model spectra for the specified dither and detector to the specified HDF5 file.

        Parameters
        ----------

        hdf5_file: H5PY File object
            The HDF5 file object (referencing the root of the file).

        dither: int
            The dither index, an integer (1, 2, 3, 4), specifying the exposure number within the dither sequence.

        detector: int
           The detector on which the spectrum of interest is located (1, ..., 16).
        """
        models = self._hdf5_models.get_nested_dict(dither, detector, plural_key='ob_ids')

        model_group = hdf5_file.create_group(DecontaminatedSpectraCollection.MODEL_SPECTRA_LABEL)

        model_group.attrs['info'] = ("Contains the model spectra that were subtracted away from the original "
                                     "spectra. These are referenced in the table of 'contaminants' associated "
                                     "with each decontaminated spectrum cutout.")
        self._add_models_to_hdf5_group(model_group, models)

    def _ensure_initialization_from_file(self):
        if self._initialization_method != self.InitMethod.FROM_FILE:
            raise RuntimeError("This functionality is only available when loading the collection from a file.")

    def _add_1d_model_spectra_to_hdf5_file(self, hdf5_file, dither, detector):
        """
        Adds all model 1D spectra for the specified dither and detector to the specified HDF5 file.

        Parameters
        ----------

        hdf5_file: H5PY File object
            The HDF5 file object (referencing the root of the file).

        dither: int
            The dither index, an integer (1, 2, 3, 4), specifying the exposure number within the dither sequence.

        detector: int
           The detector on which the spectrum of interest is located (1, ..., 16).
        """
        models = self._hdf5_1d_models.get_subdict(dither, detector)

        model_group = hdf5_file.create_group(DecontaminatedSpectraCollection.MODEL_1D_SPECTRA_LABEL)

        if self._one_dim_wavelengths is not None:
            wavelength_set = model_group.create_dataset(f"wavelengths", data=self._one_dim_wavelengths)
            wavelength_set.attrs['unit'] = 'Angstroms'

        model_group.attrs['info'] = ("Contains the 1-dimensional version of the model spectra that were subtracted"
                                     "away from the original spectra.")
        model_group.attrs['flux_unit'] = 'erg/s/cm^2/Angstrom'
        self._add_1d_models_to_hdf5_group(model_group, models)

    def _fetch_1d_spec(self, object_id: int) -> Union[np.ndarray, None]:
        spec_1d = None
        path = f'{object_id}/flux'
        for h5_files in self._hdf5_files.values():
            for part in h5_files:
                group = part[self.MODEL_1D_SPECTRA_LABEL]
                if path in group:
                    spec_1d = np.array(group[path])
                    break
        return spec_1d

    def _fetch_1d_wavelengths(self) -> Union[np.ndarray, None]:
        for h5_files in self._hdf5_files.values():
            group = h5_files[0][self.MODEL_1D_SPECTRA_LABEL]
            if 'wavelengths' in group:
                return np.array(group['wavelengths'])
        return None

    @staticmethod
    def _add_spectrum_to_hdf5_group(group, spectrum):
        spec_container = group.create_group(str(spectrum.id))
        spec_container.attrs['x_offset'] = spectrum.x_offset
        spec_container.attrs['y_offset'] = spectrum.y_offset
        spec_container.attrs['object_id'] = spectrum.id
        spec_container.attrs['y_mag'] = spectrum.y_mag
        spec_container.attrs['j_mag'] = spectrum.j_mag
        spec_container.attrs['h_mag'] = spectrum.h_mag
        spec_container.attrs['y_flux'] = spectrum.y_flux
        spec_container.attrs['j_flux'] = spectrum.j_flux
        spec_container.attrs['h_flux'] = spectrum.h_flux

        spec_container.create_dataset('science', data=spectrum.science,
                                      compression=DecontaminatedSpectraCollection.COMPRESSION,
                                      compression_opts=DecontaminatedSpectraCollection.COMPRESS_LEVEL,
                                      shuffle=True)

        spec_container.create_dataset('variance', data=spectrum.variance,
                                      compression=DecontaminatedSpectraCollection.COMPRESSION,
                                      compression_opts=DecontaminatedSpectraCollection.COMPRESS_LEVEL,
                                      shuffle=True)

        spec_container.create_dataset('mask', data=spectrum.mask,
                                      compression=DecontaminatedSpectraCollection.COMPRESSION,
                                      compression_opts=DecontaminatedSpectraCollection.COMPRESS_LEVEL,
                                      shuffle=True)

        if spectrum.kernel is not None:
            kernel = spec_container.create_dataset('kernel', data=spectrum.kernel,
                                                   compression=DecontaminatedSpectraCollection.COMPRESSION,
                                                   compression_opts=DecontaminatedSpectraCollection.COMPRESS_LEVEL,
                                                   shuffle=True)
            if None not in (spectrum.gaussian_sigma_major, spectrum.gaussian_sigma_minor, spectrum.gaussian_theta):
                kernel.attrs['gaussian_sigma_major'] = spectrum.gaussian_sigma_major
                kernel.attrs['gaussian_sigma_minor'] = spectrum.gaussian_sigma_minor
                kernel.attrs['gaussian_theta'] = spectrum.gaussian_theta

        spec_container.create_dataset('contaminants', data=spectrum.contaminants)
        spec_container.create_dataset('boundary', data=spectrum.boundary)
        location = spec_container.create_group("location")
        location.attrs['order'] = spectrum.location.getOrder()
        location.attrs['wavelength_start'] = spectrum.location.getWavelengthStart()
        location.attrs['wavelength_end'] = spectrum.location.getWavelengthEnd()
        location.attrs['gwa_tilt'] = spectrum.location.getGWATilt()
        location.attrs['extra_tilt'] = spectrum.location.getExtraTilt()
        x_pivot, y_pivot = spectrum.location.getPivot()
        location.attrs['x_pivot'] = x_pivot
        location.attrs['y_pivot'] = y_pivot
        ref_position = location.create_dataset('ref_position', data=spectrum.location.getLambdaRefPosition())
        ref_position.attrs['ref_wavelength'] = spectrum.location.getLambdaRefValue()

        ids_model = spectrum.location.getIDSModel()
        ids_coeffs = location.create_dataset('ids_coeffs', data=ids_model.getCoeffs())
        ids_coeffs.attrs['wavelength_start'] = ids_model.getStart()
        ids_coeffs.attrs['wavelength_end'] = ids_model.getStop()

        crv_model = spectrum.location.getCRVModel()
        crv_coeffs = location.create_dataset('crv_coeffs', data=crv_model.getCoeffs())
        crv_coeffs.attrs['wavelength_start'] = crv_model.getStart()
        crv_coeffs.attrs['wavelength_end'] = crv_model.getStop()

    @staticmethod
    def _add_models_to_hdf5_group(group, models):
        """
        Adds a set of models to an HDF5 group.

        Parameters
        ----------

        group: H5py Group
            The HDF5 group into which the models will be added.

        models: NestedDict
            A nested dictionary in the format {object_id: {order: ModelSpectrum}}
        """
        if models is not None and models.root is not None:
            for object_id in models.ob_ids():
                model_container = group.create_group(str(object_id))
                model_container.attrs['object_id'] = object_id
                for model_spec in models.values(object_id):
                    if model_spec.pixels is not None:
                        order = model_container.create_dataset(f"order_{model_spec.order}",
                                                               data=model_spec.pixels,
                                                               compression=DecontaminatedSpectraCollection.COMPRESSION,
                                                               compression_opts=DecontaminatedSpectraCollection.COMPRESS_LEVEL,
                                                               shuffle=True)
                        order.attrs['x_offset'] = model_spec.x_offset
                        order.attrs['y_offset'] = model_spec.y_offset
                        order.attrs['order'] = model_spec.order
                        order.attrs['object_id'] = object_id
                        order.attrs['interp'] = model_spec.interpolated

    @staticmethod
    def _add_1d_models_to_hdf5_group(group, models):
        """
        Adds a set of 1D models to an HDF5 group.

        Parameters
        ----------

        group: H5py Group
            The HDF5 group into which the models will be added.

        models: NestedDict
            A nested dictionary in the format {object_id: {order: ModelSpectrum}}
        """
        if models is not None and len(models) > 0:
            for object_id, flux in models.items():
                model_container = group.create_group(str(object_id))
                model_container.attrs['object_id'] = int(object_id)
                model_container.create_dataset(f"flux",
                                               data=flux,
                                               shuffle=True)
