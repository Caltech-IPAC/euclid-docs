[docs]class PsfModel:
    """Extracts the parameters for the two-component Gaussian fit to the wavelength-dependent PSF."""

    def __init__(self, filenames: List[str], pixel_scale: float):
        self._pixel_scale = pixel_scale
        self._psf_tables = self._make_psf_tables(filenames)


[docs]    def get_parameters(self,
                       grism: str,
                       detector_id: str,
                       wavelength: float) -> Tuple[Tuple[float, float], Tuple[float, float]]:
        """
        Determines the parameters of the model PSF, given as a mixture of 2D Gaussians

        Parameters
        ----------

        grism: str
            The name of the grism.

        detector_id: str
            The detector ID number, as a string (example: '11', '23', '44')

        wavelength: float
            The wavelength, in units of Angstroms.

        Returns
        -------

        psf_parameters: Tuple[Tuple[float, float], Tuple[float, float]]

        A tuple of ((psf1_amplitude, sigma1), (psf2_amplitude, sigma2)).
        """
        table = self._psf_tables[detector_id][grism]

        wavs, amp, sig1, sig2 = table.T

        amp_interp = float(np.interp(wavelength, wavs, amp))
        sig1_interp = float(np.interp(wavelength, wavs, sig1))
        sig2_interp = float(np.interp(wavelength, wavs, sig2))

        return (amp_interp, sig1_interp), (1 - amp_interp, sig2_interp)


    def _make_single_table(self, filename: str) -> Dict[str, np.ndarray]:
        psf_parameters = dict()
        with fits.open(filename) as f:
            for ext in f:
                head = ext.header
                if 'ORDER' in head and head['ORDER'] == 1:
                    grism_id = head['GRISM_ID']
                    if grism_id not in psf_parameters:
                        psf_parameters[grism_id] = []

                    wavelength_in_angstroms = 1e4 * head['WLTH_WFM']
                    amplitude = head['G2PSFA']
                    sigma_1 = head['G2PSFS1'] / self._pixel_scale
                    sigma_2 = head['G2PSFS2'] / self._pixel_scale
                    psf_parameters[grism_id].append((wavelength_in_angstroms, amplitude, sigma_1, sigma_2))

        for grism in psf_parameters:
            psf_parameters[grism] = np.array(psf_parameters[grism])

        return psf_parameters

    def _make_psf_tables(self, files):
        psf_table = dict()
        for filename in files:
            detector_id = self._get_detector_id(filename)
            psf_table[detector_id] = self._make_single_table(filename)

        return psf_table

    @staticmethod
    def _get_detector_id(filename: str):
        if re.match(r'.*_RGSASBUILD_PSFs_..\.fits$', filename) is None:
            raise ValueError("The PSF filename does not match the expected pattern.")
        return filename[-7:-5]
