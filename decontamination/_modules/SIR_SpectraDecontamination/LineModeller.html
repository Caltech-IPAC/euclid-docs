"""
Contains the LineModeller class, which is used for finding and modelling emission lines in an observed spectrum.
"""

[docs]class LineModeller(object):
    """
    The LineModeller class creates models of spectral emission lines.

    The class takes segments of 1-D spectra from multiple dithers, searches for the presence of lines, and
    creates models of any lines that are detected.
    """

    def __init__(self, measured_spectra: List[Tuple[np.ndarray]]):
        """
        Stores the flux arrays and corresponding wavelengths of the spectra that may potentially
        contain emission lines, as well as the uncertainty of each flux value.

        Parameters
        ----------

        measured_spectra: List[Tuple[np.ndarray]]
            A list of tuples containing three 1D NumPy arrays: (wavelengths, fluxes, flux_uncertainties).
        """
        self._input_spectra = measured_spectra
        self._line_models = []
        self._models_created = False

    @property
    def line_models(self):
        if not self._models_created:
            self._make_models()
        return None if len(self._line_models) == 0 else self._line_models


[docs]    def mask(self, wavelengths):
        """
        Indicates whether or not the specified wavelengths correspond to emission line regions.

        Parameters
        ----------

        wavelengths: 1-D NumPy array
            The wavelengths that will be checked.

        Returns
        -------

        mask: NumPy array of bools
            For each entry in the `wavelengths` array, if the wavelength is within an emission line region, the entry
            in the mask is False; otherwise the entry is True.

        Raises
        ------

        TypeError: if the wavelengths parameter is not of the expected type.
        """
        if not isinstance(wavelengths, np.ndarray):
            raise TypeError("Expected wavelengths to be a NumPy array")

        result = np.full(wavelengths.shape, True)

        if not self._models_created:
            self._make_models()

        if len(self._line_models) == 0:
            return result

        for line in self._line_models:
            min_wav, max_wav = line.wavelength_range
            line_mask = (wavelengths > min_wav) & (wavelengths < max_wav)
            result[line_mask] = False

        return result


    def _make_models(self):
        """
        Combines lines from different observations (if possible).

        Side Effects
        ------------

        sets `self._models_created` and `self._line_models`
        """
        if self._models_created:
            return

        line_candidates = []

        for wavelength, flux, uncertainty in self._input_spectra:
            line_candidates += self._find_lines(wavelength, flux, uncertainty)

        grouped_line_candidates = self._group_lines(line_candidates)
        grouped_lines = self._reject_spurious_lines(grouped_line_candidates)

        for line_group in grouped_lines:
            wavelengths = [observation[0] for observation in line_group]
            fluxes = [observation[1] for observation in line_group]
            self._append_line_model(self._line_models, wavelengths, fluxes)

        self._models_created = True

    def _group_lines(self, raw_lines: List[Tuple[np.ndarray]]) -> List[List[Tuple[np.ndarray]]]:
        """
        Takes the set of all spectral regions that appear to contain lines and groups them so that
        all segments corresponding to a unique line are grouped together into a list.

        Parameters
        ----------
        raw_lines: List[Tuple[np.ndarray]]

        Returns
        -------

        line_candidates: List[List[Tuple[np.ndarray]]]
            A list containing lists of tuples of (wavelength, flux, uncertainty). The tuples in
            the sub-lists contain spectral segments that all correspond to the same emission line.
        """
        all_lines = [{'raw_line': line, 'has_been_sorted': False} for line in raw_lines]

        line_candidates = []

        for line in all_lines:
            if not line['has_been_sorted']:
                line['has_been_sorted'] = True
                this_line = line['raw_line']
                this_group = [this_line]
                for other in all_lines:
                    if other is not line:
                        other_line = other['raw_line']
                        if self._lines_match(this_line, other_line) and not other['has_been_sorted']:
                            this_group.append(other_line)
                line_candidates.append(this_group)

        return line_candidates

    def _reject_spurious_lines(self,
                               line_candidates: List[List[Tuple[np.ndarray]]]) -> List[List[Tuple[np.ndarray]]]:
        """
        Rejects a line if it appears to not be real.

        A line is only considered real if it is observed in more than one dither, except in the
        case in which the peak flux is 10 x larger than the flux uncertainty. If a line is only
        detected in a single dither, but it could have been detected in more than one dither,
        then it is spurious, even if it has a 10-sigma singnificance.

        Parameters
        ----------
        line_candidates: List[List[Tuple[np.ndarray]]]
            The output of the _group_lines() method.

        Returns
        -------

        real_lines: List[List[Tuple[np.ndarray]]]
            The same as the input, but lines that are determined to be spurious have been removed.
        """
        real_lines = []

        for candidate in line_candidates:
            if len(candidate) == 1:
                wavelengths, fluxes, uncertainties = candidate[0]
                peak_index = fluxes.argmax()
                peak_wav = wavelengths[peak_index]
                maximum_possible_detections = self._count_successful_measurements(peak_wav)
                if maximum_possible_detections < 2:
                    peak_flux = fluxes[peak_index]
                    uncertainty_at_peak = uncertainties[peak_index]
                    significance = peak_flux / uncertainty_at_peak
                    if significance >= 10.0:
                        real_lines.append(candidate)
            else:
                real_lines.append(candidate)

        return real_lines

    def _count_successful_measurements(self, wavelength: float) -> int:
        """
        Counts the number of times that the specified wavelength was successfully measured.

        Parameters
        ----------

        wavelength: float
            The wavelength of interest

        Returns
        -------

        number_of_measurements: int
            The number of times that the specified wavelength occurs in a region of finite uncertainty
            in self._input_spectra.
        """
        count = 0

        for wav, _, uncertainty in self._input_spectra:
            if len(wav) > 0:
                nearest_index = np.argmin(np.fabs(wav - wavelength))
                if nearest_index < 5 or len(wav) - nearest_index < 6:
                    # the line couldn't have been captured in this dither because it's on the edge or beyond the edge
                    # of the available region.
                    continue
                if np.all(np.isfinite(uncertainty[nearest_index - 4: nearest_index + 5])):
                    count += 1
        return count

    @staticmethod
    def _lines_match(raw_line1: Tuple[np.ndarray], raw_line2: Tuple[np.ndarray]) -> bool:
        """
        Determines whether the two input line candidates are observations of the same emission line.

        Parameters
        ----------
        raw_line1: Tuple[np.ndarray]
            The first line, as a tuple of three NumPy arrays: (wavelengths, fluxes, uncertainties)

        raw_line2: Tuple[np.ndarray]
            The first line, as a tuple of three NumPy arrays: (wavelengths, fluxes, uncertainties)

        Returns
        -------

        match: bool
            True if raw_lines1 and raw_lines2 appear to be two observations of the same line.
            Otherwise, false.
        """
        wavs1, fluxes1, _ = raw_line1
        wavs2, fluxes2, _ = raw_line2

        peak_wav1 = float(wavs1[fluxes1.argmax()])
        peak_wav2 = float(wavs2[fluxes2.argmax()])
        dwav = 2.5 * math.fabs(wavs1[1] - wavs1[0])

        return math.fabs(peak_wav1 - peak_wav2) < dwav

    @staticmethod
    def _append_line_model(model_list: List[LineModel],
                           wavelengths: List[np.ndarray],
                           fluxes: List[np.ndarray]):
        """
        Attempts to create a `LineModel` from the input `wavelengths` and `fluxes`. If this succeeds, the `LineModel`
        object is appended to the list of models, `model_list`; otherwise nothing happens.

        Parameters
        ----------

        model_list: list
            A (potentially empty) list of `LineModel` objects.

        wavelengths: list
           A list of wavelength vectors [wavelengths1, wavelengths2]. The list must contain at least one array of
           wavelengths.

        fluxes: list
            A list of the flux vectors corresponding to the wavelength vectors listed in wavelengths. This is expected
            be given in units of detector counts per Angstrom.

        Side Effects
        ------------

        Appends a `LineModel` object to the `model_list` if a LineModel can be created from the input wavelengths and
        fluxes.
        """
        try:
            line_model = LineModel(wavelengths, fluxes)
        except ValueError:
            line_model = None

        if line_model is not None:
            model_list.append(line_model)

    @classmethod
    def _find_lines(cls, wavelengths, flux_density, flux_uncertainty):
        """
        Identifies regions of a collapsed 2D spectrum that contain emission lines and returns the wavelengths and
        flux densities in the line regions.

        Parameters
        ----------

        wavelengths: 1-D NumPy array
            Wavelengths of the entries in `flux_density`, in Angstroms.

        flux_density: 1-D NumPy array
            The flux densities of the spectrum, measured in detector counts per Angstrom.

        flux_uncertainty: 1-D NumPy array
            The uncertainty of each entry in the `flux_density` array.

        Returns
        -------

        lines: list
            Returns a list of tuples of NumPy arrays, which looks like this:

                [(wavelengths1, fluxes1, uncertainties1), (wavelengths2, fluxes2, uncertainties2), ...,
                 (wavelengthsN, fluxesN, uncertaintiesN)]

           where the wavelengths, fluxes, and flux uncertainties are NumPy arrays.
        """
        indices = cls._find_spikes(flux_density, spike_width=2, uncertainty=flux_uncertainty, padding=4)

        # if there are obviously no lines, return empty list

        if len(indices) == 0:
            return []

        # identify indices of the borders between separate lines:

        separators = np.where(np.diff(indices) > 1)[0]

        lines = [indices] if len(separators) == 0 else np.split(indices, separators + 1)

        output = []

        for line_indices in lines:
            if len(line_indices) > 1:
                line_flux = flux_density[(line_indices,)]
                clipped_line_flux = np.clip(line_flux, 0, None)
                uncertainty = flux_uncertainty[(line_indices,)]
                wavs = wavelengths[(line_indices,)]
                if wavs.min() > DetectorGrism.MIN_WAV_INNER and wavs.max() < DetectorGrism.MAX_WAV_INNER:
                    # avoid anomaly at the end-points
                    output.append((wavelengths[(line_indices,)], clipped_line_flux, uncertainty))

        return output

    @staticmethod
    def _find_spikes(signal, uncertainty, spike_width, padding):
        """
        Locates regions of a signal that exceed the local mean by 5 times the local flux uncertainty.

        Parameters
        ----------

        signal: np.ndarray
            Contains the raw, noisy signal, which potentially contains a spike (a 5-sigma outlier).

        uncertainty: np.ndarray
            The expected standard deviation of the noise in the signal, in the absence of any spikes.

        spike_width: int
            The expected width of the spike in the signal.

        padding: int
            The number of samples to the left and right of the spike that should be included in the output.

        Returns
        -------

        spike_indices: NumPy array of integers
            A list of integer indices corresponding to the locations of spikes in the signal. This includes the
            central position of the spike as well as +/- `padding` samples to the left and right of the spike.
        """

        if spike_width < 1:
            raise ValueError('spike_width must be >= 1.')

        # perform median filtering to smooth the uncertainty.

        width = 3 * spike_width

        # the kernel size for the median filter has to be odd, so:

        kernel_size = width + 1 if width % 2 == 0 else width

        smoothed_uncertainty = medfilt(uncertainty, kernel_size)

        # identify regions in which the signal exceeds the the uncertainty by 5 sigma

        spike = (signal - 5 * smoothed_uncertainty) > 0

        conv_kernel = np.ones(spike_width) / spike_width

        try:
            spike_indices = np.where(np.convolve(spike, conv_kernel, mode='same') == 1)[0]
            has_spike = len(spike_indices) > 0
        except ValueError:
            has_spike = False

        if not has_spike:
            return np.array([])

        indices = set()

        imin = spike_width + 1
        imax = len(signal) - spike_width

        for idx in spike_indices:
            ileft = max(imin, idx - padding)
            iright = min(imax, idx + padding)
            for i in range(ileft, iright):
                indices.add(i)

        index_list = list(indices)
        index_list.sort()

        return np.array(index_list)
