    MIN_WAV = 12450.0

    def __init__(self, decon_spectra: 'DecontaminatedSpectraCollection'):
        if decon_spectra.mdb is None:
            raise RuntimeError("The spectra collection needs to contain a reference to the Mdb.")

        self._collection = decon_spectra
        self._quality_params = {}



[docs]    def add_statistics(self, dither: int, detector: int):
        """Compute the statistics to the decontaminated spectra collection data product

        Args:
            dither (int): The dither number (1-4) of the data product
            detector (int): The detector number (1-16) of the data product
        """

        # number of spectra
        name = 'n_spectra'
        self._quality_params[name] = {'Key': name,
                                      'Value':len(list(self._collection.get_spectra(dither, detector))),
                                      'Description': 'Total number of decontaminated spectra in this file',
                                      'Unit': ''}

        self._add_mean_contaminants(dither, detector)
        self._add_interpolated_fraction(dither, detector)
        self._add_median_residual_statistics(dither, detector)
        self._add_residual_significance_statistics(dither, detector)




    def _add_mean_contaminants(self, dither:int, detector:int) -> None:
        """Computes the mean number of contaminants per spectrum and adds this to the data product.

        Args:
            dither (int): The dither number (1-4) of the data product
            detector (int): The detector number (1-16) of the data product
        """
        n_contams = 0
        n_specs = 0

        for spec in self._collection.get_spectra(dither, detector):
            n_contams += spec.contaminants.size
            n_specs += 1
        value = 0.0 if not n_specs else n_contams / n_specs

        name = 'mean_contaminants'
        self._quality_params[name] = {'Key': name,
                                      'Value': value,
                                      'Description': 'Mean number of contaminants per decontaminated spectrum',
                                      'Unit': ''}


    def _add_interpolated_fraction(self, dither: int, detector: int) -> None:
        """Computes the fraction of spectral models that were computed using only photometry.

        Args:
            dither (int): The dither number (1-4) of the data product
            detector (int): The detector number (1-16) of the data product
        """
        n_interpolated = 0
        n_total = 0

        for spec in self._collection.get_spectra(dither, detector):
            if spec.model is not None:
                n_total += 1
                if spec.model.interpolated:
                    n_interpolated += 1

        interpolated_fraction = n_interpolated
        if n_total > 0:
           interpolated_fraction = n_interpolated / n_total

        name = 'interpolated_frac'
        self._quality_params[name] = {'Key': name,
                                      'Value': interpolated_fraction,
                                      'Description': 'Fraction of model spectra created using interpolating photometry',
                                      'Unit': ''}

        name = 'measured_frac'
        self._quality_params[name] = {'Key': name,
                                      'Value': 1.0 - interpolated_fraction,
                                      'Description': 'Fraction of model spectra created using spectral measurements',
                                      'Unit': ''}


    def _add_median_residual_statistics(self, dither: int, detector: int) -> None:
        """Adds statistics regarding the median pixel residual med(decontaminated - model) distribution.

        Args:
            dither (int): The dither number (1-4) of the data product
            detector (int): The detector number (1-16) of the data product
        """
        median_residuals = []
        for spec in self._collection.get_spectra(dither, detector):
            if spec.model is not None:
                residual = spec.science - spec.model.pixels
                mask = (spec.model.pixels > 0.0) & np.isfinite(residual) & ~np.isnan(residual)
                median_residuals.append(np.median(residual[mask]))

        median_residuals = np.array(median_residuals)[~np.isnan(median_residuals)]

        if len(median_residuals) == 0:
            median_residuals = [-9e9]

        dist_description = 'the distribution of median(science - model).'

        name = 'q1_residual'
        self._quality_params[name] = {'Key': name,
                                      'Value': np.quantile(median_residuals, 0.25),
                                      'Description':f'Lower quartile (25th percentile) of {dist_description}',
                                      'Unit': ''}
        name = 'median_residual'
        self._quality_params[name] = {'Key': name,
                                      'Value': np.median(median_residuals),
                                      'Description': f'Median value of {dist_description}',
                                      'Unit': ''}
        name = 'q3_residual'
        self._quality_params[name] = {'Key': name,
                                      'Value':  np.quantile(median_residuals, 0.75),
                                      'Description': f'Upper quartile (75th percentile) of {dist_description}',
                                      'Unit': ''}


    def _add_residual_significance_statistics(self, dither: int, detector: int) -> None:
        """Computes the mean of the median residual significance: med(decontaminated - model) / sigma).
        
        Args:
            dither (int): The dither number (1-4) of the data product
            detector (int): The detector number (1-16) of the data product
        """
        median_residual = []
        residual_significance = []
        median_noise = []
        for spec in self._collection.get_spectra(dither, detector):
            if spec.model is not None:
                nan_mask = np.isnan(spec.variance)
                variance = np.copy(spec.variance)
                variance[nan_mask] = np.inf
                variance[variance <= 0] = np.inf
                residual = spec.science - spec.model.pixels
                significance = residual / np.sqrt(variance)
                finite = np.isfinite(variance) & np.isfinite(residual) & np.isfinite(significance)
                mask = (spec.model.pixels > 0.0) & finite
                if mask.sum() > 0:
                    med_res = np.median(residual[mask])
                    res_sig = np.median(significance[mask])
                    med_noise = np.median(np.sqrt(variance[mask]))
                    if not np.any(np.isnan([med_res, res_sig, med_noise])):
                        median_residual.append(med_res)
                        residual_significance.append(res_sig)
                        median_noise.append(med_noise)

        residual_significance = np.array(residual_significance)[~np.isnan(residual_significance)]

        if len(residual_significance) == 0:
            residual_significance = [-9e9]

        if len(median_noise) == 0:
            median_noise = [0.0]

        median_residual = np.array(median_residual)
        median_noise = np.array(median_noise)

        dist_description = 'the distribution of median(science - model) / sigma.'

        name = 'mean_residual_significance'
        self._quality_params[name] = {'Key': name,
                                      'Value': np.mean(residual_significance),
                                      'Description': f'Mean of {dist_description}',
                                      'Unit': ''}

        name = 'median_residual_significance'
        self._quality_params[name] = {'Key': name,
                                      'Value': np.median(residual_significance),
                                      'Description': f'Median of {dist_description}',
                                      'Unit': ''}

        name = 'median_sigma'
        self._quality_params[name] = {'Key': name,
                                      'Value': np.median(median_noise),
                                      'Description': 'Median of the sqrt(variance)',
                                      'Unit': ''}

        within_sigma = np.fabs(median_residual) < median_noise
        if within_sigma.size > 0:
            one_sigma_fraction = sum(within_sigma) / within_sigma.size
        else:
            one_sigma_fraction = -1.0

        name = 'fraction_within_sigma'
        self._quality_params[name] = {'Key': name,
                                      'Value': one_sigma_fraction,
                                      'Description': 'The fraction of median residuals within 1-sigma of zero',
                                      'Unit': ''}


        status = within_sigma.size > 0 and one_sigma_fraction > 0.9

        status_message = "PASS: No major issues were detected."

        if not status:
            status_message = "FAIL: A major issue was detected; "
            if within_sigma.size:
                status_message += "no finite (non-NaN) spectra were found. "

            if 0.0 < one_sigma_fraction <= 0.9:
                status_message += "The residuals are not consistent with the uncertainty."

        name = 'QUALITY_STATUS'
        self._quality_params[name] = {'Key': name,
                                      'Value': status,
                                      'Description': status_message,
                                      'Unit': ''}
