    """
    Measures 1-D spectra from 2-D spectra, potentially by combining data from multiple dithers.
    """

    # the nominal width of a wavelength bin in units of Angstroms
    WAVELENGTH_BIN_SIZE = 1500

    def __init__(self, spectrum_modeller):
        """
        Constructs the SpectrumMeasurer.

        Parameters
        ----------

        spectrum_modeller: SpectrumModeller
            The object responsible for measuring spectra
        """
        self._modeller = spectrum_modeller


[docs]    def measure_spectra(self, object_info: 'AstroObjectInfo') -> Union[Tuple[np.ndarray, LineModeller], None]:
        """
        Measures all of the spectra associated with the object_info.

        Parameters
        ----------
        object_info: AstroObjectInfo
            The AstroObjectInfo containing all information about the star or galaxy of interest.

        Returns
        -------

        measurements: tuple or None
            Returns None if no measurements could be made. Otherwise, returns a tuple of
            (continuum_flux, emission_lines), where:

            continuum_flux: np.ndarray
                A 1-D array of the fluxes at the specified wavelengths, representing the smooth continuum of
                the spectrum.

            emission_lines: LineModeller or None
                A LineModeller instance describing any lines that were detected in the spectrum. If no lines were
                found, then None.
        """
        if object_info.has_been_measured:
            raise RuntimeError('This spectrum has been measured already!')

        spectra = [self._measure_spectrum(spec) for spec in object_info.spectra.values() if spec.order == 1]

        measured_spectra = list(filter(lambda s: s is not None, spectra))

        if len(measured_spectra) == 0:
            return None

        model_continuum, lines = self._model_1d_spectrum(measured_spectra, object_info)

        if lines.line_models is not None:
            # mask out the line regions from the measured_spectra, then model the continuum and lines again because the
            # lines skew the initial continuum measurement upward.
            masked_spectra = []
            for wavs, flux, dflux in measured_spectra:
                mask = lines.mask(wavs)
                masked_spectra.append((wavs[mask], flux[mask], dflux[mask]))

            model_continuum, lines = self._model_1d_spectrum(masked_spectra, object_info)

        return model_continuum, lines


    def _model_1d_spectrum(self, measured_spectra: list,
                           object_info: 'AstroObjectInfo') -> Tuple[np.ndarray, Union[LineModeller, None]]:
        """
        Creates the 1-D continuum model and the LineModeller for the set of measured_spectra

        Parameters
        ----------

        measured_spectra: list
            A list of tuples produced by _measure_spectrum(). The tuples are (wavelengths,
            measured_fluxes, flux_uncertainties)

        object_info: AstroObjectInfo
            The AstroObjectInfo associated with the spectrum being measured. This will be used to
            access the continuum if the continuum could not be modelled from available data.

        Returns
        -------

        spectrum_model: Tuple[np.ndarray, Union[LineModeller, None]]

        A tuple containing the continuum (np.ndarray) and the line models (LineModeller).
        """
        # perform the smoothing / binning

        binned_fluxes = (self._compute_binned_fluxes(*spec) for spec in measured_spectra)

        binned_fluxes = [bf for bf in binned_fluxes if bf is not None]

        if len(binned_fluxes) == 0:
            object_info.interp_mask = True
            return object_info.continuum, LineModeller(measured_spectra)

        bin_wavelengths, bin_fluxes, bin_uncertainties = np.concatenate(binned_fluxes, axis=1)

        # If there are fewer than 3 bins available, use the continuum from the photometry
        if len(bin_wavelengths) < 3 or np.any(bin_wavelengths <= 0):
            object_info.interp_mask = True
            continuum = object_info.continuum
        else:
            # fit the binned fluxes to estimate the continuum
            continuum = self._modeller.power_law_interp(bin_wavelengths,
                                                        bin_fluxes,
                                                        self._modeller.lambdas,
                                                        bin_uncertainties)
            # If the continuum couldn't be computed, it's now None, so fall-back on the
            # broad-band model:
            if continuum is None:
                object_info.interp_mask = True
                continuum = object_info.continuum
            else:
                # the continuum is based on the binned fluxes.
                object_info.interp_mask = False

        #plt.errorbar(bin_wavelengths, bin_fluxes, yerr=bin_uncertainties, capsize=3, linestyle="None", marker='o', markersize=1, fillstyle='none', label='measured')
        # subtract continuum from each spectrum, so that the LineModeller can model the lines.
        for spec in measured_spectra:
            wavs, fluxes, uncertainties = spec
            fluxes -= np.interp(wavs, self._modeller.lambdas, continuum)

        return continuum, LineModeller(measured_spectra)

    def _measure_spectrum(self, spec: 'SpectrumInfo') -> Union[Tuple[np.ndarray, np.ndarray, np.ndarray], None]:
        """
        Measures a single spectrum.

        Parameters
        ----------
        spec: SpectrumInfo
            The SpectrumInfo instance associated with the spectrum.

        Returns
        -------
        measurement: tuple of 3 np.ndarray objects or None
            A tuple of (wavelengths, fluxes, flux_uncertainties) which are 1-dimensional NumPy arrays of equal length.
            Fluxes and uncertainties are reported in the physical units [erg/s/cm^2/A] and wavelengths are in Angstroms.
            If no measurement was possible (e.g., because there are too few clean pixels), then None is returned.
        """
        search_grid = self._modeller.search_grids[spec.dither, spec.detector]
        contaminated_pixel_mask = self._identify_contaminated_pixels(spec, search_grid)

        correction_output = self._measure_correction(spec, contaminated_pixel_mask)

        if correction_output is None:
            return None

        wavelengths, correction, variance = correction_output

        model_wavs = self._modeller.lambdas
        model_flux = spec.astro_object_info.continuum

        interpolated_model_flux = np.interp(wavelengths, model_wavs, model_flux)

        measured_fluxes = correction * interpolated_model_flux

        flux_uncertainties = np.sqrt(variance) * interpolated_model_flux

        return wavelengths, measured_fluxes, flux_uncertainties

    @staticmethod
    def _measure_correction(spec: 'SpectrumInfo',
                            contaminated_pixel_mask: np.ndarray) -> Union[np.ndarray, None]:
        """
        Measures the correction factor by which the model flux needs to be multiplied in order
        to obtain the observed spectrum.

        Parameters
        ----------
        spec: SpectrumInfo
            The SpectrumInfo object associated with the spectrum that is being measured.

        contaminated_pixel_mask: np.ndarray
            A 2-D NumPy array with True values indicating that the pixel is contaminated.

        Returns
        -------

        correction_info: np.ndarray or None
            A 3xN NumPy array with the following rows:

            row 0: wavelengths
                wavelength values at which the correction was measured.

            row 1: correction
                flux correction values.

            row 2: variance
                the variance of each of the correction values.
        """
        if spec.model_spec is None:
            spec.make_2d_model()

        model2d = spec.model_spec
        observed = spec.spectrum_cutout.original_spectrum.get_science()
        flags = spec.spectrum_cutout.original_spectrum.get_mask()
        bad_pixel_mask = (flags & Flags.INVALID) == Flags.INVALID

        usable_pixels = ~(contaminated_pixel_mask | bad_pixel_mask)

        conv_kernel = spec.convolution_kernel

        variance = list()
        correction = list()
        wavelengths = list()

        # the threshold value is used to select the pixels in the kernel that significantly contribute
        threshold = 0.0002 * conv_kernel.const_kernel.max()

        for entry in conv_kernel.history:
            i, wav, kernel, region, _ = entry
            wavelengths.append(wav)
            mask = kernel > threshold
            mask = mask & usable_pixels[region]
            spec_region = observed[region]
            usable_spec_pixels = spec_region[mask]
            model_region = model2d.pixels[region]
            usable_model_pixels = model_region[mask]
            spec_sum = np.sum(usable_spec_pixels)
            model_sum = np.sum(usable_model_pixels)
            if usable_spec_pixels.size > 3 and spec_sum > 0.0 and model_sum > 0.0:
                beta = spec_sum / model_sum
                correction.append(min(beta, 500.0))
                residual = usable_spec_pixels - beta * usable_model_pixels
                var = np.sum(np.square(residual)) / (model_sum * model_sum)
                variance.append(var)
            else:
                correction.append(1.0)
                variance.append(np.inf)

        spec.clear_convolution_kernel()

        if np.any(np.isfinite(variance)):
            return np.array([wavelengths, correction, variance], dtype=np.float32)
        else:
            return None

    @classmethod
    def _compute_binned_fluxes(cls,
                               wavelengths: np.ndarray,
                               fluxes: np.ndarray,
                               uncertainties: np.ndarray) -> Union[np.ndarray, None]:
        """
        Computes binned fluxes from large regions of the input spectrum. By including more pixels,
        the uncertainty of the binned fluxes is lower than the uncertainty in the high-resolution
        spectrum that is provided as input.

        Parameters
        ----------
        wavelengths: np.ndarray
            A 1-D array of wavelengths, in Angstroms. This is assumed to be given in order of increasing wavelength.

        fluxes: np.ndarray
            A 1-D array of the measured fluxes at the specified wavelengths

        uncertainties: np.ndarray
            A 1-D array of the uncertainties in the flux measurements.

        Returns
        -------
        binned_measurements: tuple or None
            A tuple of (wavelengths, flux_measurements, flux_uncertainties), where:

            bin_wavelengths of the bins: np.ndarray
                A 1-D array of wavelengths, in Angstroms

            bin_fluxes: np.ndarray
                A 1-D array of the binned measured fluxes corresponding to the output wavelengths.

            bin_flux_uncertainties: np.ndarray
                A 1-D array of the uncertainties in the flux measurements.

            None: if fewer than 90% of the input flux values were positive.

            Raises
            ------
            ValueError if the arrays are not sorted in order of increasing wavelength.
        """
        # empirically-determined minimum number of individual flux measurements that are useful to combine in a bin,
        # plus 1 so that < can be used instead of <=.
        min_usable_length = 9

        if len(wavelengths) < min_usable_length:
            return None

        if wavelengths[1] < wavelengths[0]:
            raise ValueError("The Input arrays must already be sorted in order of increasing wavelength.")

        # clip off the ends. We are only interested in the more reliable, inner region of the spectrum

        inner = (wavelengths > DetectorGrism.MIN_WAV_INNER) & (wavelengths < DetectorGrism.MAX_WAV_INNER)
        wavelengths = wavelengths[inner]
        fluxes = fluxes[inner]
        uncertainties = uncertainties[inner]

        positive = fluxes > 0

        finite_weights = np.isfinite(uncertainties).sum()

        if fluxes.size < min_usable_length or sum(positive) < 0.99 * len(fluxes) or finite_weights < min_usable_length:
            # there are too few measurements to be useful OR more than 1 percent of the fluxes are negative,
            # which likely indicates that this spectrum has very low signal / noise, and binning is not going to
            # help very much; the broadband fluxes should be used.
            return None

        min_wav = wavelengths[0]
        max_wav = wavelengths[-1]
        wavelength_extent = max_wav - min_wav
        weights = 1 / np.square(uncertainties)

        if wavelength_extent < 1.5 * cls.WAVELENGTH_BIN_SIZE:
            # just compute a single (wavelength, flux, dflux) triplet when the spectral segment is short
            bin_wavelengths = np.array([Utils.effective_wavelength(wavelengths, fluxes * weights)], dtype=np.float32)
            bin_fluxes = np.array([Utils.weighted_mean(fluxes, weights, wavelengths)], dtype=np.float32)
            weighted_sample_std = Utils.weighted_std(fluxes, weights, wavelengths)
            bin_dflux = np.array([weighted_sample_std], dtype=np.float32)
        else:
            n_bins = math.ceil(wavelength_extent / cls.WAVELENGTH_BIN_SIZE)
            counts, _ = np.histogram(wavelengths, bins=n_bins)
            bin_wavelengths = np.empty((n_bins,), dtype=np.float32)
            bin_fluxes = np.empty((n_bins,), dtype=np.float32)
            bin_dflux = np.empty((n_bins,), dtype=np.float32)
            lower_bound = 0

            for i, c in enumerate(counts):
                fluxes_i = fluxes[lower_bound: lower_bound + c]
                wavelengths_i = wavelengths[lower_bound: lower_bound + c]
                weights_i = weights[lower_bound: lower_bound + c]
                bin_wavelengths[i] = Utils.effective_wavelength(wavelengths_i, fluxes_i * weights_i)
                bin_fluxes[i] = Utils.weighted_mean(fluxes_i, weights_i, wavelengths_i)
                bin_dflux[i] = Utils.weighted_std(fluxes_i, weights_i, wavelengths_i)
                lower_bound += c

        mask = (bin_wavelengths < max_wav) & (bin_wavelengths > min_wav)
        mask &= ~np.isnan(bin_fluxes)
        mask &= ~np.isnan(bin_dflux)

        if sum(mask) > 0:
            output = np.array((bin_wavelengths[mask], bin_fluxes[mask], bin_dflux[mask]), dtype=np.float32)
        else:
            output = None

        return output

    @staticmethod
    def _identify_contaminated_pixels(spec: 'SpectrumInfo', search_grid: 'SearchGrid') -> np.ndarray:
        """
        Masks the pixels in the specified spectrum that are contaminated by other spectra.

        Parameters
        ----------
        spec: SpectrumInfo
            The SpectrumInfo instance associated with the spectrum of interest.

        search_grid: 'searchGrid'
            The SearchGrid for the relevant detector.

        Returns
        -------
        contaminated_pixels: np.ndarray
            A boolean 2D NumPy array, with True values corresponding to contaminated pixels and
            False corresponding to clean, uncontaminated pixels.
        """
        # the x and y coordinates of the pixels in the bounding box of the spectrum
        x_image, y_image = Utils.grid_coords_in_bbox(spec.bbox)

        # search for only the real contaminants, excluding ones that fall within the bounding box
        # but do not actually contaminate the spectrum (which can happen when the spectrum is tilted).
        contaminants = search_grid.find_contaminants(spec.searchgrid_bbox,
                                                     ignore_idx=spec.object_id,
                                                     mag_limit=spec.astro_object_info.mag('J') + 2.5)

        contam_mask = np.full(shape=x_image.shape, fill_value=False)

        for contam in contaminants:
            contam_mask |= search_grid.mask_grid_bbox(x_image, y_image, contam.searchgrid_bbox)

        _, _, width, height = spec.bbox

        return contam_mask.reshape((height, width), order='F')
