"""
Contains the `SpectrumCutout` class which is used for passing data from the `SpectraDecontaminator` class to the
`DecontaminatedSpectraCollection` object.
"""
import numpy as np

from SIR_SpectraDecontamination.Utils import verify_2d_numpy_array
from SIR_SpectraDecontamination.DetectorRegion import DetectorRegion



[docs]class SpectrumCutout(object):
    """
    Stores all layers and metadata for a single decontaminated spectrum.
    """
    def __init__(self, spectrum_info):
        """
        Constructs the object; creates all three layers and stores the cutout in the science layer.

        Parameters
        ----------

        spectrum_info: SpectrumInfo.SpectrumInfo
            The SpectrumInfo object associated with this cutout.
        """
        # the center and offset can be determined from the object info, dither, and detector

        if spectrum_info.nisp_detector_data is None:
            raise ValueError('The associated SpectrumInfo object must contain pixel data from a NISP Detector.')

        self._spec_info = spectrum_info
        self._astro_object_info = spectrum_info.astro_object_info

        self._observed = None  # will hold the the original NISP detector exposure, with 3 layers

        # these three are just the NumPy arrays, not wrapped up in a NISP detector exposure
        self._sci = None
        self._var = None
        self._flags = None
        self._offset = spectrum_info.get_offset()
        self._contaminants = None

    def __del__(self):
        self.free_memory()


[docs]    def free_memory(self):
        self._observed = None
        self._sci = None
        self._var = None
        self._flags = None
        self._contaminants = None


    @property
    def spec_info(self):
        return self._spec_info

    @property
    def has_contaminants(self):
        """Returns true if this object is contaminated at all."""
        return self._contaminants is not None and len(self._contaminants) > 0

    @property
    def is_decontaminated(self):
        """
        Returns True if the object has been decontaminated.
        """
        return (self._var is not None) and (self._flags is not None) and (self._sci is not None)

    @property
    def original_spectrum(self):
        """
        Gets the original, unmodified NispDetectorExposure cutout.

        Note that the NispDetectorExposure cutout is only created at this point, when the data is being requested.
        Thus, the cropping procedure is performed lazily. This reduces memory utilization and computation time.

        Returns
        -------
            cropped: NispDetectorExposure
                A cropped NispDetectorExposure object, containing only the relevant region of the larger NISP detector.
        """
        if self._observed is None:
            # the NISP cutout hasn't been created yet; create it!
            self._observed = DetectorRegion(self._spec_info.nisp_detector_data, self._spec_info.bbox)
        return self._observed

    @property
    def science(self):
        """Gets the updated (decontaminated) science layer."""
        return self._sci

    @science.setter
    def science(self, decon_spec):
        verify_2d_numpy_array(decon_spec)
        self._sci = decon_spec

    @property
    def variance(self):
        """Gets the updated variance layer (i.e.,  the variance of the decontaminated spectrum)"""
        return self._var

    @variance.setter
    def variance(self, var_layer):
        verify_2d_numpy_array(var_layer)
        self._var = var_layer

    @property
    def flags(self):
        """Gets the updated mask layer (with decontamination flags added)."""
        return self._flags

    @flags.setter
    def flags(self, flags):
        verify_2d_numpy_array(flags, np.uint32)
        self._flags = flags

    @property
    def contaminants(self):
        """
        Gets a list of spectra that overlap with this spectrum (i.e., the spectra that potentially contaminate this
        spectrum).

        Returns
        -------

        List of SpectrumInfo objects.
        """
        return self._contaminants

    @contaminants.setter
    def contaminants(self, contams):
        self._contaminants = contams
