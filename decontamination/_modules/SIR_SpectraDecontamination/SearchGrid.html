[docs]class SearchGrid(object):
    """
    Manages a data structure that can be used to quickly search for overlapping boxes.
    """
    def __init__(self, spectrum_info_list, tilt_angle=0, grid_res=34):
        """
        Constructs a SearchGrid.

        Parameters
        ----------

        spectrum_info_list: list

            A list of `SpectrumInfo` objects falling within a single detector.

        tilt: float
            The angle, in degrees, between the mean +wavelength direction and the +x axis of the detector.

        grid_res: int (optional)
            The number of search grid cells in the x and y directions. The total number of search grid cells will
            scale as the square of this number, grid_res^2, and the number of pixels in each cell dimension scales
            inversely with grid_res.

        Raises
        ------

        ValueError
            If the entries in the spectrum_info_list do not all share the same DetectorGrism object.
        """
        if len(spectrum_info_list) == 0:
            self._grid = None
            return

        detector_grism_info = spectrum_info_list[0].grism_model
        for spec in spectrum_info_list:
            if spec.grism_model is not detector_grism_info:
                raise ValueError('All spectra added to the grid must belong to the same detector, which means they all'
                                 ' share the same detector info. It appears that spectra from more than one detector '
                                 'have been added to the `spectrum_info_list`.')

        self._tilt_angle = np.radians(tilt_angle)

        x_corners = np.array([0.0, detector_grism_info.xres, detector_grism_info.xres, 0.0])
        y_corners = np.array([0.0, 0.0, detector_grism_info.yres, detector_grism_info.yres])

        x_rot, y_rot = self._rotate(x_corners, y_corners, -self._tilt_angle)

        left = math.floor(x_rot.min())
        bottom = math.floor(y_rot.min())
        right = math.ceil(x_rot.max())
        top = math.ceil(y_rot.max())

        grid_width = right - left
        grid_height = top - bottom

        self._shift = (left, bottom)

        # the number of grid cells in the x and y directions
        self._grid_resolution = grid_res

        # the total number of grid cells
        self._ncells = self._grid_resolution * self._grid_resolution

        # the width of each grid cell in pixels
        self._dw = math.ceil(grid_width / self._grid_resolution)

        # the height of each grid cell, in pixels
        self._dh = math.ceil(grid_height / self._grid_resolution)

        # the height of the grid, in units of pixels
        self._grid_height = self._grid_resolution * self._dh

        # the width of the grid, in units of pixels
        self._grid_width = self._grid_resolution * self._dw

        # the grid
        self._grid = np.ndarray(self._ncells, dtype=list)

        # initialize the lists within the grid cells:
        self._grid[:] = [list() for _ in range(self._ncells)]

        # populate the grid
        for spec in spectrum_info_list:

            # identify cells in the region of the model_spec
            spec.searchgrid_bbox = self.get_grid_bbox(spec.boundary[:, 0], spec.boundary[:, 1])

            cells = self._cells_in_region(spec.searchgrid_bbox)

            # populate these grid cells with spectrum info
            for cell in cells:
                self._grid[cell].append(spec)


[docs]    def find_contaminants(self, grid_box, ignore_idx=None, mag_limit=np.inf, mag_band='J') -> List['SpectrumInfo']:
        """
        Creates a list of the contaminants that overlap the specified box.

        Parameters
        ----------

        grid_box: tuple
            A bounding box, specified as a tuple (left, bottom, width, height), in grid coordinates.

        ignore_idx: int (optional)
            If specified, the object with this index will be ignored; it will not be added to the list
            of contaminants.

        mag_limit: float
            The maximum magnitude to include as a contaminant.

        mag_band: str

        Returns
        -------

        contaminating spectra: list
            A list of contaminating spectra (`SpectrumInfo` objects) that overlap the specified box.
        """
        if self._grid is None:
            return []

        # Identify the cells in the search grid that this box overlaps
        cells = self._cells_in_region(grid_box)

        # Now, identify all potential contaminating boxes by looking at the contents of all of the cells:
        candidates = set()

        for cell_idx in cells:
            candidates_in_cell = self._grid[cell_idx]
            for can in candidates_in_cell:
                bright_enough = can.astro_object_info.invalid_mag or can.astro_object_info.mag(mag_band) < mag_limit
                if can.object_id != ignore_idx and bright_enough:
                    candidates.add(can)

        contaminants = list()

        # check to see which of the candidate boxes actually overlap with the box of interest:
        for can in candidates:
            if self._overlaps(grid_box, can.searchgrid_bbox):
                contaminants.append(can)

        # contaminants now contains info about the boxes that definitely overlap with the box of interest
        return contaminants



[docs]    def get_grid_bbox(self, x_image: Iterable[float], y_image: Iterable[float]) -> Tuple[int]:
        """
        Returns the bounding box of the input coordinates in the coordinate system of the grid.

        Parameters
        ----------

        x_image: Iterable[float]
            The x-coordinates of the points that will be transformed to the coordinate system of the grid.

        y_image: Iterable[float]
            The x-coordinates of the points that will be transformed to the coordinate system of the grid.

        Returns
        -------

        grid_bbox: Tuple[int]
            The tuple of ints (left, bottom, width, height) of the bounding box of the points specified by
            the x_image and y_image coordinates, in the coordinate system of the grid.
        """
        x = np.array(x_image)
        y = np.array(y_image)

        if x.size != y.size or x.size < 3:
            raise ValueError("The input arrays must be of the same length and describe at least 3 points.")

        x_grid, y_grid = self._transform_to_grid_coordinates(x, y)

        left = max(0, math.floor(x_grid.min()))
        right = min(self._grid_width, math.ceil(x_grid.max()))
        bottom = max(0, math.floor(y_grid.min()))
        top = min(self._grid_height, math.ceil(y_grid.max()))

        return left, bottom, right - left, top - bottom



[docs]    def mask_grid_bbox(self, x_image, y_image, grid_box: tuple):
        """
        Masks pixels falling within the SearchGrid bounding box, grid_box, given pixel
        coordinates.

        Parameters
        ----------

        x_image: np.ndarray
            An array of x-coordinates of pixels in the image.

        y_image: np.ndarray
            An array of x-coordinates of pixels in the image.

        grid_box: tuple
            The bounding box, specified in grid coordinates (left, bottom, width, height)

        Returns
        -------

        in_box_mask: np.ndarray
            A Boolean array the same size of x_image and y_image, with True entries
            corresponding to coordinates that fall within the bounding box (grid_box).
        """
        x_grid, y_grid = self._transform_to_grid_coordinates(x_image, y_image)

        return Utils.mask_in_bounding_box(x_grid, y_grid, grid_box)


    def _cell_box(self, idx):
        """
        Computes the bounding box of the specified grid cell

        Parameters
        ----------

        idx: int
            The index of the cell within the search grid

        Returns
        -------

        bounding box: tuple
            The bounding box of the cell (left, bottom , width, height) with the specified index, in image coordinates.
        """
        j = idx // self._grid_resolution

        i = idx % self._grid_resolution

        left = i * self._dw

        bottom = j * self._dh

        return left, bottom, self._dw, self._dh

    def _cells_in_region(self, grid_box):
        """
        Identifies all boxes in the region defined by grid_box.

        Parameters
        ----------

        grid_box: tuple
            The bounding box, specified in grid coordinates (left, bottom, width, height)

        Returns
        -------

        indices: list
            A list of indices of grid cells that overlap the specified box.
        """
        left, bottom, width, height = grid_box

        i_min = max(int(math.floor(left / self._dw)), 0)
        j_min = max(int(math.floor(bottom / self._dh)), 0)
        i_max = min(int(math.ceil((left + width) / self._dw)) + 1, self._grid_resolution)
        j_max = min(int(math.ceil((bottom + height) / self._dh)) + 1, self._grid_resolution)

        ids = []

        for j in range(j_min, j_max):
            for i in range(i_min, i_max):
                cell_id = self._grid_resolution * j + i
                cell_box = self._cell_box(cell_id)
                if self._overlaps(cell_box, grid_box):
                    ids.append(cell_id)

        return ids

    def _transform_to_grid_coordinates(self, x_image: 'np.ndarray', y_image: 'np.ndarray') -> 'np.ndarray':
        """
        Transforms the input image coordinates to grid coordinates.

        x_image: np.ndarray
            An array of x-coordinates of pixels in the image.

        y_image: np.ndarray
            An array of x-coordinates of pixels in the image.

        Returns
        -------

        x_grid, y_grid: Tuple
            A tuple of np.ndarray objects containing the x- and y-coordinates in the coordinate
            system of the grid
        """
        if x_image.size != y_image.size:
            raise ValueError("The input arrays must be of the same length.")

        rot_x, rot_y = self._rotate(x_image, y_image, -self._tilt_angle)

        x_grid = rot_x - self._shift[0]
        y_grid = rot_y - self._shift[1]

        return x_grid, y_grid

    @staticmethod
    def _overlaps(box_a, box_b):
        """'
        Tests whether boxes intersect / overlap

        Parameters
        ----------

        box_a: tuple
            A bounding box, specified as a tuple (left, bottom, width, height)

        box_b: tuple
            A bounding box, specified as a tuple (left, bottom, width, height)

        Returns
        -------

        overlaps: bool
            True if box1 and box2 overlap; False otherwise.
        """
        intersection_a, intersection_b = Utils.bounding_indices(box_a, box_b)

        return not (intersection_a is None or intersection_b is None)

    @staticmethod
    @jit
    def _rotate(x, y, theta):
        """
        Rotates the points with the specified x and y vectors by theta, counter-clockwise and returns
        the coordinates of the rotated points.
        """
        st = math.sin(theta)
        ct = math.cos(theta)

        xp = x * ct - y * st
        yp = x * st + y * ct

        return xp, yp
