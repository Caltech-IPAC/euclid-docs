    """ Computes a / b, ignoring division by zero: div0( [-1, 0, 1], 0 ) -> [0, 0, 0] """
    with np.errstate(divide='ignore', invalid='ignore'):
        c = np.true_divide(a, b)

    Parameters

[docs]def weighted_mean(quantity, weights, wavelengths):
    """
    Computes the weighted mean of a quantity that is a function of wavelength.

    Parameters
    ----------
    quantity: np.ndarray
        1D NumPy array containing samples of the quantity that is being averaged.

    weights: np.ndarray
        1D NumPy array of weights corresponding to the samples.

    wavelengths: np.ndarray
        1D NumPy array of wavelengths corresponding to the samples.

    Returns
    -------

    weighted_mean: float

    The weighted mean of the quantity.
    """
    if np.all(weights <= 0.0):
        return np.nan

    if len(quantity) < 2:
        return quantity[0]

    return trapz(quantity * weights, wavelengths) / trapz(weights, wavelengths)




[docs]def weighted_std(quantity, weights, wavelengths):
    """
    The weighted standard deviation of a quantity that is a function of wavelength.

    Parameters
    ----------
    quantity: np.ndarray
        1D NumPy array containing samples of the quantity that is being averaged.

    weights: np.ndarray
        1D NumPy array of weights corresponding to the samples.

    wavelengths: np.ndarray
        1D NumPy array of wavelengths corresponding to the samples.

    Returns
    -------

    weighted_std: float

    The weighted standard deviation of the quantity.
    """
    mean_square = weighted_mean(np.square(quantity), weights, wavelengths)
    squared_mean = weighted_mean(quantity, weights, wavelengths)**2

    if np.any(np.isnan([mean_square, squared_mean])):
        return np.nan

    var = mean_square - squared_mean

    if var < 0:
        return np.nan

    return np.sqrt(var)

    """
    Integrates a filter_transmission profile against a spectrum to obtain the flux of the bandpass.

    Parameters
    ----------

    wavelengths: 1D NumPy array

    fluxes: 1D NumPy array

def prob_in_range(lower, upper, mu, sigma):
    """
    Computes the probability between x = lower and x = upper of the Gaussian, centered at x = mu.

    Parameters
    ----------

    lower: float
        The lower limit of the integral.

    upper: float
        The upper limit of the integral

        The standard deviation of the Gaussian distribution

    -------

    The integral of the Gaussian distribution between x = lower to x = upper.
    """
    return cdf(upper, mu, sigma) - cdf(lower, mu, sigma)

def gauss_kernel(canvas, center, sigma):
    """
    Computes a 2D Gaussian, centered at `center`.

    Parameters
    ----------

    canvas: NumPy.ndarray
        A square 2D NumPy array in which the 2D Gaussian kernel will be painted.

    center: tuple,
        The pixel coordinates (x, y) of the center of the kernel.

    sigma: tuple
        The axes of the resulting elliptical Gaussian kernel, (major, minor).

    Returns
    -------

    A 2D NumPy array containing the Gaussian Kernel centered at the specified position.
    """
    y_res, x_res = canvas.shape

    if y_res != x_res:
        raise ValueError("The image must be square.")

    res = x_res

    center_x, center_y = center

    center_x += 0.5
    center_y += 0.5

    major_sigma, minor_sigma = sigma

    if minor_sigma > major_sigma:
        raise ValueError("The major axis must equal to or larger than the minor axis.")

    # compute the total mass in each row of the array

    row_sum = np.zeros(res)

    for j in range(res):
        row_sum[j] = prob_in_range(j, j + 1.0, center_y, minor_sigma)

    # compute the 1D kernel that will be used for distributing the light along the major axis

    row_kernel = np.zeros(res)

    for i in range(res):
        row_kernel[i] = prob_in_range(i, i + 1.0, center_x, major_sigma)

    # compute the value of each pixel:

    for j in range(res):
        row = canvas[j]
        for i in range(res):
            row[i] = row_kernel[i] * row_sum[j]

    return canvas

def rebin3(image: np.ndarray) -> np.ndarray:
    """
    Reduces the resolution of an image by a factor of 3.

    Each pixel in the output image is the sum of 3 x 3 pixels
    in the input image.

    Parameters
    ----------

    image: np.ndarray
        The input image

    Returns
    -------

    rebinned: np.ndarray
        An image in which each pixel is the sum of 3 x 3 pixels in
        the input image
    """
    y_dim, x_dim = image.shape

    y_output = y_dim // 3
    x_output = x_dim // 3

    output = np.zeros((y_output, x_output), dtype=np.float32)

    for j in range(y_output):
        for i in range(x_output):
            i_input = 3 * i
            j_input = 3 * j
            v = image[j_input, i_input]
            v += image[j_input, i_input + 1]
            v += image[j_input, i_input + 2]

            v += image[j_input + 1, i_input]
            v += image[j_input + 1, i_input + 1]
            v += image[j_input + 1, i_input + 2]

            v += image[j_input + 2, i_input]
            v += image[j_input + 2, i_input + 1]
            v += image[j_input + 2, i_input + 2]
            output[j, i] = v

    return output

def bilinear_shift(image: np.ndarray, shift_vector: Tuple[float, float]) -> np.ndarray:
    """
    Shifts a 2D image by the specified shift vector using bilinear interpolation.

    Parameters
    ----------

    image: np.ndarray
        A 2D NumPy array that will be shifted.

    shift_vector: tuple
        A tuple of (y_shift, x_shift) specifying how much the image will be shifted in
        the y and x directions.

    Returns
    -------

    shifted: np.ndarray
         A 2D NumPy array with the same dimensions as the input image, but with the
         contents shifted by the specified amount.
    """
    shift_y, shift_x = shift_vector

    shift_y = -shift_y
    shift_x = -shift_x

    dy_lower = math.floor(shift_y)
    y_m = shift_y - dy_lower
    m_y = 1.0 - y_m

    dx_left = math.floor(shift_x)
    x_m = shift_x - dx_left
    m_x = 1.0 - x_m

    w11 = m_x * m_y
    w21 = m_x * y_m
    w12 = x_m * m_y
    w22 = x_m * y_m

    y_size, x_size = image.shape

    canvas = np.zeros(shape=image.shape, dtype=np.float32)

    for j in range(y_size):
        j1 = j + dy_lower
        j2 = j1 + 1
        if j1 < 0 or j2 >= y_size:
            continue
        for i in range(x_size):
            i1 = i + dx_left
            i2 = i1 + 1
            if i1 < 0 or i2 >= x_size:
                continue
            canvas[j, i] = w11 * image[j1, i1] + w21 * image[j2, i1] + w12 * image[j1, i2] + w22 * image[j2, i2]

    return canvas

def shift_horizontal(image, x_shift):
    y_size, x_size = image.shape

    x_shift = -x_shift

    int_shift = math.floor(x_shift)
    dx = x_shift - int_shift

    dxm1 = dx - 1
    dxm2 = dx - 2
    dxp1 = dx + 1

    wl0 = 0.5 * dx * dxm1
    wl1 = - dxp1 * dxm1
    wl2 = 0.5 * dxp1 * dx

    wr0 = 0.5 * dxm1 * dxm2
    wr1 = - dx * dxm2
    wr2 = 0.5 * dx * dxm1

    norm = 1.0 / (dxm1 * dxm1 + dx * dx + dx - dxm1)
    left_weight = norm * (dxm1 * dxm1 - dxm1)
    right_weight = norm * (dx * dx + dx)

    out = np.zeros(image.shape, dtype=np.float32)

    for j in range(y_size):
        for i in range(x_size):
            i_0 = i + int_shift - 1
            i_1 = i + int_shift
            i_2 = i + int_shift + 1
            i_3 = i + int_shift + 2
            if i_0 < 0 or i_3 >= x_size:
                continue
            f0 = image[j, i_0]
            f1 = image[j, i_1]
            f2 = image[j, i_2]
            f3 = image[j, i_3]
            f_left = wl0 * f0 + wl1 * f1 + wl2 * f2
            f_right = wr0 * f1 + wr1 * f2 + wr2 * f3
            out[j, i] = right_weight * f_right + left_weight * f_left
    return out

def convolve2d(stamp: np.ndarray, kernel: np.ndarray) -> np.ndarray:
    """
    Convolves a stamp image with a convolution kernel.

    Parameters
    ----------

    stamp: np.ndarray
        The (2D) image array to be convolved.

    kernel: np.ndarray
        The (2D) convolution kernel.

    Returns
    -------

    convolved: np.ndarray
        A NumPy array the same shape as the input stamp image, convolved with the kernel.
    """
    y_stamp, x_stamp = stamp.shape
    y_kernel, x_kernel = kernel.shape

    x_kernel_shift = - x_kernel // 2 + 1
    y_kernel_shift = - y_kernel // 2 + 1

    out = np.zeros((y_stamp, x_stamp), dtype=np.float32)

    for j in range(y_stamp):
        for i in range(x_stamp):
            v = stamp[j, i]
            if v < 1e-9:
                continue
            for jk in range(y_kernel):
                jo = j + y_kernel_shift + jk
                if jo < 0 or jo >= y_stamp:
                    continue
                for ik in range(x_kernel):
                    io = i + x_kernel_shift + ik
                    if io < 0 or io >= x_stamp:
                        continue
                    out[jo, io] += v * kernel[jk, ik]
    return out

    image = np.zeros((dim, dim))

def bounding_indices(box1, box2):
    """
    Computes the bounding indices of the region of overlap of two boxes.

    Parameters
    ----------

    box1:
        A tuple of (left, bottom, width, height)
    box2:
        A tuple of (left, bottom, width, height)

    Returns
    -------

    bounding_indices_1, bounding_indices_2: two tuples
        (left1, bottom1, right1, top1), (left2, bottom2, right2, top2) which are the bounding indices
        of the area of overlap in the coordinate system of box1 and the coordinate system of box2, respectively.
    """
    left1, bottom1, width1, height1 = box1
    left2, bottom2, width2, height2 = box2

    # the offset between the corners of the two boxes is:

    x_offset = left1 - left2
    y_offset = bottom1 - bottom2

    # the coordinates of box1's boundaries, in the coordinate system of box2 are:

    left = x_offset  # 0 + x_offset
    bottom = y_offset  # 0 + y_offset
    right = width1 + x_offset
    top = height1 + y_offset

    # clip the boundaries so that we are limited to the region within box2:

    left2 = max(0, left)
    bottom2 = max(0, bottom)
    right2 = min(width2, right)
    top2 = min(height2, top)

    # the coordinates of the region of overlap, in the coordinate system of box1:

    left1 = max(0, left2 - x_offset)
    right1 = min(width1, right2 - x_offset)
    bottom1 = max(0, bottom2 - y_offset)
    top1 = min(height1, top2 - y_offset)

    boxes_intersect = left1 < right1 and bottom1 < top1 and left2 < right2 and bottom2 < top2

    if boxes_intersect:
        return (left1, bottom1, right1, top1), (left2, bottom2, right2, top2)
    else:
        # the boxes do not intersect.
        return None, None




[docs]def apply_overlay(overlay, canvas, operation):
    """
    Applies the `overlay` values to a `canvas` by applying the specified `operation` in the region in which the overlay
    intersects the canvas.

    Parameters
    ----------

    overlay: tuple
        (offset, 2D array), where the offset is a tuple of (left, bottom) specifying the position of the lower-left
        pixel of the array within the detector and the 2D array contains the data which will be applied to the canvas,
        using the specified operation.

    canvas: tuple
        (offset, 2D array), where the offset is a tuple of (left, bottom) specifying the position of the lower-left
        pixel of the array within the detector and the 2D array contains the data which will be updated by this
        procedure.

    operation: function
        A function accepting two NumPy ndarray objects of the same shape and returning a third ndarray of the same
        shape as its input arguments. For example, numpy.add, numpy.fmax, numpy.fmin, numpy.subtract, numpy.multiply.
        Note that using non-commutative operations could yield unexpected results. The function is called with the
        `canvas` as its first argument and `overlay` as the second argument.

    Side Effects
    ------------

    Modifies the `canvas` by replacing elements in the intersection area with the result of the specified `operation`.
    """
    (overlay_left, overlay_bottom), overlay_img = overlay
    (canvas_left, canvas_bottom), canvas_img = canvas

    overlay_height, overlay_width = overlay_img.shape
    canvas_height, canvas_width = canvas_img.shape

    overlay_box = (overlay_left, overlay_bottom, overlay_width, overlay_height)
    canvas_box = (canvas_left, canvas_bottom, canvas_width, canvas_height)

    overlay_boundaries, canvas_boundaries = bounding_indices(overlay_box, canvas_box)

    if overlay_boundaries is None or canvas_boundaries is None:
        # the overlay does not intersect the canvas
        return

    ovr_left, ovr_bottom, ovr_right, ovr_top = overlay_boundaries
    can_left, can_bottom, can_right, can_top = canvas_boundaries

    canvas_region = (slice(can_bottom, can_top), slice(can_left, can_right))
    overlay_region = (slice(ovr_bottom, ovr_top), slice(ovr_left, ovr_right))

    canvas_img[canvas_region] = operation(canvas_img[canvas_region], overlay_img[overlay_region])

def mask_in_bounding_box(x, y, bbox):
    """
    Identifies which points with the specified x and y values lie within the bounding box

    Parameters
    -----------

    x: np.ndarray of floats
        x-coordinates of points to be tested.

    y: np.ndarray of floats
        y-coordinates of points to be tested.

    bbox: tuple of ints
        The tuple, (left, bottom, width, height)

    Returns
    -------

    inside_bbox: np.ndarray of Bools

    True if the corresponding point is inside of the bbox, else False.
    """
    left, bottom, width, height = bbox

    right = left + width
    top = bottom + height

    in_x = (x >= left) & (x < right)
    in_y = (y >= bottom) & (y < top)

    return in_x & in_y

    can_left, can_bottom, can_right, can_top = canvas_boundaries

    mask_overlap = mask_img[msk_bottom: msk_top, msk_left: msk_right]

    canvas_img[can_bottom:can_top, can_left:can_right][mask_overlap] |= flag

def normalize_image(image):
    """
    Clips and normalizes an image.

    Parameters
    ----------

    image: NumPy.ndarray

    Returns
    -------

    normalized_stamp: NumPy.ndarray or None

    A normalized version of the input image with all negative values clipped to zero, None if the
    pixels are all zero or nearly zero. The image is normalized such that the sum of all pixels is 1.0.
    """
    clipped = np.clip(image, 0.0, np.inf)

    image_sum = clipped.sum()

    if image_sum > 1e-9:
        return clipped / image_sum
    else:
        return None

def normalize_image_fast(image):
    """
    Normalizes an image that has already been clipped.

    Parameters
    ----------

    image: NumPy.ndarray

    Returns
    -------

    normalized_stamp: NumPy.ndarray or None

    A normalized version of the input image, None if the pixels are all zero or negative. The image is
    normalized such that the sum of all pixels is 1.0.
    """
    image_sum = 0

    for i in range(image.size):
        image_sum += image.flat[i]

    if image_sum < 1e-9:
        return None

    isum = 1.0 / image_sum
    for i in range(image.size):
        image.flat[i] *= isum

    return image

def gaussian_moments(data: np.ndarray) -> Tuple[float, float]:
    """
    Computes the mean and standard deviation of a 1-D data array.

    Parameters
    ----------

    data: np.ndarray
        A 1-D array

    Returns
    -------

    moments: tuple
        A tuple of (mean, sigma)
    """
    x = np.arange(data.size)
    mean = np.sum(x * data) / np.sum(data)
    sigma = np.sqrt(abs(np.sum(data * (x - mean) ** 2) / np.sum(data)))
    return mean, sigma

def gaussian_centroid(image: np.ndarray) -> Tuple[float, float]:
    """
    Computes the coordinates of the centroid of the image by fitting Gaussian profiles in x and y.

    Parameters
    ----------

    image: NumPy.ndarray
        A 2-D array of pixels.

    Returns
    -------

    centroid: tuple

    A tuple of (x_center, y_center).
    """
    x_center, _ = gaussian_moments(image.sum(axis=0))

    y_center, _ = gaussian_moments(image.sum(axis=1))

    return x_center, y_center

[docs]def spectral_axis(location):
    """
    Determines the spectral axis---the axis along which the light is dispersed .

    Parameters
    ----------

    location: LocationSpectrum

    Returns
    -------

    spectral_axis: int

    Returns 0 if the spectrum is dispersed along the x-axis; 1 if the spectrum is dispersed along the y axis.
    """
    l_min = location.getWavelengthStart()

    l_max = location.getWavelengthEnd()

    max_position = location.computePosition(l_max, 0).getPosition()
    min_position = location.computePosition(l_min, 0).getPosition()

    spectral_direction = max_position - min_position

    return int(abs(spectral_direction[0]) < abs(spectral_direction[1]))




[docs]def compute_wavelengths(location, bbox):
    """
    Computes the central wavelength of each pixel.

    Each pixel corresponds to a range of wavelength values, due to the spectral width of the pixel itself and
    the spectral size of the dispersing object. This function computes the nominal central wavelength value of
    a given pixel.

    Parameters
    ----------

    location: LocationSpectrum
        The LocationSpectrum associated with the spectrum of interest

    bbox: tuple
        The bounding box of the spectrum; a tuple of (left, bottom, width, height)

    Returns
    -------

    2D NumPy array
        The central wavelength of each pixel in the decontaminated spectrum.
    """
    # Determine the (wavelength, row) pairs for which to compute pixel positions:

    left, bottom, width, height = bbox

    l_min = location.getWavelengthStart()

    l_max = location.getWavelengthEnd()

    spatial_extent = location.getApertureSize()

    spectral_extent = bbox[spectral_axis(location) + 2]

    dl = 30 if spectral_extent > 40 else 6

    lambda_points = np.arange(l_min, l_max, dl)

    spatial_points = range(0, spatial_extent)

    # Compute the pixel position corresponding to each wavelength and store as
    # 3D coordinates: [(x_0, y_0, wavelength_0)
    #                  (x_1, y_1, wavelength_1)
    #                             â®
    #                  (x_N, y_N, wavelength_N)]

    samples = np.empty((len(lambda_points) * spatial_extent, 3), dtype=np.float32)

    for i, (row, wav) in enumerate(product(spatial_points, lambda_points)):
        x_pos, y_pos = location.computePosition(wav, row).getPosition() - np.array((left, bottom))
        samples[i, :] = x_pos, y_pos, wav

    return interpolate_2d_grid(width, height, samples)

def is_left(p0: tuple, p1: tuple, p2: tuple) -> float:
    """
    Determines whether point2 (p2) is to the left of the line defined by p0 and p1.

    Parameters
    ----------

    p0: tuple
        (x0, y0) coordinates of point 0.

    p1: tuple
        (x1, y1) coordinates of point 1.

    p2: tuple
        (x1, y1) coordinates of point 2.

    Returns
    -------

    left: float
        left > 0 if p2 is to the left of the line, p0 - p1.
        left == 0 if the three points are co-linear
        left < 0 if p2 is to the right of the line, p0 - p1.
    """
    return (p1[0] - p0[0]) * (p2[1] - p0[1]) - (p2[0] - p0[0]) * (p1[1] - p0[1])

def contains(loop, p):
    """
    Determines whether a point is contained within a polygon.

    Uses Dan Sunday's optimized winding number algorithm (https://en.wikipedia.org/wiki/Point_in_polygon)
    to test if the point, p, is located inside of the polygon defined by the vertices listed in loop.

    Parameters
    ----------

    loop: no.ndarray
        An array of points defining the vertices of a polygon, in order [(x0, y0), (x1, y1), ...]

    p: tuple
        The coordinates of a point (x, y)

    Returns
    -------

    contains: bool
        True if the point, p, is within the loop (polygon); False otherwise.
    """
    n = len(loop)
    winding_number = 0

    for i in range(-1, n - 1):
        if loop[i, 1] <= p[1]:
            if loop[i + 1, 1] > p[1]:
                if is_left((loop[i, 0], loop[i, 1]), (loop[i + 1, 0], loop[i + 1, 1]), (p[0], p[1])) > 0:
                    winding_number += 1
        else:
            if loop[i + 1, 1] <= p[1]:
                if is_left((loop[i, 0], loop[i, 1]), (loop[i + 1, 0], loop[i + 1, 1]), (p[0], p[1])) < 0:
                    winding_number -= 1
    return winding_number != 0

def rasterize(canvas, offset, loop):
    """
    Creates a raster image of a polygon on a Boolean canvas.

    Note that the center of the pixel must be contained within the polygon in order to be marked as True.
    If pixels which partially overlap the polygon should be marked as True as well, then consider expanding
    the polygon.

    Parameters
    ----------

    canvas: np.ndarray
        A 2D boolean NumPy array which will contain the rasterized polygon.

    offset: tuple:
        The (x, y) offset from the lower-left corner of the canvas to the origin of the coordinate system.

    loop: no.ndarray
        An array of points defining the vertices of a polygon, in order [(x0, y0), (x1, y1), ...]

    Returns
    -------

    canvas: np.ndarray
        The canvas with the rasterized polygon, with True inside of the polygon; False outside.
    """
    xoff, yoff = offset
    height, width = canvas.shape
    for j in range(height):
        jp = yoff + j
        for i in range(width):
            ip = xoff + i
            canvas[j, i] = contains(loop, (ip, jp))
    return canvas
