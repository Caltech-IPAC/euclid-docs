    """
    Handles the process of actually running the SpectraDecontaminator.

    The Executor runs the decontamination code on a set of detectors, or a section of a single detector,
    depending on the arguments provided.
    """

    FINISHED_ALL_PARTS = 23
    """Exit code indicating that this is the final part in the series of parts of a detector."""

    COMMAND_LINE_ERROR = 2
    """Exit code indicating that there is an error in the command line."""

    FILE_COUNTER = 'DECON_FILE_COUNTER'
    FILE_PART = 'DECON_FILE_PART_NUMBER'
    DITHER = 'DECON_DITHER_NUMBER'
    DETECTOR = 'DECON_DETECTOR_NUMBER'

    def __init__(self, args: 'argparse.Namespace', logger: 'logging.Logger'):
        """
        Parameters
        ----------

        args: argparse.Namespace
            The Namespace instance, containing data provided via the command line interface.

        logger: logging.Logger
           A logging.Logger instance for logging progress data.
        """
        self._args = args
        self._logger = logger

        work_dir = args.workdir
        self._workdir = args.workdir
        self._data_dir = os.path.join(work_dir, 'data')

        config_filename = self._get_conf_filename(work_dir, args.config)

        self._config = ConfigObj(config_filename)

        self._use_relflux = self._config['OPTIONS'].as_bool('use_relflux')

        location_tables_json_filename = os.path.join(work_dir, args.loc_tables)

        if args.relflux_flat != 'None' and self._use_relflux:
            if args.relflux_flat.lower().endswith('.xml'):
                rel_flux_dpd = DpdSirRelativeFluxScaling(os.path.join(args.workdir, args.relflux_flat))
                fits_name = os.path.join(work_dir, 'data', rel_flux_dpd.get_sens_fits_file())
                self._relflux_flat = fits.open(fits_name)
                logger.info(f"Loaded relative flux flat file: {fits_name}")

            elif args.relflux_flat.lower().endswith('.fits'):
                self._relflux_flat = fits.open(args.relflux_flat)
                logger.info(f"Loaded relative flux flat file: {args.relflux_flat}")
            else:
                logger.warning("The relative flux filename argument appears to be invalid. Ignoring.")
        else:
            self._relflux_flat = None

            if not self._use_relflux and args.relflux_flat != 'None':
                logger.warning("A relative flux flat filename was provided, but use_relflux was not set in the "
                               "configuration file. The flat will be ignored. If this is not what you wanted to do, then "
                               "set use_relflux = True in the config file, under [OPTIONS].")

            if self._use_relflux and args.relflux_flat == 'None':
                logger.error("use_relflux was set to True, but no filename was provided via --relflux_flat.")
                sys.exit(self.COMMAND_LINE_ERROR)

        if args.relflux_flat == 'None' and not self._use_relflux:
            logger.warning("The relative flux correction will not be performed. Add --relflux_flat <filename> and set "
                           "use_relflux = True in the config file, under [OPTIONS] if you want to use the relative "
                           "flux flat when building model spectra.")

        self._sens = None

        if args.sens == 'None':
            logger.warning("The calibrated grism sensitivity curve was not provided; The value in the MDB will be used "
                           "instead. If this is a mistake, provide the sensitivity using --sens.")
        elif os.path.isfile(os.path.join(work_dir, args.sens)):
            sens_dpd_filename = os.path.join(work_dir, args.sens)
            logger.info(f"loading sensitivity data product, {sens_dpd_filename}")
            with open(sens_dpd_filename) as f:
                sens_xml = f.read()
            sens_dpd = absoluteflux_stub.CreateFromDocument(sens_xml)
            sens_fits_filename = sens_dpd.Data.DataStorage.DataContainer.FileName
            logger.info(f"loading sensitivity file, {os.path.join(work_dir, sens_fits_filename)}")
            self._sens = fits.open(os.path.join(self._data_dir, sens_fits_filename))
        else:
            raise FileNotFoundError(f"The sensitivity data product, {args.sens}, could not be found.")

        if args.cutouts != 'None' and not args.passthrough:
            cutouts_json_filename = os.path.join(work_dir, args.cutouts)
            self._cutouts = CutoutCollection(work_dir, cutouts_json_filename, logger)
            self._cutouts.use_gaussian_fit = args.gaussian_profile
        else:
            self._cutouts = None

        nisp_exposures_json_filename = os.path.join(work_dir, args.sci_frames)

        logger.info(f"Loading the input files, beginning with the MDB file, {args.mdb}")

        self._mdb = DpdMdbDataBase(args.mdb, workdir=work_dir)

        self._location_table_info = LocationTableSetInfo(work_dir,
                                                         location_tables_json_filename,
                                                         args.config,
                                                         logger)

        lt_list_ids = self._location_table_info.get_lt_identifiers()
        self._exposures = ExposureSet(work_dir, nisp_exposures_json_filename, lt_list_ids, logger)
 
        self._check_input_files()

        self._config.data_dir = self._data_dir

        self._det_str = args.detectors

        if args.pipeline:
            detector = list(self._location_table_info.metadata.values())[0][0].getDetectorName()
            self._det_str = f"*.{detector}"
            logger.info(f"Detector {detector} will be decontaminated.")
        else:
            logger.info(f"Using detector specifier {self._det_str}")

        if self._args.max_spectra == 0:
            self._run_full(self._det_str)
        elif self._args.max_spectra > 0:
            file_part = os.getenv(self.FILE_PART)

    def _run_full(self, det_str: str):
        """
        Runs the code on a full detector or set of detectors, all in one process.

        This is the main mode of operation when all of the data will fit into system memory because this is much
        more computationally efficient, due to the re-use of model contaminants.

        Parameters
        ----------

        det_str: str
            A string specifying which detectors should be decontaminated. Refer to the documentation for
            SpectraDecontaminator.run() for more details.
        """
        mode = 'overwrite' if self._args.overwrite else 'create'

        location_tables = LocationTableSet(self._location_table_info, self._logger)
        if self._cutouts is not None:
            self._cutouts._make_index(location_tables.objects_ids)        

        if len(location_tables.empty_detectors) > 0:
            self._logger.info("WARNING: One or more of the LocationTables contained no spectra. Exiting.")
            sys.exit(0)

        self._logger.info("Finished loading input files. Preparing to construct the SpectraDecontaminator.")

        decon = SpectraDecontaminator(self._mdb,
                                      self._config,
                                      self._exposures,
                                      location_tables,
                                      self._relflux_flat,
                                      self._sens,
                                      self._cutouts,
                                      self._args.limit_band,
                                      self._args.limit_mag,
                                      self._args.zeroth_order_limit,
                                      self._args.second_order_limit,
                                      self._args.padding)

        decon_spectra = decon.run(det_str,
                                  decont_order=self._args.order,
                                  passthrough=self._args.passthrough,
                                  skip_measuring=self._args.no_measure)

        decon_spectra.mdb = self._mdb

        self._logger.info(f"Saving a file containing {len(decon_spectra)} decontaminated spectra...")
#        decon_spectra.save(self._data_dir, self._args.dec_spectra, location_tables, mode, parallel=True)
        decon_spectra.write(self._workdir, self._args.dec_spectra, location_tables, mode, parallel=True)

    def _run_parts(self, det_str: str):
        """
        Runs the code on sections of a detector by launching subprocesses.

        Parameters
        ----------

        det_str: str
            A string specifying which detectors should be decontaminated. Refer to the documentation for
            SpectraDecontaminator.run() for more details.

        """
        detector_map = SpectraDecontaminator.parse_detectors(det_str, self._location_table_info)

        file_counter = 0

        for dither in detector_map.keys():
            for detector in detector_map[dither]:
                part = 0
                while True:
                    part += 1
                    file_counter += 1
                    self._logger.info(f"Beginning on detector {detector}, part {part}.")
                    exit_code = self._run_part_in_subprocess(self._args, dither, detector, part, file_counter)
                    if exit_code == self.FINISHED_ALL_PARTS:
                        self._logger.info(f"Finished multi-part decontamination for detector {dither}.{detector}.")
                        self._concatenate_in_subprocess(dither, detector)
                        break
                    elif exit_code != 0:
                        self._logger.info(f"Exiting detector {detector}, part {part} with error code {exit_code}.")
                        sys.exit(exit_code)

    def _run_one_part(self, file_part):
        """Runs the code on a single part of a detector."""

        dither = os.getenv(self.DITHER)
        detector = os.getenv(self.DETECTOR)
        file_number = os.getenv(self.FILE_COUNTER)

        if dither is None or detector is None or file_number is None:
            raise RuntimeError(f"When {self.FILE_PART} is set, {self.DITHER}, {self.DETECTOR}, "
                               f"and {Executor.FILE_COUNTER} must also be set.")

        location_tables = LocationTableSet(self._location_table_info, self._logger)
        if self._cutouts is not None:
            self._cutouts._make_index(location_tables.objects_ids)        

        decon = SpectraDecontaminator(self._mdb,
                                      self._config,
                                      self._exposures,
                                      location_tables,
                                      self._relflux_flat,
                                      self._sens,
                                      self._cutouts,
                                      self._args.limit_band,
                                      self._args.limit_mag,
                                      self._args.zeroth_order_limit,
                                      self._args.second_order_limit,
                                      self._args.padding)

        decon_spectra = decon.run_part(dither=int(dither),
                                       detector=int(detector),
                                       part=int(file_part),
                                       max_spectra=self._args.max_spectra,
                                       decont_order=self._args.order,
                                       passthrough=self._args.passthrough,
                                       skip_measuring=self._args.no_measure)
        decon_spectra.mdb = self._mdb

        if self._args.overwrite:
            mode = 'overwrite'
        else:
            mode = 'create' if not os.path.isfile(self._args.dec_spectra) else 'append'

        self._logger.info(f"Saving a file containing {len(decon_spectra)} decontaminated spectra...")

        decon_spectra.write(self._workdir, self._args.dec_spectra, location_tables, mode, parallel=False)

        if decon_spectra.is_final_part:
            sys.exit(self.FINISHED_ALL_PARTS)

    def _check_input_files(self) -> None:
        """
        Checks that the input files are consistent with one another and makes sure that the files are monochromatic.

        Creates basic descriptions of each location table and exposure: (dither, observation ID, grism name, tilt).
        Then verifies that each table has a corresponding exposure with the same description. An exception is raised
        if no matching values of the dither, observation ID, grism name, and tilt can be found.

        Also raises an exception if a mixture of red and blue grisms were provided as input.

        Raises
        ------

        ValueError:

        Raised if the dither, exposure ID, grism, and, tilt in the location tables do not match any of
        the exposures OR if a mixture of RGS and BGS data files are provided.
        """
        table_descriptions = set()
        exposure_descriptions = set()
        grism_color = set()

        for dither, detector, tables in self._location_table_info.metadata.items:
            for table in tables:
                descr = (dither, table.getObservationID(), table.getGWAPositionName(), table.getGWATilt())
                table_descriptions.add(descr)
                color = ''.join(filter(lambda c: not c.isdigit(), table.getGWAPositionName()))
                grism_color.add(color)

        for dither, exposure in self._exposures:
            metadata = exposure.getMetadata()           
            descr = (dither, metadata.get('OBS_ID'), metadata.get('GWA_POS'), metadata.get('GWA_TILT'))
            exposure_descriptions.add(descr)
            color = ''.join(filter(lambda c: not c.isdigit(), metadata.get('GWA_POS')))
            grism_color.add(color)

        if not table_descriptions.issubset(exposure_descriptions):
            message = "Comparing (dither number, OBS_ID, grism, tilt) of the tables and exposures. "
            for table in table_descriptions:
                if not table in exposure_descriptions:
                    message += f"There is no exposure corresponding to the table: {table}.\n"
                    message += f"The exposures that we have are: {exposure_descriptions}."
            raise ValueError(message)

        if len(grism_color) > 1:
            # at this point, the set should only contain "RGS" or "BGS"; if the color set is > 1, there is an issue.
            raise ValueError("Input files must use the red-only or blue-only grisms. A mixture of blue and red "
                             "grism data cannot be processed.")

    def _get_conf_filename(self, work_dir: str, sir_config: str):
        """
        Gets the fully-qualified filename of the .ini file containing decontamination parameters.

        Parameters
        ----------

        work_dir: str
            The working directory.

        sir_config: str
            The filename of the XML file containing configuration parameters for the SIR Pipeline.

        Returns
        -------

        factory_parameter_filename: str
            Fully-qualified name of the .ini parameter file for the SpectraDecontamination class.
        """
        config_set = DpdSirConfigurationSet(filename=os.path.join(work_dir, sir_config))
        parameter_file = config_set.get_parameter_file('SpectraDecontamination')
        ini_file = os.path.join(work_dir, 'data', parameter_file)

        if not os.path.isfile(ini_file):
            raise FileNotFoundError(f"No parameter file {ini_file}")

        return ini_file

    def _concatenate_in_subprocess(self, dither: int, detector: int):
        self._logger.info(f"Concatenating file parts...")
        p = mp.Process(target=concatenate, args=(self._args, dither, detector))
        p.start()
        p.join()

    @staticmethod
    def _run_part_in_subprocess(prog_args, dither, detector, part, file_counter):
        command = ['SIR_SpectraDecontamination',
                   '--workdir', prog_args.workdir,
                   '--sci_frames', prog_args.sci_frames,
                   '--loc_tables', prog_args.loc_tables,
                   '--padding', str(prog_args.padding),
                   '--mdb', prog_args.mdb,
                   '--relflux_flat', prog_args.relflux_flat,
                   '--sens', prog_args.sens,
                   '--cutouts', prog_args.cutouts,
                   '--limit_mag', str(prog_args.limit_mag),
                   '--limit_band', prog_args.limit_band,
                   '--detectors', prog_args.detectors,
                   '--zeroth_order_limit', str(prog_args.zeroth_order_limit),
                   '--second_order_limit', str(prog_args.second_order_limit),
                   '--order', str(prog_args.order),
                   '--max_spectra', str(prog_args.max_spectra),
                   '--dec_spectra', prog_args.dec_spectra]

        if prog_args.overwrite and file_counter == 1:
            command.append('--overwrite')

        if prog_args.no_measure:
            command.append('--no_measure')

        if prog_args.passthrough:
            command.append('--passthrough')

        if prog_args.pipeline:
            command.append('--pipeline')

        if prog_args.logdir is not None:
            command.append('--logdir')
            command.append(prog_args.logdir)

        if prog_args.config is not None:
            command.append('--config')
            command.append(prog_args.config)

        env = os.environ

        env[Executor.FILE_COUNTER] = str(file_counter)
        env[Executor.FILE_PART] = str(part)
        env[Executor.DITHER] = str(dither)
        env[Executor.DETECTOR] = str(detector)

        completed = subprocess.run(command, env=env)

        return completed.returncode
