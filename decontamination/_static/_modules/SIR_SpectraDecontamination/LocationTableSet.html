[docs]class LocationTableSet:
    """
    Loads and manages a set of h5LocationTable files.
    """

    def __init__(self, table_info: 'LocationTableSetInfo', logger=None):
        """
        Loads the h5LocationTable files.

        Parameters
        ----------

        table_info: LocationTableSetInfo
            The LocationTableSetInfo object corresponding to the set of LocationTable files that
            will be loaded.

        logger: logging.Logger
            [optional] If provided, the class will print progress messages to the log.

        Raises
        ------

        FileNotFoundError: if the input file does not exist.
        """
        self._info = table_info
        self._logger = logger

        self._tables = []

        self._log("Loading the location tables specified in ", table_info.json_filename)

        self._empty_detectors = set()

        ini_file = self._info.location_factory_params

        for idx in table_info.selection:
            hdf5_filename = table_info.hdf5_filenames[idx]
            self._log("Loading", hdf5_filename)
            table = H5LocationTable.load(hdf5_filename)
            dither = table.getDither()
            detector = table.getDetectorNumber()

            if table.getNumberOfSpectra(1) == 0:
                self._log(f"Ignoring dither {dither}, {table.getDetectorName()} because it is empty.")
                self._empty_detectors.add((dither, detector))
            else:
                table.lo_factory = LocationObjectFactory(table.opt_model, ini_file, 'DECONTAMINATION')
                self._tables.append(table)

        self._log(f"Loaded:\n{self}")

    @property
    def info(self) -> 'LocationTableSetInfo':
        return self._info

    @property
    def empty_detectors(self):
        return self._empty_detectors

    def __getitem__(self, table_idx):
        """

        Parameters
        ----------

        table_idx: int
            The index of the table within the list of tables.

        Returns
        -------

        table: H5LocationTable
        """
        return self._tables[table_idx]

    def __iter__(self):
        return (table for table in self._tables)

    def __len__(self):
        return len(self._tables)

    def __str__(self):
        strings = ["  Index    Dither    Detector   N_spectra",
                   "-------------------------------------------"]

        for idx, table in enumerate(self._tables):
            detector = table.getDetectorNumber()
            detector_name = 'All' if detector == 0 else DetectorModel.getDetectorName(detector)
            strings.append(f"{idx:^9}{table.getDither():^10}{detector_name:^12}{table.getNumberOfSpectra(1):^12}")

        return "\n".join(strings)

    def _log(self, *message):

        if self._logger is not None:
            self._logger.info(" ".join(message))
    
    @cached_property
    def objects_ids(self):
        _ids = [table.getIds() for table in self._tables]
        _ids = np.concatenate(_ids)
        return np.unique(_ids)
