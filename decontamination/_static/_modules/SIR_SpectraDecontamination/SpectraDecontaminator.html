    def __init__(self,
                 mdb,
                 config,
                 survey_exposures,
                 location_tables,
                 relflux_flat=None,
                 sensitivity=None,
                 cutouts=None,
                 limit_band='H',
                 limit_mag='1000',
                 zeroth_order_mag=20,
                 second_order_mag=16,
                 padding=0):
        """

        Parameters
        ----------

        mdb: DpdMdbDataBase
            A SIR Utilities Mdb class instance containing data from the official mission database and the
            additional information, specific to SIR, that is not (yet) part of the official Euclid MDB.

        config: ConfigObj
            Contains the parsed contents of the spectra decontamination .ini file.

        survey_exposures: ExposureSet
            Contains a set of NispSurveyExposure objects that have each been loaded with data from a FITS file.

        location_tables: LocationTableSet
            Contains a set of H5LocationTable objects, containing information loaded from location tables (HDF5 files)
            produced by the SIR_SpectraLocation pipeline module.

        relflux_flat: astropy.io.fits.hdu.hdulist.HDUList
            The relative flux correction master flat cube, produced by the SIR RelativeFluxCalibration pipeline.

        sensitivity: astropy.io.fits.hdu.hdulist.HDUList
            The absolute flux sensitivity function (the grism sensitivity derived from SirAbsoluteFluxScaling).

        cutouts: SIR_CreateCuts.CutoutCollection
            Contains metadata and pixel arrays corresponding to stacked, mosaics of direct image stamps.

        limit_band: str
            The name of the band that will be used when performing a magnitude cut on the input data.

        limit_mag: float
            The magnitude limit; objects with `limit_band` magnitude greater than this value will be ignored
            when performing the decontamination.

        zeroth_order_mag: float
            The magnitude limit for flagging the positions of zeroth-order spectra. Zeroth-order spectra
            that are fainter than this limit in the Y, J, and H bands will be completely ignored.

        second_order_mag: float
            The magnitude limit for flagging the positions of second-order spectra. Second-order spectra
            that are fainter than this limit in the Y, J, and H bands will be completely ignored.

        padding: int
            The number of pixels around the box specified in the `location_tables` to include in the spectral
            cutouts.
        """
        self._logger = log.getLogger('SpectraDecontaminator')

        self._logger.info("Constructing the SpectraDecontaminator.")

        self._location_tables = location_tables

        self._check_for_missing_exposures(survey_exposures, location_tables)

        # Information about the photometric bands / filters
        self._phot_info = PhotometerInfo(mdb)

        self._config = config

        self._exposures = survey_exposures

        self._relflux_flat = relflux_flat

        self._cutouts = cutouts

        AstroObjectInfo.ZEROTH_ORDER_THRESHOLD = zeroth_order_mag
        AstroObjectInfo.SECOND_ORDER_THRESHOLD = second_order_mag

        sigcont_fraction, invalid_contamination_fraction = self._get_significant_contamination_threshold()

        self._logger.info(f"sigcont_fraction = {sigcont_fraction}.")
        self._logger.info(f"invalid_contamination_fraction = {invalid_contamination_fraction}.")

        self._padding = padding

        contam_band, contamin_limit, contam_dmag = self.get_contaminant_mag_parameters(config)

        self._logger.info(f"contaminant_mag_band = {contam_band}.")
        self._logger.info(f"contaminant_mag_limit = {contamin_limit}.")
        self._logger.info(f"contaminant_dmag = {contam_dmag}.")

        self._contaminant_band = contam_band
        self._contamin_mag_limit = contamin_limit
        self._contaminant_dmag = contam_dmag

        self._shrink_exposureset()

        # A dict in the format: {dither: DetectorGrism}
        self._det = {dither: DetectorGrism(mdb, config, exposure, sensitivity) for dither, exposure in survey_exposures}

        if len(self._det) == 0:
            self._logger.error("The DetectorGrism dict is empty. Perhaps the detectors contain no spectra? Exiting.")
            sys.exit(2)

        self._logger.info("Organizing spectra by object id.")

        # A dict of the form: {object_id_as_int: AstroObjectInfo}
        self._objects_by_id = self._make_object_info_table(location_tables, limit_band, limit_mag)

        self._logger.info("Organizing spectra by dither and detector")

        # NestedDict in format {dither: {detector: [SpectrumInfo]}}
        self._spectrum_info_by_detector = self._make_spectrum_lookup_table(self._objects_by_id)

        # NestedDict in format {dither: {detector: SearchGrid}}
        self._search_grids = None

        # The SpectrumModeller object responsible for modelling spectra:
        self._modeller = None

        self._dec_counter = 0
        self._dec_call_counter = 0


[docs]    def run(self, detectors: str, decont_order: int=1, passthrough=False,
            skip_measuring: bool=False) -> DecontaminatedSpectraCollection:
        """
        Carries out the steps needed to decontaminate the first-order spectra of each object.

        Parameters
        ----------

        detectors: str
           The detector(s) which will be decontaminated. These are specified as a string of the form 'dither.detector',
           where the dither is one of `('1', '2', '3', '4')` and specifies the exposure in the dither sequence from
           which to obtain 2-D spectra that will be decontaminated. The detector is one of ('1', '2', '3', ..., '16')
           and indicates the detector within the specified exposure to decontaminate. If the string '*' is provided,
           then all detectors of all dithers will be decontaminated. If '*.1' is provided, then detector 1 of all four
           exposures (dithers 1, 2, 3, 4) will be decontaminated. Finally, if `'2.*'` is specified, then all detectors
           of dither 2 will be decontaminated.

        decont_order: int
           The spectral order to decontaminate (should be 1 or 2)

        passthrough: bool
           If True, the decontamination will not actually be performed. Cutouts of the spectra will be created and
           saved to the DecontaminatedSpectraCollection without modification. If False, the decontamination will proceed
           as usual.

        skip_measuring: bool
           If True, then the model contaminants will only be based on the broadband photometry; no effort will be made
           to measure the spectra.

        Returns
        -------

        DecontaminatedSpectraCollection instance containing the decontaminated spectra, flags, variance, and model
        contaminants.
        """
        if self._search_grids is None:
            self._search_grids = self._make_search_grids()
            self._modeller = SpectrumModeller(self._phot_info, self._det, self._search_grids, self._padding)
            if self._relflux_flat is not None:
                self._modeller.set_relflux(self._relflux_flat)
            AstroObjectInfo.WAVELENGTHS = self._modeller.lambdas

        if not passthrough:
            self._logger.info("Beginning the decontamination procedure.")
        else:
            self._logger.info("Simply cropping the spectra, since passthrough is set to True")

        # This step crops each `decont_order` spectrum in each detector of each exposure.
        # The output is a NestedDict in the format {dither: {detector: [SpectrumCutout]}}
        spectral_cutouts = self._create_spectral_cutouts(decont_order)

        # Parse the detector specification string:

        detector_map = self.parse_detectors(detectors, self._location_tables.info)

        # Remove any entries in the detector map that have no spectral data

        detector_map = self._prune_detector_map(detector_map, spectral_cutouts)

        # Create container for storing the set of decontaminated spectral cutouts:

        decontaminated_spectra = NestedDict(list, 'dither', 'detector', plural_key='dithers')

        # At this point, we have cutouts for all spectra. We can now decontaminate each spectrum:

        for dither in detector_map.keys():
            for detector in detector_map[dither]:
                cutouts = []

                self._logger.info(f"Working on dither {dither}, detector {detector}.")

                spec_cutouts = spectral_cutouts[dither, detector]

                if spec_cutouts is not None:
                    for spec in spectral_cutouts[dither, detector]:
                        self._decontaminate_spectrum(spec, passthrough, skip_measuring)
                        cutouts.append(spec)

                for spec_info in self._spectrum_info_by_detector[dither, detector]:
                    spec_info.clear_convolution_kernel()

                decontaminated_spectra[dither, detector] = cutouts

        if not passthrough:
            self._logger.info("Finished decontaminating the spectra. Now updating the variance layers.")
        else:
            self._logger.info("Finished cropping the spectra. The spectra were not decontaminated because "
                              "passthrough was set to True.")

        for dither in detector_map.keys():
            for detector in detector_map[dither]:
                if decontaminated_spectra[dither, detector] is not None:
                    self._finalize_spectra(decontaminated_spectra[dither, detector], passthrough, skip_measuring)

        self._logger.info("The variance for all spectra has been updated.")

        # delete the search grids

        self._search_grids = None

        return DecontaminatedSpectraCollection(decontaminated_spectra=decontaminated_spectra)



[docs]    def run_part(self, dither: int, detector: int, part: int, max_spectra: int=0, decont_order: int=1,
                 passthrough=False, skip_measuring: bool=False) -> Union[DecontaminatedSpectraCollection, None]:
        """
        Carries out the steps needed to decontaminate the first-order spectra of each object.

        Parameters
        ----------

        dither: int
            The dither sequence number.

        detector:
            The detector number (1-16).

        part: int
            The section of the spectra list to decontaminate. The spectra will be in groups of `max_spectra`.

        max_spectra: int
           The maximum number of spectra that will be stored in a single DecontaminatedSpectraCollection object. The
           default value (zero) indicates that an unlimited number of spectra can be stored in the object.

        decont_order: int
           The spectral order to decontaminate (should be 1 or 2)

        passthrough: bool
           If True, the decontamination will not actually be performed. Cutouts of the spectra will be created and
           saved to the DecontaminatedSpectraCollection without modification. If False, the decontamination will proceed
           as usual.

        skip_measuring: bool
           If True, then the model contaminants will only be based on the broadband photometry; no effort will be made
           to measure the spectra.

        Returns
        -------

        decontaminated_spectra: DecontaminatedSpectraCollection or None

        A DecontaminatedSpectraCollection instance containing the decontaminated spectra, flags, variance, and model
        contaminants. Returns None if the part does not exist.
        """
        if self._search_grids is None:
            self._search_grids = self._make_search_grids()
            self._modeller = SpectrumModeller(self._phot_info, self._det, self._search_grids, self._padding)
            if self._relflux_flat is not None:
                self._modeller.set_relflux(self._relflux_flat)
            AstroObjectInfo.WAVELENGTHS = self._modeller.lambdas

        if not passthrough:
            self._logger.info("Beginning the decontamination procedure.")
        else:
            self._logger.info("Simply cropping the spectra, since passthrough is set to True")

        # This step crops each `decont_order` spectrum in each detector of each exposure.
        # The output is a NestedDict in the format {dither: {detector: [SpectrumCutout]}}
        spectral_cutouts = self._create_spectral_cutouts(decont_order)

        # At this point, we have cutouts for all spectra. We can now decontaminate each spectrum:

        cutouts = []

        spec_cutouts = spectral_cutouts[dither, detector]

        if spec_cutouts is None:
            return None

        n_parts = math.ceil(len(spec_cutouts) / max_spectra)

        if part > n_parts:
            return None

        self._logger.info(f"Working on dither {dither}, detector {detector}, part {part} of {n_parts}.")

        for spec in spec_cutouts[(part - 1) * max_spectra:]:
            self._decontaminate_spectrum(spec, passthrough, skip_measuring)
            cutouts.append(spec)
            if len(cutouts) == max_spectra:
                break

        self._finalize_spectra(cutouts, passthrough, skip_measuring)
        decontaminated_spectra = NestedDict(list, 'dither', 'detector', plural_key='dithers')
        decontaminated_spectra[dither, detector] = cutouts

        for spec_info in self._spectrum_info_by_detector[dither, detector]:
            spec_info.clear_convolution_kernel()

        # delete the search grids

        self._search_grids = None

        self._logger.info("Preparing to save a section of a detector.")

        collection = DecontaminatedSpectraCollection(decontaminated_spectra=decontaminated_spectra)

        collection.is_final_part = part == n_parts

        return collection



[docs]    def make_models(self, detectors: str, pixel_threshold: float=0) -> NestedDict:
        """
        Creates model spectra for each object in the specified detectors.

        Parameters
        ----------

        detectors: str
           The detector(s) for which model spectra will be created. These are specified as a string of the form
           'dither.detector', as described in the documentation for the run() method.

        pixel_threshold: float
           The minimum pixel threshold. Models whose maximum pixel value is lower than this threshold will be ignored.

        Returns
        -------

        A NestedDict instance containing the models of spectra falling on the specified detectors.
        """
        if self._modeller is None:
            self._modeller = SpectrumModeller(photo_info=self._phot_info,
                                              detector_grisms=self._det,
                                              search_grids=None,
                                              padding=self._padding)
            if self._relflux_flat is not None:
                self._modeller.set_relflux(self._relflux_flat)
            AstroObjectInfo.WAVELENGTHS = self._modeller.lambdas

        detector_map = self.parse_detectors(detectors, self._location_tables.info)

        models = NestedDict(tuple, 'dither', 'detector', 'object_id', 'order', plural_key='dithers')

        for dither, detectors in detector_map.items():
            for detector in detectors:
                for spectrum in self._spectrum_info_by_detector[dither, detector]:
                    model = self._modeller.get_2d_model_spectrum(spectrum, skip_measuring=True)
                    if model is not None and model.pixels.max() > pixel_threshold:
                        wavelengths = Utils.compute_wavelengths(spectrum.location_spectrum, spectrum.bbox)
                        models[dither, detector, spectrum.object_id, 1] = (model, wavelengths)
        return models


    def _list_dithers(self):
        """
        Lists the dithers contained in the location tables.

        Returns
        -------

        list
            A list of integers corresponding to the dithers present in the collection.
        """
        return list(self._spectrum_info_by_detector.dithers())

    def _list_detectors(self, dither):
        """
        Lists the detectors within the specified dither that are present in the collection.

        Parameters
        ----------

        dither: int
            The dither sequence number (1, 2, 3, 4).

        Returns
        -------

        list
            A list of integers [1, 16] corresponding to the detectors of the specified dither that are
            present in the collection.
        """
        return list(self._spectrum_info_by_detector.keys(dither))

    def _find_spectra(self, dither, detector):
        """
        Identifies the SpectrumInfo objects associated with the specified detector in the specified dither.

        Parameters
        ----------

        dither: int
            The dither sequence number (1, 2, 3, 4)

        detector: int
            The detector number (1, 2, 3, ..., 16)

        Returns
        -------

        list
            A list of SpectrumInfo objects for objects that appear on the specified detector of the dither.
        """
        return self._spectrum_info_by_detector[dither, detector]

    def _make_search_grids(self):
        self._logger.info("Creating search grids.")
        search_grids = NestedDict(SearchGrid, 'dither', 'detector', plural_key='dithers')
        for dither in self._list_dithers():
            for detector in self._list_detectors(dither):
                spectra = self._find_spectra(dither, detector)
                tilt_angle = 0.0 if len(spectra) == 0 else spectra[0].location_spectrum.getGWATilt()
                search_grids[dither, detector] = SearchGrid(spectra, tilt_angle)

        return search_grids

    def _make_object_info_table(self, location_tables, limit_band, limit_mag):
        # A dict with the format {object_id: AstroObjectInfo}
        object_info = {}

        for table in location_tables:
            dither = int(table.getDither())

            for object_id in table:
                loc = table.getLocationObject(object_id)
                astro_object = loc.getAstronomicalObject()
                loc = table.create_location_object(astro_object)

                if object_id in object_info:
                    # the object is already here. Just update the spectra location info.
                    object_info[object_id].load_spectra(loc, self._det[dither], self._padding)
                else:
                    # the object has not been added to _objects_by_id yet; add it now:
                    object_data = AstroObjectInfo(loc, self._cutouts, self._phot_info, limit_band, limit_mag)

                    # If the object is bright enough, load its spectra and add it to object_info
                    if object_data.is_usable:
                        object_data.load_spectra(loc, self._det[dither], self._padding)
                        object_info[object_id] = object_data

        return object_info

    def _create_spectral_cutouts(self, decont_order):
        spec_cutouts = NestedDict(list, 'dither', 'detector', plural_key='dithers')

        for dither in self._list_dithers():
            for detector in self._list_detectors(dither):
                spec_cutouts.add_item([], dither, detector)
                spectra = self._find_spectra(dither, detector)

                for spec in spectra:
                    if spec.order == decont_order and spec.area != 0:
                        spec_cutouts[dither, detector].append(spec.spectrum_cutout)

        return spec_cutouts

    def _decontaminate_spectrum(self, spec, passthrough=False, skip_measuring=False):
        """
        Decontaminate a single spectrum.

        Parameters
        ----------

        spec: SpectrumCutout
            The SpectrumCutout object associated with the spectrum that will be decontaminated.

        passthrough: bool
           If True, the decontamination will not actually be performed. Cutouts of the spectra will be created and
           saved to the DecontaminatedSpectraCollection without modification. If False, the decontamination will proceed
           as usual.

        skip_measuring : bool
           If set to True, the model spectra will be based solely on the lower and upper band magnitudes; the process of
           attempting to measure the spectrum will be skipped.

        Side Effects
        ------------

            `spec` is updated. More specifically, the `contaminants`, `science`, and `flags` properties are set.
        """
        self._dec_call_counter += 1
        if spec.science is not None and spec.flags is not None:
            # the spectrum has already been decontaminated; return
            return

        self._dec_counter += 1
        spec_info = spec.spec_info
        dither = spec_info.dither
        detector = spec_info.detector
        object_id = spec_info.object_id
        spec_offset = spec_info.get_offset()
        relevant_searchgrid = self._search_grids[dither, detector]
        if not spec_info.astro_object_info.invalid_mag:
            relative_mag_limit = spec_info.astro_object_info.mag(self._contaminant_band) + self._contaminant_dmag
        else:
            relative_mag_limit = self._contamin_mag_limit
        contam_mag_limit = min(self._contamin_mag_limit, relative_mag_limit)
        spec.contaminants = relevant_searchgrid.find_contaminants(spec_info.searchgrid_bbox,
                                                                  ignore_idx=object_id,
                                                                  mag_limit=contam_mag_limit,
                                                                  mag_band=self._contaminant_band)

        # At this point, spec contains the cutout *and* it knows about the contaminants.

        original_science = spec.original_spectrum.get_science()
        original_flags = spec.original_spectrum.get_mask()

        # Create a blank canvas on which to accumulate the contaminating flux:

        total_contamination = np.zeros_like(original_science)

        # Create an array to store the updated flags:

        new_flags = original_flags.astype(np.uint32)

        if spec.has_contaminants and not passthrough:
            self._logger.debug(f"decontaminating {dither}.{detector}.{object_id}")

            for contam in spec.contaminants:
                # note the side effect of get_2d_model_spectrum(); it adds the model to the spectrum info.
                model_of_contaminant = self._modeller.get_2d_model_spectrum(contam, skip_measuring)

                # Add the model spectrum to the total contamination for this object, unless the
                # contaminant is a zeroth-order spectrum (None), in which case, we just flag it:

                contamination_canvas = (spec_offset, total_contamination)

                flag_canvas = (spec_offset, new_flags)

                if model_of_contaminant is not None:

                    model_overlay = (model_of_contaminant.offset, model_of_contaminant.pixels)

                    Utils.apply_overlay(overlay=model_overlay, canvas=contamination_canvas, operation=np.add)

                    interpolation_mask = (model_of_contaminant.offset, model_of_contaminant.interpolated_pixels)

                    Utils.set_flag(mask=interpolation_mask, canvas=flag_canvas, flag=Flags.MISSING)
                elif contam.order in (0, 2) or contam.astro_object_info.invalid_mag:
                    # This contaminant is a zeroth-order, a bright second-order, or an object with unknown magnitude,
                    # which means that it cannot be modelled. Flag it:
                    self._flag_zeroth(contam.bbox, spec_offset, new_flags)

                contam.clear_convolution_kernel()

            # Remove contamination:

            spec.science = original_science - total_contamination

            # Update new_flags by flagging pixels that are significantly contaminated:

            threshold, invalid_threshold = self._get_significant_contamination_threshold()

            self._flag_significantly_contaminated(spec.science,
                                                  total_contamination,
                                                  new_flags,
                                                  threshold,
                                                  invalid_threshold)

            spec.flags = new_flags
        else:
            # this spectrum had no significant contamination (or passthrough was set to True). We still have to set
            # the science and flag layers:
            spec.science = original_science
            spec.flags = original_flags

    def _update_variance(self, decon_spec, passthrough: bool=False, skip_measuring: bool=False):
        """
        Computes the updated variance layer of the `SpectrumCutout`, `spec` and updates the variance layer.

        Parameters
        ----------

        decon_spec: SpectrumCutout.SpectrumCutout
            A SpectrumCutout object whose science layer has already been decontaminated, but whose
            variance has not been updated.

        passthrough: bool
            If True, then no decontamination was done, so the variance passes through, unaltered. If false, the variance
            is updated.

        skip_measuring: bool
            If True, then the model spectra used in the decontamination are based on the broadband fluxes.
        """
        object_id = decon_spec.spec_info.object_id
        dither = decon_spec.spec_info.grism_model.dither
        detector = decon_spec.spec_info.detector

        # check to make sure that the science layer has been decontaminated:

        self._logger.debug(f"Computing variance for {dither}.{detector}.{object_id}")

        if decon_spec.science is None:
            raise RuntimeError(f"The spectrum {dither}.{detector}.{object_id} has not been decontaminated.")

        if decon_spec.variance is not None:
            raise RuntimeError(f"The variance of {dither}.{detector}.{object_id} has already been set.")

        variance = np.copy(decon_spec.original_spectrum.get_variance())

        decon_spec.variance = variance

        if passthrough or decon_spec.spec_info.model_spec is None:
            return

        variance_canvas = (decon_spec.spec_info.get_offset(), variance)

        for contam in decon_spec.contaminants:
            if contam.order in (0, 2) or contam.astro_object_info.invalid_mag:
                # The contaminant cannot be modelled because it is a zeroth-order, or a bright second-order, or it has
                # unknown or invalid magnitudes. So we set infinite variance for the region covered by this spectrum.
                _, _, width, height = contam.bbox
                var = np.empty((height, width))
                var[:] = np.inf
            else:
                if contam.spectrum_cutout.science is None:
                    self._decontaminate_spectrum(contam.spectrum_cutout, passthrough, skip_measuring)

                if contam.model_spec is None:
                    self._modeller.get_2d_model_spectrum(contam, skip_measuring=skip_measuring)

                contam_residual = contam.spectrum_cutout.science - contam.model_spec.pixels

                contam_residual_canvas = (contam.get_offset(), contam_residual)

                # contam_residual includes the result of subtracting away the model of decon_spec. We add the
                # model of decon_spec back to contam_residual and subtract away the decontaminated spectrum, so that
                # the residual here is not based at all upon the model of decon_spec.

                residual_correction = decon_spec.spec_info.model_spec.pixels - decon_spec.science
                
                residual_correction_overlay = (decon_spec.spec_info.get_offset(), residual_correction)
                
                Utils.apply_overlay(overlay=residual_correction_overlay,
                                    canvas=contam_residual_canvas,
                                    operation=np.add)

                var = np.square(contam_residual)

            contam_variance_overlay = (contam.get_offset(), var)

            Utils.apply_overlay(overlay=contam_variance_overlay, canvas=variance_canvas, operation=np.fmax)

        decon_spec.variance = variance

    def _finalize_spectra(self, spectra: List['SpectrumCutout'], passthrough: bool, skip_measuring: bool):
        """
        Updates the variance of the list of spectra objects and frees unneeded data from memory.
        """
        for spec in spectra:
            if spec.variance is None:
                self._update_variance(spec, passthrough, skip_measuring)
            spec.clear_original_spectrum()

    def _get_significant_contamination_threshold(self) -> Tuple[float, float]:
        if 'PARAMETERS' not in self._config:
            raise KeyError(f"The configuration file, {self._config.filename} is expected to include a section"
                           " called PARAMETERS.")

        if 'sigcont_fraction' in self._config['PARAMETERS']:
            sigcont_frac = float(self._config['PARAMETERS']['sigcont_fraction'])
            if sigcont_frac > 1 or sigcont_frac < 0:
                raise ValueError(f"The sigcont_fraction is expected to be in the interval [0,1], not {sigcont_frac}.")
        else:
            raise KeyError(f"The PARAMETERS section of the configuration file, {self._config.filename} is expected"
                           f" to include variable called 'sigcont_fraction' with a value in the interval [0,1].")

        if 'invalid_contamination_fraction' in self._config['PARAMETERS']:
            invalid_frac = float(self._config['PARAMETERS']['invalid_contamination_fraction'])
            if invalid_frac <= 0.0:
                raise ValueError(f"The invalid_contamination_fraction is expected to be > 0.0, not {invalid_frac}.")
        else:
            raise KeyError(f"The PARAMETERS section of the configuration file, {self._config.filename} is expected"
                           f" to include a variable called 'invalid_contamination_fraction' with value > 0.0.")

        return sigcont_frac, invalid_frac

    def parse_detectors(cls, detector_string: str, tables: 'LocationTableSetInfo') -> dict:

        detector_string = detector_string.strip().lower().replace('det_', '')

        if detector_string == '*':
            detector_string = '*.*'

        if not cls._is_valid_pattern(detector_string):
            raise ValueError("Expected a string of the form 'dither.detector' or 'dither0,dither1.detector0,detector1'")

        dither_spec, detector_spec = detector_string.split('.')

        dithers = tables.get_dithers() if dither_spec == '*' else cls._parse_dither_specifier(dither_spec)

        detector_map = {}

        if detector_spec != '*':
            for dith in dithers:
                detector_map[dith] = cls._parse_detector_specifier(detector_spec)
        else:
            for dith in dithers:
                detector_map[dith] = tables.get_detectors(dith)

        return detector_map


    @staticmethod
    def _is_valid_pattern(pattern: str) -> bool:
        pattern = pattern.lower().replace(',', 'c').replace('.', 'p').replace('*', 'a').replace('det_', '')

        acceptable_1 = re.compile('^ap([1-4]{2}c){0,15}[1-4]{2}$')
        acceptable_2 = re.compile('^(([0-9]){1,3}c){0,100}[0-9]{1,3}pa$')
        acceptable_3 = re.compile('^(([0-9]){1,3}c){0,100}[0-9]{1,3}p([1-4]{2}c){0,15}[1-4]{2}$')

        match_1 = acceptable_1.fullmatch(pattern)
        match_2 = acceptable_2.fullmatch(pattern)
        match_3 = acceptable_3.fullmatch(pattern)

        is_type_0 = pattern == 'apa'
        is_type_1 = match_1 is not None and match_1.span() == (0, len(pattern))
        is_type_2 = match_2 is not None and match_2.span() == (0, len(pattern))
        is_type_3 = match_3 is not None and match_3.span() == (0, len(pattern))

        return any((is_type_0, is_type_1, is_type_2, is_type_3))

    def _parse_detector_specifier(spec):
        id_to_number_map = DetectorModel.getIDs2Numbers()

        if Utils.represents_int(spec):
            return [id_to_number_map[spec]]
        elif spec.count(',') > 0:
            return [id_to_number_map[str(det_id)] for det_id in json.loads(f"[{spec}]")]
        else:

    @staticmethod
    def _check_for_missing_exposures(survey_exposures, location_tables):
        """
        Checks for missing exposures. If an exposure is mentioned in the location tables, but not present in the list of
        input exposures, then an error message is logged and the code exits.

        Parameters
        ----------

        survey_exposures: ExposureSet
            Contains a set of NispSurveyExposure objects that have each been loaded with data from a FITS file.

        location_tables: LocationTableSet
            Contains a set of H5LocationTable objects, containing information loaded from location tables (HDF5 files)
            produced by the SIR_SpectraLocation pipeline module.

        Raises
        ------

        UserWarning:
            Raised if the location tables refer to exposures (dithers) that are not present in survey_exposures.
        """
        missing_exposures = set()
        exposures_referenced = set()

        for table in location_tables:
            dither = table.getDither()
            exposures_referenced.add(dither)
            if dither not in survey_exposures:
                missing_exposures.add(dither)

        if len(missing_exposures) > 0:
            raise UserWarning(f"The location tables refer to exposures {exposures_referenced}, but {missing_exposures}"
                              " could not be found in the set of exposures provided.")

    @staticmethod
    def _flag_zeroth(contam_box: tuple, flag_offset: tuple, flag_layer: np.ndarray):
        """
        Flags the pixels in the mask (flag) layer, falling within the `box` as the position of a zeroth-order
        spectrum.

        Parameters
        ----------

        contam_box: tuple
            A 4-tuple (left, bottom, width, height) specifying the bounding box of a zeroth-order spectrum.

        flag_offset: tuple
            The position of the lower-left pixel of the flag layer within the detector in the format (left, bottom)

        flag_layer: 2-D NumPy array
            A 2-D array where the offset is a tuple of (left, bottom) specifying the
            position of the lower-left pixel of the array within the detector and the 2-D array contains the
            mask / flag data for a spectrum.

        Side Effects
        ------------

        Modifies flag_layer
        """
        left, bottom = flag_offset

        flag_height, flag_width = flag_layer.shape

        # the bounding box of the flag layer
        flag_box = (left, bottom, flag_width, flag_height)

        flag_boundaries, _ = Utils.bounding_indices(flag_box, contam_box)

        if flag_boundaries is not None:
            flag_left, flag_bottom, flag_right, flag_top = flag_boundaries

            flag_layer[flag_bottom:flag_top, flag_left:flag_right] |= Flags.ZERO
            flag_layer[flag_bottom:flag_top, flag_left:flag_right] |= Flags.INVALID

    @staticmethod
    def _flag_significantly_contaminated(decontaminated_spec, contamination, flag_layer, threshold, invalid_threshold):
        """
        Flags the pixels in which the contamination flux exceeds the decontaminated flux by more than 10%.

        Parameters
        ----------

        decontaminated_spec: 2-D NumPy array
            The decontaminated 2-D spectral cutout.

        contamination: 2-D NumPy array
            An image containing the total estimated contamination from all contaminants.

        flag_layer: 2-D NumPy array
            The mask or flag layer associated with the spectrum.

        threshold: float
            The threshold defining what fraction of flux contamination is considered 'significant'

        invalid_threshold: float
            The threshold defining what fraction of flux contamination is flagged as 'invalid'

        Side Effects
        ------------

        Modifies the `flag_layer` array by setting the significant contamination flag. `SIGCONT`.
        """
        sig_mask = np.fabs(contamination) > threshold * np.fabs(decontaminated_spec)
        invalid_mask = np.fabs(contamination) > invalid_threshold * np.fabs(decontaminated_spec)

        flag_layer[sig_mask] |= Flags.SIGCONT
        flag_layer[invalid_mask] |= Flags.INVALID

    @staticmethod
    def _make_spectrum_lookup_table(objects_by_id):
        # format {dither: {detector: [SpectrumInfo]}}
        spectra_by_detector = NestedDict(list, 'dither', 'detector', plural_key='dithers')

        for object_info in objects_by_id.values():
            for dither, detector, order, spec_info in object_info.spectra.items:
                if (dither, detector) not in spectra_by_detector:
                    spectra_by_detector[dither, detector] = [spec_info]
                else:
                    spectra_by_detector[dither, detector].append(spec_info)

        return spectra_by_detector

    @staticmethod
    def _prune_detector_map(detector_map: dict, cutouts: NestedDict) -> dict:
        """
        Removes entries from the detector map which are not present in the cutouts NestedDict.

        This function only has an effect if only a subset of the spectral locations are provided to the module.

        Parameters
        ----------

        detector_map: dict
            A dictionary of lists of detectors, specifying which detectors will be decontaminated.

        cutouts: NestedDict
            A nested dict with keys (dither, detector) corresponding to the spectra that are actually available.

        Returns
        -------

        pruned_detector_map: A modified dictionary in which only detectors present in the cutouts NestedDict
        are preserved; all other entries are removed, since there are no spectral cutouts in those detectors.

        Side Effects
        ------------

        Modifies detector_map
        """
        map_dithers = list(detector_map.keys())

        for dither in map_dithers:
            if dither not in cutouts.dithers():
                del detector_map[dither]
            else:
                map_detectors = tuple(detector_map[dither])

                for detector in map_detectors:
                    if cutouts[dither, detector] is None:
                        detector_map[dither].remove(detector)

        return detector_map

    def _shrink_exposureset(self):
        location_dithers = set([table.getDither() for table in self._location_tables])

        exposure_dithers = [dither for dither, _ in self._exposures]
