from EL_ImageBinding import NispSurveyExposure

from SIR_Products.DpdSirScienceFrame import DpdSirScienceFrame



[docs]class ExposureSet:
    """
    Loads and manages a set of NispSurveyExposures.
    """

    def __init__(self, workdir: str, filename: str, lt_ids: List[str], logger=None):
        """ Loads the NispSurveyExposure files.

        Args:
            workdir (str): The path to the working directory (i.e., the directory that contains the data/ directory)
            filename (str): The filename of the JSON file containing a list of objects that will be loaded.
                            The list within the JSON file should contain the names of XML files associated with NispSurveyExposure
                            files. Each NispSurveyExposure file is a FITS file, located in a subdirectory called `data/`.
            lt_ids (List[str]): List of string IDs to match the correct exposure set to the loaded LT
            logger (logging.Logger, optional): If provided, the class will print progress messages to the log. Defaults to None.

        Raises:
            ValueError: if multiple exposures with the same dither ID (DITHOBS) are provided.
        """

        self._logger = logger

        self._exposures = {}

        work_dir = os.path.abspath(workdir)

        data_dir = os.path.join(work_dir, 'data')

        self._data_dir = data_dir

        self._log("Loading the list of exposures,", os.path.abspath(filename))

        self._exposure_filenames = []

        with open(filename, 'r', encoding='utf-8') as json_file:
            # read all JSON list of XML science frame products
            for frame_xml in json.load(json_file):
                # open XML product
                frame_obj = DpdSirScienceFrame(os.path.join(work_dir, frame_xml))
                # get unique id for frame to match Location Table
                dither = frame_obj.get_dither()
                frame_id = '_'.join([str(frame_obj.get_gwa_pos()),
                                     str(frame_obj.get_gwa_tilt()),
                                     str(frame_obj.get_ptg_id()),
                                     str(frame_obj.get_obs_id()),
                                     str(dither)])

                if frame_id in lt_ids:
                    # check for dither duplication 
                    if dither in self._exposures:
                        raise ValueError(f"The file {filename} contains multiple exposures with"
                                         f"grism, tilt, PTGID, OBSID, DITHOBS {frame_id.split('_')}.")
                    frame_fits_file = frame_obj.get_frame_fits_file()
                    self._exposure_filenames.append(frame_fits_file)
                    full_path = os.path.join(data_dir, frame_fits_file)
                    self._log("Loading", full_path)
                    # NOTE: still using dither ID, rather than frame_id, asked to hange this on JAN 2024
                    self._exposures[dither] = NispSurveyExposure.readFits(full_path)

    @property
    def data_dir(self) -> str:
        """The directory dontaining the data files (FITS files, in this case)."""
        return self._data_dir

    @property
    def exposure_filenames(self) -> List[str]:
        """The list of filenames of the NISP survey exposure files that are loaded."""
        return self._exposure_filenames

    def __getitem__(self, dither: int) -> NispSurveyExposure:
        """
        Get an item by dither ID

        Parameters
        ----------

        dither: int
            The dither ID (DITHOBS) of the exposure.

        Returns
        -------

        exposure: NispSurveyExposure
            The exposure corresponding to the specified dither ID.

        Raises
        ------

        KeyError: if the dither is not present
        """
        return self._exposures[int(dither)]

    def __iter__(self):
        return ((dither, exposure) for dither, exposure in self._exposures.items())

    def __len__(self):
        return len(self._exposures)

    def __contains__(self, item):
        return item in self._exposures

    def __delitem__(self, dither: int):
        del self._exposures[int(dither)]

    def __str__(self):
        strings = ["  Observation ID    Dither             Date",
                   "-----------------------------------------------------"]

        for dither, exposure in self._exposures.items():
            metadata = exposure.getMetadata()
            obs_id = metadata.get('OBS_ID')
            date = metadata.get('DATE-OBS')
            strings.append(f"{obs_id:^21} {dither:<9} {date}")

        return "\n".join(strings)

    def _log(self, *message):

        if self._logger is not None:
            self._logger.info(" ".join(message))
