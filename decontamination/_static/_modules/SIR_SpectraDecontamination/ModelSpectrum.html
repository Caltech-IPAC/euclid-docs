[docs]class ModelSpectrum(SpectrumContainer):
    """
    A struct-like class for storing a single model of a 2D spectrum.
    """
    __slots__ = ('_order',
                 '_pixels',
                 '_interpolated_pixels',
                 '_interpolated')

    def __init__(self, h5dataset:  'h5py.Dataset'=None):
        """
        Creates an empty ModelSpectrum object.
        """
        super().__init__(h5dataset)
        self._order = None
        self._pixels = None
        self._interpolated_pixels = None
        self._interpolated = None

    def __del__(self):
        self._order = None
        self._pixels = None
        self._interpolated_pixels = None
        self._interpolated = None

    @property
    def order(self) -> int:
        """
        The spectral order of the model (1 or 2) [int].
        """
        if self._order is None and self._h5object is not None:
            self._order = self._h5object.attrs['order']
        return self._order

    @order.setter
    def order(self, order: int):
        if order not in (1, 2):
            raise ValueError("The order must be either 1 or 2.")
        self._order = int(order)

    @property
    def pixels(self):
        """
        The 2D model spectrum of the object [NumPy ndarray(float32)].
        """
        if self._pixels is None and self._h5object is not None:
            self._pixels = np.array(self._h5object)
        return self._pixels

    @pixels.setter
    def pixels(self, pixels):
        if pixels is not None:
            Utils.verify_2d_numpy_array(pixels)
            # discard any negative pixels in the model:
            self._pixels = pixels * (pixels > 1e-3)
        else:
            self._pixels = None

    @property
    def interpolated(self) -> bool:
        if self._interpolated is None and self._h5object is not None:
            self._interpolated = self._h5object.attrs['interp']

        if self._interpolated is None and self.interpolated_pixels is not None:
            self._interpolated = np.any(self.interpolated_pixels)

        return self._interpolated

    @property
    def interpolated_pixels(self):
        """
        The interpolated pixel mask as a NumPy array of bools.

        Returns
        -------

        interpolated_pixels: NumPy ndarray of bools or None

        None is returned if the data has not been set yet. Otherwise, an array of bools is returned. True values in the
        array indicate places in which the pixels of the model are at least partially based upon interpolated fluxes.
        """
        return self._interpolated_pixels

    @interpolated_pixels.setter
    def interpolated_pixels(self, mask):
        Utils.verify_2d_numpy_array(mask, bool)
        self._interpolated_pixels = mask

    @property
    def width(self):
        if self._pixels is not None:
            return self._pixels.shape[1]
        elif self._h5object is not None:
            return self._h5object.shape[1]
        else:
            return None

    @property
    def height(self):
        if self._pixels is not None:
            return self._pixels.shape[0]
        elif self._h5object is not None:
            return self._h5object.shape[0]
        else:
            return None

    def from_spec_info(cls, spec_info):
        """
        Constructs a ModelSpectrum from a SpectrumInfo object

        Parameters
        ----------

        spec_info: SpectrumInfo

        Returns
        -------

        A ModelSpectrum object corresponding to the SpectrumInfo object. If the SpectrumInfo object already has a
        ModelSpectrum, that object is returned instead of constructing a new one.
        """
        if spec_info.model_spec is not None:
            return spec_info.model_spec
        else:
            model_spec = cls()
            model_spec.id = spec_info.object_id
            model_spec.order = spec_info.order
            model_spec.offset = spec_info.get_offset()
            model_spec.dither = spec_info.dither
            model_spec.detector = spec_info.detector
            return model_spec
