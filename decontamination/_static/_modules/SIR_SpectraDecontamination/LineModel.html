"""
Contains the LineModeller class, which is used to model an emission line.
"""

from typing import List, Union
import warnings

import numpy as np
from scipy.optimize import curve_fit, OptimizeWarning



[docs]class LineModel(object):
    """This represents an emission line in a 1D spectrum."""

    def __init__(self, wavelength_vectors: List[np.ndarray], flux_vectors: List[np.ndarray]):
        """
        Constructs the line model from one or more spectra.

        Parameters
        ----------

        wavelength_vectors: List[np.ndarray]
           A list of wavelength vectors [wavelengths1, wavelengths2]. The list must contain at least one array of
           wavelengths.

        flux_vectors: List[np.ndarray]
            A list of the flux vectors corresponding to the wavelength vectors listed in wavelengths.

        Raises
        ------

        ValueError
            If too few or too many spectra are provided, or the number of wavelength elements does not match the number
            of flux elements.
        """
        if len(wavelength_vectors) != len(flux_vectors):
            raise ValueError('The number of wavelength vectors does not match the number of flux vectors.')

        if len(wavelength_vectors) == 0:
            raise ValueError('The wavelength vector list is empty.')

        min_wav = 1e15
        max_wav = 0

        for i, wav_vec in enumerate(wavelength_vectors):
            flux_vec = flux_vectors[i]
            if len(wav_vec) != len(flux_vec) or len(wav_vec) < 2:
                raise ValueError('The wavelength and flux vectors must have the same length and  must be > 1'
                                 f'len(flux_vectors[{i}]) = {len(flux_vec)}, len(wavelength_vectors[{i}])'
                                 f' = {len(wav_vec)}')

            min_wav = min(min_wav, wav_vec.min())
            max_wav = max(max_wav, wav_vec.max())

        # the minimum and maximum wavelengths of the modelled region, containing the line
        self._wavelength_range = (min_wav, max_wav)

        self._dlambda = wavelength_vectors[0][1] - wavelength_vectors[0][0]

        # the amplitude, center, and half-width of the line
        self._coeffs = self._find_coeffs(wavelength_vectors, flux_vectors)

        warnings.filterwarnings("ignore", category=OptimizeWarning)

    @property
    def wavelength_range(self):
        return self._wavelength_range


[docs]    def eval(self, wavelengths):
        """
        Evaluates the line model at the specified wavelengths.

        Parameters
        ----------

        wavelengths: 1-D NumPy array
            The wavelengths of the model, in Angstroms

        Returns
        -------

        model line fluxes: 1-D NumPy array
            The fluxes corresponding to the emission line represented by this `LineModel`, in detector counts per
            angstrom.
        """
        amplitude, center, width = self._coeffs
        x = wavelengths - center
        mask = np.fabs(x) < 5 * width ** 0.5
        output = np.zeros_like(wavelengths)
        output[mask] = self._gaussian_line(wavelengths[mask], amplitude, center, width)

        return output


    def _find_coeffs(self, wavelength_vectors, flux_vectors):
        """
        Finds the optimal amplitude, central wavelength, and half-width of the model line.

        Parameters
        ----------

        wavelength_vectors: list
           A list of wavelength vectors [wavelengths1, wavelengths2, ...]. The list must contain at least one array of
           wavelengths.

        flux_vectors: list
            A list of the flux vectors corresponding to the wavelength vectors listed in wavelength. The fluxes
            are expected to be in units of detector counts per Angstrom.

        Returns
        -------

        tuple:
            The best-fit parameters (amplitude, central_wavelength, and a width parameter)

        Raises
        ------

        ValueError: if the number of flux values and wavelength values do not agree.
        """
        wavs = np.concatenate(wavelength_vectors)
        fluxes = np.concatenate(flux_vectors)

        if wavs.size != fluxes.size:
            raise ValueError("The number of wavelength entries must match the number of flux entries")

        guess_amp = fluxes.max()
        guess_center = wavs[fluxes.argmax()]
        initial_guess = (guess_amp, guess_center, 12.0)
        bounds = ((2.5 * fluxes.min(), wavs.min(), 0.5), (1.5 * fluxes.max(), wavs.max(), 130.0))

        uncertainties = 0.1 * np.sqrt(np.std(fluxes) / (1e-50 + np.abs(fluxes - np.median(fluxes))))

        # clamp the wings to zero:

        wavs = np.concatenate([[wavs.min() - 0.1], wavs, [wavs.max()+0.1]])
        fluxes = np.concatenate([[0], fluxes, [0]])
        uncertainties = np.concatenate([[1e-50], uncertainties, [1e-50]])

        try:
            fit_params, _ = curve_fit(self._gaussian_line, wavs, fluxes,
                                      p0=initial_guess,
                                      sigma=uncertainties,
                                      bounds=bounds,
                                      method='dogbox')
        except RuntimeError:
            # FIXME: breakpoint here and check to see why the fit didn't work. fix if possible.
            fit_params = initial_guess

        amplitude, center, width = fit_params

        return amplitude, center, width

    @staticmethod
    def _gaussian_line(wav: Union[float, np.ndarray],
                       amplitude: float,
                       center: float,
                       width: float) -> Union[np.ndarray, float]:
        """
        Computes the Gaussian: amplitude * exp{-(wav - center)^2 / width)}

        Parameters
        ----------
        wav: float or np.ndarray
          The wavelength value(s) for which to compute the value of the Gaussian line profile.

        amplitude: float
            The amplitude of the Gaussain line profile.

        center: float
            The wavelength of the center of the Gaussian line profile.

        width: float
            The variance of the Gaussian line profile (proportional to the square of the width of the line)

        Returns
        -------

        line_profile: np.ndarray or float
            The value(s) of the Gaussian profile at the input wavelength values.
        """
        inverse_width = 1.0 / width

        return amplitude * np.exp(-inverse_width * (wav - center)**2)
