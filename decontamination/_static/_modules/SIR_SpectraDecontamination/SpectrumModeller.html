[docs]class SpectrumModeller(object):
    """
    Constructs one-dimensional and two-dimensional spectral models.

    Given information about the grisms and detectors (via `GrismDetector` objects for each dither), positions of objects
    and contaminants (via `SearchGrid` objects), and information about a particular physical object (via an
    `AstroObjectInfo` object), this class constructs the one-dimensional and two-dimensional spectra of an object, with
    the help of the `SpectrumMeasurer` class, which performs the low-level measurements of spectral segments.
    """

    def __init__(self, photo_info: 'PhotometerInfo',
                 detector_grisms: dict,
                 search_grids: 'NestedDict',
                 padding: int):
        """
        Constructs the SpectrumModeller.

        Parameters
        ----------

        photo_info: PhotometerInfo
            An object containing information about the photometric filters, Y, J, and H, which are used to construct
            models when there is insufficient clean spectral data in the 2D spectral images to create a more detailed
            model.

        detector_grisms: dict
            A dictionary in the format `{dither: DetectorGrism}`, where `dither` is an integer (1, 2, 3, 4)
            specifying the number of an exposure within the dither sequence and `DetectorGrism` is the
            `DetectorGrism` object associated with the specified dither.

        search_grids: NestedDict
            A NestedDict in format `{dither: {detector: SearchGrid}}`, where `dither` and `detector` are integers
            specifying the exposure number within the dither sequence and the detector number. The `SearchGrid`
            objects, stored in `search_grids`, must be fully-populated with `SpectrumInfo` objects.

        padding: int
            The number of pixels that should be inserted between the bounding box of each spectral cutout
            (specified in a `SpectrumInfo` object) and the rectangle defined by the `LocationSpectrum` object
            in the LocationTable.
        """
        self._filters = photo_info
        self._grisms = detector_grisms
        self._search_grids = search_grids
        self._padding = padding
        self._reflux = None
        self._relflux_corrector = dict()

        # Create the common wavelength vector that will be used when handling high-resolution spectra.

        if len(detector_grisms) == 0:
            raise ValueError("The dictionary of detectors must contain at least one entry. Are the detectors empty?")

        sample_grism = list(detector_grisms.values())[0]

        min_wav, max_wav = sample_grism.full_wavelength_range

        wavelength_step = 0.99 * sample_grism.angstroms_per_pixel

        self._lambdas = np.arange(min_wav, max_wav + 1, wavelength_step, dtype=np.float32)

        # Set up the SpectrumMeasurer, which will be used to measure fluxes from the 2-D spectra

        self._metric = SpectrumMeasurer(self)

        # Get information about the relevant filters, which will be used for validating the 1D model spectra as
        # well as for creating 1D model spectra, when the 2-D spectral information is inadequate.

        self._lower_band = 'Y'
        self._central_band = 'J'
        self._upper_band = 'H'
        self._lower_effective = photo_info.effective_wavelength(self._lower_band)
        self._central_effective = photo_info.effective_wavelength(self._central_band)
        self._upper_effective = photo_info.effective_wavelength(self._upper_band)
        self._lower_transmission_wavs, self._lower_transmission = photo_info.filter_transmission(self._lower_band)
        self._central_transmission_wavs, self._central_transmission = photo_info.filter_transmission(self._central_band)
        self._upper_transmission_wavs, self._upper_transmission = photo_info.filter_transmission(self._upper_band)

        self._logger = log.getLogger('SpectrumModeller')

        warnings.filterwarnings("ignore", category=OptimizeWarning)

    @property
    def lambdas(self):
        """Wavelengths used by all 1D model spectra (Angstroms)."""
        return self._lambdas

    @property
    def grisms(self):
        return self._grisms

    @property
    def search_grids(self):
        return self._search_grids

    @property
    def padding(self) -> int:
        return self._padding


[docs]    def get_2d_model_spectrum(self, spectrum: 'SpectrumInfo', skip_measuring: bool=False) -> 'ModelSpectrum':
        """
        Returns a 2D model of the specified spectrum. If the model has already been computed, then the
        previously-computed result is returned. If the model has not been computed, then it is computed and
        then returned.

        Parameters
        ----------

        spectrum : SpectrumInfo
            A SpectrumInfo object describing the 2-D spectrum.

        skip_measuring : bool
           If set to True, the model will be based solely on the lower and upper band magnitudes; the process of
           attempting to measure the spectrum will be skipped.

        Returns
        -------

        model_spec: ModelSpectrum.ModelSpectrum or None
            The modelled 2-D spectrum of the contaminant. If this is a zeroth-order spectrum, None is returned
            because we do not model zeroth-order spectra; we only flag their positions. Also returns None if the
            magnitudes are missing / invalid.

        Side Effects
        ------------

        Potentially modifies the input parameter,`spectrum`. If the `spectrum` has already been modelled, then no
        modifications are made.
        """
        if spectrum.order == 0 or spectrum.area == 0 or spectrum.astro_object_info.invalid_mag:
            return None

        if spectrum.model_spec is not None:
            # The spectrum has already been modelled, return the model.
            return spectrum.model_spec
        else:
            # The spectrum has not yet been modelled; compute the model, then return it:
            return self._make_2d_model(spectrum, skip_measuring)

        """Set the relative flux correction flat FITS file object."""

    def _get_relflux_corrector(self, spectrum: 'SpectrumInfo') -> RelativeFluxCorrectionCore:
        if self._reflux is None:
            return None
        gwa_pos_name = GrismModel.getPositionName(spectrum.location_spectrum.getGWAPosition())

        tilt = spectrum.location_spectrum.getGWATilt()

        grism_name = gwa_pos_name if tilt == 0 else f"{gwa_pos_name}{tilt:+}"

        if grism_name not in self._relflux_corrector:
            corrector = RelativeFluxCorrectionCore(self._reflux[f'{grism_name}.SCI'].data,
                                                   self._reflux[f'{grism_name}.DQ'].data)
            self._relflux_corrector[grism_name] = corrector

        return self._relflux_corrector[grism_name]

    def _make_2d_model(self, spectrum, skip_measuring: bool=False):
        """
        Builds a 2D model of the specified spectrum, using sections of clean spectra from other dithers or from
        photometric data. This function updates the `model_spec` of the input spectrum.

        Parameters
        ----------

        spectrum : SpectrumInfo.SpectrumInfo
            A SpectrumInfo object describing the 2-D spectrum.

        skip_measuring : bool
           If set to True, the model will be based solely on the lower and upper band magnitudes; the process of
           attempting to measure the spectrum will be skipped.

        Returns
        -------

        model_spectrum: ModelSpectrum
           A ModelSpectrum object with the 2-D model populated.

        Side Effects
        ------------

        Modifies the input `spectrum` object by adding the model spectrum.

        Modifies the `spectrum.object_info` by adding `.continuum`, `.lines`, and `.interp_mask`.
        """
        object_info = spectrum.astro_object_info

        corrector = self._get_relflux_corrector(spectrum)

        if corrector is not None:
            spectrum.relflux_correction, _, _ = corrector.compute_correction(self.lambdas, spectrum.location_spectrum)

        if object_info.continuum is None:
            object_info.continuum = self._continuum_from_band_fluxes(object_info)
            spectrum.make_2d_model(self.lambdas)

        if spectrum.model_spec is None:
            spectrum.make_2d_model(self.lambdas)

        original_continuum = object_info.continuum

        if not object_info.has_been_measured and not skip_measuring:
            object_info.continuum, object_info.lines, object_info.interp_mask = self._make_1d_spec(object_info)
            object_info.has_been_measured = True

            if object_info.continuum is not original_continuum or object_info.lines is not None:
                spectrum.make_2d_model(self.lambdas)

        if (object_info.has_been_measured or skip_measuring) and spectrum.model_spec is not None:
            # The final 2D model has been created, so we don't need the history of the convolution kernel.
            spectrum.clear_convolution_kernel()

        return spectrum.model_spec

    def _make_1d_spec(self, object_info: 'AstroObjectInfo') -> Tuple[np.ndarray, 'LineModeller', bool]:
        """
        Builds the 1-D model spectrum of the specified object, using clean regions of the spectra.

        Parameters
        ----------

        object_info: AstroObjectInfo
            The AstroObjectInfo object associated with the galaxy or star of interest.

        Returns
        -------

        tuple: (continuum_flux_density,  line_model, interpolation_mask)

        continuum flux density: 1-D NumPy array of floats
            The continuum flux densities corresponding to the wavelengths stored in self.lambdas.

        line_model: LineModeller
            The LineModeller object for modelling the emission lines for the spectrum.

        interpolation_mask: np.ndarray or bool
            a single bool value or a 1-D NumPy array of bools, with one entry per wavelength bin (i.e., each entry in
            self.lambdas.)
        """
        measured_fluxes = self._metric.measure_spectra(object_info)

        if measured_fluxes is not None:
            continuum, line_modeller = measured_fluxes

            lines = self._eval_lines(line_modeller, object_info)

            if self._is_model_spectrum_consistent(self.lambdas, continuum + lines, object_info):
                # The spectrum was measured and looks reasonable, but the photometry may have been used for the
                # continuum. So, the interpolation mask may still be True, if the interp_mask has been set:
                interpolation_mask = object_info.interp_mask
            else:
                continuum = object_info.continuum
                line_modeller = None
                interpolation_mask = True
        else:
            continuum = object_info.continuum
            line_modeller = None
            interpolation_mask = True

        return continuum, line_modeller, interpolation_mask

    def _is_model_spectrum_consistent(self, wavelengths, fluxes, object_info, end_slope_limit=1e-18):
        """
        Checks whether the spectrum, defined by `wavs` and `fluxes`, is consistent with the broadband
        fluxes in the catalog.

        Parameters
        ----------

        wavelengths: one-dimensional NumPy array of floats
            The wavelengths of the spectrum, in Angstroms.

        fluxes: one-dimensional NumPy array of floats
            The flux density values of the spectrum, corresponding to the wavelengths listed in `wavelengths`.

        object_info: AstroObjectInfo
            The AstroObjectInfo object associated with the spectrum.

        end_slope_limit: float
            The maximum allowed magnitude of the slope on the ends of the spectrum. This is used to discard
            models that 'blow up' on one end.

        Returns
        -------

        bool
            True if convolving the spectrum with the broadband filters yields fluxes that are consistent with the
            flux values in the photometric catalog and the end-point slopes are within the bound specified by
            `end_slope_limit`.
        """
        min_wav = self._lambdas[0]
        max_wav = self._lambdas[-1]

        if self._lower_effective < min_wav:
            # The red grism
            left_band = self._central_band
            right_band = self._upper_band
            left_transmission_wavs = self._central_transmission_wavs
            left_transmission = self._central_transmission
            right_transmission_wavs = self._upper_transmission_wavs
            right_transmission = self._upper_transmission
        elif self._upper_effective > max_wav:
            # the blue grism
            left_band = self._lower_band
            right_band = self._central_band
            left_transmission_wavs = self._lower_transmission_wavs
            left_transmission = self._lower_transmission
            right_transmission_wavs = self._central_transmission_wavs
            right_transmission = self._central_transmission
        else:
            raise ValueError("Grism wavelengths and photometric band wavelengths are incompatible.")

        nominal_left_flux = object_info.flux(left_band)
        nominal_right_flux = object_info.flux(right_band)

        left_flux_uncertainty = object_info.dflux(left_band)
        right_flux_uncertainty = object_info.dflux(right_band)

        left_consistent = self._flux_is_consistent(wavelengths, fluxes, nominal_left_flux, left_flux_uncertainty,
                                                   left_transmission_wavs,
                                                   left_transmission)

        right_consistent = self._flux_is_consistent(wavelengths, fluxes, nominal_right_flux, right_flux_uncertainty,
                                                    right_transmission_wavs,
                                                    right_transmission)

        # verify that the slopes of the left and right side of the spectrum are reasonable:
        slopes = np.diff(fluxes) / np.diff(wavelengths)

        slopes_reasonable = np.fabs(slopes[0]) < end_slope_limit and np.fabs(slopes[-1]) < end_slope_limit

        return left_consistent and right_consistent and slopes_reasonable

    @staticmethod
    def _flux_is_consistent(wavelengths, fluxes, nominal_flux, flux_uncertainty, transmission_wavs, transmission):

        model_flux = Utils.convolve_filter(wavelengths, fluxes, transmission_wavs, transmission)

        variance = (0.1 * model_flux) ** 2

        flux_uncertainty = min(3 * flux_uncertainty, np.fabs(0.1 * nominal_flux))

        sigma = np.sqrt(variance + flux_uncertainty ** 2)

        return np.fabs(nominal_flux - model_flux) < sigma

    def _continuum_from_band_fluxes(self, object_info) -> 'np.ndarray':
        """
        Creates a model of the continuum flux, based solely upon the object magnitudes.

        Parameters
        ----------

        object_info: AstroObjectInfo
            The AstroObjectInfo object associated with the spectrum.

        Returns
        -------

        continuum_flux: 1-D NumPy array of floats
            The continuum flux densities corresponding to the wavelengths stored in self.lambdas.
        """
        lower_wavelengths = np.array([self._lower_effective, self._central_effective])

        lower_phot_fluxes = np.array([object_info.flux(self._lower_band), object_info.flux(self._central_band)])

        lower_continuum = self.power_law_interp(lower_wavelengths, lower_phot_fluxes, self.lambdas)

        upper_wavelengths = np.array([self._central_effective, self._upper_effective])

        upper_phot_fluxes = np.array([object_info.flux(self._central_band), object_info.flux(self._upper_band)])

        upper_continuum = self.power_law_interp(upper_wavelengths, upper_phot_fluxes, self.lambdas)

        if lower_continuum is None or upper_continuum is None:
            # there is an issue with one of the broadband fluxes, preventing the interpolator from producing a model.
            self._logger.warning(f"Broadband fluxes of {object_info.object_id} could not be interpolated.")
            if lower_continuum is None and upper_continuum is None:
                # neither end could be fit; using a generic flat spectrum as a fallback.
                self._logger.warning(f"Using a flat dummy spectrum to model object {object_info.object_id}.")
                lower_continuum = np.ones_like(self.lambdas) * 1e-18
                upper_continuum = lower_continuum
            elif lower_continuum is None:
                lower_continuum = upper_continuum
            elif upper_continuum is None:
                upper_continuum = lower_continuum

        if object_info.has_2mass_mags:
            # In this case, only the 2MASS J and H fluxes are used, which happens to be the upper_continuum
            # this will need to be updated when the actual 2MASS filter transmission curves are used because
            # the effective wavelength of the filters are slightly different from the Euclid filters.
            return upper_continuum

        lower_mask = self.lambdas < self._central_effective

        upper_mask = self.lambdas >= self._central_effective

        return lower_mask * lower_continuum + upper_mask * upper_continuum


[docs]    @staticmethod
    def power_law_interp(known_wavelengths: np.ndarray,
                         known_fluxes: np.ndarray,
                         interp_wavelengths: np.ndarray,
                         known_flux_uncertainties: Union[np.ndarray, None] = None):
        """
        Interpolates the known values, at the points given by `known_wavelengths`, assuming that the data follows a
        piecewise power-law. Extrapolates beyond the known_wavelengths range so that fluxes are computed for all
        values of interp_wavelengths are computed.

        Parameters
        ----------

        known_wavelengths: 1-D NumPy array
            Wavelength values associated with known flux values in `known_fluxes`.

        known_fluxes: 1-D NumPy array
            The known flux values, at the corresponding `known_wavelengths`.

        interp_wavelengths: 1-D NumPy array
            Independent values for which you want to compute the interpolated or extrapolated
            flux values.

        known_flux_uncertainties: 1-D NumPy array or None
            The uncertainties of the flux values in `known_fluxes`.

        Returns
        -------

        y_interp: NumPy ndarray or None

        The interpolated values of the flux at the wavelengths specified by the `interp_wavelengths` input array.
        If the data being interpolated consisted of zeros or there were not at least 2 data points in the interior
        of the region.

        Raises
        ------

        ValueError
            Raised if there are fewer than 2 data points in the input arrays or any of the wavelengths are negative.
        """
        if len(known_wavelengths) != len(known_fluxes) < 2:
            raise ValueError("The known input data must contain at least two data points and "
                             "the lengths of the input arrays must match.")

        if np.any(known_wavelengths <= 0) or np.any(interp_wavelengths <= 0):
            raise ValueError("wavelength values must be > 0.")

        # Make sure that known fluxes are > 0

        valid_ids = np.where(known_fluxes > 0.0)

        if len(valid_ids[0]) < 2:
            # cannot interpolate with these data points
            return None

        known_fluxes = known_fluxes[valid_ids]
        known_wavelengths = known_wavelengths[valid_ids]

        xval = np.log(known_wavelengths)
        yval = np.log(known_fluxes)
        yval_error = None if known_flux_uncertainties is None else known_flux_uncertainties[valid_ids] / np.abs(yval)

        def line(variable, intercept, slope):
            return intercept + slope * variable

        try:
            fit_params, _ = curve_fit(line, xval, yval, sigma=yval_error)
        except RuntimeError:
            fit_params = None

        if fit_params is not None:
            best_fit_intercept, best_fit_slope = fit_params

            y = line(np.log(interp_wavelengths), best_fit_intercept, best_fit_slope)

            # if y.max() would lead to extremely large fluxes, return None
            return np.exp(y) if y.max() < -5 else None
        else:
            return None


    @staticmethod
    def _eval_lines(line_modeller: 'LineModeller', object_info: 'AstroObjectInfo') -> Union[np.ndarray, None]:
        lines = np.zeros_like(object_info.WAVELENGTHS)
        if line_modeller is not None and line_modeller.line_models is not None:
            for line_model in line_modeller.line_models:
                lines += line_model.eval(object_info.WAVELENGTHS)
        return lines
