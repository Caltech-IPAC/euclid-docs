Contains the DetectorRegion class, which references a region of a detector without duplicating data.

import SIR_SpectraDecontamination.Utils as Utils



[docs]class DetectorRegion(object):
    """
    Manages a region of a NISP Detector Exposure in a memory-efficient way.
    """

    __slots__ = ['_detector',
                 '_bbox',
                 '_region',
                 '__weakref__']

    def __init__(self, nisp_detector, bbox):
        """
        Constructs the object.

        Parameters
        ----------

        nisp_detector:
            The NISP Detector from which the region is being fetched.

        bbox: tuple
            The bounding box, (left, bottom, width, height), of the region of interest in detector pixel
            coordinates. If the bbox extends beyond the bounds of the detector, only the region within the detector
            will be considered.
        """

        self._detector = nisp_detector

        det_width = nisp_detector.getScience().getWidth()
        det_height = nisp_detector.getScience().getHeight()

        overlap_box, _ = Utils.bounding_indices((0, 0, det_width, det_height), bbox)

        if overlap_box is None:
            raise ValueError(f"The bounding box {bbox} must be at least partially within the detector")

        left, bottom, right, top = overlap_box

        # the bounding box of the bbox region in the coordinate system of the detector:
        self._bbox = (left, bottom, right - left, top - bottom)

        # the 2D region (y-slice, x-slice) within the detector corresponding to the bbox
        self._region = (slice(bottom, top), slice(left, right))

    def __del__(self):
        self._detector = None
        self._bbox = None
        self._region = None

    @property
    def nisp_detector(self):
        """Returns a reference to the fill NISP detector."""
        return self._detector

    @property
    def bbox(self):
        """Returns the bounding box of the region of interest within the detector."""
        return self._bbox
