    def __init__(self,
                 collection: 'DecontaminatedSpectraCollection',
                 spectrum_cutout: SpectrumCutout=None,
                 h5group: 'h5py.Group'=None):
        """
        Creates a DecontaminatedSpectrum object from a SpectrumCutout object.

        Parameters
        ----------

        spectrum_cutout: SpectrumCutout
            A `SpectrumCutout object`, created by the SpectraDecontaminator.
        """
        super().__init__(h5group)
        self._collection = collection
        self._contamination = None
        self._wavelengths = None
        self._kernel = None
        self._gaussian_sigma_major = None
        self._gaussian_sigma_minor = None
        self._gaussian_theta = None
        self._boundary = None

        if spectrum_cutout is not None:
            xoff, yoff = spectrum_cutout.spec_info.get_offset()

            if spectrum_cutout.contaminants is not None:

                n_contaminants = len(spectrum_cutout.contaminants)

                contaminants = np.zeros((n_contaminants,), dtype=[('id', np.int64), ('order', np.int8)])

                for i, contam in enumerate(spectrum_cutout.contaminants):
                    contaminants[i]['id'] = contam.object_id
                    contaminants[i]['order'] = contam.order
            else:
                contaminants = None

            self.id = spectrum_cutout.spec_info.object_id
            self.dither = spectrum_cutout.spec_info.dither
            self.detector = spectrum_cutout.spec_info.detector
            self.x_offset = xoff
            self.y_offset = yoff

            self.science = spectrum_cutout.science
            self.variance = spectrum_cutout.variance
            self.mask = spectrum_cutout.flags
            self.kernel = spectrum_cutout.spec_info.const_kernel
            if self.kernel is not None:
                self.gaussian_sigma_major = spectrum_cutout.spec_info.gaussian_sigma_major
                self.gaussian_sigma_minor = spectrum_cutout.spec_info.gaussian_sigma_minor
                self.gaussian_theta = spectrum_cutout.spec_info.gaussian_theta

            self.contaminants = contaminants
            self.location = spectrum_cutout.spec_info.location_spectrum
            self._boundary = spectrum_cutout.spec_info.boundary
            self.y_mag = spectrum_cutout.spec_info.astro_object_info.mag('Y')
            self.j_mag = spectrum_cutout.spec_info.astro_object_info.mag('J')
            self.h_mag = spectrum_cutout.spec_info.astro_object_info.mag('H')
            self.y_flux = spectrum_cutout.spec_info.astro_object_info.flux('Y')
            self.j_flux = spectrum_cutout.spec_info.astro_object_info.flux('J')
            self.h_flux = spectrum_cutout.spec_info.astro_object_info.flux('H')
        else:
            assert h5group is not None, "An HDF5 Group must be provided because no SpectrumCutout was provided."
            self._science = None
            self._variance = None
            self._mask = None
            self._contaminants = None
            self._location = None
            self._boundary = None
            self._y_mag = None
            self._j_mag = None
            self._h_mag = None
            self._y_flux = None
            self._j_flux = None
            self._h_flux = None

    @property
    def science(self):
        """
        The decontaminated science layer [NumPy ndarray, float32].
        """
        if self._science is None and self._h5object is not None:
            self._science = np.array(self._h5object['science'])
        return self._science

    @science.setter
    def science(self, sci):
        Utils.verify_2d_numpy_array(sci)
        self._science = sci

    @property
    def variance(self):
        """
        Variance of the decontaminated science layer [NumPy ndarray, float32].
        """
        if self._variance is None and self._h5object is not None:
            self._variance = np.array(self._h5object['variance'])
        return self._variance

    @variance.setter

    @property
    def mask(self) -> np.ndarray:
        """
        The mask layer, containing the decontamination flags as well as the flags that were present in the original
        image [NumPy ndarray, uint32]. Pixels that are outside of the spectrogram are flagged as NULL (Flags.VOID) to
        indicate that these pixels are not real; they are only fillers.
        """
        if self._mask is None and self._h5object is not None:
            mask = np.array(self._h5object['mask'])
            mask[self.variance == 0] = Flags.VOID
            mask[self.variance == np.inf] = Flags.INVALID
            self._mask = mask
        return self._mask

    @mask.setter
    def mask(self, mask: np.ndarray):
        Utils.verify_2d_numpy_array(mask)
        if self._variance is not None:
            mask[self.variance == 0] = Flags.VOID
            mask[self.variance == np.inf] = Flags.INVALID
        else:
            raise RuntimeError("The variance of this spectrum must be set before the mask can be set.")
        self._mask = mask

    @property
    def contamination(self) -> np.ndarray:
        """
        The total contamination for this spectrum (the combined model spectra of all spectra that overlap with this
        spectrum) [NumPy ndarray, float32].
        """
        if self._contamination is None:
            self._contamination = self._compute_total_contamination()
        return self._contamination

    @property

[docs]    def rasterize_polygon(self, polygon: Polygon) -> np.ndarray:
        """
        Creates a rasterized mask of a polygon: pixels within the polygon are True. False otherwise.

        This is useful for displaying which pixels are contaminated by a spectrum: simply provide the polygon
        of the contaminating spectrum to see which pixels are contaminated by that spectrum.

        Parameters
        ---------

        polygon: Polygon
            A Shapely Polygon object that will be masked. The vertex coordinates should be detector coordinates.

        Returns
        -------

        mask: np.ndarray
            An array with the same dimensions as the pixel layers of the spectrum. Pixels are set to True inside
            of the polygon, False elsewhere.
        """
        mask = np.zeros(shape=self.science.shape, dtype=bool)
        # The polygon is expanded by half of the maximum diameter of a pixel to account for pixels on the edge.
        # 0.5 * sqrt(2) is approximately 0.7071
        vertices = np.array(polygon.buffer(0.7071, join_style=2).boundary.coords)[:-1]
        return Utils.rasterize(mask, self.offset, vertices)


    def _compute_total_contamination(self):
        """
        Compute the total contamination for the spectrum

        Returns
        -------

        2D NumPy array
           The sum of all of the contaminants of the specified spectrum for each pixel of the spectrum.

        Raises
        ------

        ValueError
            If a model mentioned in the table of contaminants cannot be located.
        """
        contamination = np.zeros_like(self.science)
        contamination_canvas = (self.x_offset, self.y_offset), contamination

        for contaminant in self.contaminants:
            model_id = contaminant['id']
            model_order = contaminant['order']

            # we do not attempt to model the zeroth-order spectra:
            if model_order != 0:
                contam = self.collection.get_model(self.dither, self.detector, model_id, model_order)
                if contam is not None:
                    contaminant_overlay = (contam.x_offset, contam.y_offset), contam.pixels
                    Utils.apply_overlay(overlay=contaminant_overlay, canvas=contamination_canvas, operation=np.add)

        return contamination

    def _location_from_h5group(self, group: 'h5py.Group') -> LocationSpectrum:
        location_group = group['location']
        gwa_tilt = int(location_group.attrs['gwa_tilt'])
        extra_tilt = float(location_group.attrs['extra_tilt'])
        reference_position = location_group['ref_position']
        reference_wavelength = float(reference_position.attrs['ref_wavelength'])

        lrange = LambdaRange()
        l_start = float(location_group.attrs['wavelength_start'])
        l_end = float(location_group.attrs['wavelength_end'])
        lrange.setWavelengthStart(l_start)
        lrange.setWavelengthEnd(l_end)

        ids = ChebLocalModel(np.array(location_group['ids_coeffs']))
        ids_start = float(location_group['ids_coeffs'].attrs['wavelength_start'])
        ids_end = float(location_group['ids_coeffs'].attrs['wavelength_end'])
        ids.setRange(ids_start, ids_end)

        crv = ChebLocalModel(np.array(location_group['crv_coeffs']))
        crv_start = float(location_group['crv_coeffs'].attrs['wavelength_start'])
        crv_end = float(location_group['crv_coeffs'].attrs['wavelength_end'])
        crv.setRange(crv_start, crv_end)

        detector_model = self._collection.detector_models[(self.dither, self.detector)]

        x_pivot = float(location_group.attrs['x_pivot'])
        y_pivot = float(location_group.attrs['y_pivot'])

        loc = LocationSpectrum()
        loc.setObjectID(self.id)
        loc.setDetector(DetectorModel.getDetectorName(self.detector))
        loc.setFieldID(int(group.file.attrs[H5LocationTable.FIELD_ID_LABEL]))
        loc.setObservationID(int(group.file.attrs[H5LocationTable.OBSERVATION_ID_LABEL]))
        loc.setPointingID(int(group.file.attrs[H5LocationTable.POINTING_ID_LABEL]))
        loc.setGWAPosition(GrismModel.getPositionNumber(group.file.attrs[H5LocationTable.GWA_POSITION_LABEL]))
        loc.setOrder(1)
        loc.setLambdaRange(lrange)
        loc.setIDSModel(ids)
        loc.setCRVModel(crv)
        loc.setLambdaRefValue(reference_wavelength)
        loc.setLambdaRefPosition(np.array(reference_position))
        loc.setGWATilt(gwa_tilt)
        loc.setExtraTilt(extra_tilt)
        loc.setDetectorModel(detector_model)
        loc.setPivot(x_pivot, y_pivot)

        return loc
